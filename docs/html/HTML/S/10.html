  63 /*
  64  * A simple function to strip the ethernet/ip/tcp header from an 
  65  * openflow packet. 
  66  * @param b the data of the packet
  67  * @param len the size of the packet
  68  */
  69 int 
  70 parse_ip_packet_header(const void *b, int len, struct flow *fl) {
  71   // assume we have ethernet packets.
  72   // skip first bytes of the ether because they are
  73   // in the simple case of static length.
  74   if (len < sizeof(struct ether_header))
  75     return -1;
  76   b = b + sizeof(struct ether_header);
  77   len -= sizeof(struct ether_header);
  78   if (len < sizeof(struct iphdr))
  79     return -1;
  80   struct iphdr *ip_p = (struct iphdr *) b;
  81   if (len < 4*ip_p->ihl)
  82     return -1;
  83   b = b + 4*ip_p->ihl;
  84   len -=  4*ip_p->ihl;
  85   
  86   if(ip_p->protocol != IPPROTO_TCP)
  87     return -1;
  88   fl->nw_src = ip_p->saddr;
  89   fl->nw_dst = ip_p->daddr;
  90   
  91   if(len <  sizeof(struct tcphdr))
  92     return -1;
  93   struct tcphdr *tcp_p = (struct tcphdr *)b;
  94   if (len < 4*tcp_p->doff)
  95     return -1;
  96   b = b + 4*tcp_p->doff;
  97   len -=  4*tcp_p->doff;
  98   fl->tp_src = tcp_p->source;
  99   fl->tp_dst = tcp_p->dest;
 100   return sizeof(struct ether_header) + 4*ip_p->ihl + 4*tcp_p->doff;
 101 }

 157 /*
 158  * A function the creates a simple flow modification message 
 159  * based on the content of the  flow structure and the mask details.
 160  * @param ofp The bufer where we create the packet.
 161  * @param command the type of message we want to create. 
 162  * @param flow The flow structure from we create the match rule.
 163  * @param mask T
 164  */
 165 int
 166 make_flow_mod(void *ofp, uint16_t command, uint32_t len, 
 167               struct flow *flow) {
 168   struct ofp_flow_mod *ofm = ofp;
 169   memset(ofp, 0, len);
 170   ofm->header.version = OFP_VERSION;
 171   ofm->header.type = OFPT_FLOW_MOD;
 172   ofm->header.length = htons(len);
 173   ofm->match.wildcards = htonl(flow->mask);
 174   ofm->match.in_port = flow->in_port;
 175   memcpy(ofm->match.dl_src, flow->dl_src, sizeof ofm->match.dl_src);
 176   memcpy(ofm->match.dl_dst, flow->dl_dst, sizeof ofm->match.dl_dst);
 177   ofm->match.dl_vlan = flow->dl_vlan;
 178   ofm->match.dl_type = flow->dl_type;
 179   ofm->match.nw_src = flow->nw_src;
 180   ofm->match.nw_dst = flow->nw_dst;
 181   ofm->match.nw_proto = flow->nw_proto;
 182   ofm->match.tp_src = flow->tp_src;
 183   ofm->match.tp_dst = flow->tp_dst;
 184   ofm->command = htons(command);
 185   return 0;
 186 }

 188 int
 189 append_data_to_flow(const  void *b, struct pcap_pkthdr hdr) {
 190   size_t len = hdr.caplen;
 191   //struct ofp_header *ofp = NULL;
 192   struct flow fl;
 193   int dir = 0;
 194   
 195   //since this is a packet capture, strip packet from all the l1-l4 headers.
 196   int start = parse_ip_packet_header(b, len, &fl);
 197   if(ntohs(fl.tp_src) < ntohs(fl.tp_dst))
 198     dir = 1; //switch to controller 
 199   if(len - start == 0) 
 200     return -1;
 201 
 202   b += start;
 203   len -= start;
 204   while(buff_size[dir] < content_length[dir] + len) {
 205     buff_size[dir] += INITIAL_BUF_SIZE;
 206     buff[dir] = realloc(buff[dir], buff_size[dir]);
 207   }
 208   //append new packet to the buffer
 209   memcpy(buff[dir] + content_length[dir], b, len);
 210   content_length[dir] += len;
 211   
 212   last_pkt_ts.tv_sec = hdr.ts.tv_sec;
 213   last_pkt_ts.tv_usec = hdr.ts.tv_usec;
 214 
 215   return dir;
 216 }

 337   }

 341     memmove(buff[dir], buff[dir] + count, (content_length[dir] - count));
 342     content_length[dir] -= count;
 343   } else
 344     content_length[dir] = 0;
 345   return ret;

 356 make_ofp_flow_add(void **buferp, struct flow *fl, uint32_t out_port,

 361   *buferp = xmalloc(len);
 362   if(make_flow_mod(*buferp, OFPFC_ADD, len, fl) < 0 )

 364   struct ofp_flow_mod *ofm = *buferp;
 365   p = (struct ofp_action_output *)ofm->actions;
 366   ofm->idle_timeout = htons(idle_timeout);
 367   ofm->hard_timeout = htons(OFP_FLOW_PERMANENT);
 368   ofm->buffer_id = htonl(-1); //buffer_id);

 371   p->len = htons(8);
 372   p->port = htons(out_port);
 373   p->max_len = htons(0);

 538 uint32_t
 539 extract_pkt_id(const char *b, int len) {
 540   struct ether_header *ether = (struct ether_header *)b;
 541   struct ether_vlan_header *ether_vlan = (struct ether_vlan_header *)b;
 542   
 543   //  printf("%x %x\n",ntohl(ether->ether_type),ntohl(ether_vlan->ether_type));
 544 
 545   if( (ntohs(ether->ether_type) == 0x8100) && (ntohs(ether_vlan->ether_type) == 0x0800)) {
 546     b = b + sizeof(struct ether_vlan_header);
 547     len -= sizeof(struct ether_vlan_header);
 548   } else if(ntohs(ether->ether_type) == 0x0800) {
