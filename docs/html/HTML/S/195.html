  52 static void
  53 flush_working(struct datapath *dp)
  54 {
  55         struct sw_flow_key key;
  56         int num_deleted = 0;
  57 
  58         memset(&key, 0, sizeof(key));
  59         key.wildcards = OFPFW_ALL;
  60         num_deleted = chain_delete(dp->chain, &key, OFPP_NONE, 0, 0, 0);
  61 }

  63 static int
  64 protection_callback(struct sw_flow *flow, void *private_)
  65 {
  66         struct emerg_flow_context *private
  67                 = (struct emerg_flow_context *)private_;
  68         struct sw_flow_actions *actions = flow->sf_acts;
  69         struct ofp_match match;
  70         struct sw_flow *tgtflow = NULL;
  71         int error = 0;
  72 
  73         tgtflow = flow_alloc(flow->sf_acts->actions_len);
  74         if (tgtflow == NULL)
  75                 return -ENOBUFS;
  76 
  77         /* Dup w/o idle and hard timeout. */
  78         memset(&match, 0, sizeof(match));
  79         flow_fill_match(&match, &flow->key.flow, flow->key.wildcards);
  80         flow_extract_match(&tgtflow->key, &match);
  81         /* Fill out flow. */
  82         tgtflow->priority = flow->priority;
  83         tgtflow->idle_timeout = OFP_FLOW_PERMANENT;
  84         tgtflow->hard_timeout = OFP_FLOW_PERMANENT;
  85         tgtflow->send_flow_rem = flow->send_flow_rem;
  86         tgtflow->emerg_flow = 0;
  87         flow_setup_actions(tgtflow, actions->actions, actions->actions_len);
  88 
  89         error = chain_insert(private->dp->chain, tgtflow, 0);
  90         if (error)
  91                 flow_free(tgtflow);
  92 
  93         return error;
  94 }

  96 static void
  97 do_protection(struct datapath *dp)
  98 {
  99         struct emerg_flow_context private;
 100         struct sw_flow_key key;
 101         struct sw_table_position position;
 102         struct sw_table *table = dp->chain->emerg_table;
 103         int error = 0;
 104 
 105         memset(&private, 0, sizeof(private));
 106         private.dp = dp;
 107         memset(&key, 0, sizeof(key));
 108         key.wildcards = OFPFW_ALL;
 109         memset(&position, 0, sizeof(position));
 110 
 111         error = table->iterate(table, &key, OFPP_NONE,
 112                                &position, protection_callback, &private);
 113 }

 115 int
 116 private_recv_msg(struct datapath *dp, const struct sender *sender UNUSED,
 117                  const void *ofph)
 118 {
 119         struct private_vxhdr *vxhdr = (struct private_vxhdr *)ofph;
 120         struct private_vxopt *vxopt = (struct private_vxopt *)(vxhdr + 1);
 121         int error = 0;
 122 
 123         switch (ntohs(vxopt->pvo_type)) {
 124         case PRIVATEOPT_PROTOCOL_STATS_REQUEST:
 125         case PRIVATEOPT_PROTOCOL_STATS_REPLY:
 126                 break;
 127         case PRIVATEOPT_EMERG_FLOW_PROTECTION:
 128                 flush_working(dp);
 129                 do_protection(dp);
 130                 break;
 131         case PRIVATEOPT_EMERG_FLOW_RESTORATION:
 132                 /* Nothing to do because we assume that a re-connected
 133                  * controller will do flush current working flow table. */
 134                 break;
 135         default:
 136                 error = -EINVAL;
 137         }
 138 
 139         return error;
 140 }
