 106 /* Searches the 'n_names' in 'names'.  Returns the index of a match for
 107  * 'target', or 'n_names' if no name matches. */
 108 static size_t
 109 search_name_array(const char *target, const char **names, size_t n_names) 
 110 {
 111     size_t i;
 112 
 113     for (i = 0; i < n_names; i++) {
 114         assert(names[i]);

 268 /* Sets the name of the log file used by VLF_FILE to 'file_name', or to the
 269  * default file name if 'file_name' is null.  Returns 0 if successful,
 270  * otherwise a positive errno value. */
 271 int
 272 vlog_set_log_file(const char *file_name)
 273 {
 274     char *old_log_file_name;
 275     enum vlog_module module;
 276     int error;
 277 
 278     /* Close old log file. */
 279     if (log_file) {
 280         VLOG_INFO("closing log file");
 281         fclose(log_file);
 282         log_file = NULL;
 283     }
 284 
 285     /* Update log file name and free old name.  The ordering is important
 286      * because 'file_name' might be 'log_file_name' or some suffix of it. */
 287     old_log_file_name = log_file_name;
 288     log_file_name = (file_name
 289                      ? xstrdup(file_name)
 290                      : xasprintf("%s/%s.log", ofp_logdir, program_name));
 291     free(old_log_file_name);
 292     file_name = NULL;           /* Might have been freed. */
 293 
 294     /* Open new log file and update min_levels[] to reflect whether we actually
 295      * have a log_file. */
 296     log_file = fopen(log_file_name, "a");
 297     for (module = 0; module < VLM_N_MODULES; module++) {
 298         update_min_level(module);
 299     }
 300 
 301     /* Log success or failure. */
 302     if (!log_file) {
 303         VLOG_WARN("failed to open %s for logging: %s",
 304                   log_file_name, strerror(errno));
 305         error = errno;
 306     } else {
 307         VLOG_INFO("opened log file %s", log_file_name);
 308         error = 0;
 309     }
 310 
 311     return error;
 312 }

 323 /* Set debugging levels:
 324  *
 325  *  mod[:facility[:level]] mod2[:facility[:level]] ...
 326  *
 327  * Return null if successful, otherwise an error message that the caller must
 328  * free().
 329  */
 330 char *
 331 vlog_set_levels_from_string(const char *s_)
 332 {
 333     char *save_ptr;
 334     char *s = xstrdup(s_);
 335     char *module, *facility;
 336 
 337     for (module = strtok_r(s, ": \t", &save_ptr); module != NULL;
 338          module = strtok_r(NULL, ": \t", &save_ptr)) {
 339         enum vlog_module e_module;
 340         enum vlog_facility e_facility;
 341 
 342         facility = strtok_r(NULL, ":", &save_ptr);
 343 
 344         if (!facility || !strcmp(facility, "ANY")) {
 345             e_facility = VLF_ANY_FACILITY;
 346         } else {
 347             e_facility = vlog_get_facility_val(facility);
 348             if (e_facility >= VLF_N_FACILITIES) {
 349                 char *msg = xasprintf("unknown facility \"%s\"", facility);
 350                 free(s);
 351                 return msg;
 352             }
 353         }
 354 
 355         if (!strcmp(module, "PATTERN")) {
 356             vlog_set_pattern(e_facility, save_ptr);
 357             break;
 358         } else {
 359             char *level;
 360             enum vlog_level e_level;
 361 
 362             if (!strcmp(module, "ANY")) {
 363                 e_module = VLM_ANY_MODULE;
 364             } else {
 365                 e_module = vlog_get_module_val(module);
 366                 if (e_module >= VLM_N_MODULES) {
 367                     char *msg = xasprintf("unknown module \"%s\"", module);
 368                     free(s);
 369                     return msg;
 370                 }
 371             }
 372 
 373             level = strtok_r(NULL, ":", &save_ptr);
 374             e_level = level ? vlog_get_level_val(level) : VLL_DBG;
 375             if (e_level >= VLL_N_LEVELS) {
 376                 char *msg = xasprintf("unknown level \"%s\"", level);
 377                 free(s);
 378                 return msg;
 379             }
 380 
 381             vlog_set_levels(e_module, e_facility, e_level);
 382         }
 383     }
 384     free(s);
 385     return NULL;
 386 }

 461 static const char *
 462 fetch_braces(const char *p, const char *def, char *out, size_t out_size)
 463 {
 464     if (*p == '{') {
 465         size_t n = strcspn(p + 1, "}");
 466         size_t n_copy = MIN(n, out_size - 1);
 467         memcpy(out, p + 1, n_copy);
 468         out[n_copy] = '\0';
 469         p += n + 2;
 470     } else {
 471         strlcpy(out, def, out_size);
 472     }
 473     return p;
 474 }

 476 static void
 477 format_log_message(enum vlog_module module, enum vlog_level level,
 478                    enum vlog_facility facility, unsigned int msg_num,
 479                    const char *message, va_list args_, struct ds *s)
 480 {
 481     char tmp[128];
 482     va_list args;
 483     const char *p;
 484 
 485     ds_clear(s);
 486     for (p = facilities[facility].pattern; *p != '\0'; ) {
 487         enum { LEFT, RIGHT } justify = RIGHT;
 488         int pad = '0';
 489         size_t length, field, used;
 490 
 491         if (*p != '%') {
 492             ds_put_char(s, *p++);
 493             continue;
 494         }
 495 
 496         p++;
 497         if (*p == '-') {
 498             justify = LEFT;
 499             p++;
 500         }
 501         if (*p == '0') {
 502             pad = '0';
 503             p++;
 504         }
 505         field = 0;
 506         while (isdigit(*p)) {
 507             field = (field * 10) + (*p - '0');
 508             p++;
 509         }
 510 
 511         length = s->length;
 512         switch (*p++) {
 513         case 'A':
 514             ds_put_cstr(s, program_name);
 515             break;
 516         case 'c':
 517             p = fetch_braces(p, "", tmp, sizeof tmp);
 518             ds_put_cstr(s, vlog_get_module_name(module));
 519             break;
 520         case 'd':
 521             p = fetch_braces(p, "%Y-%m-%d %H:%M:%S", tmp, sizeof tmp);
 522             ds_put_strftime(s, tmp, NULL);
 523             break;
 524         case 'm':
 525             va_copy(args, args_);
 526             ds_put_format_valist(s, message, args);
 527             va_end(args);
 528             break;
 529         case 'N':
 530             ds_put_format(s, "%u", msg_num);
 531             break;
 532         case 'n':
 533             ds_put_char(s, '\n');
 534             break;
 535         case 'p':
 536             ds_put_cstr(s, vlog_get_level_name(level));
 537             break;
 538         case 'P':
 539             ds_put_format(s, "%ld", (long int) getpid());
 540             break;
 541         case 'r':
 542             ds_put_format(s, "%lld", time_msec() - boot_time);
 543             break;
 544         default:
 545             ds_put_char(s, p[-1]);
 546             break;
 547         }
 548         used = s->length - length;
 549         if (used < field) {
 550             size_t n_pad = field - used;
 551             if (justify == RIGHT) {
 552                 ds_put_uninit(s, n_pad);
 553                 memmove(&s->string[length + n_pad], &s->string[length], used);
 554                 memset(&s->string[length], pad, n_pad);
 555             } else {
 556                 ds_put_char_multiple(s, pad, n_pad);
 557             }
 558         }
 559     }
 560 }

 562 /* Writes 'message' to the log at the given 'level' and as coming from the
 563  * given 'module'.
 564  *
 565  * Guaranteed to preserve errno. */
 566 void
 567 vlog_valist(enum vlog_module module, enum vlog_level level,
 568             const char *message, va_list args)
 569 {
 570     bool log_to_console = levels[module][VLF_CONSOLE] >= level;
 571     bool log_to_syslog = levels[module][VLF_SYSLOG] >= level;
 572     bool log_to_file = levels[module][VLF_FILE] >= level && log_file;
 573     if (log_to_console || log_to_syslog || log_to_file) {
 574         int save_errno = errno;
 575         static unsigned int msg_num;
 576         struct ds s;
 577 
 578         ds_init(&s);
 579         ds_reserve(&s, 1024);
 580         msg_num++;
 581 
 582         if (log_to_console) {
 583             format_log_message(module, level, VLF_CONSOLE, msg_num,
 584                                message, args, &s);
 585             ds_put_char(&s, '\n');
 586             fputs(ds_cstr(&s), stderr);
 587         }
 588 
 589         if (log_to_syslog) {
 590             int syslog_level = syslog_levels[level];
 591             char *save_ptr = NULL;
 592             char *line;
 593 
 594             format_log_message(module, level, VLF_SYSLOG, msg_num,
 595                                message, args, &s);
 596             for (line = strtok_r(s.string, "\n", &save_ptr); line;
 597                  line = strtok_r(NULL, "\n", &save_ptr)) {
 598                 syslog(syslog_level, "%s", line);
 599             }
 600         }
 601 
 602         if (log_to_file) {
 603             format_log_message(module, level, VLF_FILE, msg_num,
 604                                message, args, &s);
 605             ds_put_char(&s, '\n');
 606             fputs(ds_cstr(&s), log_file);
 607             fflush(log_file);
 608         }
 609 
 610         ds_destroy(&s);
 611         errno = save_errno;
 612     }
 613 }

 625 void
 626 vlog_rate_limit(enum vlog_module module, enum vlog_level level,
 627                 struct vlog_rate_limit *rl, const char *message, ...)
 628 {
 629     va_list args;
 630 
 631     if (!vlog_is_enabled(module, level)) {
 632         return;
 633     }
 634 
 635     if (rl->tokens < VLOG_MSG_TOKENS) {
 636         time_t now = time_now();
 637         if (rl->last_fill > now) {
 638             /* Last filled in the future?  Time must have gone backward, or
 639              * 'rl' has not been used before. */
 640             rl->tokens = rl->burst;
 641         } else if (rl->last_fill < now) {
 642             unsigned int add = sat_mul(rl->rate, now - rl->last_fill);
 643             unsigned int tokens = sat_add(rl->tokens, add);
 644             rl->tokens = MIN(tokens, rl->burst);
 645             rl->last_fill = now;
 646         }
 647         if (rl->tokens < VLOG_MSG_TOKENS) {
 648             if (!rl->n_dropped) {
 649                 rl->first_dropped = now;
 650             }
 651             rl->n_dropped++;
 652             return;
 653         }
 654     }
 655     rl->tokens -= VLOG_MSG_TOKENS;
 656 
 657     va_start(args, message);
 658     vlog_valist(module, level, message, args);
 659     va_end(args);
 660 
 661     if (rl->n_dropped) {
 662         vlog(module, level,
 663              "Dropped %u messages in last %u seconds due to excessive rate",
 664              rl->n_dropped, (unsigned int) (time_now() - rl->first_dropped));
 665         rl->n_dropped = 0;
 666     }
 667 }

 673            "  -v, --verbose=MODULE[:FACILITY[:LEVEL]]  set logging levels\n"
 674            "  -v, --verbose           set maximum verbosity level\n"
 675            "  --log-file[=FILE]       enable logging to specified FILE\n"
 676            "                          (default: %s/%s.log)\n",
