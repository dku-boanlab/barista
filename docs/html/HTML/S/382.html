  28 #ifndef __IPV6_UTILS_H__

  31 struct e_in6_addr {
  32         guint8   bytes[16];             /* 128 bit IP6 address */
  33 };
  34 
  35 /*
  36  * Unicast Scope
  37  * Note that we must check topmost 10 bits only, not 16 bits (see RFC2373).
  38  */
  39 #define E_IN6_IS_ADDR_LINKLOCAL(a)      \
  40         (((a)->bytes[0] == 0xfe) && (((a)->bytes[1] & 0xc0) == 0x80))
  41 #define E_IN6_IS_ADDR_SITELOCAL(a)      \
  42         (((a)->bytes[0] == 0xfe) && (((a)->bytes[1] & 0xc0) == 0xc0))
  43 
  44 /*
  45  * Multicast
  46  */
  47 #define E_IN6_IS_ADDR_MULTICAST(a)      ((a)->bytes[0] == 0xff)
  48 
  49 #endif /* __IPV6_UTILS_H__ */
