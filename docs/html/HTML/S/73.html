 129 start(struct oflops_context * ctx) {  
 130   struct flow *fl = (struct flow*)xmalloc(sizeof(struct flow));
 131   fl_probe = (struct flow*)xmalloc(sizeof(struct flow));
 132   void *b; //somewhere to store message data
 133   int res, len;
 134   struct timeval now;
 135   char msg[1024];
 136 
 137   //init h
 138   TAILQ_INIT(&head); 
 139 
 140   //Initialize pap-based  tcp flow reassembler for the communication 
 141   //channel
 142   msg_init();  
 143   snprintf(msg, 1024,  "Intializing module %s", name());
 144 
 145   get_mac_address(ctx->channels[OFLOPS_DATA1].dev, local_mac);
 146   printf("%s: %02x:%02x:%02x:%02x:%02x:%02x\n", ctx->channels[OFLOPS_DATA2].dev,
 147          (unsigned char)local_mac[0], (unsigned char)local_mac[1], (unsigned char)local_mac[2], 
 148          (unsigned char)local_mac[3], (unsigned char)local_mac[4], (unsigned char)local_mac[5]);
 149 
 150   //log when I start module
 151   gettimeofday(&now, NULL);
 152   oflops_log(now, GENERIC_MSG, msg);
 153   oflops_log(now,GENERIC_MSG , cli_param);
 154 
 155   //start openflow session with switch
 156   make_ofp_hello(&b);
 157   res = oflops_send_of_mesgs(ctx, b, sizeof(struct ofp_hello));
 158   free(b);  
 159   
 160   //send a message to clean up flow tables. 
 161   printf("cleaning up flow table...\n");
 162   res = make_ofp_flow_del(&b);
 163   res = oflops_send_of_mesg(ctx, b);  
 164   free(b);
 165

 171   if(table == 0)
 172     fl->mask = 0; //if table is 0 the we generate an exact match */

 189   res = oflops_send_of_mesg(ctx, b);

 193   memcpy(fl_probe, fl, sizeof(struct flow));

 197    */
 198   //send the flow modyfication command in 30 seconds.

 201   oflops_schedule_timer_event(ctx,&now, SND_ACT);

 220 int destroy(struct oflops_context *ctx) {
 221   FILE *out;
 222   struct entry *np;
 223   int  min_id[] = {INT_MAX, INT_MAX, INT_MAX};
 224   int ix[] = {0,0,0};
 225   int max_id[] = {INT_MIN, INT_MIN, INT_MIN}, ch, i;
 226   char msg[1024];
 227   struct timeval now;
 228   double **data;
 229   uint32_t mean, std, median;
 230   float loss;
 231 
 232   //get what time we start printin output
 233   gettimeofday(&now, NULL);
 234 
 235   //open log file if required
 236   if(print) {
 237     out = fopen(logfile, "w");
 238     if(out == NULL)
 239       perror_and_exit("fopen_logfile", 1);
 240   }
 241 
 242   //init tmp data storage
 243   data = xmalloc(3*sizeof(double *));
 244   for(ch = 0; ch < 3; ch++) {
 245     data[ch] = xmalloc(count[ch]*sizeof(double));
 246   }
 247 
 248   // for every measurement save the delay in the appropriate entry on the 
 249   // measurement matrix
 250   for (np = head.tqh_first; np != NULL; np = np->entries.tqe_next) {
 251     ch = np->ch - 1;
 252     min_id[ch] = (np->id < min_id[ch])?np->id:min_id[ch];
 253     max_id[ch] = (np->id > max_id[ch])?np->id:max_id[ch];
 254     data[ch][ix[ch]++] = time_diff(&np->snd, &np->rcv);
 255     //print also packet details on otuput if required
 256     if(print)
 257       if(fprintf(out, "%lu;%lu.%06lu;%lu.%06lu;%d\n", 
 258                  (long unsigned int)np->id,  
 259                  (long unsigned int)np->snd.tv_sec, 
 260                  (long unsigned int)np->snd.tv_usec,
 261                  (long unsigned int)np->rcv.tv_sec, 
 262                  (long unsigned int)np->rcv.tv_usec,  np->ch) < 0)  
 263         perror_and_exit("fprintf fail", 1); 
 264 
 265     free(np);
 266   }
 267   
 268   for(ch = 0; ch < 3; ch++) {
 269     if(ix[ch] > 0) {
 270       gsl_sort (data[ch], 1, ix[ch]);
 271       mean = (uint32_t)gsl_stats_mean(data[ch], 1, ix[ch]);
 272       std = (uint32_t)sqrt(gsl_stats_variance(data[ch], 1, ix[ch]));
 273       median = (uint32_t)gsl_stats_median_from_sorted_data (data[ch], 1, ix[ch]);
 274       loss = (float)ix[ch]/(float)(max_id[ch] - min_id[ch]);
 275       
 276       snprintf(msg, 1024, "statistics:port:%d:%u:%u:%u:%.4f:%d", 
 277                ctx->channels[ch + 1].of_port, mean, median, std, loss, count[ch]);
 278       printf("statistics:port:%d:%u:%u:%u:%.4f:%d\n", 
 279              ctx->channels[ch + 1].of_port, mean, median, std, loss, count[ch]);
 280       oflops_log(now, GENERIC_MSG, msg);
 281     }
 282   }
 283   
 284   free(data[0]);
 285   data[0] = xmalloc(echo_data_count * sizeof(double));
 286   for(i = 1; i <= echo_data_count; i++) {
 287     snprintf(msg, 1024, "OFP_ECHO:%d:%ld.%06ld:%ld.%06ld", i, echo_data[i].snd.tv_sec, echo_data[i].snd.tv_usec, 
 288              echo_data[i].rcv.tv_sec, echo_data[i].rcv.tv_usec);
 289     oflops_log(now, GENERIC_MSG, msg);
 290     data[0][i-1] = time_diff(& echo_data[i].snd, &echo_data[i].rcv);
 291   }
 292   if(echo_data_count > 0) {
 293     gsl_sort (data[0], 1, echo_data_count);
 294     mean = (uint32_t)gsl_stats_mean(data[0], 1, echo_data_count);
 295     std = (uint32_t)sqrt(gsl_stats_variance(data[0], 1, echo_data_count));
 296     median = (uint32_t)gsl_stats_median_from_sorted_data (data[0], 1, echo_data_count);
 297     printf("statistics:echo:%u:%u:%u:%d\n", mean, median, std, echo_data_count);
 298     snprintf(msg, 1024, "statistics:echo:%u:%u:%u:%d", mean, median, std, echo_data_count);
 299     oflops_log(now, GENERIC_MSG, msg);
 300   }
 301 
 302   return 0;
 303 }

 306  * @param ctx pointer to opaque context
 307  * @param te pointer to timer event

 309 int handle_timer_event(struct oflops_context * ctx, struct timer_event *te) {  
 310   char *str = te->arg; 
 311   int len, i, inc=1;
 312   void *b;
 313   struct timeval now;
 314   struct in_addr ip_addr;
 315 
 316   //terminate process 
 317   if (strcmp(str, BYESTR) == 0) {
 318     printf("terminating test....\n");
 319     oflops_end_test(ctx,1);
 320     finished = 0;    
 321     return 0;    
 322   } else if (strcmp(str, SND_ACT) == 0) {
 323     //first create new rules
 324     send_flow_mod = 1;
 325     if(table == 1)
 326       inc = inc << 8;
 327     memcpy(fl_probe->dl_src, local_mac, 6); 
 328     memcpy(fl_probe->dl_dst, "\x00\x15\x17\x7b\x92\x0a", 6);
 329     fl_probe->in_port = htons(ctx->channels[OFLOPS_DATA1].of_port);
 330     ip_addr.s_addr = inet_addr(network);
 331     for(i=0; i< flows; i++) {
 332       ip_addr.s_addr =  ntohl(ip_addr.s_addr);
 333       ip_addr.s_addr += inc;
 334       fl_probe->nw_dst =  htonl(ip_addr.s_addr);
 335       len = make_ofp_flow_add(&b, fl_probe, ctx->channels[OFLOPS_DATA2].of_port, 1, 1200);
 336       oflops_send_of_mesgs(ctx, b, len);
 337       free(b);
 338     }
 339     fl_probe->nw_dst =  inet_addr("10.1.1.2");
 340     len = make_ofp_flow_modify_output_port(&b, fl_probe, ctx->channels[OFLOPS_DATA3].of_port, 
 341                                            1, 1200);
 342     oflops_send_of_mesg(ctx, b);
 343     free(b);
 344     printf("sending flow modifications ....\n");
 345 
 346   } else if(strcmp(str, SNMPGET) == 0) {
 347     for(i = 0; i < ctx->cpuOID_count; i++) {
 348       oflops_snmp_get(ctx, ctx->cpuOID[i], ctx->cpuOID_len[i]);
 349     }
 350     for(i=0;i<ctx->n_channels;i++) {
 351       oflops_snmp_get(ctx, ctx->channels[i].inOID, ctx->channels[i].inOID_len);
 352       oflops_snmp_get(ctx, ctx->channels[i].outOID, ctx->channels[i].outOID_len);
 353     }      
 354     gettimeofday(&now, NULL);
 355     add_time(&now, 1, 0);
 356     oflops_schedule_timer_event(ctx,&now, SNMPGET);
 357   } else if(strcmp(str, SEND_ECHO_REQ) == 0) {
 358     struct ofp_header *p;
 359     make_ofp_echo_req(&p);
 360     p->xid = htonl(++echo_data_count);
 361     oflops_send_of_mesg(ctx, p);
 362     free(p);
 363     gettimeofday(&now, NULL);
 364     add_time(&now, 4, 0);
 365     oflops_schedule_timer_event(ctx,&now, SEND_ECHO_REQ);
 366     
 367   }
 368   return 0;
 369 }

 388 /** Handle pcap event.

 393 int 
 394 handle_pcap_event(struct oflops_context *ctx, struct pcap_event * pe, oflops_channel_name ch) {
 395   struct pktgen_hdr *pktgen;
 396   int dir, len;
 397   struct ofp_header *ofp;
 398   struct pcap_event *ofp_msg;
 399   struct ofp_error_msg *err_p = NULL;
 400   char msg[1024];
 401 
 402   if (ch == OFLOPS_CONTROL) {
 403     dir = append_data_to_flow(pe->data,pe->pcaphdr);
 404     while(contains_next_msg(dir) > 0) {
 405       len = get_next_msg(dir, &ofp_msg);
 406       ofp = (struct ofp_header *)ofp_msg->data;
 407       switch(ofp->type) {
 408       case OFPT_FLOW_MOD:
 409         if((send_flow_mod) && (!stored_flow_mod_time)) {
 410           memcpy(&flow_mod_timestamp, &pe->pcaphdr.ts, sizeof(struct timeval));
 411           stored_flow_mod_time = 1;
 412         }
 413         oflops_log(pe->pcaphdr.ts,OFPT_FLOW_MOD_ADD, "flow modification send");
 414         break;  
 415       case OFPT_ERROR:
 416         err_p = (struct ofp_error_msg *)ofp;
 417         snprintf(msg, 1024, "OFPT_ERROR(type: %d, code: %d)", ntohs(err_p->type), ntohs(err_p->code));
 418         oflops_log(pe->pcaphdr.ts, OFPT_ERROR_MSG, msg);
 419         fprintf(stderr, "%s\n", msg);
 420         break;   
 421       case  OFPT_ECHO_REQUEST:
 422         err_p = (struct ofp_error_msg *)ofp;
 423         printf("%ld.%06ld: OFPT_ECHO_REQUEST %d\n", pe->pcaphdr.ts.tv_sec, pe->pcaphdr.ts.tv_usec, ntohl(ofp->xid));
 424         if((ntohl(ofp->xid) < 100) && (ntohl(ofp->xid) > 0))
 425           memcpy(&echo_data[ntohl(ofp->xid)].snd, &pe->pcaphdr.ts, sizeof(struct timeval));
 426         break;   
 427       case  OFPT_ECHO_REPLY:
 428         err_p = (struct ofp_error_msg *)ofp;
 429         printf("%ld.%06ld: OFPT_ECHO_REPLY %d\n", pe->pcaphdr.ts.tv_sec, pe->pcaphdr.ts.tv_usec, ntohl(ofp->xid));
 430         if( (ntohl(ofp->xid) < 100) && (ntohl(ofp->xid) > 0))
 431           memcpy(&echo_data[ntohl(ofp->xid)].rcv, &pe->pcaphdr.ts, sizeof(struct timeval));
 432         break;  
 433       }
 434     }
 435   } else if ((ch == OFLOPS_DATA1) || (ch == OFLOPS_DATA2) || (ch == OFLOPS_DATA3)) {
 436     struct flow fl;
 437     struct timeval now;
 438     pktgen = extract_pktgen_pkt(pe->data, pe->pcaphdr.caplen, &fl);
 439     if((ch == OFLOPS_DATA3) && (!first_pkt)) {
 440       oflops_log(pe->pcaphdr.ts, GENERIC_MSG, "FIRST_PKT_RCV");
 441       printf("INSERT_DELAY:%d\n", time_diff(&flow_mod_timestamp, &pe->pcaphdr.ts));
 442       oflops_log(pe->pcaphdr.ts, GENERIC_MSG, msg);
 443       snprintf(msg, 1024, "INSERT_DELAY:%d", time_diff(&flow_mod_timestamp, &pe->pcaphdr.ts));
 444       oflops_log(pe->pcaphdr.ts, GENERIC_MSG, msg);
 445       first_pkt = 1;
 446       gettimeofday(&now, NULL);
 447       add_time(&now, 1, 0);
 448       oflops_schedule_timer_event(ctx,&now, BYESTR);
 449     }
 450     if(htonl(pktgen->seq_num) % 100000 == 0)
 451       printf("data packet received %d\n", htonl(pktgen->seq_num));
 452     
 453     struct entry *n1 = malloc(sizeof(struct entry));
 454     n1->snd.tv_sec = htonl(pktgen->tv_sec);
 455     n1->snd.tv_usec = htonl(pktgen->tv_usec);
 456     memcpy(&n1->rcv, &pe->pcaphdr.ts, sizeof(struct timeval));
 457     n1->id = htonl(pktgen->seq_num);
 458     n1->ch = ch;
 459     count[ch - 1]++;
 460     TAILQ_INSERT_TAIL(&head, n1, entries);
 461   }
 462   return 0;
 463 }

 493 handle_snmp_event(struct oflops_context * ctx, struct snmp_event * se) {
 494   netsnmp_variable_list *vars;
 495   int len = 1024, i;
 496   char msg[1024], log[1024];
 497   struct timeval now;
 498 
 499   for(vars = se->pdu->variables; vars; vars = vars->next_variable)  {
 500     snprint_value(msg, len, vars->name, vars->name_length, vars);
 501 
 502     for (i = 0; i < ctx->cpuOID_count; i++) {
 503       if((vars->name_length == ctx->cpuOID_len[i]) &&
 504          (memcmp(vars->name, ctx->cpuOID[i],  ctx->cpuOID_len[i] * sizeof(oid)) == 0) ) {
 505         snprintf(log, len, "cpu:%ld:%d:%s",
 506                  se->pdu->reqid, 
 507                  (int)vars->name[ vars->name_length - 1], msg);
 508         oflops_log(now, SNMP_MSG, log);
 509       }
 510     }
 511 
 512     for(i=0;i<ctx->n_channels;i++) {
 513       if((vars->name_length == ctx->channels[i].inOID_len) &&
 514          (memcmp(vars->name, ctx->channels[i].inOID,  
 515                  ctx->channels[i].inOID_len * sizeof(oid)) == 0) ) {
 516         snprintf(log, len, "port:rx:%ld:%d:%s",  
 517                  se->pdu->reqid, 
 518                  (int)(int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 519         oflops_log(now, SNMP_MSG, log);
 520         break;
 521       }
 522         
 523       if((vars->name_length == ctx->channels[i].outOID_len) &&
 524          (memcmp(vars->name, ctx->channels[i].outOID,  
 525                  ctx->channels[i].outOID_len * sizeof(oid))==0) ) {
 526         snprintf(log, len, "port:tx:%ld:%d:%s",  
 527                  se->pdu->reqid, 
 528                  (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 529         //      printf("port %d : tx %s pkts\n",  (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 530         oflops_log(now, SNMP_MSG, log);
 531         break;
 532       }
 533     } //for
 534   }// if cpu
 535   return 0;
 536 }

 561 /**
 562  * Initialization code with parameters
 563  * @param ctx 
 564  */
 565 int init(struct oflops_context *ctx, char * config_str) {
 566   char *pos = NULL;
 567   char *param = config_str;
 568   char *value = NULL;
 569   struct timeval now;
 570 
 571   //init counters
 572   finished = 0;
 573 
 574   gettimeofday(&now, NULL);
 575 
 576   cli_param = strdup(config_str);
 577 
 578 
 579   while(*config_str == ' ') {
 580     config_str++;
 581   }
 582   param = config_str;
 583   while(1) {
 584     pos = index(param, ' ');
 585 
 586     if((pos == NULL)) {
 587       if (*param != '\0') {
 588         pos = param + strlen(param) + 1;
 589       } else
 590         break;
 591     }
 592     *pos='\0';
 593     pos++;
 594     value = index(param,'=');
 595     *value = '\0';
 596     value++;
 597     //fprintf(stderr, "param = %s, value = %s\n", param, value);
 598     if(value != NULL) {
 599       if(strcmp(param, "pkt_size") == 0) {
 600         //parse int to get pkt size
 601         pkt_size = strtol(value, NULL, 0);
 602         if((pkt_size < MIN_PKT_SIZE) && (pkt_size > MAX_PKT_SIZE))
 603           perror_and_exit("Invalid packet size value", 1);
 604       }  else if(strcmp(param, "probe_rate") == 0) {
 605         //parse int to get measurement probe rate
 606         proberate = strtol(value, NULL, 0);
 607         if((proberate <= 0) || (proberate >= 1010)) 
 608           perror_and_exit("Invalid probe rate param(Value between 1 and 1010)", 1);
 609       } else if(strcmp(param, "table") == 0) {
 610         //parse int to get pkt size
 611         table = strtol(value, NULL, 0);
 612         if((table < 0) && (table > 2))  
 613           perror_and_exit("Invalid table number", 1);
 614       } else if(strcmp(param, "flows") == 0) {
 615         //parse int to get pkt size
 616         flows = strtol(value, NULL, 0);
 617         if(flows <= 0)  
 618           perror_and_exit("Invalid flow number", 1);
 619       } else if(strcmp(param, "print") == 0) {
 620         //parse int to get pkt size
 621         print = strtol(value, NULL, 0);
 622       } else 
 623         fprintf(stderr, "Invalid parameter:%s\n", param);
 624       param = pos;
 625     }
 626   } 
 627 
 628   //calculate sendind interval
 629   probe_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (proberate * mbits_to_bits);
 630   fprintf(stderr, "Sending probe interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 631           (uint32_t)probe_snd_interval, (uint32_t)pkt_size, (uint32_t)proberate);
 632   return 0;
 633 }
