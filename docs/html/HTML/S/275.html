  81 /* Checks to see if the actions requested by the flow are capable of being
  82  * done in the NF2 hardware. Returns 1 if yes, 0 for no.
  83  */
  84 int
  85 nf2_are_actions_supported(struct sw_flow *flow)
  86 {
  87         struct sw_flow_actions *sfa;
  88         size_t actions_len;
  89         uint8_t *p;
  90 
  91         sfa = flow->sf_acts;
  92         actions_len = sfa->actions_len;
  93         p = (uint8_t *)sfa->actions;
  94 
  95         while (actions_len > 0) {
  96                 struct ofp_action_header *ah = (struct ofp_action_header *)p;
  97                 struct ofp_action_output *oa = (struct ofp_action_output *)p;
  98                 size_t len = ntohs(ah->len);
  99 
 100                 NF2DEBUGMSG("Action Support Chk: Len of this action: %i\n",
 101                             len);
 102                 NF2DEBUGMSG("Action Support Chk: Len of actions    : %i\n",
 103                             actions_len);
 104 
 105                 // Currently, output port(s) action, dl_src/dst rewrite actions
 106                 // are suported.
 107                 if (!(ntohs(ah->type) == OFPAT_OUTPUT
 108                       || ntohs(ah->type) == OFPAT_SET_DL_SRC
 109                       || ntohs(ah->type) == OFPAT_SET_DL_DST)) {
 110                         NF2DEBUGMSG
 111                                 ("Flow action type %#0x not supported in hardware\n",
 112                                  ntohs(ah->type));
 113                         return 0;
 114                 }
 115                 // Only support ports 1-4(incl. IN_PORT), ALL, FLOOD.
 116                 // Let CONTROLLER/LOCAL fall through
 117                 if ((ntohs(ah->type) == OFPAT_OUTPUT)
 118                     && (!((ntohs(oa->port) >= PORT_BASE)
 119                           && (ntohs(oa->port) <= MAX_IFACE))
 120                         && !(ntohs(oa->port) == OFPP_ALL)
 121                         && !(ntohs(oa->port) == OFPP_FLOOD)
 122                         && !(ntohs(oa->port) == OFPP_IN_PORT))) {
 123                         NF2DEBUGMSG
 124                                 ("Flow action output port %#0x is not supported in hardware\n",
 125                                  ntohs(oa->port));
 126                         return 0;
 127                 }
 128                 p += len;
 129                 actions_len -= len;
 130         }
 131 
 132         return 1;
 133 }

 249 /* Setup the wildcard table by adding static flows that will handle
 250  * misses by sending them up to the cpu ports, and handle packets coming
 251  * back down from the cpu by sending them out the corresponding port.
 252  */
 253 int
 254 nf2_write_static_wildcard(void)
 255 {
 256         nf2_of_entry_wrap entry;
 257         nf2_of_mask_wrap mask;
 258         nf2_of_action_wrap action;
 259         int i;
 260         struct net_device *dev;
 261 
 262         dev = nf2_get_net_device();
 263         if (dev == NULL)
 264                 return 1;
 265 
 266         memset(&entry, 0x00, sizeof(nf2_of_entry_wrap));
 267         memset(&mask, 0xFF, sizeof(nf2_of_mask_wrap));
 268         // Only non-wildcard section is the source port
 269         mask.entry.src_port = 0;
 270         memset(&action, 0, sizeof(nf2_of_action_wrap));
 271 
 272         // write the catch all entries to send to the cpu
 273         for (i = 0; i < 4; ++i) {
 274                 entry.entry.src_port = i * 2;
 275                 action.action.forward_bitmask = 0x1 << ((i * 2) + 1);
 276                 nf2_write_of_wildcard(dev, (OPENFLOW_WILDCARD_TABLE_SIZE - 4)
 277                                       + i, &entry, &mask, &action);
 278         }
 279 
 280         // write the entries to send out packets coming from the cpu
 281         for (i = 0; i < 4; ++i) {
 282                 entry.entry.src_port = (i * 2) + 1;
 283                 action.action.forward_bitmask = 0x1 << (i * 2);
 284                 nf2_write_of_wildcard(dev, (OPENFLOW_WILDCARD_TABLE_SIZE - 8)
 285                                       + i, &entry, &mask, &action);
 286         }
 287 
 288         nf2_free_net_device(dev);
 289         return 0;
 290 }

 292 /* Populate a nf2_of_entry_wrap with entries from a struct sw_flow. */
 293 void
 294 nf2_populate_of_entry(nf2_of_entry_wrap *key, struct sw_flow *flow)
 295 {
 296         int vlan_vid;
 297         int vlan_pcp;
 298         int i;
 299 
 300         key->entry.transp_dst = ntohs(flow->key.tp_dst);
 301         key->entry.transp_src = ntohs(flow->key.tp_src);
 302         key->entry.ip_proto = flow->key.nw_proto;
 303         key->entry.ip_dst = ntohl(flow->key.nw_dst);
 304         key->entry.ip_src = ntohl(flow->key.nw_src);
 305         key->entry.eth_type = ntohs(flow->key.dl_type);
 306         // Blame Jad for applying endian'ness to character arrays
 307         for (i = 0; i < 6; ++i) {
 308                 key->entry.eth_dst[i] = flow->key.dl_dst[5 - i];
 309         }
 310         for (i = 0; i < 6; ++i) {
 311                 key->entry.eth_src[i] = flow->key.dl_src[5 - i];
 312         }
 313 
 314         key->entry.src_port = (ntohs(flow->key.in_port) - PORT_BASE) * 2;
 315 
 316         if (ntohs(flow->key.dl_vlan) == 0xffff) {
 317                 key->entry.vlan_id = 0xffff;
 318         } else {
 319                 vlan_vid = VID_BITMASK & ntohs(flow->key.dl_vlan);
 320                 vlan_pcp = PCP_BITMASK
 321                         & ((uint16_t)(flow->key.dl_vlan_pcp) << PCP_BITSHIFT);
 322                 key->entry.vlan_id = vlan_pcp | vlan_vid;
 323         }
 324 }

 326 static uint32_t
 327 make_nw_wildcard(int n_wild_bits)
 328 {
 329         n_wild_bits &= (1u << OFPFW_NW_SRC_BITS) - 1;
 330         return n_wild_bits < 32 ? ((1u << n_wild_bits) - 1) : 0xFFFFFFFF;
 331 }

 333 /* Populate a nf2_of_mask_wrap with entries from a struct sw_flow's wildcards. */
 334 void
 335 nf2_populate_of_mask(nf2_of_mask_wrap *mask, struct sw_flow *flow)
 336 {
 337         int vlan_vid = 0;
 338         int vlan_pcp = 0;
 339         int i;
 340 
 341         if (OFPFW_IN_PORT & flow->key.wildcards) {
 342                 mask->entry.src_port = 0xFF;
 343         }
 344         if (OFPFW_DL_VLAN & flow->key.wildcards) {
 345                 vlan_vid = 0x0FFF;
 346         }
 347         if (OFPFW_DL_VLAN_PCP & flow->key.wildcards) {
 348                 vlan_pcp = 0xE000;
 349         }
 350         mask->entry.vlan_id = vlan_pcp | vlan_vid;
 351         if (OFPFW_DL_SRC & flow->key.wildcards) {
 352                 for (i = 0; i < 6; ++i) {
 353                         mask->entry.eth_src[i] = 0xFF;
 354                 }
 355         }
 356         if (OFPFW_DL_DST & flow->key.wildcards) {
 357                 for (i = 0; i < 6; ++i) {
 358                         mask->entry.eth_dst[i] = 0xFF;
 359                 }
 360         }
 361         if (OFPFW_DL_TYPE & flow->key.wildcards)
 362                 mask->entry.eth_type = 0xFFFF;
 363         if ((OFPFW_NW_SRC_ALL & flow->key.wildcards)
 364             || (OFPFW_NW_SRC_MASK & flow->key.wildcards))
 365                 mask->entry.ip_src = make_nw_wildcard
 366                         (flow->key.wildcards >> OFPFW_NW_SRC_SHIFT);
 367         if ((OFPFW_NW_DST_ALL & flow->key.wildcards)
 368             || (OFPFW_NW_DST_MASK & flow->key.wildcards))
 369                 mask->entry.ip_dst = make_nw_wildcard
 370                         (flow->key.wildcards >> OFPFW_NW_DST_SHIFT);
 371         if (OFPFW_NW_PROTO & flow->key.wildcards)
 372                 mask->entry.ip_proto = 0xFF;
 373         if (OFPFW_TP_SRC & flow->key.wildcards)
 374                 mask->entry.transp_src = 0xFFFF;
 375         if (OFPFW_TP_DST & flow->key.wildcards)
 376                 mask->entry.transp_dst = 0xFFFF;
 377 
 378         mask->entry.pad = 0x0000;
 379 }

 381 static void
 382 populate_action_output(nf2_of_action_wrap *action, nf2_of_entry_wrap *entry,
 383                        uint8_t *flowact)
 384 {
 385         uint16_t port = 0;
 386         struct ofp_action_output *actout = (struct ofp_action_output *)flowact;
 387         int i;
 388 
 389         port = ntohs(actout->port);
 390         NF2DEBUGMSG("Action Type: %i Output Port: %i\n",
 391                     ntohs(actout->type), port);
 392 
 393         if ((port >= PORT_BASE) && (port <= MAX_IFACE)) {
 394                 // Bitmask for output port(s), evens are phys odds cpu
 395                 action->action.forward_bitmask
 396                         |= (1 << ((port - PORT_BASE) * 2));
 397                 NF2DEBUGMSG("Output Port: %i Forward Bitmask: %x\n",
 398                             port, action->action.forward_bitmask);
 399         } else if (port == OFPP_IN_PORT) {
 400                 // Send out to input port
 401                 action->action.forward_bitmask
 402                         |= (1 << (entry->entry.src_port));
 403                 NF2DEBUGMSG("Output Port = Input Port  Forward Bitmask: %x\n",
 404                             action->action.forward_bitmask);
 405         } else if (port == OFPP_ALL || port == OFPP_FLOOD) {
 406                 // Send out all ports except the source
 407                 for (i = 0; i < 4; ++i) {
 408                         if ((i * 2) != entry->entry.src_port) {
 409                                 // Bitmask for output port(s), evens are
 410                                 // phys odds cpu
 411                                 action->action.forward_bitmask
 412                                         |= (1 << (i * 2));
 413                                 NF2DEBUGMSG
 414                                         ("Output Port: %i Forward Bitmask: %x\n",
 415                                          port, action->action.forward_bitmask);
 416                         }
 417                 }
 418         }
 419 }

 445 /* Populate an nf2_of_action_wrap. */
 446 void
 447 nf2_populate_of_action(nf2_of_action_wrap *action, nf2_of_entry_wrap *entry,
 448                        nf2_of_mask_wrap *mask, struct sw_flow *flow)
 449 {
 450         struct sw_flow_actions *sfa;
 451         size_t actions_len;
 452         uint8_t *p;
 453 
 454         sfa = flow->sf_acts;
 455         actions_len = sfa->actions_len;
 456         p = (uint8_t *)sfa->actions;
 457 
 458         // zero it out for now
 459         memset(action, 0, sizeof(nf2_of_action_wrap));
 460         action->action.nf2_action_flag = 0;
 461 
 462         while (actions_len > 0) {
 463                 struct ofp_action_header *acth = (struct ofp_action_header *)p;
 464                 size_t len = ntohs(acth->len);
 465 
 466                 NF2DEBUGMSG("Action Populate: Len of this action: %i\n", len);
 467                 NF2DEBUGMSG("Action Populate: Len of actions    : %i\n",
 468                             actions_len);
 469 
 470                 if (acth->type == htons(OFPAT_OUTPUT)) {
 471                         populate_action_output(action, entry, p);
 472                 } else if (acth->type == htons(OFPAT_SET_DL_SRC)) {
 473                         populate_action_set_dl_src(action, p);
 474                 } else if (acth->type == htons(OFPAT_SET_DL_DST)) {
 475                         populate_action_set_dl_dst(action, p);
 476                 }
 477                 p += len;
 478                 actions_len -= len;
 479         }
 480 }

 505 /* Hashes the entry to find where it should exist in the exact table
 506  * returns NULL on failure
 507  */
 508 static struct nf2_flow *
 509 get_free_exact(nf2_of_entry_wrap *entry)
 510 {
 511         unsigned int poly1 = 0x04C11DB7;
 512         unsigned int poly2 = 0x1EDC6F41;
 513         struct nf2_flow *sfw = NULL;
 514         unsigned int hash = 0x0;
 515         unsigned int index = 0x0;
 516         struct crc32 crc;
 517 
 518         crc32_init(&crc, poly1);
 519         hash = crc32_calculate(&crc, entry, sizeof(nf2_of_entry_wrap));
 520 
 521         // the bottom 15 bits of hash == the index into the table
 522         index = 0x7FFF & hash;
 523 
 524         // if this index is free, grab it
 525         sfw = exact_free_list[index];
 526         exact_free_list[index] = NULL;
 527 
 528         if (sfw != NULL) {
 529                 return sfw;
 530         }
 531         // try the second index
 532         crc32_init(&crc, poly2);
 533         hash = crc32_calculate(&crc, entry, sizeof(nf2_of_entry_wrap));
 534         // the bottom 15 bits of hash == the index into the table
 535         index = 0x7FFF & hash;
 536 
 537         // if this index is free, grab it
 538         sfw = exact_free_list[index];
 539         exact_free_list[index] = NULL;
 540 
 541         // return whether its good or not
 542         return sfw;
 543 }

 545 /* Get the first free position in the wildcard hardware table
 546  * to write into.

 549 get_free_wildcard(void)
 550 {
 551         struct nf2_flow *sfw = NULL;

 555         if (list_empty(&wildcard_free_list)) {

 563 
 564         return sfw;
 565 }

 580 /* Returns 1 if this flow contains an action outputting to all ports except
 581  * input port, 0 otherwise. We support OFPP_ALL and OFPP_FLOOD actions, however
 582  * since we do not perform the spanning tree protocol (STP) then OFPP_FLOOD is
 583  * equivalent to OFPP_ALL.
 584  */
 585 static int
 586 is_action_forward_all(struct sw_flow *flow)
 587 {
 588         struct sw_flow_actions *sfa;
 589         size_t actions_len;
 590         uint8_t *p;
 591 
 592         sfa = flow->sf_acts;
 593         actions_len = sfa->actions_len;
 594         p = (uint8_t *)sfa->actions;
 595 
 596         while (actions_len > 0) {
 597                 struct ofp_action_header *ah = (struct ofp_action_header *)p;
 598                 struct ofp_action_output *oa = (struct ofp_action_output *)p;
 599                 size_t len = ntohs(ah->len);
 600 
 601                 NF2DEBUGMSG("Fwd Action Chk: Action type: %x\n",
 602                             ntohs(ah->type));
 603                 NF2DEBUGMSG("Fwd Action Chk: Output port: %x\n",
 604                             ntohs(oa->port));
 605                 NF2DEBUGMSG("Fwd Action Chk: Len of this action: %i\n", len);
 606                 NF2DEBUGMSG("Fwd Action Chk: Len of actions    : %i\n",
 607                             actions_len);
 608                 // Currently only support the output port(s) action
 609                 if (ntohs(ah->type) == OFPAT_OUTPUT
 610                     && (ntohs(oa->port) == OFPP_ALL
 611                         || ntohs(oa->port) == OFPP_FLOOD)) {
 612                         return 1;
 613                 }
 614                 p += len;
 615                 actions_len -= len;
 616         }
 617 
 618         return 0;
 619 }

 621 /* Attempts to build and write the flow to hardware.
 622  * Returns 0 on success, 1 on failure.
 623  */
 624 int
 625 nf2_build_and_write_flow(struct sw_flow *flow)
 626 {
 627         struct nf2_flow *sfw = NULL;
 628         struct nf2_flow *sfw_next = NULL;
 629         struct net_device *dev;
 630         int num_entries = 0;
 631         int i, table_type;
 632         nf2_of_entry_wrap key;
 633         nf2_of_mask_wrap mask;
 634         nf2_of_action_wrap action;
 635 
 636         memset(&key, 0, sizeof(nf2_of_entry_wrap));
 637         memset(&mask, 0, sizeof(nf2_of_mask_wrap));
 638         memset(&action, 0, sizeof(nf2_of_action_wrap));
 639 
 640         dev = nf2_get_net_device();
 641         if (dev == NULL) {
 642                 // failure getting net device
 643                 NF2DEBUGMSG("Failure getting net device struct\n");
 644                 return 1;
 645         }
 646 
 647         table_type = nf2_get_table_type(flow);
 648         switch (table_type) {
 649         default:
 650                 break;
 651 
 652         case NF2_TABLE_EXACT:
 653                 NF2DEBUGMSG("---Exact Entry---\n");
 654                 nf2_populate_of_entry(&key, flow);
 655                 nf2_populate_of_action(&action, &key, NULL, flow);
 656                 sfw = get_free_exact(&key);
 657                 if (sfw == NULL) {
 658                         NF2DEBUGMSG
 659                                 ("Collision getting free exact match entry\n");
 660                         // collision
 661                         nf2_free_net_device(dev);
 662                         return 1;
 663                 }
 664                 // set the active bit on this entry
 665                 key.entry.pad = 0x8000;
 666                 nf2_write_of_exact(dev, sfw->pos, &key, &action);
 667                 flow->private = (void *)sfw;
 668                 break;
 669 
 670         case NF2_TABLE_WILDCARD:
 671                 NF2DEBUGMSG("---Wildcard Entry---\n");
 672                 // if action is all out and source port is wildcarded
 673                 if ((is_action_forward_all(flow)) &&
 674                     (flow->key.wildcards & OFPFW_IN_PORT)) {
 675                         NF2DEBUGMSG("Grab four wildcard tables\n");
 676                         if (!(sfw = get_free_wildcard())) {
 677                                 NF2DEBUGMSG("No free wildcard entries found.");
 678                                 // no free entries
 679                                 nf2_free_net_device(dev);
 680                                 return 1;
 681                         }
 682                         // try to get 3 more positions
 683                         for (i = 0; i < 3; ++i) {
 684                                 if (!(sfw_next = get_free_wildcard())) {
 685                                         break;
 686                                 }
 687                                 list_add_tail(&sfw_next->node, &sfw->node);
 688                                 ++num_entries;
 689                         }
 690 
 691                         if (num_entries < 3) {
 692                                 // failed to get enough entries, return them and exit
 693                                 nf2_delete_private((void *)sfw);
 694                                 nf2_free_net_device(dev);
 695                                 return 1;
 696                         }
 697 
 698                         nf2_populate_of_entry(&key, flow);
 699                         nf2_populate_of_mask(&mask, flow);
 700 
 701                         // set first entry's src port to 0, remove wildcard mask on src
 702                         key.entry.src_port = 0;
 703                         mask.entry.src_port = 0;
 704                         nf2_populate_of_action(&action, &key, &mask, flow);
 705                         nf2_write_of_wildcard(dev, sfw->pos, &key, &mask,
 706                                               &action);
 707 
 708                         i = 1;
 709                         sfw_next = list_entry(sfw->node.next,
 710                                               struct nf2_flow, node);
 711                         // walk through and write the remaining 3 entries
 712                         while (sfw_next != sfw) {
 713                                 key.entry.src_port = i * 2;
 714                                 nf2_populate_of_action(&action, &key, &mask,
 715                                                        flow);
 716                                 nf2_write_of_wildcard(dev, sfw_next->pos, &key,
 717                                                       &mask, &action);
 718                                 sfw_next = list_entry(sfw_next->node.next,
 719                                                       struct nf2_flow, node);
 720                                 ++i;
 721                         }
 722                         flow->private = (void *)sfw;
 723                 } else {
 724                         /* Get a free position here, and write to it */
 725                         if ((sfw = get_free_wildcard())) {
 726                                 nf2_populate_of_entry(&key, flow);
 727                                 nf2_populate_of_mask(&mask, flow);
 728                                 nf2_populate_of_action(&action, &key, &mask,
 729                                                        flow);
 730                                 if (nf2_write_of_wildcard
 731                                     (dev, sfw->pos, &key, &mask, &action)) {
 732                                         // failure writing to hardware
 733                                         nf2_add_free_wildcard(sfw);
 734                                         NF2DEBUGMSG
 735                                                 ("Failure writing to hardware\n");
 736                                         nf2_free_net_device(dev);
 737                                         return 1;
 738                                 } else {
 739                                         // success writing to hardware, store the position
 740                                         flow->private = (void *)sfw;
 741                                 }
 742                         } else {
 743                                 // hardware is full, return 0
 744                                 NF2DEBUGMSG("No free wildcard entries found.");
 745                                 nf2_free_net_device(dev);
 746                                 return 1;
 747                         }
 748                 }
 749                 break;
 750         }
 751 
 752         nf2_free_net_device(dev);
 753         return 0;
 754 }

 756 void
 757 nf2_delete_private(void *private)
 758 {
 759         struct nf2_flow *sfw = (struct nf2_flow *)private;
 760         struct nf2_flow *sfw_next;
 761         struct list_head *next;
 762 
 763         switch (sfw->type) {
 764         default:
 765                 break;
 766 
 767         case NF2_TABLE_EXACT:
 768                 nf2_clear_of_exact(sfw->pos);
 769                 nf2_add_free_exact(sfw);
 770                 break;
 771 
 772         case NF2_TABLE_WILDCARD:
 773                 while (!list_empty(&sfw->node)) {
 774                         next = sfw->node.next;
 775                         sfw_next = list_entry(next, struct nf2_flow, node);
 776                         list_del_init(&sfw_next->node);
 777                         // Immediately zero out the entry in hardware
 778                         nf2_clear_of_wildcard(sfw_next->pos);
 779                         // add it back to the pool
 780                         nf2_add_free_wildcard(sfw_next);
 781                 }
 782                 // zero the core entry
 783                 nf2_clear_of_wildcard(sfw->pos);
 784                 // add back the core entry
 785                 nf2_add_free_wildcard(sfw);
 786                 break;
 787         }
 788 }

 790 int
 791 nf2_modify_acts(struct sw_table *swt, struct sw_flow *flow)
 792 {
 793         struct nf2_flow *sfw = (struct nf2_flow *)flow->private;
 794         struct net_device *dev;
 795         nf2_of_entry_wrap key;
 796         nf2_of_mask_wrap mask;
 797         nf2_of_action_wrap action;
 798 
 799         memset(&key, 0, sizeof(nf2_of_entry_wrap));
 800         memset(&mask, 0, sizeof(nf2_of_mask_wrap));
 801         memset(&action, 0, sizeof(nf2_of_action_wrap));
 802 
 803         dev = nf2_get_net_device();
 804         if (dev == NULL)
 805                 return 0;
 806 
 807         switch (sfw->type) {
 808         default:
 809                 break;
 810 
 811         case NF2_TABLE_EXACT:
 812                 nf2_populate_of_entry(&key, flow);
 813                 nf2_populate_of_action(&action, &key, NULL, flow);
 814                 key.entry.pad = 0x8000;
 815                 nf2_modify_write_of_exact(dev, sfw->pos, &action);
 816                 break;
 817 
 818         case NF2_TABLE_WILDCARD:
 819                 if (flow->key.wildcards & OFPFW_IN_PORT) {
 820                         nf2_free_net_device(dev);
 821                         return 0;
 822                 }
 823                 nf2_populate_of_entry(&key, flow);
 824                 nf2_populate_of_mask(&mask, flow);
 825                 nf2_populate_of_action(&action, &key, &mask, flow);
 826                 nf2_modify_write_of_wildcard(dev, sfw->pos,
 827                                              &key, &mask, &action);
 828                 break;
 829         }
 830 
 831         nf2_free_net_device(dev);
 832         return 1;
 833 }

 835 uint64_t
 836 nf2_get_packet_count(struct net_device *dev, struct nf2_flow *sfw)
 837 {
 838         uint32_t count = 0;
 839         uint32_t hw_count = 0;
 840         uint64_t total = 0;
 841         struct nf2_flow *sfw_next = NULL;
 842 
 843         switch (sfw->type) {
 844         default:
 845                 break;
 846 
 847         case NF2_TABLE_EXACT:
 848                 count = nf2_get_exact_packet_count(dev, sfw->pos);
 849                 total = count;
 850                 break;
 851 
 852         case NF2_TABLE_WILDCARD:
 853                 sfw_next = sfw;
 854                 do {
 855                         hw_count = nf2_get_wildcard_packet_count(dev,
 856                                                                  sfw_next->pos);
 857                         if (hw_count >= sfw_next->hw_packet_count) {
 858                                 count = hw_count - sfw_next->hw_packet_count;
 859                                 sfw_next->hw_packet_count = hw_count;
 860                         } else {
 861                                 // wrapping occurred
 862                                 count = (MAX_INT_32 - sfw_next->hw_packet_count)
 863                                         + hw_count;
 864                                 sfw_next->hw_packet_count = hw_count;
 865                         }
 866                         total += count;
 867                         sfw_next = list_entry(sfw_next->node.next,
 868                                               struct nf2_flow, node);
 869                 } while (sfw_next != sfw);
 870                 break;
 871         }
 872 
 873         NF2DEBUGMSG("Return nf2_get_packet_count value: %llu\n", total);
 874         return total;
 875 }
