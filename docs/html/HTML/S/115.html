  59 int ptrack_add_of_entry( struct ptrack_list * ptl, void * data, int len, struct pcap_pkthdr hdr)
  60 {
  61         struct ptrack_entry * ent;      
  62         int add_count=0;
  63         int add_len;
  64         int offset = tcp_payload_offset(data, len);
  65         if(offset >= len)       // nothing to add
  66                 return add_count;
  67         len-=offset;    // jump to just the tcp payload part
  68         data+=offset;
  69         while(len>0 ) 
  70         {
  71                 struct ofp_header * ofph= data;
  72                 add_len= ntohs(ofph->length);
  73                 if(add_len > len)
  74                 {
  75                         fprintf(stderr, "ptrack_add_entry: unable to track timestamp for partial openflow message\n");
  76                         return add_count;
  77                 }
  78                 ent = &ptl->entries[ptl->head];
  79                 ent->data = malloc_and_check(add_len);
  80                 memcpy(ent->data, data, add_len);
  81                 ent->len = add_len;
  82                 ent->hdr=hdr;
  83                 ptl->head++;
  84                 if(ptl->head >= PTRACK_MAX_LEN)
  85                         ptl->head=0;
  86                 if(ptl->count < PTRACK_MAX_LEN)
  87                         ptl->count++;
  88                 add_count++;
  89                 len -=add_len;
  90                 data += add_len;
  91         }
  92         return add_count;
  93 }

 140 static unsigned int tcp_payload_offset(void * data, int len)
 141 {
 142          int offset=0;
 143          struct iphdr * ip;
 144          struct tcphdr * tcp;
 145          struct ether_header * eth = data+offset;
 146          if(eth->ether_type == htons(ETHERTYPE_VLAN))
 147          {
 148                 offset+=4;
 149                 eth = data+offset;
 150          }
 151          if((offset>=len)||(eth->ether_type != htons(ETHERTYPE_IP)))
 152                  return INT_MAX;
 153          offset+=sizeof(struct ether_header);
 154          ip = data + offset;
 155          if((offset>=len) || (ip->protocol != IPPROTO_TCP))
 156                  return INT_MAX;
 157          offset += ip->ihl *4;
 158          tcp = data + offset;
 159          offset += tcp->doff*4;
 160          return offset;
 161 }
