  48 /**
  49  * \brief The main function
  50  * \param activated The activation flag of this component
  51  * \param argc The number of arguments
  52  * \param argv Arguments
  53  */
  54 int switch_mgmt_main(int *activated, int argc, char **argv)
  55 {
  56     LOG_INFO(SWITCH_MGMT_ID, "Init - Switch management");
  57 
  58     num_switches = 0;
  59 
  60     memset(&switches, 0, sizeof(switch_table_t));
  61 
  62     switch_table = (switch_t **)MALLOC(sizeof(switch_t *) * __DEFAULT_TABLE_SIZE);
  63     if (switch_table == NULL) {
  64         PERROR("malloc");
  65         return -1;
  66     }
  67 
  68     memset(switch_table, 0, sizeof(switch_t *) * __DEFAULT_TABLE_SIZE);
  69 
  70     pthread_rwlock_init(&switches.lock, NULL);
  71 
  72     sw_q_init();
  73 
  74     activate();
  75 
  76     return 0;
  77 }

  79 /**
  80  * \brief The cleanup function
  81  * \param activated The activation flag of this component
  82  */
  83 int switch_mgmt_cleanup(int *activated)
  84 {
  85     LOG_INFO(SWITCH_MGMT_ID, "Clean up - Switch management");
  86 
  87     deactivate();
  88 
  89     pthread_rwlock_wrlock(&switches.lock);
  90 
  91     switch_t *curr = switches.head;
  92     while (curr != NULL) {
  93         switch_t *tmp = curr;
  94         curr = curr->next;
  95         switch_table[tmp->fd] = NULL;
  96         FREE(tmp);
  97     }
  98 
  99     pthread_rwlock_unlock(&switches.lock);
 100     pthread_rwlock_destroy(&switches.lock);
 101 
 102     FREE(switch_table);
 103 
 104     sw_q_destroy();
 105 
 106     return 0;
 107 }

 109 /**
 110  * \brief Function to print all switches
 111  * \param cli The CLI pointer
 112  */
 113 static int switch_listup(cli_t *cli)
 114 {
 115     int cnt = 0;
 116 
 117     cli_print(cli, "<Switch List>");
 118 
 119     cli_print(cli, "  The total number of switches: %d", num_switches);
 120 
 121     pthread_rwlock_rdlock(&switches.lock);
 122 
 123     switch_t *curr = switches.head;
 124 
 125     while (curr != NULL) {
 126         cli_print(cli, "  Switch #%d", ++cnt);
 127         cli_print(cli, "    Datapath ID: %lu", curr->dpid);
 128         cli_print(cli, "    Location: %s", (curr->remote) ? "remote" : "local");
 129         cli_print(cli, "    Current OF version: %s", (curr->version == 0x4) ? "1.3" : "1.0");
 130         cli_print(cli, "    Number of tables: %u", curr->n_tables);
 131         cli_print(cli, "    Number of buffers: %u", curr->n_buffers);
 132         cli_print(cli, "    Capabilities: 0x%x", curr->capabilities);
 133         cli_print(cli, "    Actions: 0x%x", curr->actions);
 134         cli_print(cli, "    Manufacturer: %s", curr->mfr_desc);
 135         cli_print(cli, "    Hardware: %s", curr->hw_desc);
 136         cli_print(cli, "    Software: %s", curr->sw_desc);
 137         cli_print(cli, "    Serial number: %s", curr->serial_num);
 138         cli_print(cli, "    Datapath: %s", curr->dp_desc);
 139         if (curr->pkt_count) {
 140             cli_print(cli, "    # of packets: %lu", curr->pkt_count);
 141             cli_print(cli, "    # of bytes: %lu", curr->byte_count);
 142             cli_print(cli, "    # of flows: %u", curr->flow_count);
 143         } else {
 144             cli_print(cli, "    # of packets: 0");
 145             cli_print(cli, "    # of bytes: 0");
 146             cli_print(cli, "    # of flows: 0");
 147         }
 148 
 149         curr = curr->next;
 150     }
 151 
 152     pthread_rwlock_unlock(&switches.lock);
 153 
 154     return 0;
 155 }

 162 static int switch_showup(cli_t *cli, char *dpid_str)
 163 {
 164     int cnt = 0;
 165 
 166     uint64_t dpid = strtoull(dpid_str, NULL, 0);
 167 
 168     cli_print(cli, "<Switch Information>");
 169 
 170     pthread_rwlock_rdlock(&switches.lock);
 171 
 172     switch_t *curr = switches.head;
 173 
 174     while (curr != NULL) {
 175         if (curr->dpid == dpid) {
 176             cli_print(cli, "    Datapath ID: %lu", curr->dpid);
 177             cli_print(cli, "    Location: %s", (curr->remote) ? "remote" : "local");
 178             cli_print(cli, "    Current OF version: %s", (curr->version == 0x4) ? "1.3" : "1.0");
 179             cli_print(cli, "    Number of tables: %u", curr->n_tables);
 180             cli_print(cli, "    Number of buffers: %u", curr->n_buffers);
 181             cli_print(cli, "    Capabilities: 0x%x", curr->capabilities);
 182             cli_print(cli, "    Actions: 0x%x", curr->actions);
 183             cli_print(cli, "    Manufacturer: %s", curr->mfr_desc);
 184             cli_print(cli, "    Hardware: %s", curr->hw_desc);
 185             cli_print(cli, "    Software: %s", curr->sw_desc);
 186             cli_print(cli, "    Serial number: %s", curr->serial_num);
 187             cli_print(cli, "    Datapath: %s", curr->dp_desc);
 188             if (curr->pkt_count) {
 189                 cli_print(cli, "    # of packets: %lu", curr->pkt_count);
 190                 cli_print(cli, "    # of bytes: %lu", curr->byte_count);
 191                 cli_print(cli, "    # of flows: %u", curr->flow_count);
 192             } else {
 193                 cli_print(cli, "    # of packets: 0");
 194                 cli_print(cli, "    # of bytes: 0");
 195                 cli_print(cli, "    # of flows: 0");
 196             }
 197 
 198             cnt++;
 199 
 200             break;
 201         }
 202 
 203         curr = curr->next;
 204     }
 205 
 206     pthread_rwlock_unlock(&switches.lock);
 207 
 208     if (!cnt)
 209         cli_print(cli, "  No switch whose datapath ID is %lu", dpid);
 210 
 211     return 0;
 212 }

 214 /**
 215  * \brief The CLI function
 216  * \param cli The CLI pointer
 217  * \param args Arguments
 218  */
 219 int switch_mgmt_cli(cli_t *cli, char **args)
 220 {
 221     if (args[0] != NULL && strcmp(args[0], "list") == 0) {
 222         if (args[1] != NULL && strcmp(args[1], "switches") == 0 && args[2] == NULL) {
 223             switch_listup(cli);
 224             return 0;
 225         }
 226     } else if (args[0] != NULL && strcmp(args[0], "show") == 0) {
 227         if (args[1] != NULL && args[2] == NULL) {
 228             switch_showup(cli, args[1]);
 229             return 0;
 230         }
 231     }
 232 
 233     cli_print(cli, "<Available Commands>");
 234     cli_print(cli, "  switch_mgmt list switches");
 235     cli_print(cli, "  switch_mgmt show [datapath ID]");
 236 
 237     return 0;
 238 }

 240 /**
 241  * \brief The handler function
 242  * \param ev Read-only event
 243  * \param ev_out Read-write event (if this component has the write permission)
 244  */
 245 int switch_mgmt_handler(const event_t *ev, event_out_t *ev_out)
 246 {
 247     switch (ev->type) {
 248     case EV_SW_NEW_CONN:
 249         PRINT_EV("EV_SW_NEW_CONN\n");
 250         {
 251             LOG_DEBUG(SWITCH_MGMT_ID, "Accepted (FD=%d)", ev->sw->fd);
 252         }
 253         break;
 254     case EV_SW_EXPIRED_CONN:
 255         PRINT_EV("EV_SW_EXPIRED_CONN\n");
 256         {
 257             const switch_t *sw = ev->sw;
 258 
 259             pthread_rwlock_wrlock(&switches.lock);
 260 
 261             if (switch_table[sw->fd] == NULL) {
 262                 pthread_rwlock_unlock(&switches.lock);
 263                 break;
 264             }
 265 
 266             switch_t *curr = switch_table[sw->fd];
 267             switch_t *tmp = curr;
 268 
 269             if (curr->prev != NULL && curr->next != NULL) {
 270                 curr->prev->next = curr->next;
 271                 curr->next->prev = curr->prev;
 272             } else if (curr->prev == NULL && curr->next != NULL) {
 273                 switches.head = curr->next;
 274                 curr->next->prev = NULL;
 275             } else if (curr->prev != NULL && curr->next == NULL) {
 276                 switches.tail = curr->prev;
 277                 curr->prev->next = NULL;
 278             } else if (curr->prev == NULL && curr->next == NULL) {
 279                 switches.head = NULL;
 280                 switches.tail = NULL;
 281             }
 282 
 283             switch_table[sw->fd] = NULL;
 284 
 285             pthread_rwlock_unlock(&switches.lock);
 286 
 287             if (tmp->dpid == 0) { // closed without handshake
 288                 LOG_DEBUG(SWITCH_MGMT_ID, "Closed (FD=%d)", sw->fd);
 289             } else { // disconnected
 290                 num_switches--;
 291 
 292                 switch_t out = {0};
 293 
 294                 out.dpid = tmp->dpid;
 295                 out.fd = sw->fd;
 296 
 297                 ev_sw_disconnected(SWITCH_MGMT_ID, &out);
 298 
 299                 LOG_INFO(SWITCH_MGMT_ID, "Disconnected (FD=%d, DPID=%lu)", sw->fd, tmp->dpid);
 300 
 301                 sw_enqueue(tmp);
 302             }
 303         }
 304         break;
 305     case EV_SW_UPDATE_CONFIG:
 306         PRINT_EV("EV_SW_UPDATE_CONFIG\n");
 307         {
 308             const switch_t *sw = ev->sw;
 309 
 310             pthread_rwlock_rdlock(&switches.lock);
 311 
 312             int i, pass = FALSE;
 313             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 314                 if (switch_table[i] == NULL)
 315                     continue;
 316 
 317                 if (switch_table[i]->dpid == sw->dpid) {
 318                     pass = TRUE;
 319                     break;
 320                 }
 321             }
 322 
 323             pthread_rwlock_unlock(&switches.lock);
 324 
 325             if (pass == FALSE) {
 326                 switch_t *new = sw_dequeue();
 327                 if (new == NULL) {
 328                     PERROR("sw_dequeue");
 329                     return -1;
 330                 }
 331 
 332                 new->dpid = sw->dpid;
 333                 new->fd = sw->fd;
 334                 new->xid = 0;
 335                 new->version = sw->version;
 336                 new->n_tables = sw->n_tables;
 337                 new->n_buffers = sw->n_buffers;
 338                 new->capabilities = sw->capabilities;
 339                 new->actions = sw->actions;
 340 
 341                 new->remote = sw->remote;
 342 
 343                 new->prev = new->next = NULL;
 344 
 345                 pthread_rwlock_wrlock(&switches.lock);
 346 
 347                 if (switches.head == NULL) {
 348                     switches.head = new;
 349                     switches.tail = new;
 350                 } else {
 351                     new->prev = switches.tail;
 352                     switches.tail->next = new;
 353                     switches.tail = new;
 354                 }
 355 
 356                 switch_table[sw->fd] = new;
 357 
 358                 num_switches++;
 359 
 360                 pthread_rwlock_unlock(&switches.lock);
 361 
 362                 switch_t out = {0};
 363 
 364                 out.dpid = sw->dpid;
 365                 out.fd = sw->fd;
 366 
 367                 out.remote = sw->remote;
 368 
 369                 ev_sw_connected(SWITCH_MGMT_ID, &out);
 370 
 371                 LOG_INFO(SWITCH_MGMT_ID, "Connected (FD=%d, DPID=%lu)", sw->fd, sw->dpid);
 372             }
 373         }
 374         break;
 375     case EV_SW_UPDATE_DESC:
 376         PRINT_EV("EV_SW_UPDATE_DESC\n");
 377         {
 378             const switch_t *sw = ev->sw;
 379 
 380             pthread_rwlock_wrlock(&switches.lock);
 381 
 382             if (switch_table[sw->fd] != NULL) {
 383                 strncpy(switch_table[sw->fd]->mfr_desc, sw->mfr_desc, 256);
 384                 strncpy(switch_table[sw->fd]->hw_desc, sw->hw_desc, 256);
 385                 strncpy(switch_table[sw->fd]->sw_desc, sw->sw_desc, 256);
 386                 strncpy(switch_table[sw->fd]->serial_num, sw->serial_num, 32);
 387                 strncpy(switch_table[sw->fd]->dp_desc, sw->dp_desc, 256);
 388             }
 389 
 390             pthread_rwlock_unlock(&switches.lock);
 391         }
 392         break;
 393     case EV_SW_DISCONNECTED:
 394         PRINT_EV("EV_SW_DISCONNECTED\n");
 395         {
 396             const switch_t *sw = ev->sw;
 397 
 398             if (sw->remote == FALSE)
 399                 break;
 400 
 401             pthread_rwlock_wrlock(&switches.lock);
 402 
 403             if (switch_table[sw->fd] == NULL) {
 404                 pthread_rwlock_unlock(&switches.lock);
 405                 break;
 406             }
 407 
 408             switch_t *curr = switch_table[sw->fd];
 409             switch_t *tmp = curr;
 410 
 411             if (curr->prev != NULL && curr->next != NULL) {
 412                 curr->prev->next = curr->next;
 413                 curr->next->prev = curr->prev;
 414             } else if (curr->prev == NULL && curr->next != NULL) {
 415                 switches.head = curr->next;
 416                 curr->next->prev = NULL;
 417             } else if (curr->prev != NULL && curr->next == NULL) {
 418                 switches.tail = curr->prev;
 419                 curr->prev->next = NULL;
 420             } else if (curr->prev == NULL && curr->next == NULL) {
 421                 switches.head = NULL;
 422                 switches.tail = NULL;
 423             }
 424 
 425             switch_table[sw->fd] = NULL;
 426 
 427             num_switches--;
 428 
 429             pthread_rwlock_unlock(&switches.lock);
 430 
 431             LOG_INFO(SWITCH_MGMT_ID, "Disconnected (FD=%d, DPID=%lu)", sw->fd, tmp->dpid);
 432 
 433             sw_enqueue(tmp);
 434         }
 435         break;
 436     case EV_DP_AGGREGATE_STATS:
 437         PRINT_EV("EV_DP_AGGREGATE_STATS\n");
 438         {
 439             const flow_t *flow = ev->flow;
 440 
 441             pthread_rwlock_wrlock(&switches.lock);
 442 
 443             switch_t *curr = switches.head;
 444             while (curr != NULL) {
 445                 if (curr->dpid == flow->dpid) {
 446                     curr->pkt_count += flow->pkt_count;
 447                     curr->byte_count += flow->byte_count;
 448                     curr->flow_count += flow->flow_count;
 449                     break;
 450                 }
 451                 curr = curr->next;
 452             }
 453 
 454             pthread_rwlock_unlock(&switches.lock);
 455         }
 456         break;
 457     case EV_SW_GET_DPID:
 458         PRINT_EV("EV_SW_GET_DPID\n");
 459         {
 460             switch_t *sw = ev_out->sw_data;
 461 
 462             pthread_rwlock_rdlock(&switches.lock);
 463 
 464             if (switch_table[sw->fd] != NULL) {
 465                 sw->dpid = switch_table[sw->fd]->dpid;
 466             } else {
 467                 switch_t *curr = switches.head;
 468                 while (curr != NULL) {
 469                     if (curr->fd == sw->fd) {
 470                         sw->dpid = curr->dpid;
 471                         break;
 472                     }
 473                     curr = curr->next;
 474                 }
 475             }
 476 
 477             pthread_rwlock_unlock(&switches.lock);
 478         }
 479         break;
 480     case EV_SW_GET_FD:
 481         PRINT_EV("EV_SW_GET_FD\n");
 482         {
 483             switch_t *sw = ev_out->sw_data;
 484 
 485             pthread_rwlock_rdlock(&switches.lock);
 486 
 487             switch_t *curr = switches.head;
 488             while (curr != NULL) {
 489                 if (curr->dpid == sw->dpid) {
 490                     sw->fd = curr->fd;
 491                     break;
 492                 }
 493                 curr = curr->next;
 494             }
 495 
 496             pthread_rwlock_unlock(&switches.lock);
 497         }
 498         break;
 499     case EV_SW_GET_XID:
 500         PRINT_EV("EV_SW_GET_XID\n");
 501         {
 502             switch_t *sw = ev_out->sw_data;
 503 
 504             pthread_rwlock_rdlock(&switches.lock);
 505 
 506             if (sw->fd) {
 507                 sw->xid = switch_table[sw->fd]->xid++;
 508             } else {
 509                 switch_t *curr = switches.head;
 510                 while (curr != NULL) {
 511                     if (curr->dpid == sw->dpid) {
 512                         sw->xid = curr->xid++;
 513                         break;
 514                     }
 515                     curr = curr->next;
 516                 }
 517             }
 518 
 519             pthread_rwlock_unlock(&switches.lock);
 520         }
 521         break;
 522     case EV_SW_GET_VERSION:
 523         PRINT_EV("EV_SW_GET_VERSION\n");
 524         {
 525             switch_t *sw = ev_out->sw_data;
 526 
 527             pthread_rwlock_rdlock(&switches.lock);
 528 
 529             if (sw->fd) {
 530                 sw->version = switch_table[sw->fd]->version;
 531             } else {
 532                 switch_t *curr = switches.head;
 533                 while (curr != NULL) {
 534                     if (curr->dpid == sw->dpid) {
 535                         sw->version = curr->version;
 536                         break;
 537                     }
 538                     curr = curr->next;
 539                 }
 540             }
 541 
 542             pthread_rwlock_unlock(&switches.lock);
 543         }
 544         break;
 545     case EV_SW_GET_INFO:
 546         PRINT_EV("EV_SW_GET_INFO\n");
 547         {
 548             switch_t *sw = ev_out->sw_data;
 549 
 550             pthread_rwlock_rdlock(&switches.lock);
 551 
 552             switch_t *curr = switches.head;
 553 
 554             while (curr != NULL) {
 555                 if (curr->dpid == sw->dpid) {
 556                     memmove(sw, curr, sizeof(switch_t));
 557                     sw->port = sw->fd = sw->xid = 0;
 558                     sw->prev = sw->next = NULL;
 559                     break;
 560                 }
 561                 curr = curr->next;
 562             }
 563 
 564             pthread_rwlock_unlock(&switches.lock);
 565         }
 566         break;
 567     case EV_SW_GET_ALL_INFO:
 568         PRINT_EV("EV_SW_GET_ALL_INFO\n");
 569         {
 570             switch_t *sw = ev_out->sw_data;
 571 
 572             sw = (switch_t *)MALLOC(sizeof(switch_t) * num_switches);
 573             if (sw == NULL) {
 574                 break;
 575             }
 576 
 577             pthread_rwlock_rdlock(&switches.lock);
 578 
 579             int idx = 0;
 580             switch_t *curr = switches.head;
 581 
 582             while (curr != NULL) {
 583                 memmove(&sw[idx], curr, sizeof(switch_t));
 584                 sw->port = sw->fd = sw->xid = 0;
 585 
 586                 if (idx == 0) {
 587                     sw[idx].prev = NULL;
 588 
 589                     if (num_switches > 1) {
 590                         sw[idx].next = &sw[idx + 1];
 591                     } else {
 592                         sw[idx].next = NULL;
 593                     }
 594                 } else {
 595                     sw[idx].prev = &sw[idx - 1];
 596 
 597                     if (idx < num_switches - 1) {
 598                         sw[idx].next = &sw[idx + 1];
 599                     } else {
 600                         sw[idx].next = NULL;
 601                     }
 602                 }
 603 
 604                 idx++;
 605 
 606                 curr = curr->next;
 607             }
 608 
 609             pthread_rwlock_unlock(&switches.lock);
 610         }
 611         break;
 612     default:
 613         break;
 614     }
 615 
 616     return 0;
 617 }
