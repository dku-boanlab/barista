  47 static struct arp_header *
  48 pull_arp(struct ofpbuf *packet)
  49 {
  50     if (packet->size >= ARP_ETH_HEADER_LEN) {
  51         struct arp_eth_header *arp = packet->data;
  52         return ofpbuf_pull(packet, ARP_ETH_HEADER_LEN);
  53     }
  54     return NULL;
  55 }

  57 static struct ip_header *
  58 pull_ip(struct ofpbuf *packet)
  59 {
  60     if (packet->size >= IP_HEADER_LEN) {
  61         struct ip_header *ip = packet->data;
  62         int ip_len = IP_IHL(ip->ip_ihl_ver) * 4;
  63         if (ip_len >= IP_HEADER_LEN && packet->size >= ip_len) {
  64             return ofpbuf_pull(packet, ip_len);
  65         }
  66     }
  67     return NULL;
  68 }

  70 static struct tcp_header *
  71 pull_tcp(struct ofpbuf *packet) 
  72 {
  73     if (packet->size >= TCP_HEADER_LEN) {
  74         struct tcp_header *tcp = packet->data;
  75         int tcp_len = TCP_OFFSET(tcp->tcp_ctl) * 4;
  76         if (tcp_len >= TCP_HEADER_LEN && packet->size >= tcp_len) {
  77             return ofpbuf_pull(packet, tcp_len);
  78         }
  79     }
  80     return NULL;
  81 }

 107 /* Returns 1 if 'packet' is an IP fragment, 0 otherwise. */
 108 int
 109 flow_extract(struct ofpbuf *packet, uint16_t in_port, struct flow *flow)
 110 {
 111     struct ofpbuf b = *packet;
 112     struct eth_header *eth;
 113     int retval = 0;
 114 
 115     memset(flow, 0, sizeof *flow);
 116     flow->dl_vlan = htons(OFP_VLAN_NONE);
 117     flow->in_port = htons(in_port);
 118 
 119     packet->l2 = b.data;
 120     packet->l3 = NULL;
 121     packet->l4 = NULL;
 122     packet->l7 = NULL;
 123 
 124     eth = pull_eth(&b);
 125     if (eth) {
 126         if (ntohs(eth->eth_type) >= OFP_DL_TYPE_ETH2_CUTOFF) {
 127             /* This is an Ethernet II frame */
 128             flow->dl_type = eth->eth_type;
 129         } else {
 130             /* This is an 802.2 frame */
 131             struct llc_header *llc = ofpbuf_at(&b, 0, sizeof *llc);
 132             struct snap_header *snap = ofpbuf_at(&b, sizeof *llc,
 133                                                  sizeof *snap);
 134             if (llc == NULL) {
 135                 return 0;
 136             }
 137             if (snap
 138                 && llc->llc_dsap == LLC_DSAP_SNAP
 139                 && llc->llc_ssap == LLC_SSAP_SNAP
 140                 && llc->llc_cntl == LLC_CNTL_SNAP
 141                 && !memcmp(snap->snap_org, SNAP_ORG_ETHERNET,
 142                            sizeof snap->snap_org)) {
 143                 flow->dl_type = snap->snap_type;
 144                 ofpbuf_pull(&b, LLC_SNAP_HEADER_LEN);
 145             } else {
 146                 flow->dl_type = htons(OFP_DL_TYPE_NOT_ETH_TYPE);
 147                 ofpbuf_pull(&b, sizeof(struct llc_header));
 148             }
 149         }
 150 
 151         /* Check for a VLAN tag */
 152         if (flow->dl_type == htons(ETH_TYPE_VLAN)) {
 153             struct vlan_header *vh = pull_vlan(&b);
 154             if (vh) {
 155                 flow->dl_type = vh->vlan_next_type;
 156                 flow->dl_vlan = vh->vlan_tci & htons(VLAN_VID_MASK);
 157                 flow->dl_vlan_pcp = (uint8_t)((ntohs(vh->vlan_tci) >> VLAN_PCP_SHIFT)
 158                                                & VLAN_PCP_BITMASK);
 159             }
 160         }
 161         memcpy(flow->dl_src, eth->eth_src, ETH_ADDR_LEN);
 162         memcpy(flow->dl_dst, eth->eth_dst, ETH_ADDR_LEN);
 163 
 164         packet->l3 = b.data;
 165         if (flow->dl_type == htons(ETH_TYPE_IP)) {
 166             const struct ip_header *nh = pull_ip(&b);
 167             if (nh) {
 168                 flow->nw_tos = nh->ip_tos & 0xfc;
 169                 flow->nw_proto = nh->ip_proto;
 170                 flow->nw_src = nh->ip_src;
 171                 flow->nw_dst = nh->ip_dst;
 172                 packet->l4 = b.data;
 173                 if (!IP_IS_FRAGMENT(nh->ip_frag_off)) {
 174                     if (flow->nw_proto == IP_TYPE_TCP) {
 175                         const struct tcp_header *tcp = pull_tcp(&b);
 176                         if (tcp) {
 177                             flow->tp_src = tcp->tcp_src;
 178                             flow->tp_dst = tcp->tcp_dst;
 179                             packet->l7 = b.data;
 180                         } else {
 181                             /* Avoid tricking other code into thinking that
 182                              * this packet has an L4 header. */
 183                             flow->nw_proto = 0;
 184                         }
 185                     } else if (flow->nw_proto == IP_TYPE_UDP) {
 186                         const struct udp_header *udp = pull_udp(&b);
 187                         if (udp) {
 188                             flow->tp_src = udp->udp_src;
 189                             flow->tp_dst = udp->udp_dst;
 190                             packet->l7 = b.data;
 191                         } else {
 192                             /* Avoid tricking other code into thinking that
 193                              * this packet has an L4 header. */
 194                             flow->nw_proto = 0;
 195                         }
 196                     } else if (flow->nw_proto == IP_TYPE_ICMP) {
 197                         const struct icmp_header *icmp = pull_icmp(&b);
 198                         if (icmp) {
 199                             flow->icmp_type = htons(icmp->icmp_type);
 200                             flow->icmp_code = htons(icmp->icmp_code);
 201                             packet->l7 = b.data;
 202                         } else {
 203                             /* Avoid tricking other code into thinking that
 204                              * this packet has an L4 header. */
 205                             flow->nw_proto = 0;
 206                         }
 207                     }
 208                 } else {
 209                     retval = 1;
 210                 }
 211             }
 212         } else if (flow->dl_type == htons(ETH_TYPE_ARP)) {
 213             const struct arp_eth_header *arp = pull_arp(&b);
 214             if (arp) {
 215                 if (arp->ar_pro == htons(ARP_PRO_IP) && arp->ar_pln == IP_ADDR_LEN) {
 216                     flow->nw_src = arp->ar_spa;
 217                     flow->nw_dst = arp->ar_tpa;
 218                 }
 219                 flow->nw_proto = ntohs(arp->ar_op) & 0xff;
 220             }
 221         }
 222     }
 223     return retval;
 224 }

 226 void
 227 flow_fill_match(struct ofp_match *to, const struct flow *from,
 228                 uint32_t wildcards)
 229 {
 230     to->wildcards = htonl(wildcards);
 231     to->in_port = from->in_port;
 232     to->dl_vlan = from->dl_vlan;
 233     memcpy(to->dl_src, from->dl_src, ETH_ADDR_LEN);
 234     memcpy(to->dl_dst, from->dl_dst, ETH_ADDR_LEN);
 235     to->dl_type = from->dl_type;
 236     to->nw_tos = from->nw_tos;
 237     to->nw_proto = from->nw_proto;
 238     to->nw_src = from->nw_src;
 239     to->nw_dst = from->nw_dst;
 240     to->tp_src = from->tp_src;
 241     to->tp_dst = from->tp_dst;
 242     to->dl_vlan_pcp = from->dl_vlan_pcp;
 243 }

 245 void
 246 flow_print(FILE *stream, const struct flow *flow) 
 247 {
 248     fprintf(stream,
 249             "port %04x vlan-vid %04x vlan-pcp %02x src-mac "ETH_ADDR_FMT" "
 250             "dst-mac "ETH_ADDR_FMT" frm-type %04x ip-tos %02x ip-proto %02x "
 251             "src-ip "IP_FMT" dst-ip "IP_FMT" tp-src %d tp-dst %d",
 252             ntohs(flow->in_port), ntohs(flow->dl_vlan), flow->dl_vlan_pcp,
 253             ETH_ADDR_ARGS(flow->dl_src), ETH_ADDR_ARGS(flow->dl_dst),
 254             ntohs(flow->dl_type),
 255             flow->nw_tos, flow->nw_proto,
 256             IP_ARGS(&flow->nw_src), IP_ARGS(&flow->nw_dst),
 257             ntohs(flow->tp_src), ntohs(flow->tp_dst));
 258 }
