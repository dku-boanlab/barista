 114 static int
 115 stream_recv(struct vconn *vconn, struct ofpbuf **bufferp)
 116 {
 117     struct stream_vconn *s = stream_vconn_cast(vconn);
 118     struct ofpbuf *rx;
 119     size_t want_bytes;
 120     ssize_t retval;
 121 
 122     if (s->rxbuf == NULL) {
 123         s->rxbuf = ofpbuf_new(1564);
 124     }
 125     rx = s->rxbuf;
 126 
 127 again:
 128     if (sizeof(struct ofp_header) > rx->size) {
 129         want_bytes = sizeof(struct ofp_header) - rx->size;
 130     } else {
 131         struct ofp_header *oh = rx->data;
 132         size_t length = ntohs(oh->length);
 133         if (length < sizeof(struct ofp_header)) {
 134             VLOG_ERR_RL(&rl, "received too-short ofp_header (%zu bytes)",
 135                         length);
 136             return EPROTO;
 137         }
 138         want_bytes = length - rx->size;
 139         if (!want_bytes) {
 140             *bufferp = rx;
 141             s->rxbuf = NULL;
 142             return 0;
 143         }
 144     }
 145     ofpbuf_prealloc_tailroom(rx, want_bytes);
 146 
 147     retval = read(s->fd, ofpbuf_tail(rx), want_bytes);
 148     if (retval > 0) {
 149         rx->size += retval;
 150         if (retval == want_bytes) {
 151             if (rx->size > sizeof(struct ofp_header)) {
 152                 *bufferp = rx;
 153                 s->rxbuf = NULL;
 154                 return 0;
 155             } else {
 156                 goto again;
 157             }
 158         }
 159         return EAGAIN;
 160     } else if (retval == 0) {
 161         if (rx->size) {
 162             VLOG_ERR_RL(&rl, "connection dropped mid-packet");
 163             return EPROTO;
 164         } else {
 165             return EOF;
 166         }
 167     } else {
 168         return errno;
 169     }
 170 }

 180 static void
 181 stream_do_tx(int fd UNUSED, short int revents UNUSED, void *vconn_)
 182 {
 183     struct vconn *vconn = vconn_;
 184     struct stream_vconn *s = stream_vconn_cast(vconn);
 185     ssize_t n = write(s->fd, s->txbuf->data, s->txbuf->size);
 186     if (n < 0) {
 187         if (errno != EAGAIN) {
 188             VLOG_ERR_RL(&rl, "send: %s", strerror(errno));
 189             stream_clear_txbuf(s);
 190             return;
 191         }
 192     } else if (n > 0) {
 193         ofpbuf_pull(s->txbuf, n);
 194         if (!s->txbuf->size) {
 195             stream_clear_txbuf(s);
 196             return;
 197         }
 198     }
 199     s->tx_waiter = poll_fd_callback(s->fd, POLLOUT, stream_do_tx, vconn);
 200 }

 202 static int
 203 stream_send(struct vconn *vconn, struct ofpbuf *buffer)
 204 {
 205     struct stream_vconn *s = stream_vconn_cast(vconn);
 206     ssize_t retval;
 207 
 208     if (s->txbuf) {
 209         return EAGAIN;
 210     }
 211 
 212     retval = write(s->fd, buffer->data, buffer->size);
 213     if (retval == buffer->size) {
 214         ofpbuf_delete(buffer);
 215         return 0;
 216     } else if (retval >= 0 || errno == EAGAIN) {
 217         leak_checker_claim(buffer);
 218         s->txbuf = buffer;
 219         if (retval > 0) {
 220             ofpbuf_pull(buffer, retval);
 221         }
 222         s->tx_waiter = poll_fd_callback(s->fd, POLLOUT, stream_do_tx, vconn);
 223         return 0;
 224     } else {
 225         return errno;
 226     }
 227 }

 229 static void
 230 stream_wait(struct vconn *vconn, enum vconn_wait_type wait)
 231 {
 232     struct stream_vconn *s = stream_vconn_cast(vconn);
 233     switch (wait) {
 234     case WAIT_CONNECT:
 235         poll_fd_wait(s->fd, POLLOUT);
 236         break;
 237 
 238     case WAIT_SEND:
 239         if (!s->txbuf) {
 240             poll_fd_wait(s->fd, POLLOUT);
 241         } else {
 242             /* Nothing to do: need to drain txbuf first. */
 243         }
 244         break;
 245 
 246     case WAIT_RECV:
 247         poll_fd_wait(s->fd, POLLIN);
 248         break;
 249 
 250     default:
 251         NOT_REACHED();
 252     }
 253 }

 284 int
 285 new_pstream_pvconn(const char *name, int fd,
 286                   int (*accept_cb)(int fd, const struct sockaddr *,
 287                                    size_t sa_len, struct vconn **),
 288                   struct pvconn **pvconnp)
 289 {
 290     struct pstream_pvconn *ps;
 291     int retval;
 292 
 293     retval = set_nonblocking(fd);
 294     if (retval) {
 295         close(fd);
 296         return retval;
 297     }
 298 
 299     if (listen(fd, 10) < 0) {
 300         int error = errno;
 301         VLOG_ERR("%s: listen: %s", name, strerror(error));
 302         close(fd);
 303         return error;
 304     }
 305 
 306     ps = xmalloc(sizeof *ps);
 307     pvconn_init(&ps->pvconn, &pstream_pvconn_class, name);
 308     ps->fd = fd;
 309     ps->accept_cb = accept_cb;
 310     *pvconnp = &ps->pvconn;
 311     return 0;
 312 }

 322 static int
 323 pstream_accept(struct pvconn *pvconn, struct vconn **new_vconnp)
 324 {
 325     struct pstream_pvconn *ps = pstream_pvconn_cast(pvconn);
 326     struct sockaddr_storage ss;
 327     socklen_t ss_len = sizeof ss;
 328     int new_fd;
 329     int retval;
 330 
 331     new_fd = accept(ps->fd, (struct sockaddr *) &ss, &ss_len);
 332     if (new_fd < 0) {
 333         int retval = errno;
 334         if (retval != EAGAIN) {
 335             VLOG_DBG_RL(&rl, "accept: %s", strerror(retval));
 336         }
 337         return retval;
 338     }
 339 
 340     retval = set_nonblocking(new_fd);
 341     if (retval) {
 342         close(new_fd);
 343         return retval;
 344     }
 345 
 346     return ps->accept_cb(new_fd, (const struct sockaddr *) &ss, ss_len,
 347                          new_vconnp);
 348 }
