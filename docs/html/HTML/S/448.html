  30 /*
  31  * RTP Payload types
  32  * Table B.2 / H.225.0
  33  * Also RFC 1890, and
  34  *
  35  *      http://www.iana.org/assignments/rtp-parameters
  36  */
  37 #define PT_PCMU         0       /* RFC 1890 */
  38 #define PT_1016         1       /* RFC 1890 */
  39 #define PT_G721         2       /* RFC 1890 */
  40 #define PT_GSM          3       /* RFC 1890 */
  41 #define PT_G723         4       /* From Vineet Kumar of Intel; see the Web page */
  42 #define PT_DVI4_8000    5       /* RFC 1890 */
  43 #define PT_DVI4_16000   6       /* RFC 1890 */
  44 #define PT_LPC          7       /* RFC 1890 */
  45 #define PT_PCMA         8       /* RFC 1890 */
  46 #define PT_G722         9       /* RFC 1890 */
  47 #define PT_L16_STEREO   10      /* RFC 1890 */
  48 #define PT_L16_MONO     11      /* RFC 1890 */
  49 #define PT_QCELP        12      /* Qualcomm Code Excited Linear Predictive coding? */
  50 #define PT_CN           13      /* RFC 3389 */
  51 #define PT_MPA          14      /* RFC 1890, RFC 2250 */
  52 #define PT_G728         15      /* RFC 1890 */
  53 #define PT_DVI4_11025   16      /* from Joseph Di Pol of Sun; see the Web page */
  54 #define PT_DVI4_22050   17      /* from Joseph Di Pol of Sun; see the Web page */
  55 #define PT_G729         18
  56 #define PT_CN_OLD       19      /* Payload type reserved (old version Comfort Noise) */
  57 #define PT_CELB         25      /* RFC 2029 */
  58 #define PT_JPEG         26      /* RFC 2435 */
  59 #define PT_NV           28      /* RFC 1890 */
  60 #define PT_H261         31      /* RFC 2032 */
  61 #define PT_MPV          32      /* RFC 2250 */
  62 #define PT_MP2T         33      /* RFC 2250 */
  63 #define PT_H263         34      /* from Chunrong Zhu of Intel; see the Web page */
