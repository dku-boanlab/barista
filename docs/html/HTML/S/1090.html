 583 static int cli_parse_line(const char *line, char *words[], int max_words)
 584 {
 585     int nwords = 0;
 586     const char *p = line;
 587     const char *word_start = 0;
 588     int inquote = 0;
 589 
 590     while (*p)
 591     {
 592         if (!isspace(*p))
 593         {
 594             word_start = p;
 595             break;
 596         }
 597         p++;
 598     }
 599 
 600     while (nwords < max_words - 1)
 601     {
 602         if (!*p || *p == inquote || (word_start && !inquote && (isspace(*p) || *p == '|')))
 603         {
 604             if (word_start)
 605             {
 606                 int len = p - word_start;
 607 
 608                 memcpy(words[nwords] = malloc(len + 1), word_start, len);
 609                 words[nwords++][len] = 0;
 610             }
 611 
 612             if (!*p)
 613                 break;
 614 
 615             if (inquote)
 616                 p++; /* skip over trailing quote */
 617 
 618             inquote = 0;
 619             word_start = 0;
 620         }
 621         else if (*p == '"' || *p == '\'')
 622         {
 623             inquote = *p++;
 624             word_start = p;
 625         }
 626         else
 627         {
 628             if (!word_start)
 629             {
 630                 if (*p == '|')
 631                 {
 632                     if (!(words[nwords++] = strdup("|")))
 633                         return 0;
 634                 }
 635                 else if (!isspace(*p))
 636                     word_start = p;
 637             }
 638 
 639             p++;
 640         }
 641     }
 642 
 643     return nwords;
 644 }

 691         if (commands->parent && commands->parent->callback)
 692             cli_error(cli, "%-20s %s", cli_command_name(cli, commands->parent),
 693                       (commands->parent->help != NULL ? commands->parent->help : ""));
 694 
 695         for (c = commands; c; c = c->next)
 696         {
 697             if (strncasecmp(c->command, words[start_word], l) == 0
 698                 && (c->callback || c->children)
 699                 && cli->privilege >= c->privilege
 700                 && (c->mode == cli->mode || c->mode == MODE_ANY))
 701                     cli_error(cli, "  %-20s %s", c->command, (c->help != NULL ? c->help : ""));
 702         }
 703 
 704         return CLI_OK;
 705     }
 706 
 707     for (c = commands; c; c = c->next)
 708     {
 709         if (cli->privilege < c->privilege)
 710             continue;
 711 
 712         if (strncasecmp(c->command, words[start_word], c->unique_len))
 713             continue;
 714 
 715         if (strncasecmp(c->command, words[start_word], strlen(words[start_word])))
 716             continue;
 717 
 718         AGAIN:
 719         if (c->mode == cli->mode || (c->mode == MODE_ANY && again_any != NULL))
 720         {
 721             int rc = CLI_OK;
 722             int f;
 723             struct cli_filter **filt = &cli->filters;
 724 
 725             // Found a word!
 726             if (!c->children)
 727             {
 728                 // Last word
 729                 if (!c->callback)
 730                 {
 731                     cli_error(cli, "No callback for \"%s\"", cli_command_name(cli, c));
 732                     return CLI_ERROR;
 733                 }
 734             }
 735             else
 736             {
 737                 if (start_word == c_words - 1)
 738                 {
 739                     if (c->callback)
 740                         goto CORRECT_CHECKS;
 741 
 742                     cli_error(cli, "Incomplete command");
 743                     return CLI_ERROR;
 744                 }
 745                 rc = cli_find_command(cli, c->children, num_words, words, start_word + 1, filters);
 746                 if (rc == CLI_ERROR_ARG)
 747                 {
 748                     if (c->callback)
 749                     {
 750                         rc = CLI_OK;
 751                         goto CORRECT_CHECKS;
 752                     }
 753                     else
 754                     {
 755                         cli_error(cli, "Invalid %s \"%s\"", commands->parent ? "argument" : "command",
 756                                   words[start_word]);
 757                     }
 758                 }
 759                 return rc;
 760             }
 761 
 762             if (!c->callback)
 763             {
 764                 cli_error(cli, "Internal server error processing \"%s\"", cli_command_name(cli, c));
 765                 return CLI_ERROR;
 766             }
 767 
 768             CORRECT_CHECKS:
 769             for (f = 0; rc == CLI_OK && filters[f]; f++)
 770             {
 771                 int n = num_words;
 772                 char **argv;
 773                 int argc;
 774                 int len;
 775 
 776                 if (filters[f+1])
 777                 n = filters[f+1];
 778 
 779                 if (filters[f] == n - 1)
 780                 {
 781                     cli_error(cli, "Missing filter");
 782                     return CLI_ERROR;
 783                 }
 784 
 785                 argv = words + filters[f] + 1;
 786                 argc = n - (filters[f] + 1);
 787                 len = strlen(argv[0]);
 788                 if (argv[argc - 1][strlen(argv[argc - 1]) - 1] == '?')
 789                 {
 790                     if (argc == 1)
 791                     {
 792                         int i;
 793                         for (i = 0; filter_cmds[i].cmd; i++)
 794                             cli_error(cli, "  %-20s %s", filter_cmds[i].cmd, filter_cmds[i].help );
 795                     }
 796                     else
 797                     {
 798                         if (argv[0][0] != 'c') // count
 799                             cli_error(cli, "  WORD");
 800 
 801                         if (argc > 2 || argv[0][0] == 'c') // count
 802                             cli_error(cli, "  <cr>");
 803                     }
 804 
 805                     return CLI_OK;
 806                 }
 807 
 808                 if (argv[0][0] == 'b' && len < 3) // [beg]in, [bet]ween
 809                 {
 810                     cli_error(cli, "Ambiguous filter \"%s\" (begin, between)", argv[0]);
 811                     return CLI_ERROR;
 812                 }
 813                 *filt = calloc(sizeof(struct cli_filter), 1);
 814 
 815                 if (!strncmp("include", argv[0], len) || !strncmp("exclude", argv[0], len) ||
 816                     !strncmp("grep", argv[0], len) || !strncmp("egrep", argv[0], len))
 817                     rc = cli_match_filter_init(cli, argc, argv, *filt);
 818                 else if (!strncmp("begin", argv[0], len) || !strncmp("between", argv[0], len))
 819                     rc = cli_range_filter_init(cli, argc, argv, *filt);
 820                 else if (!strncmp("count", argv[0], len))
 821                     rc = cli_count_filter_init(cli, argc, argv, *filt);
 822                 else
 823                 {
 824                     cli_error(cli, "Invalid filter \"%s\"", argv[0]);
 825                     rc = CLI_ERROR;
 826                 }
 827 
 828                 if (rc == CLI_OK)
 829                 {
 830                     filt = &(*filt)->next;
 831                 }
 832                 else
 833                 {
 834                     free(*filt);
 835                     *filt = 0;
 836                 }
 837             }
 838 
 839             if (rc == CLI_OK)
 840                 rc = c->callback(cli, cli_command_name(cli, c), words + start_word + 1, c_words - start_word - 1);
 841 
 842             while (cli->filters)
 843             {
 844                 struct cli_filter *filt = cli->filters;
 845 
 846                 // call one last time to clean up
 847                 filt->filter(cli, NULL, filt->data);
 848                 cli->filters = filt->next;
 849                 free(filt);
 850             }
 851 
 852             return rc;
 853         }
 854         else if (cli->mode > MODE_CONFIG && c->mode == MODE_CONFIG)
 855         {
 856             // command matched but from another mode,
 857             // remember it if we fail to find correct command
 858             again_config = c;
 859         }
 860         else if (c->mode == MODE_ANY)
 861         {
 862             // command matched but for any mode,
 863             // remember it if we fail to find correct command
 864             again_any = c;
 865         }
 866     }
 867 
 868     // drop out of config submode if we have matched command on MODE_CONFIG
 869     if (again_config)
 870     {
 871         c = again_config;
 872         cli_set_configmode(cli, MODE_CONFIG, NULL);
 873         goto AGAIN;
 874     }
 875     if (again_any)
 876     {
 877         c = again_any;
 878         goto AGAIN;
 879     }
 880 
 881     if (start_word == 0)
 882         cli_error(cli, "Invalid %s \"%s\"", commands->parent ? "argument" : "command", words[start_word]);
 883 
 884     return CLI_ERROR_ARG;
 885 }

 887 int cli_run_command(struct cli_def *cli, const char *command)
 888 {
 889     int r;
 890     unsigned int num_words, i, f;
 891     char *words[CLI_MAX_LINE_WORDS] = {0};
 892     int filters[CLI_MAX_LINE_WORDS] = {0};
 893 
 894     if (!command) return CLI_ERROR;
 895     while (isspace(*command))
 896         command++;
 897 
 898     if (!*command) return CLI_OK;
 899 
 900     num_words = cli_parse_line(command, words, CLI_MAX_LINE_WORDS);
 901     for (i = f = 0; i < num_words && f < CLI_MAX_LINE_WORDS - 1; i++)
 902     {
 903         if (words[i][0] == '|')
 904         filters[f++] = i;
 905     }
 906 
 907     filters[f] = 0;
 908 
 909     if (num_words)
 910         r = cli_find_command(cli, cli->commands, num_words, words, 0, filters);
 911     else
 912         r = CLI_ERROR;
 913 
 914     for (i = 0; i < num_words; i++)
 915         free(words[i]);
 916 
 917     if (r == CLI_QUIT)
 918         return r;
 919 
 920     return CLI_OK;
 921 }

 923 static int cli_get_completions(struct cli_def *cli, const char *command, char **completions, int max_completions)
 924 {
 925     struct cli_command *c;
 926     struct cli_command *n;
 927     int num_words, save_words, i, k=0;
 928     char *words[CLI_MAX_LINE_WORDS] = {0};
 929     int filter = 0;
 930 
 931     if (!command) return 0;
 932     while (isspace(*command))
 933         command++;
 934 
 935     save_words = num_words = cli_parse_line(command, words, sizeof(words)/sizeof(words[0]));
 936     if (!command[0] || command[strlen(command)-1] == ' ')
 937         num_words++;
 938 
 939     if (!num_words)
 940         goto out;
 941 
 942     for (i = 0; i < num_words; i++)
 943     {
 944         if (words[i] && words[i][0] == '|')
 945             filter = i;
 946     }
 947 
 948     if (filter) // complete filters
 949     {
 950         unsigned len = 0;
 951 
 952         if (filter < num_words - 1) // filter already completed
 953             goto out;
 954 
 955         if (filter == num_words - 1)
 956             len = strlen(words[num_words-1]);
 957 
 958         for (i = 0; filter_cmds[i].cmd && k < max_completions; i++)
 959         {
 960             if (!len || (len < strlen(filter_cmds[i].cmd) && !strncmp(filter_cmds[i].cmd, words[num_words - 1], len)))
 961                 completions[k++] = (char *)filter_cmds[i].cmd;
 962         }
 963 
 964         completions[k] = NULL;
 965         goto out;
 966     }
 967 
 968     for (c = cli->commands, i = 0; c && i < num_words && k < max_completions; c = n)
 969     {
 970         n = c->next;
 971 
 972         if (cli->privilege < c->privilege)
 973             continue;
 974 
 975         if (c->mode != cli->mode && c->mode != MODE_ANY)
 976             continue;
 977 
 978         if (words[i] && strncasecmp(c->command, words[i], strlen(words[i])))
 979             continue;
 980 
 981         if (i < num_words - 1)
 982         {
 983             if (strlen(words[i]) < c->unique_len)
 984                 continue;
 985 
 986             n = c->children;
 987             i++;
 988             continue;
 989         }
 990 
 991         completions[k++] = c->command;
 992     }
 993 
 994 out:
 995     for (i = 0; i < save_words; i++)
 996         free(words[i]);
 997 
 998     return k;
 999 }

1069 int cli_loop(struct cli_def *cli, int sockfd)
1070 {
1071     unsigned char c;
1072     int n, l, oldl = 0, is_telnet_option = 0, skip = 0, esc = 0;
1073     int cursor = 0, insertmode = 1;
1074     char *cmd = NULL, *oldcmd = 0;
1075     char *username = NULL, *password = NULL;
1076 
1077     cli_build_shortest(cli, cli->commands);
1078     cli->state = STATE_LOGIN;
1079 
1080     cli_free_history(cli);
1081     if (cli->telnet_protocol)
1082     {
1083         static const char *negotiate =
1084             "\xFF\xFB\x03"
1085             "\xFF\xFB\x01"
1086             "\xFF\xFD\x03"
1087             "\xFF\xFD\x01";
1088         _write(sockfd, negotiate, strlen(negotiate));
1089     }
1090 
1091     if ((cmd = malloc(CLI_MAX_LINE_LENGTH)) == NULL)
1092         return CLI_ERROR;
1093 
1094     if (!(cli->client = fdopen(sockfd, "w+")))
1095         return CLI_ERROR;
1096 
1097     setbuf(cli->client, NULL);
1098     if (cli->banner)
1099         cli_error(cli, "%s", cli->banner);
1100 
1101     // Set the last action now so we don't time immediately
1102     if (cli->idle_timeout)
1103         time(&cli->last_action);
1104 
1105     /* start off in unprivileged mode */
1106     cli_set_privilege(cli, PRIVILEGE_UNPRIVILEGED);
1107     cli_set_configmode(cli, MODE_EXEC, NULL);
1108 
1109     /* no auth required? */
1110     if (!cli->users && !cli->auth_callback)
1111         cli->state = STATE_NORMAL;
1112 
1113     while (1)
1114     {
1115         signed int in_history = 0;
1116         int lastchar = 0;
1117         struct timeval tm;
1118 
1119         cli->showprompt = 1;
1120 
1121         if (oldcmd)
1122         {
1123             l = cursor = oldl;
1124             oldcmd[l] = 0;
1125             cli->showprompt = 1;
1126             oldcmd = NULL;
1127             oldl = 0;
1128         }
1129         else
1130         {
1131             memset(cmd, 0, CLI_MAX_LINE_LENGTH);
1132             l = 0;
1133             cursor = 0;
1134         }
1135 
1136         memcpy(&tm, &cli->timeout_tm, sizeof(tm));
1137 
1138         while (1)
1139         {
1140             int sr;
1141             fd_set r;
1142             if (cli->showprompt)
1143             {
1144                 if (cli->state != STATE_PASSWORD && cli->state != STATE_ENABLE_PASSWORD)
1145                     _write(sockfd, "\r\n", 2);
1146 
1147                 switch (cli->state)
1148                 {
1149                     case STATE_LOGIN:
1150                         _write(sockfd, "Username: ", strlen("Username: "));
1151                         break;
1152 
1153                     case STATE_PASSWORD:
1154                         _write(sockfd, "Password: ", strlen("Password: "));
1155                         break;
1156 
1157                     case STATE_NORMAL:
1158                     case STATE_ENABLE:
1159                         show_prompt(cli, sockfd);
1160                         _write(sockfd, cmd, l);
1161                         if (cursor < l)
1162                         {
1163                             int n = l - cursor;
1164                             while (n--)
1165                                 _write(sockfd, "\b", 1);
1166                         }
1167                         break;
1168 
1169                     case STATE_ENABLE_PASSWORD:
1170                         _write(sockfd, "Password: ", strlen("Password: "));
1171                         break;
1172 
1173                 }
1174 
1175                 cli->showprompt = 0;
1176             }
1177 
1178             FD_ZERO(&r);
1179             FD_SET(sockfd, &r);
1180 
1181             if ((sr = select(sockfd + 1, &r, NULL, NULL, &tm)) < 0)
1182             {
1183                 /* select error */
1184                 if (errno == EINTR)
1185                     continue;
1186 
1187                 perror("select");
1188                 l = -1;
1189                 break;
1190             }
1191 
1192             if (sr == 0)
1193             {
1194                 /* timeout every second */
1195                 if (cli->regular_callback && cli->regular_callback(cli) != CLI_OK)
1196                 {
1197                     l = -1;
1198                     break;
1199                 }
1200 
1201                 if (cli->idle_timeout)
1202                 {
1203                     if (time(NULL) - cli->last_action >= cli->idle_timeout)
1204                     {
1205                         if (cli->idle_timeout_callback)
1206                         {
1207                             // Call the callback and continue on if successful
1208                             if (cli->idle_timeout_callback(cli) == CLI_OK)
1209                             {
1210                                 // Reset the idle timeout counter
1211                                 time(&cli->last_action);
1212                                 continue;
1213                             }
1214                         }
1215                         // Otherwise, break out of the main loop
1216                         l = -1;
1217                         break;
1218                     }
1219                 }
1220 
1221                 memcpy(&tm, &cli->timeout_tm, sizeof(tm));
1222                 continue;
1223             }
1224 
1225             if ((n = read(sockfd, &c, 1)) < 0)
1226             {
1227                 if (errno == EINTR)
1228                     continue;
1229 
1230                 perror("read");
1231                 l = -1;
1232                 break;
1233             }
1234 
1235             if (cli->idle_timeout)
1236                 time(&cli->last_action);
1237 
1238             if (n == 0)
1239             {
1240                 l = -1;
1241                 break;
1242             }
1243 
1244             if (skip)
1245             {
1246                 skip--;
1247                 continue;
1248             }
1249 
1250             if (c == 255 && !is_telnet_option)
1251             {
1252                 is_telnet_option++;
1253                 continue;
1254             }
1255 
1256             if (is_telnet_option)
1257             {
1258                 if (c >= 251 && c <= 254)
1259                 {
1260                     is_telnet_option = c;
1261                     continue;
1262                 }
1263 
1264                 if (c != 255)
1265                 {
1266                     is_telnet_option = 0;
1267                     continue;
1268                 }
1269 
1270                 is_telnet_option = 0;
1271             }
1272 
1273             /* handle ANSI arrows */
1274             if (esc)
1275             {
1276                 if (esc == '[')
1277                 {
1278                     /* remap to readline control codes */
1279                     switch (c)
1280                     {
1281                         case 'A': /* Up */
1282                             c = CTRL('P');
1283                             break;
1284 
1285                         case 'B': /* Down */
1286                             c = CTRL('N');
1287                             break;
1288 
1289                         case 'C': /* Right */
1290                             c = CTRL('F');
1291                             break;
1292 
1293                         case 'D': /* Left */
1294                             c = CTRL('B');
1295                             break;
1296 
1297                         default:
1298                             c = 0;
1299                     }
1300 
1301                     esc = 0;
1302                 }
1303                 else
1304                 {
1305                     esc = (c == '[') ? c : 0;
1306                     continue;
1307                 }
1308             }
1309 
1310             if (c == 0) continue;
1311             if (c == '\n') continue;
1312 
1313             if (c == '\r')
1314             {
1315                 if (cli->state != STATE_PASSWORD && cli->state != STATE_ENABLE_PASSWORD)
1316                     _write(sockfd, "\r\n", 2);
1317                 break;
1318             }
1319 
1320             if (c == 27)
1321             {
1322                 esc = 1;
1323                 continue;
1324             }
1325 
1326             if (c == CTRL('C'))
1327             {
1328                 _write(sockfd, "\a", 1);
1329                 continue;
1330             }
1331 
1332             /* back word, backspace/delete */
1333             if (c == CTRL('W') || c == CTRL('H') || c == 0x7f)
1334             {
1335                 int back = 0;
1336 
1337                 if (c == CTRL('W')) /* word */
1338                 {
1339                     int nc = cursor;
1340 
1341                     if (l == 0 || cursor == 0)
1342                         continue;
1343 
1344                     while (nc && cmd[nc - 1] == ' ')
1345                     {
1346                         nc--;
1347                         back++;
1348                     }
1349 
1350                     while (nc && cmd[nc - 1] != ' ')
1351                     {
1352                         nc--;
1353                         back++;
1354                     }
1355                 }
1356                 else /* char */
1357                 {
1358                     if (l == 0 || cursor == 0)
1359                     {
1360                         _write(sockfd, "\a", 1);
1361                         continue;
1362                     }
1363 
1364                     back = 1;
1365                 }
1366 
1367                 if (back)
1368                 {
1369                     while (back--)
1370                     {
1371                         if (l == cursor)
1372                         {
1373                             cmd[--cursor] = 0;
1374                             if (cli->state != STATE_PASSWORD && cli->state != STATE_ENABLE_PASSWORD)
1375                                 _write(sockfd, "\b \b", 3);
1376                         }
1377                         else
1378                         {
1379                             int i;
1380                             cursor--;
1381                             if (cli->state != STATE_PASSWORD && cli->state != STATE_ENABLE_PASSWORD)
1382                             {
1383                                 for (i = cursor; i <= l; i++) cmd[i] = cmd[i+1];
1384                                 _write(sockfd, "\b", 1);
1385                                 _write(sockfd, cmd + cursor, strlen(cmd + cursor));
1386                                 _write(sockfd, " ", 1);
1387                                 for (i = 0; i <= (int)strlen(cmd + cursor); i++)
1388                                     _write(sockfd, "\b", 1);
1389                             }
1390                         }
1391                         l--;
1392                     }
1393 
1394                     continue;
1395                 }
1396             }
1397 
1398             /* redraw */
1399             if (c == CTRL('L'))
1400             {
1401                 int i;
1402                 int cursorback = l - cursor;
1403 
1404                 if (cli->state == STATE_PASSWORD || cli->state == STATE_ENABLE_PASSWORD)
1405                     continue;
1406 
1407                 _write(sockfd, "\r\n", 2);
1408                 show_prompt(cli, sockfd);
1409                 _write(sockfd, cmd, l);
1410 
1411                 for (i = 0; i < cursorback; i++)
1412                     _write(sockfd, "\b", 1);
1413 
1414                 continue;
1415             }
1416 
1417             /* clear line */
1418             if (c == CTRL('U'))
1419             {
1420                 if (cli->state == STATE_PASSWORD || cli->state == STATE_ENABLE_PASSWORD)
1421                     memset(cmd, 0, l);
1422                 else
1423                     cli_clear_line(sockfd, cmd, l, cursor);
1424 
1425                 l = cursor = 0;
1426                 continue;
1427             }
1428 
1429             /* kill to EOL */
1430             if (c == CTRL('K'))
1431             {
1432                 if (cursor == l)
1433                     continue;
1434 
1435                 if (cli->state != STATE_PASSWORD && cli->state != STATE_ENABLE_PASSWORD)
1436                 {

1579                     }

1677                         cmd[i + 1] = cmd[i];
1678                     // Write what we've just added
1679                     cmd[cursor] = c;
1680 
1681                     _write(sockfd, &cmd[cursor], l - cursor + 1);

1686                 else
1687                 {
1688                     cmd[cursor] = c;
1689                 }

1695                 if (c == '?' && cursor == l)

1699                     oldl = cursor = l - 1;

1702                 _write(sockfd, &c, 1);
1703             }
1704

1706             oldl = 0;

1708         }
1709

1711 
1712         if (cli->state == STATE_LOGIN)
1713         {
1714             if (l == 0) continue;
1715 
1716             /* require login */
1717             free_z(username);
1718             if (!(username = strdup(cmd)))
1719                 return 0;

1721             cli->showprompt = 1;
1722         }
1723         else if (cli->state == STATE_PASSWORD)

1729             if (!(password = strdup(cmd)))
1730                 return 0;

1740                 for (u = cli->users; u; u = u->next)
1741                 {

1743                     {
1744                         allowed++;
1745                         break;
1746                     }
1747                 }
1748             }
1749 
1750             if (allowed)
1751             {
1752                 cli_error(cli, " ");
1753                 cli->state = STATE_NORMAL;
1754             }
1755             else
1756             {
1757                 cli_error(cli, "\n\nAccess denied");
1758                 free_z(username);
1759                 free_z(password);
1760                 cli->state = STATE_LOGIN;
1761             }
1762 
1763             cli->showprompt = 1;
1764         }
1765         else if (cli->state == STATE_ENABLE_PASSWORD)
1766         {
1767             int allowed = 0;
1768             if (cli->enable_password)
1769             {
1770                 /* check stored static enable password */
1771                 if (pass_matches(cli->enable_password, cmd))
1772                     allowed++;
1773             }
1774 
1775             if (!allowed && cli->enable_callback)
1776             {
1777                 /* check callback */
1778                 if (cli->enable_callback(cmd))
1779                     allowed++;
1780             }
1781 
1782             if (allowed)
1783             {
1784                 cli_print(cli, "");
1785                 cli->state = STATE_ENABLE;
1786                 cli_set_privilege(cli, PRIVILEGE_PRIVILEGED);
1787             }
1788             else
1789             {
1790                 cli_error(cli, "\n\nAccess denied");
1791                 cli->state = STATE_NORMAL;
1792             }
1793         }
1794         else
1795         {
1796             if (l == 0) continue;
1797             if (cmd[l - 1] != '?' && strcasecmp(cmd, "history") != 0)
1798                 cli_add_history(cli, cmd);
1799 
1800             if (cli_run_command(cli, cmd) == CLI_QUIT)
1801                 break;
1802         }
1803 
1804         // Update the last_action time now as the last command run could take a
1805         // long time to return
1806         if (cli->idle_timeout)
1807             time(&cli->last_action);
1808     }
1809 
1810     cli_free_history(cli);
1811     free_z(username);
1812     free_z(password);
1813     free_z(cmd);
1814 
1815     fclose(cli->client);
1816     cli->client = 0;
1817     return CLI_OK;
1818 }

1863 static void _print(struct cli_def *cli, int print_mode, const char *format, va_list ap)
1864 {
1865     va_list aq;
1866     int n;
1867     char *p;
1868 
1869     if (!cli) return; // sanity check
1870 
1871     while (1)
1872     {
1873         va_copy(aq, ap);
1874         if ((n = vsnprintf(cli->buffer, cli->buf_size, format, ap)) == -1)
1875             return;
1876 
1877         if ((unsigned)n >= cli->buf_size)
1878         {
1879             cli->buf_size = n + 1;
1880             cli->buffer = realloc(cli->buffer, cli->buf_size);
1881             if (!cli->buffer)
1882                 return;
1883             va_end(ap);
1884             va_copy(ap, aq);
1885             continue;
1886         }
1887         break;
1888     }
1889 
1890 
1891     p = cli->buffer;
1892     do
1893     {
1894         char *next = strchr(p, '\n');
1895         struct cli_filter *f = (print_mode & PRINT_FILTERED) ? cli->filters : 0;
1896         int print = 1;
1897 
1898         if (next)
1899             *next++ = 0;
1900         else if (print_mode & PRINT_BUFFERED)
1901             break;
1902 
1903         while (print && f)
1904         {
1905             print = (f->filter(cli, p, f->data) == CLI_OK);
1906             f = f->next;
1907         }
1908         if (print)
1909         {
1910             if (cli->print_callback)
1911                 cli->print_callback(cli, p);
1912             else if (cli->client)
1913                 fprintf(cli->client, "%s\r\n", p);
1914         }
1915 
1916         p = next;
1917     } while (p);
1918 
1919     if (p && *p)
1920     {
1921         if (p != cli->buffer)
1922         memmove(cli->buffer, p, strlen(p));
1923     }
1924     else *cli->buffer = 0;
1925 }
