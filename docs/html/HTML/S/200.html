  44 /* Identification data for a flow. */
  45 struct sw_flow_key {
  46     struct flow flow;           /* Flow data (in network byte order). */
  47     uint32_t wildcards;         /* Wildcard fields (in host byte order). */
  48     uint32_t nw_src_mask;       /* 1-bit in each significant nw_src bit. */
  49     uint32_t nw_dst_mask;       /* 1-bit in each significant nw_dst bit. */
  50 };
  51 
  52 struct sw_flow_actions {
  53     size_t actions_len;
  54     struct ofp_action_header actions[0];
  55 };
  56 
  57 struct sw_flow {
  58     struct sw_flow_key key;
  59 
  60     uint64_t cookie;            /* Opaque controller-issued identifier. */
  61     uint16_t priority;          /* Only used on entries with wildcards. */
  62     uint16_t idle_timeout;      /* Idle time before discarding (seconds). */
  63     uint16_t hard_timeout;      /* Hard expiration time (seconds) */
  64     uint64_t used;              /* Last used time. */
  65     uint64_t created;           /* When the flow was created. */
  66     uint64_t packet_count;      /* Number of packets seen. */
  67     uint64_t byte_count;        /* Number of bytes seen. */
  68     uint8_t reason;             /* Reason flow removed (one of OFPRR_*). */
  69     uint8_t send_flow_rem;      /* Send a flow removed to the controller */
  70     uint8_t emerg_flow;         /* Emergency flow indicator */
  71 
  72     struct sw_flow_actions *sf_acts;
  73 
  74     /* Private to table implementations. */
  75     struct list node;
  76     struct list iter_node;
  77     unsigned long int serial;
  78 
  79     void *private;              /* Cookie for tables */
  80 };
