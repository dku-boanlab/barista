  81 static int app_event_type(const char *event)

  83     int i;

  85         if (strcmp(event, app_ev_list[i]) == 0)

 245 /**
 246  * \brief Function to load the available events of each role
 247  * \param role_file Application role file
 248  */
 249 static int load_events_for_roles(char *role_file)
 250 {
 251     json_t *json;
 252     json_error_t error;
 253 
 254     ALOG_INFO(RBAC_ID, "Application role file: %s", role_file);
 255 
 256     if (access(role_file, F_OK)) {
 257         ALOG_ERROR(RBAC_ID, "No application role file whose name is '%s'", role_file);
 258         return -1;
 259     }
 260 
 261     char *raw = str_read(role_file);
 262     char *role = str_preproc(raw);
 263 
 264     json = json_loads(role, 0, &error);
 265     if (!json) {
 266         PERROR("json_loads");
 267         return -1;
 268     } else if (!json_is_array(json)) {
 269         PERROR("json_is_array");
 270         json_decref(json);
 271         return -1;
 272     }
 273 
 274     int i;
 275     for (i=0; i<json_array_size(json); i++) {
 276         json_t *data = json_array_get(json, i);
 277 
 278         char name[__CONF_WORD_LEN] = {0};
 279         json_t *j_name = json_object_get(data, "name");
 280         if (json_is_string(j_name)) {
 281             strcpy(name, json_string_value(j_name));
 282         }
 283 
 284         int role = 0;
 285         if (strlen(name) == 0) {
 286             PRINTF("no role type\n");
 287             return -1;
 288         } else {
 289             if (strcmp(name, "network") == 0)
 290                 role = APP_NETWORK;
 291             else if (strcmp(name, "management") == 0)
 292                 role = APP_MANAGEMENT;
 293             else if (strcmp(name, "security") == 0)
 294                 role = APP_SECURITY;
 295             else if (strcmp(name, "admin") == 0)
 296                 role = APP_ADMIN;
 297             else
 298                 continue;
 299         }
 300 
 301         strcpy(role_to_event[role].name, name);
 302 
 303         // set inbound events
 304         json_t *events = json_object_get(data, "events");
 305         if (json_is_array(events)) {
 306             int j;
 307             for (j=0; j<json_array_size(events); j++) {
 308                 json_t *event = json_array_get(events, j);
 309 
 310                 if (app_event_type(json_string_value(event)) == AV_NUM_EVENTS) {
 311                     PRINTF("wrong app event name\n");
 312                     return -1;
 313                 }
 314             }
 315 
 316             for (j=0; j<json_array_size(events); j++) {
 317                 json_t *event = json_array_get(events, j);
 318 
 319                 role_to_event[role].event[app_event_type(json_string_value(event))] = 1;
 320             }
 321         }
 322     }
 323 
 324     return 0;
 325 }

 343 /**
 344  * \brief Function to check the incoming event of the corresponding application
 345  * \param id Application ID
 346  * \param type App event
 347  */
 348 static int check_app_role(uint32_t id, uint16_t type)
 349 {
 350     int i;
 351     for (i=0; i<app_to_role_cnt; i++) {
 352         if (app_to_role[i].id == id) { // found the matched app
 353             if (app_to_role[i].role == APP_BASE) // skip base
 354                 return -1;
 355             else if (role_to_event[app_to_role[i].role].event[type] == 1) // allow or deny
 356                 return 1;
 357             else
 358                 return 0;
 359         }
 360     }
 361 
 362     return 0;
 363 }

 367 /**
 368  * \brief The main function
 369  * \param activated The activation flag of this application
 370  * \param argc The number of arguments
 371  * \param argv Arguments
 372  */
 373 int rbac_main(int *activated, int argc, char **argv)
 374 {
 375     ALOG_INFO(RBAC_ID, "Init - Role-based access control");
 376 
 377     app_to_role = (app2role_t *)MALLOC(sizeof(app2role_t) * __MAX_APPLICATIONS);
 378     if (app_to_role == NULL) {
 379         PERROR("malloc");
 380         return -1;
 381     }
 382 
 383     role_to_event = (role2ev_t *)MALLOC(sizeof(role2ev_t) * NUM_APP_ROLES);
 384     if (role_to_event == NULL) {
 385         PERROR("malloc");
 386         return -1;
 387     }
 388 
 389     reset_app_roles();
 390     
 391     // load app roles
 392     if (load_app_roles(app_file) < 0)
 393         return -1;
 394 
 395     // load app events for each role
 396     if (load_events_for_roles(role_file) < 0)
 397         return -1;
 398 
 399     activate();
 400 
 401     return 0;
 402 }

 430 /**
 431  * \brief The handler function
 432  * \param av Read-only app event
 433  * \param av_out Read-write app event (if this application has the write permission)
 434  */
 435 int rbac_handler(const app_event_t *av, app_event_out_t *av_out)
 436 {
 437     if (check_app_role(av->id, av->type)) {
 438         return 0;
 439     }
 440 
 441     int i;
 442     for (i=0; i<app_to_role_cnt; i++) {
 443         if (app_to_role[i].id == av->id) {
 444             ALOG_WARN(RBAC_ID, "Unauthorized access (app: %s (%u), type: %s)", app_to_role[i].name, av->id, app_ev_list[av->type]);
 445             break;
 446         }
 447     }
 448 
 449     return -1;
 450 }
