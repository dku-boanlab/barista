 117 int init(struct oflops_context *ctx, char * config_str) {
 118   char *pos = NULL;
 119   char *param = config_str;
 120   char *value = NULL;
 121   double exponent;
 122 
 123   printf("log initialized\n");
 124 
 125   //init measurement queue
 126   TAILQ_INIT(&head); 
 127   TAILQ_INIT(&stats_head); 
 128 
 129   while(*config_str == ' ') {
 130     config_str++;
 131   }
 132   param = config_str;
 133   while(1) {
 134     pos = index(param, ' ');
 135 
 136     if((pos == NULL)) {
 137       if (*param != '\0') {
 138         pos = param + strlen(param) + 1;
 139       } else
 140         break;
 141     }
 142     *pos='\0';
 143     pos++;
 144     value = index(param,'=');
 145     *value = '\0';
 146     value++;
 147     if(value != NULL) {
 148       if(strcmp(param, "flows") == 0) {
 149         flows = atoi(value);
 150         if(flows <= 0)
 151           perror_and_exit("Invalid flow number",1);
 152       } else if(strcmp(param, "query") == 0) {
 153         query = atoi(value);
 154         if(query <= 0)
 155           perror_and_exit("Invalid flow number",1);
 156 
 157         exponent = log2(query);
 158         if(exponent - floor(exponent) != 0) {
 159           printf("query=%d, exponent=%f, floor exponent:%f\n", query, exponent, floor(exponent));
 160           query = (int)pow(2, ceil(exponent));
 161           printf("query size must be a power of 2. converting to %d\n", query);
 162         }
 163 
 164       } else if(strcmp(param, "network") == 0) {
 165         network = (char *)xmalloc(strlen(value) + 1);
 166         strcpy(network, value);
 167       } else if(strcmp(param, "pkt_size") == 0) {
 168         //parse int to get pkt size
 169         pkt_size = strtol(value, NULL, 0);
 170         if((pkt_size < MIN_PKT_SIZE) && (pkt_size > MAX_PKT_SIZE))  {
 171           perror_and_exit("Invalid packet size value", 1);
 172         }
 173       } else if(strcmp(param, "data_rate") == 0) {
 174         //parse int to get pkt size
 175         datarate = strtol(value, NULL, 0);
 176         if((datarate <= 0) || (datarate > 1010))  {
 177           perror_and_exit("Invalid data rate param(Values between 1 and 1010)", 1);
 178         }
 179       } else if(strcmp(param, "probe_rate") == 0) {
 180         //parse int to get pkt size
 181         proberate = strtol(value, NULL, 0);
 182         if((proberate <= 0) || (proberate >= 1010)) {
 183           perror_and_exit("Invalid probe rate param(Value between 1 and 1010)", 1);
 184         }
 185       } else if(strcmp(param, "print") == 0) {
 186         //parse int to get pkt size
 187         print = strtol(value, NULL, 0);
 188       } else {
 189         fprintf(stderr, "Invalid parameter:%s\n", param);
 190       }
 191       param = pos;
 192     }
 193   } 
 194 
 195   //calculating interpacket gap
 196   data_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (datarate * mbits_to_bits);
 197   fprintf(stderr, "Sending data interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 198           (uint32_t)data_snd_interval, (uint32_t)pkt_size, (uint32_t)datarate);
 199   probe_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (proberate * mbits_to_bits);
 200   fprintf(stderr, "Sending probe interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 201           (uint32_t)probe_snd_interval, (uint32_t)pkt_size, (uint32_t)proberate);
 202 
 203   return 0;
 204 }

 240     ch = 2*(np->ch - 1) + i;
 241     ix[ch]++; 
 242     min_id[ch] = (np->id < min_id[ch])?np->id:min_id[ch];
 243     max_id[ch] = (np->id > max_id[ch])?np->id:max_id[ch];
 244     data[ch][ix[ch]] = (double) time_diff(&np->snd, &np->rcv);

 246   }
 247 
 248   for(ch = 0; ch < 6; ch++) {
 249       if(ix[ch] == 0) continue;

 252       std = (uint32_t)sqrt(gsl_stats_variance(data[ch], 1, ix[ch]));
 253       median = (uint32_t)gsl_stats_median_from_sorted_data (data[ch], 1, ix[ch]);

 255       snprintf(msg, 1024, "statistics:port:%d.%d:%u:%u:%u:%.4f:%d",

 257       printf("%s\n", msg);
 258       oflops_log(now, GENERIC_MSG, msg);
 259       
 260   }
 261 
 262   ix[0] = 0;
 263   min_id[0] =  INT_MAX;
 264   max_id[0] =  INT_MIN;
 265   free(data[0]);
 266   data[0] = (double *)malloc(sizeof(double)*(stats_count));
 267 
 268   for (stats_np = stats_head.tqh_first; 
 269        stats_np != NULL; stats_np = stats_np->entries.tqe_next) {
 270     delay = time_diff(&now, &stats_np->rcv);
 271     memcpy(&now,&stats_np->rcv, sizeof(struct timeval));
 272     if((print) && (ix[0] > 0)) {
 273       snprintf(msg, 1024, "STATS_REPLY:%d:%lu.%06lu:%u", 
 274                stats_np->id, stats_np->rcv.tv_sec, stats_np->rcv.tv_usec, delay);
 275       oflops_log(now, GENERIC_MSG, msg);
 276       
 277     }
 278     
 279     if(ix[0] > 0) {
 280       data[0][ix[0] - 1] = delay;
 281     }
 282     ix[0]++;
 283   }
 284 
 285   ix[0]--; //we have added 1 on the last round which we have to remove
 286   if(ix[0] > 0) {
 287       gsl_sort (data[0], 1, ix[0]);
 288       mean = (uint32_t)gsl_stats_mean(data[0], 1, ix[0]);
 289       std = (uint32_t)sqrt(gsl_stats_variance(data[0], 1, ix[0]));
 290       median = (uint32_t)gsl_stats_median_from_sorted_data (data[0], 1, ix[0]);
 291       loss = (float)ix[0]/(float)(max_id[0] - min_id[0]);
 292       snprintf(msg, 1024, "statistics:stats:%u:%u:%u:%.4f:%d",

 294       printf("%s\n", msg);
 295       oflops_log(now, GENERIC_MSG, msg);
 296   } else {
 297       oflops_log(now, GENERIC_MSG, "stats_stats:fail");
 298   }
 299   return 0;

 475   netsnmp_variable_list *vars;
 476   int i, len = 1024;
 477   char msg[1024], log[1024];
 478   struct timeval now;
 479 
 480   for(vars = se->pdu->variables; vars; vars = vars->next_variable)  {

 482 
 483     for (i = 0; i < ctx->cpuOID_count; i++) {
 484       if((vars->name_length == ctx->cpuOID_len[i]) &&
 485          (memcmp(vars->name, ctx->cpuOID[i],  ctx->cpuOID_len[i] * sizeof(oid)) == 0) ) {
 486         snprintf(log, len, "cpu : %s %%", msg);
 487         oflops_log(now, SNMP_MSG, log);
 488       }
 489     }
 490     
 491     for(i=0;i<ctx->n_channels;i++) {
 492       if((vars->name_length == ctx->channels[i].inOID_len) &&
 493          (memcmp(vars->name, ctx->channels[i].inOID,  
 494                  ctx->channels[i].inOID_len * sizeof(oid)) == 0) ) {
 495         snprintf(log, len, "port %d : rx %s pkts",  
 496                  (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], 
 497                  msg);
 498         oflops_log(now, SNMP_MSG, log);
 499         break;
 500       }
 501       
 502       if((vars->name_length == ctx->channels[i].outOID_len) &&
 503          (memcmp(vars->name, ctx->channels[i].outOID,  
 504                  ctx->channels[i].outOID_len * sizeof(oid))==0) ) {
 505         snprintf(log, len, "port %d : tx %s pkts",  
 506                  (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 507         oflops_log(now, SNMP_MSG, log);
 508         break;
 509       }
 510     } //for
 511   }// if cpu
 512   return 0;
 513 }

 533 /** Handle pcap event.
 534  * @param ctx pointer to opaque context
 535  * @param pe pcap event
 536  * @param ch enumeration of channel that pcap event is triggered
 537  */
 538 int handle_pcap_event(struct oflops_context *ctx, struct pcap_event *pe,
 539                       oflops_channel_name ch) {
 540     if ( (ch == OFLOPS_DATA3) || (ch == OFLOPS_DATA2) || (ch == OFLOPS_DATA1)){
 541     struct pktgen_hdr *pktgen;
 542     pktgen = extract_pktgen_pkt((unsigned char *)pe->data, pe->pcaphdr.caplen, NULL);
 543     if(pktgen == NULL) //skip non IP packets
 544       return 0;
 545 
 546     struct entry *n1 = malloc(sizeof(struct entry));
 547     n1->snd.tv_sec = htonl(pktgen->tv_sec);
 548     n1->snd.tv_usec = htonl(pktgen->tv_usec);
 549     memcpy(&n1->rcv, &pe->pcaphdr.ts, sizeof(struct timeval));
 550     n1->id = htonl(pktgen->seq_num);
 551     n1->ch = ch;
 552     count[ch - 1]++;
 553     TAILQ_INSERT_TAIL(&head, n1, entries);
 554   }
 555   return 0;
 556 }

 599 int 
 600 dummy_parse_packet(void *data, int len) {
 601   // assume we have ethernet packets.
 602   // skip first bytes of the ether because they are
 603   // in the simple case of static length.
 604   struct ether_header *eth = (struct ether_header *) data;
 605   if (len < sizeof(struct ether_header))
 606     return -1;
 607   data += sizeof(struct ether_header);
 608   if(ntohs(eth->ether_type) != ETHERTYPE_IP) {
 609     return -1;
 610   }
 611   if (len < sizeof(struct iphdr))
 612     return -1;
 613   struct iphdr *ip_p = (struct iphdr *)data;
 614   if (len < 4*ip_p->ihl)
 615     return -1;
 616   data += 4*ip_p->ihl;
 617 
 618   if(ip_p->protocol != IPPROTO_UDP)
 619     return -1;
 620 
 621   if(len <  sizeof(struct tcphdr))
 622     return -1;
 623   struct udphdr *udp_p = (struct udphdr *)data;
 624   if (len < sizeof(struct udphdr))
 625     return -1;
 626 
 627   if((ntohs(udp_p->source) == 8080) && 
 628      (ntohs(udp_p->dest) == 8080))
 629     return 0;
 630   else 
 631     return -1;
 632 
 633 };

 635 int
 636 of_event_other(struct oflops_context *ctx, const struct ofp_header * ofph) {
 637   struct timeval now;
 638   char msg[100];
 639   struct ofp_error_msg *err_p;
 640   int len, res;
 641   void *b;
 642 
 643   if(ofph->type == OFPT_STATS_REPLY) {
 644     struct ofp_stats_reply *ofpr = (struct ofp_stats_reply *)ofph;
 645     if(ntohs(ofpr->type) == OFPST_AGGREGATE) {
 646       sprintf(msg, "%d", ntohl(ofph->xid));
 647       gettimeofday(&now, NULL);
 648       oflops_log(now, OFPT_STATS_REPLY_FLOW, msg);
 649       if((ntohs(ofpr->flags) & OFPSF_REPLY_MORE) == 0) {
 650         len = make_ofp_aggr_flow_stats(&b, trans_id++);
 651 
 652         struct ofp_aggregate_stats_request *reqp = (struct ofp_aggregate_stats_request *)
 653           (b + sizeof(struct ofp_stats_request));
 654         reqp->match.wildcards = htonl(OFPFW_IN_PORT |  OFPFW_DL_VLAN |   OFPFW_DL_DST |
 655                                       OFPFW_DL_SRC |  OFPFW_DL_TYPE | OFPFW_DL_VLAN_PCP |
 656                                       OFPFW_NW_PROTO | OFPFW_TP_SRC | OFPFW_TP_DST |
 657                                       OFPFW_NW_TOS | (32 << OFPFW_NW_SRC_SHIFT) |
 658                                       ((query_exponent) << OFPFW_NW_DST_SHIFT));
 659 
 660         uint32_t flow_netmask = (ntohl(inet_addr(network)) & ((0xFFFFFFFF)<<flows_exponent));
 661         
 662         //in case query range is smaller that flow range, round robin around ips for more
 663         // variability.
 664         if(query_exponent < flows_exponent) 
 665           flow_netmask += (stats_count%(0x1 <<(flows_exponent-query_exponent)) << query_exponent);
 666         reqp->match.nw_dst = htonl(flow_netmask);
 667 
 668         res = oflops_send_of_mesg(ctx, b);
 669         free(b);
 670       }
 671       struct stats_entry *n1 = malloc(sizeof(struct stats_entry));
 672       memcpy(&n1->rcv, &now, sizeof(struct timeval));
 673       n1->id =  ntohl(ofph->xid);
 674       stats_count++;
 675       TAILQ_INSERT_TAIL(&stats_head, n1, entries); 
 676     }
 677   } else if (ofph->type == OFPT_ERROR) {
 678     err_p = (struct ofp_error_msg *)ofph;
 679     sprintf(msg, "OFPT_ERROR(type: %d, code: %d)", ntohs(err_p->type), ntohs(err_p->code));
 680     fprintf(stderr, "%s\n", msg);
 681     perror_and_exit(msg, 1);
 682   }
 683   return 0;
 684 }
