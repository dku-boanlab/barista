  42 static uint16_t
  43 validate_output(struct datapath *dp UNUSED, const struct sw_flow_key *key, 
  44         const struct ofp_action_header *ah) 
  45 {
  46     struct ofp_action_output *oa = (struct ofp_action_output *)ah;
  47 
  48     /* To prevent loops, make sure there's no action to send to the
  49      * OFP_TABLE virtual port.
  50      */
  51     if (oa->port == htons(OFPP_NONE) || 
  52             (!(key->wildcards & OFPFW_IN_PORT) 
  53                     && oa->port == key->flow.in_port)) {
  54         return OFPBAC_BAD_OUT_PORT;
  55     }
  56     return ACT_VALIDATION_OK;
  57 }

  59 static uint16_t
  60 validate_queue(struct datapath *dp UNUSED, const struct sw_flow_key *key UNUSED,
  61                const struct ofp_action_header *ah)
  62 {
  63     struct ofp_action_enqueue *ea = (struct ofp_action_enqueue *)ah;
  64 
  65         /* Only physical ports may have queues. */
  66     if (ntohs(ea->port) > OFPP_MAX && ntohs(ea->port) != OFPP_IN_PORT) {
  67         return OFPBAC_BAD_OUT_PORT;
  68     }
  69     return ACT_VALIDATION_OK;
  70 }

  74           size_t max_len, int out_port, uint32_t queue_id,

  84 /* Modify vlan tag control information (TCI).  Only sets the TCI bits
  85  * indicated by 'mask'.  If no vlan tag is present, one is added.
  86  */
  87 static void
  88 modify_vlan_tci(struct ofpbuf *buffer, struct sw_flow_key *key,
  89         uint16_t tci, uint16_t mask)
  90 {
  91     struct vlan_eth_header *veh;
  92 
  93     if (key->flow.dl_vlan != htons(OFP_VLAN_NONE)) {
  94         /* Modify vlan id, but maintain other TCI values */
  95         veh = buffer->l2;
  96         veh->veth_tci &= ~htons(mask);
  97         veh->veth_tci |= htons(tci);
  98     } else {
  99         /* Insert new vlan id. */
 100         struct eth_header *eh = buffer->l2;
 101         struct vlan_eth_header tmp;
 102         memcpy(tmp.veth_dst, eh->eth_dst, ETH_ADDR_LEN);
 103         memcpy(tmp.veth_src, eh->eth_src, ETH_ADDR_LEN);
 104         tmp.veth_type = htons(ETH_TYPE_VLAN);
 105         tmp.veth_tci = htons(tci);
 106         tmp.veth_next_type = eh->eth_type;
 107 
 108         veh = ofpbuf_push_uninit(buffer, VLAN_HEADER_LEN);
 109         memcpy(veh, &tmp, sizeof tmp);
 110         buffer->l2 = (char*)buffer->l2 - VLAN_HEADER_LEN;
 111     }
 112 
 113     key->flow.dl_vlan = veh->veth_tci & htons(VLAN_VID_MASK);
 114     key->flow.dl_vlan_pcp = (uint8_t)((ntohs(veh->veth_tci) >> VLAN_PCP_SHIFT)
 115                                       & VLAN_PCP_BITMASK);
 116 }

 119 /* Remove an existing vlan header if it exists. */
 120 static void
 121 vlan_pull_tag(struct ofpbuf *buffer)
 122 {
 123     struct vlan_eth_header *veh = buffer->l2;
 124 
 125     if (veh->veth_type == htons(ETH_TYPE_VLAN)) {
 126         struct eth_header tmp;
 127 
 128         memcpy(tmp.eth_dst, veh->veth_dst, ETH_ADDR_LEN);
 129         memcpy(tmp.eth_src, veh->veth_src, ETH_ADDR_LEN);
 130         tmp.eth_type = veh->veth_next_type;
 131 
 132         buffer->size -= VLAN_HEADER_LEN;
 133         buffer->data = (char*)buffer->data + VLAN_HEADER_LEN;
 134         buffer->l2 = (char*)buffer->l2 + VLAN_HEADER_LEN;
 135         memcpy(buffer->data, &tmp, sizeof tmp);
 136     }
 137 }

 181 static void
 182 set_nw_addr(struct ofpbuf *buffer, struct sw_flow_key *key, 
 183         const struct ofp_action_header *ah)
 184 {
 185     struct ofp_action_nw_addr *na = (struct ofp_action_nw_addr *)ah;
 186     uint16_t eth_proto = ntohs(key->flow.dl_type);
 187 
 188     if (eth_proto == ETH_TYPE_IP) {
 189         struct ip_header *nh = buffer->l3;
 190         uint8_t nw_proto = key->flow.nw_proto;
 191         uint32_t new, *field;
 192 
 193         new = na->nw_addr;
 194         field = na->type == htons(OFPAT_SET_NW_SRC) ? &nh->ip_src : &nh->ip_dst;
 195         if (nw_proto == IP_TYPE_TCP) {
 196             struct tcp_header *th = buffer->l4;
 197             th->tcp_csum = recalc_csum32(th->tcp_csum, *field, new);
 198         } else if (nw_proto == IP_TYPE_UDP) {
 199             struct udp_header *th = buffer->l4;
 200             if (th->udp_csum) {
 201                 th->udp_csum = recalc_csum32(th->udp_csum, *field, new);
 202                 if (!th->udp_csum) {
 203                     th->udp_csum = 0xffff;
 204                 }
 205             }
 206         }
 207         nh->ip_csum = recalc_csum32(nh->ip_csum, *field, new);
 208         *field = new;
 209     }
 210 }

 212 static void
 213 set_nw_tos(struct ofpbuf *buffer, struct sw_flow_key *key, 
 214            const struct ofp_action_header *ah)
 215 {
 216     struct ofp_action_nw_tos *nt = (struct ofp_action_nw_tos *)ah;
 217     uint16_t eth_proto = ntohs(key->flow.dl_type);
 218 
 219    if (eth_proto == ETH_TYPE_IP) {
 220        struct ip_header *nh = buffer->l3;
 221        uint8_t new, *field;
 222 
 223        /* JeanII : Set only 6 bits, don't clobber ECN */
 224        new = (nt->nw_tos & 0xFC) | (nh->ip_tos & 0x03);
 225 
 226        /* Get address of field */
 227        field = &nh->ip_tos;
 228 
 229         /* jklee : ip tos field is not included in TCP pseudo header.
 230          * Need magic as update_csum() don't work with 8 bits. */
 231        nh->ip_csum = recalc_csum32(nh->ip_csum, htons((uint16_t)*field),
 232                                    htons((uint16_t)new));
 233 
 234        /* Change the IP ToS bits */
 235        *field = new;
 236     }
 237 }

 254             th->tcp_csum = recalc_csum16(th->tcp_csum, *field, new);
 255             *field = new;
 256         } else if (nw_proto == IP_TYPE_UDP) {
 257             struct udp_header *th = buffer->l4;
 258             field = ta->type == htons(OFPAT_SET_TP_SRC) ? &th->udp_src : &th->udp_dst;
 259             th->udp_csum = recalc_csum16(th->udp_csum, *field, new);
 260             *field = new;
 261         }
 262     }

 352 /* Validate built-in OpenFlow actions.  Either returns ACT_VALIDATION_OK
 353  * or an OFPET_BAD_ACTION error code. */
 354 static uint16_t 
 355 validate_ofpat(struct datapath *dp, const struct sw_flow_key *key, 
 356         const struct ofp_action_header *ah, uint16_t type, uint16_t len)
 357 {
 358     uint16_t ret = ACT_VALIDATION_OK;
 359     const struct openflow_action *act = &of_actions[type];
 360 
 361     if ((len < act->min_size) || (len > act->max_size)) {
 362         return OFPBAC_BAD_LEN;
 363     }
 364 
 365     if (act->validate) {
 366         ret = act->validate(dp, key, ah);
 367     }
 368 
 369     return ret;
 370 }

 372 /* Validate vendor-defined actions.  Either returns ACT_VALIDATION_OK
 373  * or an OFPET_BAD_ACTION error code. */
 374 static uint16_t 
 375 validate_vendor(struct datapath *dp UNUSED, const struct sw_flow_key *key UNUSED, 
 376         const struct ofp_action_header *ah, uint16_t len)
 377 {
 378     struct ofp_action_vendor_header *avh;
 379     int ret = ACT_VALIDATION_OK;
 380 
 381     if (len < sizeof(struct ofp_action_vendor_header)) {
 382         return OFPBAC_BAD_LEN;
 383     }
 384 
 385     avh = (struct ofp_action_vendor_header *)ah;
 386 
 387     switch(ntohl(avh->vendor)) {
 388     default:
 389         return OFPBAC_BAD_VENDOR;
 390     }
 391 
 392     return ret;
 393 }

 395 /* Validates a list of actions.  If a problem is found, a code for the
 396  * OFPET_BAD_ACTION error type is returned.  If the action list validates, 
 397  * ACT_VALIDATION_OK is returned. */
 398 uint16_t 
 399 validate_actions(struct datapath *dp, const struct sw_flow_key *key,
 400         const struct ofp_action_header *actions, size_t actions_len)
 401 {
 402     uint8_t *p = (uint8_t *)actions;
 403     int err;
 404 
 405     while (actions_len >= sizeof(struct ofp_action_header)) {
 406         struct ofp_action_header *ah = (struct ofp_action_header *)p;
 407         size_t len = ntohs(ah->len);
 408         uint16_t type;
 409 
 410         /* Make there's enough remaining data for the specified length
 411          * and that the action length is a multiple of 64 bits. */
 412         if ((actions_len < len) || (len % 8) != 0) {
 413             return OFPBAC_BAD_LEN;
 414         }
 415 
 416         type = ntohs(ah->type);
 417         if (type < ARRAY_SIZE(of_actions)) {
 418             err = validate_ofpat(dp, key, ah, type, len);
 419             if (err != ACT_VALIDATION_OK) {
 420                 return err;
 421             }
 422         } else if (type == OFPAT_VENDOR) {
 423             err = validate_vendor(dp, key, ah, len);
 424             if (err != ACT_VALIDATION_OK) {
 425                 return err;
 426             }
 427         } else {
 428             return OFPBAC_BAD_TYPE;
 429         }
 430 
 431         p += len;
 432         actions_len -= len;
 433     }
 434 
 435     /* Check if there's any trailing garbage. */
 436     if (actions_len != 0) {
 437         return OFPBAC_BAD_LEN;
 438     }
 439 
 440     return ACT_VALIDATION_OK;
 441 }

 446         const struct ofp_action_header *ah, uint16_t type)

 455 /* Execute a vendor-defined action against 'buffer'. */
 456 static void
 457 execute_vendor(struct ofpbuf *buffer UNUSED, const struct sw_flow_key *key UNUSED, 
 458         const struct ofp_action_header *ah)
 459 {
 460     struct ofp_action_vendor_header *avh 
 461             = (struct ofp_action_vendor_header *)ah;
 462 
 463     switch(ntohl(avh->vendor)) {
 464     default:
 465         /* This should not be possible due to prior validation. */
 466         printf("attempt to execute action with unknown vendor: %#x\n", 
 467                 ntohl(avh->vendor));
 468         break;
 469     }
 470 }

 472 /* Execute a list of actions against 'buffer'. */
 473 void execute_actions(struct datapath *dp, struct ofpbuf *buffer,
 474              struct sw_flow_key *key,
 475              const struct ofp_action_header *actions, size_t actions_len,
 476              int ignore_no_fwd)
 477 {
 478     /* Every output action needs a separate clone of 'buffer', but the common
 479      * case is just a single output action, so that doing a clone and then
 480      * freeing the original buffer is wasteful.  So the following code is
 481      * slightly obscure just to avoid that. */
 482     int prev_port;
 483     uint32_t prev_queue;
 484     size_t max_len = UINT16_MAX;
 485     uint16_t in_port = ntohs(key->flow.in_port);
 486     uint8_t *p = (uint8_t *)actions;
 487 
 488     prev_port = -1;
 489     prev_queue = 0;
 490 
 491     /* The action list was already validated, so we can be a bit looser
 492      * in our sanity-checking. */
 493     while (actions_len > 0) {
 494         struct ofp_action_header *ah = (struct ofp_action_header *)p;
 495         size_t len = htons(ah->len);
 496 
 497         if (prev_port != -1) {
 498             do_output(dp, ofpbuf_clone(buffer), in_port, max_len,
 499                       prev_port, prev_queue, ignore_no_fwd);
 500             prev_port = -1;
 501         }
 502 
 503         if (ah->type == htons(OFPAT_OUTPUT)) {
 504             struct ofp_action_output *oa = (struct ofp_action_output *)p;
 505             prev_port = ntohs(oa->port);
 506             prev_queue = 0; /* using the default best-effort queue */
 507             max_len = ntohs(oa->max_len);
 508         } else if (ah->type == htons(OFPAT_ENQUEUE)) {
 509             struct ofp_action_enqueue *ea = (struct ofp_action_enqueue *)p;
 510             prev_port = ntohs(ea->port);
 511             prev_queue = ntohl(ea->queue_id);
 512             max_len = 0; /* we will not send to the controller anyways - useless */
 513         } else {
 514             uint16_t type = ntohs(ah->type);
 515 
 516             if (type < ARRAY_SIZE(of_actions)) {
 517                 execute_ofpat(buffer, key, ah, type);
 518             } else if (type == OFPAT_VENDOR) {
 519                 execute_vendor(buffer, key, ah);
 520             }
 521         }
 522 
 523         p += len;
 524         actions_len -= len;
 525     }
 526     if (prev_port != -1) {
 527         do_output(dp, buffer, in_port, max_len, prev_port, prev_queue, ignore_no_fwd);
 528     } else {
 529         ofpbuf_delete(buffer);
 530     }
