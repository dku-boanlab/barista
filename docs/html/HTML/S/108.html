 279 int 
 280 destroy(struct oflops_context *ctx) {
 281   char msg[1024];
 282   struct timeval now;
 283   FILE *out = fopen(logfile, "w");
 284   struct entry *np;
 285   int  min_id[] = {INT_MAX, INT_MAX, INT_MAX};
 286   int ix[] = {0,0,0};
 287   int max_id[] = {INT_MIN, INT_MIN, INT_MIN}, ch, xid;
 288   uint32_t mean, std, median;
 289   float loss;
 290   double **data;
 291 
 292   gettimeofday(&now, NULL);
 293   printf("destroying code\n");
 294   snprintf(msg, 1024, "OFPT_ERROR_DELAY:%lu", (long unsigned)delay_false_modificaton);
 295   oflops_log(now, GENERIC_MSG, msg);
 296   snprintf(msg, 1024, "OFPT_INSERT_DELAY:%lu", (long unsigned)delay_modificaton);
 297   oflops_log(now, GENERIC_MSG, msg);
 298 
 299   //insert delay
 300   data = xmalloc(3*sizeof(double *));
 301   for(ch = 0; ch < 3; ch++) 
 302     data[ch] = xmalloc(count[ch]*sizeof(double));
 303 
 304   for (np = head.tqh_first; np != NULL; np = np->entries.tqe_next) {
 305     ch = np->ch - 1;
 306     min_id[ch] = (np->id < min_id[ch])?np->id:min_id[ch];
 307     max_id[ch] = (np->id > max_id[ch])?np->id:max_id[ch];
 308     data[ch][ix[ch]++] = time_diff(&np->snd, &np->rcv);
 309     if(print)
 310       if(fprintf(out, "%lu;%lu.%06lu;%lu.%06lu;%d\n", 
 311                  (long unsigned int)np->id,  
 312                  (long unsigned int)np->snd.tv_sec, 
 313                  (long unsigned int)np->snd.tv_usec,
 314                  (long unsigned int)np->rcv.tv_sec, 
 315                  (long unsigned int)np->rcv.tv_usec,  np->ch) < 0)  
 316         perror_and_exit("fprintf fail", 1); 
 317 
 318     //release memory
 319     free(np);
 320   }
 321 
 322   for(ch = 0; ch < 3; ch++) {
 323     if(ix[ch] == 0) continue;
 324     gsl_sort (data[ch], 1, ix[ch]);
 325     mean = (uint32_t)gsl_stats_mean(data[ch], 1, ix[ch]);
 326     std = (uint32_t)sqrt(gsl_stats_variance(data[ch], 1, ix[ch]));
 327     median = (uint32_t)gsl_stats_median_from_sorted_data (data[ch], 1, ix[ch]);
 328     loss = (float)ix[ch]/(float)(max_id[ch] - min_id[ch]);
 329 
 330     //print summarization data
 331     snprintf(msg, 1024, "statistics:port:%d:%u:%u:%u:%.4f:%d", 
 332              ctx->channels[ch + 1].of_port, mean, median, std, loss, count[ch]);
 333     printf("statistics:port:%d:%u:%u:%u:%.4f:%d\n", 
 334            ctx->channels[ch + 1].of_port, mean, median, std, loss, count[ch]);
 335     oflops_log(now, GENERIC_MSG, msg);
 336   }
 337 
 338   //print echo delay per packet information 
 339   if(print) {
 340     for (xid = 1 ; xid < trans_id; xid++) {
 341       snprintf(msg, 1024, "%u", (uint32_t)delay[xid]);
 342       oflops_log(now, OFPT_ECHO_REPLY_MSG, msg);
 343     }
 344   }
 345 
 346   //claculate echo statistcs
 347   gsl_sort (delay, 1, (trans_id - 1));
 348   mean = (uint32_t)gsl_stats_mean(delay, 1, (trans_id - 1));
 349   std = (uint32_t)sqrt(gsl_stats_variance(delay, 1, (trans_id - 1)));
 350   median = (uint32_t)gsl_stats_median_from_sorted_data (delay, 1, (trans_id - 1));
 351 
 352   printf("statistics:echo:%u:%u:%u:%d\n", mean, median, std, (trans_id-1));
 353   snprintf(msg, 1024, "statistics:echo:%u:%u:%u:%d", mean, median, std, (trans_id - 1));
 354   oflops_log(now, GENERIC_MSG, msg);
 355   return 0;
 356 }

 359  * @param ctx pointer to opaque context
 360  * @param te pointer to timer event

 362 int handle_timer_event(struct oflops_context * ctx, struct timer_event *te) {  
 363   char *str = te->arg; 
 364   int len, i;
 365   void *b;
 366   struct timeval now;
 367   struct ofp_flow_mod *ofp;
 368   struct ofp_header *ofph;
 369 
 370   //terminate process 
 371   if (strcmp(str, BYESTR) == 0) {
 372     printf("terminating test....\n");
 373     oflops_end_test(ctx,1);
 374     finished = 0;
 375     return 0;    
 376   } else if(strcmp(str, OFP_PING) == 0) {
 377     len = make_ofp_hello(&b);
 378     ofph = (struct ofp_header *)b;
 379     ofph->type = OFPT_ECHO_REQUEST;
 380     ofph->xid = htonl(trans_id++);
 381     oflops_send_of_mesgs(ctx, b, len);
 382     free(b);
 383     gettimeofday(&now, NULL);
 384     add_time(&now, 1, 0);
 385     oflops_schedule_timer_event(ctx,&now, OFP_PING);
 386 
 387 
 388   } else if  (strcmp(str, SND_FALSE_ACT) == 0) {
 389     struct ofp_action_output *cmd;
 390     int cmd_len;
 391     uint32_t tmp = fl_probe->nw_src;
 392     fl_probe->nw_src = inet_addr("10.5.5.5");
 393     cmd_len = sizeof(struct ofp_action_output);
 394     cmd = (struct ofp_action_output *)xmalloc(cmd_len);
 395     cmd->type = htons(0);
 396     cmd->len = htons(8);
 397     cmd->max_len = htons(0);
 398     cmd->port = htons(50);
 399     len = make_ofp_flow_modify(&b, fl_probe, (char *)cmd, cmd_len, 
 400                                1, 1200);
 401     ofp = (struct ofp_flow_mod *) b;
 402     ofp->buffer_id = htonl(1);
 403     send_false_mod= 1;
 404     printf("sending false modification to measure delay\n");
 405     oflops_send_of_mesg(ctx, b);
 406     free(b);
 407     free(cmd);
 408     fl_probe->nw_src = tmp;
 409   } else if (strcmp(str, SND_ACT) == 0) {
 410     len = make_ofp_flow_modify(&b, fl_probe, command, command_len, 
 411                                1, 1200);
 412     send_mod= 1;
 413     oflops_send_of_mesg(ctx, b);
 414     free(b);
 415     printf("sending correct modification to measure delay\n");
 416   } else if(strcmp(str, SNMPGET) == 0) {
 417     for(i=0;i<ctx->cpuOID_count;i++) {
 418       oflops_snmp_get(ctx, ctx->cpuOID[i], ctx->cpuOID_len[i]);
 419     }
 420     for(i=0;i<ctx->n_channels;i++) {
 421       oflops_snmp_get(ctx, ctx->channels[i].inOID, ctx->channels[i].inOID_len);
 422       oflops_snmp_get(ctx, ctx->channels[i].outOID, ctx->channels[i].outOID_len);
 423     }    
 424     gettimeofday(&now, NULL);
 425     add_time(&now, 10, 0);
 426     oflops_schedule_timer_event(ctx,&now, SNMPGET);
 427   }
 428   return 0;
 429 }

 448 /** Handle pcap event.
 449  * @param ctx pointer to opaque context
 450  * @param pe pcap event
 451  * @param ch enumeration of channel that pcap event is triggered
 452  */
 453 int 
 454 handle_pcap_event(struct oflops_context *ctx, struct pcap_event * pe, oflops_channel_name ch) {
 455   struct pktgen_hdr *pktgen;
 456   int dir, len;
 457   struct ofp_header *ofp;
 458   struct pcap_event *ofp_msg;
 459   struct ofp_error_msg *err_p = NULL;
 460   char msg[1024];
 461 
 462   if (ch == OFLOPS_CONTROL) {
 463     dir = append_data_to_flow(pe->data,pe->pcaphdr);
 464     while(contains_next_msg(dir) > 0) {
 465       len = get_next_msg(dir, &ofp_msg);
 466       ofp = (struct ofp_header *)ofp_msg->data;
 467       switch(ofp->type) {
 468       case OFPT_FLOW_MOD:
 469         if (send_false_mod) {
 470           if(send_mod) {
 471             memcpy(&true_modification, &pe->pcaphdr.ts, sizeof(struct timeval));
 472           } else {
 473             memcpy(&false_modification, &pe->pcaphdr.ts, sizeof(struct timeval));
 474           }
 475         }
 476         if(send_mod) {
 477           oflops_log(pe->pcaphdr.ts,OFPT_FLOW_MOD_ADD, "flow modification send");
 478         }
 479         break;
 480       case OFPT_ECHO_REQUEST:
 481         if(ntohl(ofp->xid) < 100) 
 482           memcpy(&ofp_ping_timestamp[ntohl(ofp->xid)], &pe->pcaphdr.ts, sizeof(struct timeval));
 483         break;
 484       case OFPT_ECHO_REPLY:
 485         if(ntohl(ofp->xid) < 100 && ntohl(ofp->xid) > 0) {
 486           delay[ntohl(ofp->xid)] = time_diff(&ofp_ping_timestamp[ntohl(ofp->xid)], &pe->pcaphdr.ts);
 487         }
 488         break;    
 489       case OFPT_ERROR:
 490         err_p = (struct ofp_error_msg *)ofp;
 491         if(send_false_mod)  {
 492           delay_false_modificaton = time_diff(&false_modification, &pe->pcaphdr.ts);
 493         } else {
 494           snprintf(msg, 1024, "%d:%d", ntohs(err_p->type), ntohs(err_p->code));
 495           oflops_log(pe->pcaphdr.ts, OFPT_ERROR_MSG, msg);
 496           fprintf(stderr, "OFPT_ERROR_MSG:%s\n", msg);
 497           break;   
 498         }
 499       }
 500     }
 501   } else if ((ch == OFLOPS_DATA1) || (ch == OFLOPS_DATA2) || (ch == OFLOPS_DATA3)) {
 502     struct flow fl;
 503     pktgen = extract_pktgen_pkt(pe->data, pe->pcaphdr.caplen, &fl);
 504     if((ch == OFLOPS_DATA3) && (!first_pkt)) {
 505       delay_modificaton = time_diff(&true_modification, &pe->pcaphdr.ts);
 506       oflops_log(pe->pcaphdr.ts, GENERIC_MSG, "PACKET_NEW_PORT");
 507       //printf("first packet on new port\n");
 508       first_pkt = 1;
 509     }
 510     if(htonl(pktgen->seq_num) % 100000 == 0)
 511       printf("data packet received %d\n", htonl(pktgen->seq_num));
 512     
 513     struct entry *n1 = malloc(sizeof(struct entry));
 514     n1->snd.tv_sec = htonl(pktgen->tv_sec);
 515     n1->snd.tv_usec = htonl(pktgen->tv_usec);
 516     memcpy(&n1->rcv, &pe->pcaphdr.ts, sizeof(struct timeval));
 517     n1->id = htonl(pktgen->seq_num);
 518     n1->ch = ch;
 519     count[ch - 1]++;
 520     TAILQ_INSERT_TAIL(&head, n1, entries);
 521   }
 522   return 0;
 523 }

 550 handle_snmp_event(struct oflops_context * ctx, struct snmp_event * se) {
 551   netsnmp_variable_list *vars;
 552   int len = 1024;
 553   char msg[1024], out_buf[1024];
 554   struct timeval now;
 555   int i;
 556   gettimeofday(&now, NULL);
 557 
 558   for(vars = se->pdu->variables; vars; vars = vars->next_variable)  {
 559     snprint_value(msg, len, vars->name, vars->name_length, vars);
 560     for (i = 0; i < ctx->cpuOID_count; i++) {
 561       if((vars->name_length == ctx->cpuOID_len[i]) &&
 562          (memcmp(vars->name, ctx->cpuOID[i],  ctx->cpuOID_len[i] * sizeof(oid)) == 0) ) {
 563         snprintf(out_buf, len, "cpu : %s %%", msg);

 641 /*
 642  * read the first 2 hex characters and return the byte.
 643  */
 644 uint8_t
 645 read_hex(const char *data) {
 646   uint8_t ret = 0;
 647   int i;
 648 
 649   for(i = 0 ; i < 2 ; i++) {
 650     ret = ret << 4;
 651     if((*data >= 'A') && (*data <= 'F')) {
 652       ret += 10 + (*data - 'A');
 653     } else if((*data >= 'a') && (*data <= 'f')) {
 654       ret += 10 + ((*data) - 'a');
 655     } else if((*data >= '0') && (*data <= '9')) {
 656       ret += ((*data) - '0');
 657     }
 658     data++;
 659   }
 660   return ret;
 661 }

 663 /**
 664  * Initialization code with parameters
 665  * @param ctx 
 666  */
 667 int init(struct oflops_context *ctx, char * config_str) {
 668   char *pos = NULL;
 669   char *param = config_str;
 670   int len = strlen(config_str);
 671   char *value = NULL;
 672   char *action;
 673 
 674   //init counters
 675   finished = 0;
 676 
 677   struct timeval now;
 678   gettimeofday(&now, NULL);
 679 
 680   cli_param = strdup(config_str);
 681 
 682 
 683   while(*config_str == ' ') {
 684     config_str++;
 685   }
 686   param = config_str;
 687   while(1) {
 688     pos = index(param, ' ');
 689 
 690     if((pos == NULL)) {
 691       if (*param != '\0') {
 692         pos = param + strlen(param) + 1;
 693       } else
 694         break;
 695     }
 696     *pos='\0';
 697     pos++;
 698     value = index(param,'=');
 699     *value = '\0';
 700     value++;
 701     //fprintf(stderr, "param = %s, value = %s\n", param, value);
 702     if(value != NULL) {
 703       if(strcmp(param, "pkt_size") == 0) {
 704         //parse int to get pkt size
 705         pkt_size = strtol(value, NULL, 0);
 706         if((pkt_size < MIN_PKT_SIZE) && (pkt_size > MAX_PKT_SIZE))
 707           perror_and_exit("Invalid packet size value", 1);
 708       } else if(strcmp(param, "data_rate") == 0) {
 709         //parse int to get rate of background data
 710         datarate = strtol(value, NULL, 0);
 711         if((datarate < 0) || (datarate > 1010))
 712           perror_and_exit("Invalid data rate param(Values between 1 and 1010)", 1);
 713       }  else if(strcmp(param, "probe_rate") == 0) {
 714         //parse int to get measurement probe rate
 715         proberate = strtol(value, NULL, 0);
 716         if((proberate <= 0) || (proberate >= 1010)) 
 717           perror_and_exit("Invalid probe rate param(Value between 1 and 1010)", 1);
 718       }  else if(strcmp(param, "action") == 0) {
 719         char *p = value;
 720         while((*p != ' ') && (*p != '\0') && (config_str + len > p)) {
 721           action = p;
 722           //find where value ends and set it to null to extract the string.
 723           p = index(p, ',');
 724           if(p == NULL) {
 725             p = config_str + len + 1;
 726             *p='\0';
 727           } else {
 728             *p = '\0'; 
 729             p++;
 730           }
 731           
 732           //set null char to split action param and action value
 733           param = index(action, '/');
 734           if(param != NULL) {
 735             *param = '\0';
 736             param++;
 737           }
 738 
 739           //check if action value is correct and append it at the end of the action list
 740           if(*action >= '0' && *action <= '9') {
 741               append_action((*action) - '0', param);
 742             } else if (*action == 'a') {
 743               append_action(10, param);
 744             } else { 
 745               printf("invalid action: %1s", action); 
 746               continue;
 747             }     
 748         }
 749       } else if(strcmp(param, "table") == 0) {
 750         //parse int to get pkt size
 751         table = strtol(value, NULL, 0);
 752         if((table < 0) && (table > 2))  
 753           perror_and_exit("Invalid table number", 1);
 754       } else if(strcmp(param, "flows") == 0) { 
 755         //parse int to get pkt size
 756         flows = strtol(value, NULL, 0);
 757         if(flows <= 0)  
 758           perror_and_exit("Invalid flow number", 1);
 759       }else if(strcmp(param, "print") == 0) { 
 760         //parse int to check whether per packet statistics should be stored 
 761         print = strtol(value, NULL, 0);
 762       } else 
 763         fprintf(stderr, "Invalid parameter:%s\n", param);
 764       param = pos;
 765     }
 766   } 
 767 
 768   //calculate sendind interval
 769   if(datarate > 0) {
 770     data_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (datarate * mbits_to_bits);
 771     fprintf(stderr, "Sending data interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 772             (uint32_t)data_snd_interval, (uint32_t)pkt_size, (uint32_t)datarate);
 773   } else {
 774     fprintf(stderr, "background data probe is disabled\n");
 775     data_snd_interval = 0;
 776   }
 777   probe_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (proberate * mbits_to_bits);
 778   fprintf(stderr, "Sending probe interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 779       (uint32_t)probe_snd_interval, (uint32_t)pkt_size, (uint32_t)proberate);
 780 
 781   //by default the new rule should be redirected to port 2 to make the measurement easier
 782   /* fprintf(stderr, "by default output packet to port 1\n"); */
 783 /*   command_len += sizeof(struct ofp_action_output); */
 784 /*   command = realloc(command, command_len); */
 785 /*   act_out = (struct ofp_action_output *) */
 786 /*     (command + (command_len - sizeof(struct ofp_action_output))); */
 787 /*   act_out->type = htons(0); */
 788 /*   act_out->len = htons(8); */
 789 /*   act_out->max_len = htons(0); */
 790 /*   act_out->port = htons(ctx->channels[OFLOPS_DATA3].of_port); */
 791   return 0;
 792 }
