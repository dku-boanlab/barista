  43 void
  44 random_init(void)
  45 {
  46     static bool inited = false;
  47     if (!inited) {
  48         struct timeval tv;
  49         inited = true;
  50         if (gettimeofday(&tv, NULL) < 0) {
  51             ofp_fatal(errno, "gettimeofday");
  52         }
  53         srand(tv.tv_sec ^ tv.tv_usec);
  54     }
  55 }
  56 
  57 void
  58 random_bytes(void *p_, size_t n)
  59 {
  60     uint8_t *p = p_;
  61     random_init();
  62     while (n--) {
  63         *p++ = rand();
  64     }
  65 }

  67 uint8_t
  68 random_uint8(void)

  74 uint16_t
  75 random_uint16(void)
  76 {
  77     if (RAND_MAX >= UINT16_MAX) {
  78         random_init();
  79         return rand();
  80     } else {
  81         uint16_t x;
  82         random_bytes(&x, sizeof x);
  83         return x;
  84     }
  85 }

  87 uint32_t
  88 random_uint32(void)
  89 {
  90     if (RAND_MAX >= UINT32_MAX) {
  91         random_init();
  92         return rand();
  93     } else if (RAND_MAX == INT32_MAX) {
  94         random_init();
  95         return rand() | ((rand() & 1u) << 31);
  96     } else {
  97         uint32_t x;
  98         random_bytes(&x, sizeof x);
  99         return x;
 100     }
 101 }
