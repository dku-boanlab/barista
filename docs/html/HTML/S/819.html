  48 static void
  49 fault_handler(int sig_nr)
  50 {
  51     VLOG_EMER("Caught signal %d.", sig_nr);
  52     log_backtrace();
  53     fflush(stdout);
  54     fflush(stderr);
  55 
  56     signal(sig_nr, SIG_DFL);
  57     raise(sig_nr);
  58 }

  60 void
  61 log_backtrace(void)
  62 {
  63 #define STACK_DEPTH_LIMIT       128
  64 #ifdef OLD_LOG_BACKTRACE
  65     /* During the loop:
  66 
  67        frame[0] points to the next frame.
  68        frame[1] points to the return address. */
  69     void **frame;
  70     int stack_depth = 0;
  71     for (frame = __builtin_frame_address(0);
  72          frame != NULL && frame[0] != NULL
  73          && stack_depth < STACK_DEPTH_LIMIT;
  74          frame = frame[0], ++stack_depth) {
  75         Dl_info addrinfo;
  76         if (!dladdr(frame[1], &addrinfo) || !addrinfo.dli_sname) {
  77             fprintf(stderr, "  0x%08"PRIxPTR"\n", (uintptr_t) frame[1]);
  78         } else {
  79             fprintf(stderr, "  0x%08"PRIxPTR" (%s+0x%x)\n",
  80                     (uintptr_t) frame[1], addrinfo.dli_sname,
  81                     (char *) frame[1] - (char *) addrinfo.dli_saddr); 
  82         }
  83     }
  84 #else
  85     /* Use glibc functions to print backtrace */
  86     void *buffer[STACK_DEPTH_LIMIT];
  87     backtrace(buffer, STACK_DEPTH_LIMIT);
  88     backtrace_symbols_fd(buffer, STACK_DEPTH_LIMIT, fileno(stderr));
  89 #endif
  90     fflush(stderr);
  91 }

  95 void
  96 register_fault_handlers(void)
  97 {
  98     signal(SIGABRT, fault_handler);
  99     signal(SIGBUS, fault_handler);
 100     signal(SIGFPE, fault_handler);
 101     signal(SIGILL, fault_handler);
 102     signal(SIGSEGV, fault_handler);
 103 }
