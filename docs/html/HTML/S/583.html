  42     range_process_marked,

  47 typedef struct packet_range_tag {
  48     /* values coming from the UI */
  49     packet_range_e  process;            /* which range to process */
  50     gboolean        process_filtered;   /* captured or filtered packets */
  51 
  52     /* user specified range(s) and, if null, error status */
  53     range_t         *user_range;
  54     convert_ret_t   user_range_status;
  55 
  56     /* calculated values */
  57     guint32  selected_packet;       /* the currently selected packet */
  58 
  59     /* current packet counts (captured) */
  60     /* cfile.count */               /* packets in capture file */
  61     /* cfile.marked_count */        /* packets marked */
  62     guint32  mark_range_cnt;        /* packets in marked range */
  63     guint32  user_range_cnt;        /* packets in user specified range */
  64 
  65     /* current packet counts (displayed) */
  66     guint32  displayed_cnt;
  67     guint32  displayed_marked_cnt;
  68     guint32  displayed_mark_range_cnt;
  69     guint32  displayed_user_range_cnt;
  70 
  71     /* "enumeration" values */
  72     gboolean marked_range_active;   /* marked range is currently processed */
  73     guint32  marked_range_left;     /* marked range packets left to do */
  74     gboolean selected_done;         /* selected packet already processed */
  75 } packet_range_t;
  76 
  77 typedef enum {
  78     range_process_this,             /* process this packet */
  79     range_process_next,             /* skip this packet, process next */
  80     range_processing_finished       /* stop processing, required packets done */
  81 } range_process_e;

  91 
  92 /* do we have to process all packets? */
  93 extern gboolean packet_range_process_all(packet_range_t *range);
  94
