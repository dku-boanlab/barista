  50 /* Attempts to append 'table' to the set of tables in 'chain'.  Returns 0 or
  51  * negative error.  If 'table' is null it is assumed that table creation failed
  52  * due to out-of-memory. */
  53 static int add_table(struct sw_chain *chain, struct sw_table *table, int emerg)
  54 {
  55     if (table == NULL)
  56         return -ENOMEM;
  57     if (chain->n_tables >= CHAIN_MAX_TABLES) {
  58         VLOG_ERR("too many tables in chain\n");
  59         table->destroy(table);
  60         return -ENOBUFS;
  61     }
  62     if (emerg)
  63         chain->emerg_table = table;
  64     else
  65         chain->tables[chain->n_tables++] = table;
  66     return 0;
  67 }

  69 /* Creates and returns a new chain.  Returns NULL if the chain cannot be
  70  * created. */
  71 struct sw_chain *chain_create(struct datapath *dp)
  72 {
  73     struct sw_chain *chain = calloc(1, sizeof *chain);
  74     if (chain == NULL)
  75         return NULL;
  76 
  77     chain->dp = dp;
  78 #if defined(OF_HW_PLAT)
  79     if (dp && dp->hw_drv) {
  80         if (add_table(chain, (struct sw_table *)dp->hw_drv, 0) != 0) {
  81             VLOG_ERR("Could not attach HW table to chain\n");
  82         }
  83     }
  84 #endif
  85     if (add_table(chain, table_hash2_create(0x1EDC6F41, TABLE_HASH_MAX_FLOWS,
  86                                             0x741B8CD7, TABLE_HASH_MAX_FLOWS),
  87                                             0)
  88         || add_table(chain, table_linear_create(TABLE_LINEAR_MAX_FLOWS), 0)
  89         || add_table(chain, table_linear_create(TABLE_LINEAR_MAX_FLOWS), 1)) {
  90         chain_destroy(chain);
  91         return NULL;
  92     }
  93 
  94     return chain;
  95 }

  97 /* Searches 'chain' for a flow matching 'key', which must not have any wildcard
  98  * fields.  Returns the flow if successful, otherwise a null pointer. */
  99 struct sw_flow *
 100 chain_lookup(struct sw_chain *chain, const struct sw_flow_key *key, int emerg)
 101 {
 102     int i;
 103 
 104     assert(!key->wildcards);
 105 
 106     if (emerg) {
 107         struct sw_table *t = chain->emerg_table;
 108         struct sw_flow *flow = t->lookup(t, key);
 109         t->n_lookup++;
 110         if (flow) {
 111             t->n_matched++;
 112             return flow;
 113         }
 114     } else {
 115         for (i = 0; i < chain->n_tables; i++) {
 116             struct sw_table *t = chain->tables[i];
 117             struct sw_flow *flow = t->lookup(t, key);
 118             t->n_lookup++;
 119             if (flow) {
 120                 t->n_matched++;
 121                 return flow;
 122             }
 123         }
 124     }
 125 
 126     return NULL;
 127 }

 129 /* Inserts 'flow' into 'chain', replacing any duplicate flow.  Returns 0 if
 130  * successful or a negative error.
 131  *
 132  * If successful, 'flow' becomes owned by the chain, otherwise it is retained
 133  * by the caller. */
 134 int
 135 chain_insert(struct sw_chain *chain, struct sw_flow *flow, int emerg)
 136 {
 137     int i;
 138 
 139     if (emerg) {
 140         struct sw_table *t = chain->emerg_table;
 141         if (t->insert(t, flow))
 142             return 0;
 143     } else {
 144         for (i = 0; i < chain->n_tables; i++) {
 145             struct sw_table *t = chain->tables[i];
 146             if (t->insert(t, flow))
 147                 return 0;
 148         }
 149     }
 150 
 151     return -ENOBUFS;
 152 }

 154 /* Modifies actions in 'chain' that match 'key'.  If 'strict' set, wildcards 
 155  * and priority must match.  Returns the number of flows that were modified.
 156  *
 157  * Expensive in the general case as currently implemented, since it requires
 158  * iterating through the entire contents of each table for keys that contain
 159  * wildcards.  Relatively cheap for fully specified keys. */
 160 int
 161 chain_modify(struct sw_chain *chain, const struct sw_flow_key *key,
 162         uint16_t priority, int strict,
 163         const struct ofp_action_header *actions, size_t actions_len,
 164         int emerg)
 165 {
 166     int count = 0;
 167     int i;
 168 
 169     if (emerg) {
 170         struct sw_table *t = chain->emerg_table;
 171         count += t->modify(t, key, priority, strict, actions, actions_len);
 172     } else {
 173         for (i = 0; i < chain->n_tables; i++) {
 174             struct sw_table *t = chain->tables[i];
 175             count += t->modify(t, key, priority, strict, actions, actions_len);
 176         }
 177     }
 178 
 179     return count;
 180 }

 182 /* Checks whether the chain has an entry with the same priority which conflicts
 183  * with 'key'. If 'strict' set, wildcards should also match. If 'strict' is not 
 184  * set, comparison is done 'module wildcards'.
 185  *
 186  * Returns 'true' if such an entry exists, 'false' otherwise. */
 187 int
 188 chain_has_conflict(struct sw_chain *chain, const struct sw_flow_key *key,
 189                    uint16_t priority, int strict)
 190 {
 191     int i;
 192 
 193     for (i = 0; i < chain->n_tables; i++) {
 194         struct sw_table *t = chain->tables[i];
 195         if (t->has_conflict(t, key, priority, strict)) {
 196             return true;
 197         }
 198     }
 199 
 200     return false;
 201 }

 203 /* Deletes from 'chain' any and all flows that match 'key'.  If 'out_port' 
 204  * is not OFPP_NONE, then matching entries must have that port as an 
 205  * argument for an output action.  If 'strict" is set, then wildcards and 
 206  * priority must match.  Returns the number of flows that were deleted.
 207  *
 208  * Expensive in the general case as currently implemented, since it requires
 209  * iterating through the entire contents of each table for keys that contain
 210  * wildcards.  Relatively cheap for fully specified keys. */
 211 int
 212 chain_delete(struct sw_chain *chain, const struct sw_flow_key *key,
 213              uint16_t out_port, uint16_t priority, int strict, int emerg)
 214 {
 215     int count = 0;
 216     int i;
 217 
 218     if (emerg) {
 219         struct sw_table *t = chain->emerg_table;
 220         count += t->delete(chain->dp, t, key, out_port, priority, strict);
 221     } else {
 222         for (i = 0; i < chain->n_tables; i++) {
 223             struct sw_table *t = chain->tables[i];
 224             count += t->delete(chain->dp, t, key, out_port, priority, strict);
 225         }
 226     }
 227 
 228     return count;
 229 }

 231 /* Deletes timed-out flow entries from all the tables in 'chain' and appends
 232  * the deleted flows to 'deleted'.
 233  *
 234  * Expensive as currently implemented, since it iterates through the entire
 235  * contents of each table. */
 236 void
 237 chain_timeout(struct sw_chain *chain, struct list *deleted)
 238 {
 239     int i;
 240 
 241     for (i = 0; i < chain->n_tables; i++) {
 242         struct sw_table *t = chain->tables[i];
 243         t->timeout(t, deleted);
 244     }
 245 }

 247 /* Destroys 'chain', which must not have any users. */
 248 void
 249 chain_destroy(struct sw_chain *chain)
 250 {
 251     int i;
 252     struct sw_table *t;
 253 
 254     for (i = 0; i < chain->n_tables; i++) {
 255         t = chain->tables[i];
 256         t->destroy(t);
 257     }
 258     t = chain->emerg_table;
 259     t->destroy(t);
 260     free(chain);
 261 }
