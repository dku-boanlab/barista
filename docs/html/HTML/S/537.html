  36 #ifndef _I4B_TRACE_H_
  37 #define _I4B_TRACE_H_
  38 
  39 /*---------------------------------------------------------------------------*
  40  *      structure of the header at the beginning of every trace mbuf
  41  *---------------------------------------------------------------------------*/
  42 typedef struct {
  43         guint32 length;         /* length of the following mbuf         */
  44         gint32 unit;            /* controller unit number               */
  45         gint32 type;            /* type of channel                      */
  46 #define TRC_CH_I        0               /* Layer 1 INFO's               */
  47 #define TRC_CH_D        1               /* D channel                    */
  48 #define TRC_CH_B1       2               /* B1 channel                   */
  49 #define TRC_CH_B2       3               /* B2 channel                   */
  50         gint32 dir;             /* direction                            */
  51 #define FROM_TE 0                       /* user -> network              */
  52 #define FROM_NT 1                       /* network -> user              */
  53         gint32 trunc;           /* # of truncated bytes (frame > MCLBYTES) */
  54         guint32 count;          /* frame count for this unit/type       */
  55         guint32 ts_sec;         /* timestamp seconds */
  56         guint32 ts_usec;        /* timestamp microseconds */
  57 } i4b_trace_hdr_t;
  58 
  59 #define INFO0           0       /* layer 1 */
  60 #define INFO1_8         1
  61 #define INFO1_10        2
  62 #define INFO2           3
  63 #define INFO3           4
  64 #define INFO4_8         5
  65 #define INFO4_10        6
  66 
  67 /*---------------------------------------------------------------------------*
  68  *      ioctl via /dev/i4btrc device(s):
  69  *      get/set current trace flag settings
  70  *---------------------------------------------------------------------------*/
  71 
  72 #define I4B_TRC_GET     _IOR('T', 0, int)       /* get trace settings   */
  73 #define I4B_TRC_SET     _IOW('T', 1, int)       /* set trace settings   */
  74 
  75 #define TRACE_OFF       0x00            /* tracing off          */
  76 #define TRACE_I         0x01            /* trace L1 INFO's on   */
  77 #define TRACE_D_TX      0x02            /* trace D channel on   */
  78 #define TRACE_D_RX      0x04            /* trace D channel on   */
  79 #define TRACE_B_TX      0x08            /* trace B channel on   */
  80 #define TRACE_B_RX      0x10            /* trace B channel on   */
  81 
  82 typedef struct {
  83         gint32 rxunit;          /* unit # for rx frames */
  84         gint32 rxflags;         /* d and/or b channel   */
  85         gint32 txunit;          /* unit # for tx frames */
  86         gint32 txflags;         /* d and/or b channel   */
  87 } i4b_trace_setupa_t;
  88 
  89 #define I4B_TRC_SETA    _IOW('T', 2, i4b_trace_setupa_t) /* set analyze mode */
  90 #define I4B_TRC_RESETA  _IOW('T', 3, int)       /* reset analyze mode   */
  91 
  92 #endif /* _I4B_TRACE_H_ */
