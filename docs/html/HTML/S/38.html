  88 /********************************************************
  89  * main loop()
  90  *      1) setup poll
  91  *      2) call poll with a min timeout of the next event
  92  *      3) dispatch events as appropriate
  93  */
  94 static void test_module_loop(oflops_context *ctx, test_module *mod)
  95 {
  96   struct pollfd * poll_set;
  97   int fds = 0;
  98   int snmpblock = 0;
  99   fd_set fdset;
 100   struct timeval timeout;
 101   int ret;
 102   int len; 
 103   int ch;
 104   int n_fds=0;
 105         
 106   len = sizeof(struct pollfd) * (ctx->n_channels + 1);
 107   poll_set = malloc_and_check(len);
 108 
 109   while(!ctx->should_end )
 110     {
 111       n_fds=0;
 112       bzero(poll_set,len);
 113 
 114       //Channels poll
 115       for(ch=0; ch< ctx->n_channels; ch++)
 116         {
 117           poll_set[n_fds].fd = ctx->channels[ch].pcap_fd;
 118           poll_set[n_fds].events = 0;
 119           if( ctx->channels[ch].pcap_handle)
 120             poll_set[n_fds].events = POLLIN;
 121           //if ( msgbuf_count_buffered(ctx->channels[ch].outgoing) > 0)
 122           //    poll_set[n_fds].events |= POLLOUT;
 123           if( poll_set[n_fds].events != 0)
 124             n_fds++;
 125         }
 126       poll_set[n_fds].fd = ctx->control_fd;     // add the control channel at the end
 127       poll_set[n_fds].events = POLLIN;
 128       if ( msgbuf_count_buffered(ctx->control_outgoing) > 0)
 129         poll_set[n_fds].events |= POLLOUT;
 130       n_fds++;
 131 
 132       //SNMP poll
 133       FD_ZERO(&fdset);
 134       timeout.tv_sec = 0;
 135       timeout.tv_usec = 1;
 136       snmp_select_info(&fds, &fdset, &timeout, &snmpblock);
 137       fds = select(fds, &fdset, NULL,NULL, &timeout);
 138       if (fds)
 139         snmp_read(&fdset);
 140       //else
 141       //snmp_timeout();
 142       
 143       //Timer poll
 144       /*next_event = timer_get_next_event(ctx);
 145         while(next_event <= 0 )
 146         {
 147         timer_run_next_event(ctx);
 148         next_event = timer_get_next_event(ctx);
 149         }*/
 150       ret = poll(poll_set, n_fds, -1); //next_event);
 151 
 152       if(( ret == -1 ) && ( errno != EINTR))
 153         perror_and_exit("poll",1);
 154       else if(ret == 0 ) {
 155         //if(ctx->should_end == 1) {fprintf(stderr, "finishing poll loop\n"); break;}
 156         //                else fprintf(stderr, "not finished yet\n");
 157         continue; //timer_run_next_event(ctx);
 158       }
 159       else // found something to read
 160         {
 161           int i;        
 162           for(i=0; i<n_fds; i++)
 163             if(poll_set[i].revents & (POLLIN | POLLOUT))
 164               process_event(ctx, mod, &poll_set[i]);
 165         }
 166     }
 167 }

 179 static void process_event(oflops_context *ctx, test_module * mod, struct pollfd *pfd)
 180 {
 181   int ch;
 182   if(pfd->fd == ctx->control_fd)
 183     return process_control_event(ctx, mod, pfd);
 184   // this is inefficient, but ok since there are really typically only ~8  cases
 185   for(ch=0; ch< ctx->n_channels; ch++)
 186     if (pfd->fd == ctx->channels[ch].pcap_fd)
 187       return process_pcap_event(ctx, mod, pfd,ch);
 188   // only get here if we've screwed up somehow
 189   fprintf(stderr, "Event on unknown fd %d .. dying", pfd->fd);
 190   abort();
 191 }

 193 /***********************************************************************************************
 194  * static void process_control_event(oflops_context *ctx, test_module * mod, struct pollfd *fd);
 195  *      if POLLIN is set, read an openflow message from the control channel
 196  *      FIXME: handle a control channel reset here
 197  */
 198 static void process_control_event(oflops_context *ctx, test_module * mod, struct pollfd *pfd)
 199 {
 200   char * neobuf;
 201   static char * buf; 
 202   static int buflen   = -1;
 203   static int bufstart =  0;       // begin of unprocessed data
 204   static int bufend   =  0;       // end of unprocessed data
 205   unsigned int msglen;
 206   struct ofp_header * ofph;
 207   int count;
 208 
 209   if ( buflen == - 1 )
 210     {
 211       buflen = BUFLEN;
 212       buf = malloc_and_check(BUFLEN);
 213     }
 214   if(bufend >= buflen )   // if we've filled up our buffer, resize it
 215     {
 216       buflen *=2 ;
 217       buf = realloc_and_check(buf, buflen);
 218     }
 219 
 220   if(pfd->revents & POLLOUT)
 221     {
 222       if(msgbuf_write(ctx->control_outgoing,ctx->control_fd, 0) < 0)
 223         perror_and_exit("control write()",1);
 224     }
 225 
 226   if(!(pfd->revents & POLLIN))          // nothing to read, return
 227     return;
 228   count = read(pfd->fd,&buf[bufend], buflen - bufend);
 229   if(count < 0)
 230     {
 231       perror("process_control_event:read() ::");
 232       return ;
 233     }
 234   if(count == 0)
 235     {
 236       fprintf(stderr, "Switch Control Connection reset! wtf!?!...exiting\n");
 237       exit(0);
 238     }
 239   bufend += count;            // extend buf by amount read
 240   count = bufend - bufstart;  // re-purpose count
 241 
 242   while(count > 0 )
 243     {
 244       if(count <  sizeof(ofph))   // if we didn't get full openflow header
 245         return;                 // come back later
 246 
 247       ofph = (struct ofp_header * ) &buf[bufstart];
 248       msglen = ntohs(ofph->length);
 249       if( ( msglen > count) ||    // if we don't yet have the whole msg
 250           (buflen < (msglen + bufstart)))  // or our buffer is full
 251         return;     // get the rest on the next pass
 252 
 253       neobuf = malloc_and_check(msglen);
 254       memcpy(neobuf, ofph, msglen);
 255 
 256       switch(ofph->type)
 257         {
 258         case OFPT_PACKET_IN:
 259           mod->of_event_packet_in(ctx, (struct ofp_packet_in *)neobuf);
 260           break;
 261         case OFPT_FLOW_EXPIRED:
 262 #ifdef HAVE_OFP_FLOW_EXPIRED
 263           mod->of_event_flow_removed(ctx, (struct ofp_flow_expired *)neobuf);
 264 #elif defined(HAVE_OFP_FLOW_REMOVED)
 265           mod->of_event_flow_removed(ctx, (struct ofp_flow_removed *)neobuf);
 266 #else
 267 #error "Unknown version of openflow"
 268 #endif
 269           break;
 270         case OFPT_PORT_STATUS:
 271           mod->of_event_port_status(ctx, (struct ofp_port_status *)neobuf);
 272           break;
 273         case OFPT_ECHO_REQUEST:
 274           mod->of_event_echo_request(ctx, (struct ofp_header *)neobuf);
 275           break;
 276         default:
 277           if(ofph->type > OFPT_STATS_REPLY)   // FIXME: update for new openflow versions
 278             {
 279               fprintf(stderr, "%s:%zd :: Data buffer probably trashed : unknown openflow type %d\n",
 280                       __FILE__, __LINE__, ofph->type);
 281               abort();
 282             }
 283           mod->of_event_other(ctx, (struct ofp_header * ) neobuf);
 284           break;
 285         };
 286       free(neobuf);               
 287       bufstart += msglen;
 288       count = bufend - bufstart;  // repurpose count
 289     }       // end while()
 290 
 291   if ( bufstart >= bufend)        // if no outstanding bytes
 292     bufstart = bufend = 0;      // reset our buffer
 293 }

 297  * static void process_pcap_event(oflops_context *ctx, test_module * mod, struct pollfd *fd, oflops_channel_name ch);
 298  *      front end to oflops_pcap_handler
 299  *              make sure all of the memory is kosher before and after
 300  *              pcap's callback thing has always annoyed me
 301  */
 302 static void process_pcap_event(oflops_context *ctx, test_module * mod, struct pollfd *pfd, oflops_channel_name ch)
 303 {
 304   struct pcap_event_wrapper wrap;
 305   int count;
 306 
 307   if(pfd->revents & POLLOUT)
 308     {
 309       int err;
 310       if((err=msgbuf_write(ctx->channels[ch].outgoing,ctx->channels[ch].raw_sock, ctx->channels[ch].packet_len) < 0) && 
 311          (err != EAGAIN) && (err != EWOULDBLOCK ) && (err != EINTR))
 312         perror_and_exit("channel write()",1);
 313     }
 314   if(!(pfd->revents & POLLIN))          // nothing to read, return
 315     return;
 316   // read the next packet from the appropriate pcap socket
 317   assert(ctx->channels[ch].pcap_handle);
 318   count = pcap_dispatch(ctx->channels[ch].pcap_handle, 1, oflops_pcap_handler, (u_char *) & wrap);
 319   if((ch == OFLOPS_CONTROL) && (ctx->dump_controller)) 
 320     pcap_dump((u_char *)ctx->channels[ch].dump, &wrap.pe->pcaphdr, wrap.pe->data);
 321   if (count == 0)
 322     return;
 323   if (count < 0)
 324     {
 325       fprintf(stderr,"process_pcap_event:pcap_dispatch returned %d :: %s \n", count,
 326               pcap_geterr(ctx->channels[ch].pcap_handle));
 327       return;
 328     }
 329   // dispatch it to the test module
 330   mod->handle_pcap_event(ctx, wrap.pe, ch);
 331   // clean up our mess
 332   pcap_event_free(wrap.pe);
 333   return;
 334 }
 335 /*************************************************************************
 336  * int load_test_module(oflops_context *ctx, 
 337  *                      char * mod_filename, char * initstr);
 338  *      open this module and strip symbols out of it
 339  *      and call init() on it
 340  */
 341 int load_test_module(oflops_context *ctx, char * mod_filename, char * initstr)
 342 {
 343   void * handle;
 344   test_module * mod;
 345   mod = malloc_and_check(sizeof(*mod));
 346   bzero(mod,sizeof(*mod));
 347 
 348   // open module for dyn symbols
 349   handle = dlopen(mod_filename,RTLD_NOW);
 350   if(handle == NULL)
 351     {   
 352       fprintf(stderr,"Error reading symbols from %s : %s\n",
 353               mod_filename, dlerror());
 354       return 1;
 355     }
 356   mod->name = dlsym(handle,"name");
 357   mod->start = dlsym(handle,"start");
 358   if(!mod->name)
 359     fprintf( stderr, "Module %s does not contain a name() function\n", mod_filename);
 360   if(!mod->start)
 361     fprintf( stderr, "Module %s does not contain a start() function\n", mod_filename);
 362   if(!mod->name || !mod->start)
 363     {
 364       free(mod);
 365       dlclose(handle);
 366       return 1; // fail for now
 367     }
 368 
 369 #define symbol_fetch(X)                         \
 370   mod->X = dlsym(handle, #X);                   \
 371   if(!mod->X)                                   \
 372     mod->X = default_module_##X
 373   symbol_fetch(init);
 374   symbol_fetch(destroy);
 375   symbol_fetch(get_pcap_filter);
 376   symbol_fetch(handle_pcap_event);
 377   symbol_fetch(of_event_packet_in);
 378   symbol_fetch(of_event_flow_removed);
 379   symbol_fetch(of_event_echo_request);
 380   symbol_fetch(of_event_port_status);
 381   symbol_fetch(of_event_other);
 382   symbol_fetch(handle_timer_event);
 383   symbol_fetch(handle_snmp_event);
 384   symbol_fetch(handle_traffic_generation);
 385 #undef symbol_fetch
 386   if(ctx->n_tests >= ctx->max_tests)
 387     {
 388       ctx->max_tests *=2;
 389       ctx->tests = realloc_and_check(ctx->tests, ctx->max_tests * sizeof(struct test_modules *));
 390     }
 391   ctx->tests[ctx->n_tests++] = mod;
 392   mod->symbol_handle=handle;
 393         
 394   if(mod->init)
 395     mod->init(ctx, initstr);
 396   return 0;
 397 }
