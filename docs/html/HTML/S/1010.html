  65 /**
  66  * \brief Function to look up a host (locked)
  67  * \param list Host table mapped to a MAC address
  68  * \param ip IP address
  69  * \param mac MAC address
  70  */
  71 static int check_host(host_table_t *list, uint32_t ip, uint64_t mac)
  72 {
  73     host_t *curr = list->head;
  74 
  75     while (curr != NULL) {
  76         if (curr->ip == ip && curr->mac == mac) {
  77             return 1;
  78         } else if (curr->ip == ip && curr->mac != mac) {
  79             uint8_t orig[ETH_ALEN], in[ETH_ALEN];
  80 
  81             int2mac(curr->mac, orig);
  82             int2mac(mac, in);
  83 
  84             LOG_WARN(HOST_MGMT_ID, "Worng MAC address (original MAC: %02x:%02x:%02x:%02x:%02x:%02x, "
  85                                    "incoming MAC: %02x:%02x:%02x:%02x:%02x:%02x)",
  86                                    orig[0], orig[1], orig[2], orig[3], orig[4], orig[5],
  87                                    in[0], in[1], in[2], in[3], in[4], in[5]);
  88 
  89             return 2;
  90         } else if (curr->mac == mac && curr->ip != ip) {
  91             struct in_addr orig, in;
  92 
  93             orig.s_addr = curr->ip;
  94             in.s_addr = ip;
  95 
  96             if (strcmp(inet_ntoa(orig), inet_ntoa(in)) != 0) {
  97                 LOG_WARN(HOST_MGMT_ID, "Wrong IP address (original IP: %s, incoming IP: %s)", inet_ntoa(orig), inet_ntoa(in));
  98             } else {
  99                 curr->ip = ip;
 100             }
 101 
 102             return 3;
 103         }
 104 
 105         curr = curr->next;
 106     }
 107 
 108     return 0;
 109 }

 113 /**
 114  * \brief Function to add a new host
 115  * \param pktin PACKET_IN message
 116  */
 117 static int add_new_host(const pktin_t *pktin)
 118 {
 119 #ifdef __ENABLE_CBENCH
 120     return 0;
 121 #else /* !__ENABLE_CBENCH */
 122     if ((pktin->proto & (PROTO_ARP | PROTO_IPV4)) == 0)
 123         return 0;
 124 
 125     struct in_addr src_ip;
 126     src_ip.s_addr = pktin->src_ip;
 127 
 128     if (strncmp(inet_ntoa(src_ip), "169.254.", 8) == 0)
 129         return -1;
 130 
 131     if (strncmp(inet_ntoa(src_ip), "0.0.0.0", 7) == 0)
 132         return -1;
 133 
 134     uint64_t mac = mac2int(pktin->src_mac);
 135     uint32_t mkey = hash_func((uint32_t *)&mac, 2) % __DEFAULT_TABLE_SIZE;
 136 
 137     int ret = 0;
 138     pthread_rwlock_rdlock(&host_table[mkey].lock);
 139     ret = check_host(&host_table[mkey], pktin->src_ip, mac);
 140     pthread_rwlock_unlock(&host_table[mkey].lock);
 141 
 142     if (ret == 0) {
 143         host_t *new = host_dequeue();
 144         if (new == NULL) {
 145             return -1;
 146         }
 147 
 148         new->dpid = pktin->dpid;
 149         new->port = pktin->port;
 150 
 151         new->mac = mac;
 152         new->ip = pktin->src_ip;
 153 
 154         new->prev = new->next = NULL;
 155 
 156         pthread_rwlock_wrlock(&host_table[mkey].lock);
 157 
 158         if (host_table[mkey].head == NULL) {
 159             host_table[mkey].head = new;
 160             host_table[mkey].tail = new;
 161         } else {
 162             new->prev = host_table[mkey].tail;
 163             host_table[mkey].tail->next = new;
 164             host_table[mkey].tail = new;
 165         }
 166 
 167         num_hosts++;
 168 
 169         pthread_rwlock_unlock(&host_table[mkey].lock);
 170 
 171         ev_host_added(HOST_MGMT_ID, new);
 172 
 173         LOG_INFO(HOST_MGMT_ID, "Detected a new device (DPID: %lu, IP: %s, Mac: %02x:%02x:%02x:%02x:%02x:%02x, Port: %u)",
 174                  pktin->dpid,
 175                  inet_ntoa(src_ip),
 176                  pktin->src_mac[0], pktin->src_mac[1], pktin->src_mac[2], pktin->src_mac[3], pktin->src_mac[4], pktin->src_mac[5],
 177                  pktin->port);
 178     }
 179 
 180     return ret;
 181 #endif /* !__ENABLE_CBENCH */
 182 }

 192 int host_mgmt_main(int *activated, int argc, char **argv)
 193 {
 194     LOG_INFO(HOST_MGMT_ID, "Init - Host management");
 195 
 196     num_hosts = 0;
 197 
 198     host_table = (host_table_t *)MALLOC(sizeof(host_table_t) * __DEFAULT_TABLE_SIZE);
 199     if (host_table == NULL) {
 200         PERROR("malloc");
 201         return -1;
 202     }
 203 
 204     memset(host_table, 0, sizeof(host_table_t) * __DEFAULT_TABLE_SIZE);
 205 
 206     int i;
 207     for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 208         pthread_rwlock_init(&host_table[i].lock, NULL);
 209     }
 210 
 211     host_q_init();
 212 
 213     activate();
 214 
 215     return 0;
 216 }

 218 /**
 219  * \brief The cleanup function
 220  * \param activated The activation flag of this component
 221  */
 222 int host_mgmt_cleanup(int *activated)
 223 {
 224     LOG_INFO(HOST_MGMT_ID, "Clean up - Host management");
 225 
 226     deactivate();
 227 
 228     int i;
 229     for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 230         pthread_rwlock_wrlock(&host_table[i].lock);
 231 
 232         host_t *curr = host_table[i].head;
 233         while (curr != NULL) {
 234             host_t *tmp = curr;
 235             curr = curr->next;
 236             FREE(tmp);
 237         }
 238 
 239         pthread_rwlock_unlock(&host_table[i].lock);
 240         pthread_rwlock_destroy(&host_table[i].lock);
 241     }
 242 
 243     host_q_destroy();
 244 
 245     FREE(host_table);
 246 
 247     return 0;
 248 }

 440 /**
 441  * \brief The handler function
 442  * \param ev Read-only event
 443  * \param ev_out Read-write event (if this component has the write permission)
 444  */
 445 int host_mgmt_handler(const event_t *ev, event_out_t *ev_out)
 446 {
 447     switch (ev->type) {
 448     case EV_DP_RECEIVE_PACKET:
 449         PRINT_EV("EV_DP_RECEIVE_PACKET\n");
 450         {
 451             int ret = add_new_host(ev->pktin);
 452             if (ret != 0 && ret != 1) {
 453                 discard_packet(ev->pktin);
 454                 return -1;
 455             }
 456         }
 457         break;
 458     case EV_DP_PORT_DELETED:
 459         PRINT_EV("EV_DP_PORT_DELETED\n");
 460         {
 461             const port_t *port = ev->port;
 462 
 463             int i;
 464             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 465                 host_table_t tmp_list = {0};
 466 
 467                 pthread_rwlock_wrlock(&host_table[i].lock);
 468 
 469                 host_t *curr = host_table[i].head;
 470                 while (curr != NULL) {
 471                     if (curr->dpid == port->dpid && curr->port == port->port) {
 472                         if (tmp_list.head == NULL) {
 473                             tmp_list.head = curr;
 474                             tmp_list.tail = curr;
 475                             curr->r_next = NULL;
 476                         } else {
 477                             tmp_list.tail->r_next = curr;
 478                             tmp_list.tail = curr;
 479                             curr->r_next = NULL;
 480                         }
 481 
 482                         num_hosts--;
 483                     }
 484 
 485                     curr = curr->next;
 486                 }
 487 
 488                 curr = tmp_list.head;
 489                 while (curr != NULL) {
 490                     host_t *tmp = curr;
 491 
 492                     curr = curr->r_next;
 493 
 494                     if (tmp->prev != NULL && tmp->next != NULL) {
 495                         tmp->prev->next = tmp->next;
 496                         tmp->next->prev = tmp->prev;
 497                     } else if (tmp->prev == NULL && tmp->next != NULL) {
 498                         host_table[i].head = tmp->next;
 499                         tmp->next->prev = NULL;
 500                     } else if (tmp->prev != NULL && tmp->next == NULL) {
 501                         host_table[i].tail = tmp->prev;
 502                         tmp->prev->next = NULL;
 503                     } else if (tmp->prev == NULL && tmp->next == NULL) {
 504                         host_table[i].head = NULL;
 505                         host_table[i].tail = NULL;
 506                     }
 507 
 508                     host_t out = {0};
 509 
 510                     out.dpid = tmp->dpid;
 511                     out.port = tmp->port;
 512 
 513                     out.mac = tmp->mac;
 514                     out.ip = tmp->ip;
 515 
 516                     ev_host_deleted(HOST_MGMT_ID, &out);
 517 
 518                     struct in_addr ip_addr;
 519                     ip_addr.s_addr = out.ip;
 520 
 521                     uint8_t macaddr[6];
 522                     int2mac(out.mac, macaddr);
 523 
 524                     LOG_INFO(HOST_MGMT_ID, "Deleted a device (DPID: %lu, IP: %s, Mac: %02x:%02x:%02x:%02x:%02x:%02x, Port: %u)",
 525                              out.dpid,
 526                              inet_ntoa(ip_addr),
 527                              macaddr[0], macaddr[1], macaddr[2], macaddr[3], macaddr[4], macaddr[5],
 528                              out.port);
 529 
 530                     host_enqueue(tmp);
 531                 }
 532 
 533                 pthread_rwlock_unlock(&host_table[i].lock);
 534             }
 535         }
 536         break;
 537     case EV_SW_DISCONNECTED:
 538         PRINT_EV("EV_SW_DISCONNECTED\n");
 539         {
 540             const switch_t *sw = ev->sw;
 541 
 542             int i;

 551                         if (tmp_list.head == NULL) {
 552                             tmp_list.head = curr;
 553                             tmp_list.tail = curr;
 554                             curr->r_next = NULL;
 555                         } else {
 556                             tmp_list.tail->r_next = curr;
 557                             tmp_list.tail = curr;
 558                             curr->r_next = NULL;
 559                         }
 560 
 561                         num_hosts--;
 562                     }
 563 
 564                     curr = curr->next;
 565                 }
 566 
 567                 curr = tmp_list.head;
 568                 while (curr != NULL) {
 569                     host_t *tmp = curr;
 570 
 571                     curr = curr->r_next;
 572 
 573                     if (tmp->prev != NULL && tmp->next != NULL) {
 574                         tmp->prev->next = tmp->next;
 575                         tmp->next->prev = tmp->prev;
 576                     } else if (tmp->prev == NULL && tmp->next != NULL) {
 577                         host_table[i].head = tmp->next;
 578                         tmp->next->prev = NULL;
 579                     } else if (tmp->prev != NULL && tmp->next == NULL) {
 580                         host_table[i].tail = tmp->prev;
 581                         tmp->prev->next = NULL;
 582                     } else if (tmp->prev == NULL && tmp->next == NULL) {
 583                         host_table[i].head = NULL;
 584                         host_table[i].tail = NULL;
 585                     }
 586

 588 
 589                     out.dpid = tmp->dpid;
 590                     out.port = tmp->port;
 591

 593                     out.ip = tmp->ip;
 594 
 595                     if (sw->remote == FALSE) {
 596                         ev_host_deleted(HOST_MGMT_ID, &out);
 597                     }
 598 
 599                     struct in_addr ip_addr;
 600                     ip_addr.s_addr = out.ip;
 601 
 602                     uint8_t macaddr[6];

 605                     LOG_INFO(HOST_MGMT_ID, "Deleted a device (DPID: %lu, IP: %s, Mac: %02x:%02x:%02x:%02x:%02x:%02x, Port: %u)",

 612                 }

 615             }
 616         }
 617         break;
 618     case EV_HOST_ADDED:
 619         PRINT_EV("EV_HOST_ADDED\n");
 620         {
 621             const host_t *host = ev->host;
 622 
 623             if (host->remote == FALSE)
 624                 break;
 625

 627 
 628             int res = 0;
 629             pthread_rwlock_rdlock(&host_table[mkey].lock);
 630             res = check_host(&host_table[mkey], host->ip, host->mac);
 631             pthread_rwlock_unlock(&host_table[mkey].lock);
 632 
 633             if (res == 0) {
 634                 host_t *new = host_dequeue();
 635                 if (new == NULL) {
 636                     break;
 637                 }
 638 
 639                 new->dpid = host->dpid;
 640                 new->port = host->port;
 641 
 642                 new->ip = host->ip;
 643                 new->mac = host->mac;
 644 
 645                 new->remote = host->remote;
 646 
 647                 new->prev = new->next = NULL;
 648 
 649                 pthread_rwlock_wrlock(&host_table[mkey].lock);
 650 
 651                 if (host_table[mkey].head == NULL) {
 652                     host_table[mkey].head = new;
 653                     host_table[mkey].tail = new;
 654                 } else {
 655                     new->prev = host_table[mkey].tail;
 656                     host_table[mkey].tail->next = new;
 657                     host_table[mkey].tail = new;
 658                 }
 659 
 660                 num_hosts++;
 661 
 662                 pthread_rwlock_unlock(&host_table[mkey].lock);
 663 
 664                 struct in_addr ip_addr;
 665                 ip_addr.s_addr = new->ip;
 666 
 667                 uint8_t macaddr[6];
 668                 int2mac(new->mac, macaddr);
 669 
 670                 LOG_INFO(HOST_MGMT_ID, "Detected a new device (DPID: %lu, IP: %s, Mac: %02x:%02x:%02x:%02x:%02x:%02x, Port: %u)",
 671                          new->dpid,
 672                          inet_ntoa(ip_addr),
 673                          macaddr[0], macaddr[1], macaddr[2], macaddr[3], macaddr[4], macaddr[5],
 674                          new->port);
 675             }
 676         }
 677         break;
 678     case EV_HOST_DELETED:
 679         PRINT_EV("EV_HOST_DELETED\n");
 680         {
 681             const host_t *host = ev->host;
 682 
 683             if (host->remote == FALSE)
 684                 break;
 685 
 686             int i;
 687             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 688                 host_table_t tmp_list = {0};
 689 
 690                 pthread_rwlock_wrlock(&host_table[i].lock);
 691 
 692                 host_t *curr = host_table[i].head;
 693                 while (curr != NULL) {
 694                     if (curr->dpid == host->dpid && curr->port == host->port && curr->ip == host->ip && curr->mac == host->mac) {
 695                         if (tmp_list.head == NULL) {
 696                             tmp_list.head = curr;
 697                             tmp_list.tail = curr;
 698                             curr->r_next = NULL;
 699                         } else {
 700                             tmp_list.tail->r_next = curr;
 701                             tmp_list.tail = curr;
 702                             curr->r_next = NULL;
 703                         }
 704 
 705                         num_hosts--;
 706                     }
 707 
 708                     curr = curr->next;
 709                 }
 710 
 711                 curr = tmp_list.head;
 712                 while (curr != NULL) {
 713                     host_t *tmp = curr;
 714 
 715                     curr = curr->r_next;
 716 
 717                     if (tmp->prev != NULL && tmp->next != NULL) {
 718                         tmp->prev->next = tmp->next;
 719                         tmp->next->prev = tmp->prev;
 720                     } else if (tmp->prev == NULL && tmp->next != NULL) {
 721                         host_table[i].head = tmp->next;
 722                         tmp->next->prev = NULL;
 723                     } else if (tmp->prev != NULL && tmp->next == NULL) {
 724                         host_table[i].tail = tmp->prev;
 725                         tmp->prev->next = NULL;
 726                     } else if (tmp->prev == NULL && tmp->next == NULL) {
 727                         host_table[i].head = NULL;
 728                         host_table[i].tail = NULL;
 729                     }
 730 
 731                     struct in_addr ip_addr;
 732                     ip_addr.s_addr = tmp->ip;
 733 
 734                     uint8_t macaddr[6];
 735                     int2mac(tmp->mac, macaddr);
 736 
 737                     LOG_INFO(HOST_MGMT_ID, "Deleted a device (DPID: %lu, IP: %s, Mac: %02x:%02x:%02x:%02x:%02x:%02x, Port: %u)",
 738                              tmp->dpid,
 739                              inet_ntoa(ip_addr),
 740                              macaddr[0], macaddr[1], macaddr[2], macaddr[3], macaddr[4], macaddr[5],
 741                              tmp->port);
 742 
 743                     host_enqueue(tmp);
 744                 }
 745 
 746                 pthread_rwlock_unlock(&host_table[i].lock);
 747             }
 748         }
 749         break;
 750     case EV_HOST_GET_INFO:
 751         PRINT_EV("EV_HOST_GET_INFO\n");
 752         {
 753             host_t *host = ev_out->host_data;
 754 
 755             int i, stop = FALSE;
 756             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 757                 pthread_rwlock_rdlock(&host_table[i].lock);
 758 
 759                 host_t *curr = host_table[i].head;
 760                 while (curr != NULL) {
 761                     if (curr->ip == host->ip || curr->mac == host->mac) {
 762                         memmove(host, curr, sizeof(host_t));
 763                         host->prev = host->next = host->r_next = NULL;
 764                         stop = TRUE;
 765                         break;
 766                     }
 767 
 768                     curr = curr->next;
 769                 }
 770 
 771                 pthread_rwlock_unlock(&host_table[i].lock);
 772 
 773                 if (stop == TRUE)
 774                     break;
 775             }
 776         }
 777         break;
 778     case EV_HOST_GET_ALL_INFO:
 779         PRINT_EV("EV_HOST_GET_ALL_INFO\n");
 780         {
 781             host_t *host = ev_out->host_data;
 782 
 783             host = (host_t *)MALLOC(sizeof(host_t) * num_hosts);
 784             if (host == NULL) {
 785                 break;
 786             }
 787 
 788             int i, idx = 0;
 789             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 790                 pthread_rwlock_rdlock(&host_table[i].lock);
 791 
 792                 host_t *curr = host_table[i].head;
 793                 while (curr != NULL) {
 794                     memmove(&host[idx], curr, sizeof(host_t));
 795 
 796                     if (idx == 0) {
 797                         host[idx].prev = NULL;
 798 
 799                         if (num_hosts > 1) {
 800                             host[idx].next = &host[idx + 1];
 801                         } else {
 802                             host[idx].next = NULL;
 803                         }
 804                     } else {
 805                         host[idx].prev = &host[idx - 1];
 806 
 807                         if (idx < num_hosts - 1) {
 808                             host[idx].next = &host[idx + 1];
 809                         } else {
 810                             host[idx].next = NULL;
 811                         }
 812                     }
 813 
 814                     host[idx++].r_next = NULL;
 815 
 816                     curr = curr->next;
 817                 }
 818 
 819                 pthread_rwlock_unlock(&host_table[i].lock);
 820             }
 821         }
 822         break;
 823     default:
 824         break;
 825     }
 826 
 827     return 0;
 828 }
