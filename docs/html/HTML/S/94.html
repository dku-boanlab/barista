  60 int init(struct oflops_context *ctx, char * config_str) {
  61   char *pos = NULL;
  62   char *param = config_str;
  63   char *value = NULL;
  64   //init counters
  65   sendno = 0;
  66   TAILQ_INIT(&head);
  67   finished = 0;
  68   //open file for storing measurement
  69   measure_output = fopen("measure.log", "w");
  70 
  71   while(*config_str == ' ') {
  72     config_str++;
  73   }
  74   param = config_str;
  75   while(1) {
  76     pos = index(param, ' ');
  77 
  78     if((pos == NULL)) {
  79       if (*param != '\0') {
  80         pos = param + strlen(param) + 1;
  81       } else
  82         break;
  83     }
  84     *pos='\0';
  85     pos++;
  86     value = index(param,'=');
  87     *value = '\0';
  88     value++;
  89     if(value != NULL) {
  90       if(strcmp(param, "pkt_size") == 0) {
  91         pkt_size = atoi(value);
  92         if((pkt_size <= 70) || (pkt_size > 1500))  
  93           perror_and_exit("Invalid pkt size param(Values between 70 and 1500 bytes)", 1);
  94         
  95       } else if(strcmp(param, "duration") == 0) {
  96         duration = (uint64_t)atoi(value);
  97         if((duration < 10) )  
  98           perror_and_exit("Invalid duration param(Values larger than 10 sec)", 1);
  99         
 100       }
 101       param = pos;
 102     }
 103   } 
 104 
 105   //calculate maximum number of packet I may receive
 106   uint64_t max_pkt_count = duration*1000000000 / 
 107     ((pkt_size * byte_to_bits * sec_to_usec) / (mbits_to_bits));
 108   delay = (double *)xmalloc(max_pkt_count * sizeof(double));
 109     printf("delay_count : %llu\n", max_pkt_count);
 110 
 111   return 0;
 112 }

 133   printf("sending %d bytes\n", ret);

 137   make_ofp_feat_req(&b);
 138   printf("sending %d bytes\n", ret);

 142   printf("cleaning up flow table...\n");
 143   res = make_ofp_flow_del(&b);
 144   ret = write(ctx->control_fd, b, res);
 145   printf("sending %d bytes\n", ret);
 146   free(b);
 147 
 148   //Send a singe ruke to route the traffic we will generate
 149   bzero(fl, sizeof(struct flow));
 150   fl->mask = OFPFW_IN_PORT | OFPFW_TP_DST; 
 151   fl->in_port = htons(ctx->channels[OFLOPS_DATA1].of_port); 
 152   fl->dl_type = htons(ETHERTYPE_IP); 
 153   memcpy(fl->dl_src, local_mac, ETH_ALEN); 
 154   memcpy(fl->dl_dst, "\x00\x1e\x68\x9a\xc5\x75", ETH_ALEN); 
 155   fl->dl_vlan = 0xffff;
 156   fl->nw_proto = IPPROTO_UDP;
 157   fl->nw_src =  inet_addr("10.1.1.1");
 158   fl->nw_dst =  inet_addr("10.1.1.2");
 159   fl->tp_src = htons(8080);            
 160   fl->tp_dst = htons(8080);  
 161   len = make_ofp_flow_add(&b, fl, OFPP_IN_PORT, 0, OFP_FLOW_PERMANENT);
 162   write(ctx->control_fd, b, len);
 163   free(b);
 164 
 165   free(fl);
 166 
 167   return 0;

 186 handle_pcap_event(struct oflops_context *ctx, struct pcap_event * pe, oflops_channel_name ch) {
 187   struct pktgen_hdr *pktgen;
 188   struct timeval snd, rcv;
 189 
 190   if(ch == OFLOPS_DATA1) {
 191     struct flow fl;
 192     pktgen = extract_pktgen_pkt(pe->data, pe->pcaphdr.caplen, &fl);
 193     
 194    /*  if(htonl(pktgen->seq_num) % 100000 == 0) */
 195 /*       printf("data packet received %d\n", htonl(pktgen->seq_num)); */
 196     
 197     snd.tv_sec = htonl(pktgen->tv_sec);
 198     snd.tv_usec = htonl(pktgen->tv_usec);
 199     memcpy(&rcv, &pe->pcaphdr.ts, sizeof(struct timeval));
 200     delay[delay_count++] = (double)time_diff(&snd, &rcv);
 201   }
 202   return 0;

 205 int
 206 handle_traffic_generation (oflops_context *ctx) {
 207   struct traf_gen_det det;
 208   char msg[1024], line[1024];
 209   int datarate[]={10, 50, 100, 200, 300, 400, 
 210                   500, 600, 700, 800, 900, 1000};
 211   int i, datarate_count = 12, status;
 212   uint64_t data_snd_interval;
 213   uint32_t mean, std, median;
 214   float loss;
 215   FILE *input;
 216   struct timeval now;
 217   struct snmp_pdu *pdu, *response;
 218   struct snmp_session *ss;
 219   netsnmp_variable_list *vars;
 220 
 221   init_traf_gen(ctx);
 222 
 223   //background data
 224   strcpy(det.src_ip,"10.1.1.1");
 225   strcpy(det.dst_ip_min,"10.1.1.2");
 226   strcpy(det.dst_ip_max, "10.1.1.2");
 227   strcpy(det.mac_src,"00:00:00:00:00:00"); //"00:1e:68:9a:c5:74");
 228   strcpy(det.mac_dst,"00:1e:68:9a:c5:75");
 229   det.vlan = 0xffff;
 230   det.vlan_p = 1;
 231   det.vlan_cfi = 0;
 232   det.udp_src_port = 8080;
 233   det.udp_dst_port = 8080;
 234   det.pkt_size = pkt_size;
 235   strcpy(det.flags, "");
 236   
 237   //calculating interpacket gap
 238   for (i = 0; i < datarate_count; i++) {
 239 
 240     //initialize snmp seesion and variables
 241     if(!(ss = snmp_open(&(ctx->snmp_channel_info->session)))) {
 242       snmp_perror("snmp_open");
 243       return 1;
 244     }
 245     
 246     pdu = snmp_pdu_create(SNMP_MSG_GET);
 247     snmp_add_null_var(pdu, ctx->channels[OFLOPS_DATA1].outOID, 
 248                       ctx->channels[OFLOPS_DATA1].outOID_len);
 249     /*
 250      * Send the Request out.
 251      */    
 252     status = snmp_synch_response(ss, pdu, &response);
 253     if (status == STAT_SUCCESS && response->errstat == SNMP_ERR_NOERROR) {
 254       for(vars = response->variables; vars; vars = vars->next_variable)  {    
 255         snprint_value(msg, 1024, vars->name, vars->name_length, vars);
 256         printf("result msg: %s\n", msg);
 257       }
 258     }
 259 
 260     if (response)
 261       snmp_free_pdu(response);
 262     response = NULL;
 263     snmp_close(ss);
 264 
 265     //init packet counter 
 266     delay_count = 0;
 267     
 268     //claculate interpacket gap
 269     data_snd_interval = (pkt_size * byte_to_bits * sec_to_usec*1000) / (datarate[i] * mbits_to_bits);
 270     det.delay = data_snd_interval;
 271     //calculate packets send
 272     if(data_snd_interval)
 273       det.pkt_count = ((uint64_t)(duration*1000000000) / (data_snd_interval));
 274     else 
 275       det.pkt_count = (uint64_t)(duration*1000000000);
 276     //print sending probe details
 277     fprintf(stderr, "Sending data interval : %u nsec (pkt_size: %u bytes, rate: %u Mbits/sec %llu packets)\n", 
 278             (uint32_t)data_snd_interval, (uint32_t)pkt_size, (uint32_t)datarate[i],  det.pkt_count);
 279 
 280     //start packet generator
 281     add_traffic_generator(ctx, OFLOPS_DATA1, &det);
 282     start_traffic_generator(ctx);
 283 
 284     
 285     gsl_sort (delay, 1, delay_count);
 286     mean = (uint32_t)gsl_stats_mean(delay, 1, delay_count);
 287     std = (uint32_t)sqrt(gsl_stats_variance(delay, 1, delay_count));
 288     median = (uint32_t)gsl_stats_median_from_sorted_data (delay, 1, delay_count);
 289     loss = (float)delay_count/(float)det.pkt_count;
 290     printf("delay:%d:%u:%u:%u:%.4f:%d\n", 
 291            datarate[i], mean, median, std, loss, delay_count);
 292     snprintf(msg, 1024, "delay:%d:%u:%u:%u:%.4f:%d", 
 293              datarate[i], mean, median, std, loss, delay_count);
 294     gettimeofday(&now, NULL);
 295     oflops_log(now, GENERIC_MSG, msg);
 296     // TODO: snprintf the name of of_data1 device instead
 297     input = popen ("tail -2 /proc/net/pktgen/eth1", "r");
 298     if(input != NULL) {
 299       while(fgets(line, 1024, input) != NULL) {
 300         snprintf(msg, 1024, "pktgen:%d:%s", datarate[i], line);
 301         oflops_log(now, GENERIC_MSG, msg);
 302         printf("%s\n", msg);
 303       }
 304       pclose(input);
 305     }
 306   }
 307 
 308   oflops_end_test(ctx,1);
 309   return 1;
 310 }
