  31 /*
  32  * Structure used to represent a DDP address; gives the layout of the
  33  * data pointed to by an AT_ATALK "address" structure.
  34  */
  35 struct atalk_ddp_addr {
  36         guint16 net;
  37         guint8  node;
  38 };
  39 
  40 /*
  41  * DDP packet types.
  42  */
  43 #define DDP_RTMPDATA    0x01
  44 #define DDP_NBP         0x02
  45 #define DDP_ATP         0x03
  46 #define DDP_AEP         0x04
  47 #define DDP_RTMPREQ     0x05
  48 #define DDP_ZIP         0x06
  49 #define DDP_ADSP        0x07
  50 #define DDP_EIGRP       0x58
  51 
  52 /*
  53  * Routines to take a DDP address and generate a string.
  54  */
  55 extern gchar *atalk_addr_to_str(const struct atalk_ddp_addr *addrp);
  56 extern void atalk_addr_to_str_buf(const struct atalk_ddp_addr *addrp,
  57                                   gchar *buf, int buf_len);
  58 
  59 #ifdef __cplusplus
  60 }
  61 #endif /* __cplusplus */
