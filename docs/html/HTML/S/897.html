  92 /* State of an STP port.
  93  *
  94  * A port is in exactly one state at any given time, but distinct bits are used
  95  * for states to allow testing for more than one state with a bit mask. */
  96 enum stp_state {
  97     STP_DISABLED = 1 << 0,       /* 8.4.5: Disabled by management. */
  98     STP_LISTENING = 1 << 1,      /* 8.4.2: Not learning or relaying frames. */
  99     STP_LEARNING = 1 << 2,       /* 8.4.3: Learning but not relaying frames. */
 100     STP_FORWARDING = 1 << 3,     /* 8.4.4: Learning and relaying frames. */
 101     STP_BLOCKING = 1 << 4        /* 8.4.1: Initial boot state. */
 102 };
 103 const char *stp_state_name(enum stp_state);
