  69 static void
  70 emerg_flow_status_cb(struct status_reply *status_reply, void *context_)
  71 {
  72         struct emerg_flow_context *context = context_;
  73 
  74         status_reply_put(status_reply, "state=%s",
  75                          context->state == PRIVATEOPT_EMERG_FLOW_RESTORATION
  76                          ? "restoration"
  77                          : context->state == PRIVATEOPT_EMERG_FLOW_PROTECTION
  78                          ? "protection" : "unknown");
  79 }

  81 static void
  82 emerg_flow_periodic_cb(void *context_)
  83 {
  84         struct emerg_flow_context *context = context_;
  85         struct ofpbuf *buf = NULL;
  86         struct private_vxhdr *vxhdr = NULL;
  87         struct private_vxopt *vxopt = NULL;
  88         int error = 0;
  89 
  90         if (rconn_is_connected(context->remote_rconn)) {
  91                 if (context->state == PRIVATEOPT_EMERG_FLOW_PROTECTION) {
  92                         context->prev_state = context->state;
  93                         context->state = PRIVATEOPT_EMERG_FLOW_RESTORATION;
  94                 } else {
  95                         return;
  96                 }
  97         } else {
  98                 if (context->state == PRIVATEOPT_EMERG_FLOW_RESTORATION) {
  99                         context->prev_state = context->state;
 100                         context->state = PRIVATEOPT_EMERG_FLOW_PROTECTION;
 101                 } else {
 102                         return;
 103                 }
 104         }
 105 
 106         vxhdr = (struct private_vxhdr *)make_openflow
 107                 (sizeof(*vxhdr) + sizeof(*vxopt), OFPT_VENDOR, &buf);
 108         vxopt = (struct private_vxopt *)(vxhdr + 1);
 109         vxhdr->ofp_vxid = htonl(PRIVATE_VENDOR_ID);
 110         vxopt->pvo_type = htons(context->state);
 111         vxopt->pvo_len = htons(0);
 112 
 113         error = rconn_send(context->local_rconn, buf, NULL);
 114         if (error && error != EAGAIN) {
 115                 VLOG_WARN("send failed (%s)", strerror(error));
 116         }
 117 }

 119 void
 120 emerg_flow_start(struct secchan *secchan, const struct settings *settings,
 121                  struct switch_status *switch_status,
 122                  struct rconn *local_rconn, struct rconn *remote_rconn)
 123 {
 124         struct emerg_flow_context *context = NULL;
 125         static struct hook_class emerg_flow_hook_class = {
 126                 NULL,           /* local_packet_cb */
 127                 NULL,           /* remote_packet_cb */
 128                 emerg_flow_periodic_cb, /* periodic_cb */
 129                 NULL,           /* wait_cb */
 130                 NULL,           /* closing_cb */
 131         };
 132 
 133         context = xmalloc(sizeof(*context));
 134         context->settings = settings;
 135         context->secchan = secchan;
 136         context->local_rconn = local_rconn;
 137         context->remote_rconn = remote_rconn;
 138         context->prev_state = PRIVATEOPT_EMERG_FLOW_PROTECTION;
 139         context->state = PRIVATEOPT_EMERG_FLOW_PROTECTION;
 140 
 141         switch_status_register_category(switch_status, "emerg-flow",
 142                                         emerg_flow_status_cb, context);
 143         add_hook(secchan, &emerg_flow_hook_class, context);
 144 }
