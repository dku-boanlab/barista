  65 /** \brief The structure of a component */
  66 struct _compnt_t {

  69 
  70     char name[__CONF_WORD_LEN]; /**< Component name */

  72 
  73     int argc; /**< The number of arguments */
  74     char *argv[__CONF_ARGC + 1]; /**< Arguments */
  75 
  76     int type; /**< Type */
  77     int site; /**< Site */
  78     int role; /**< Role */
  79     int perm; /**< Permission */
  80     int status; /**< Status */
  81     int activated; /**< Activation */
  82 
  83     compnt_main_f main; /**< The main function pointer */
  84     compnt_handler_f handler; /**< The handler function pointer */
  85     compnt_cleanup_f cleanup; /**< The cleanup function pointer */
  86     compnt_cli_f cli; /**< The CLI function pointer */
  87 
  88     int in_num; /**< The number of inbound events */
  89     int in_list[__MAX_EVENTS]; /**< Inbound events */
  90 
  91     int out_num; /**< The number of outbound events */
  92     int out_list[__MAX_EVENTS]; /**< Outbound events */
  93 
  94     int num_policies; /**< The number of policies */

  96 
  97     uint64_t num_events[__MAX_EVENTS]; /**< The number of triggered times */
