  47 /**
  48  * \brief Function to send a LLDP packet using a pktout message
  49  * \param dpid Datapath ID
  50  * \param port Output port
  51  */
  52 static int send_lldp(uint64_t dpid, port_t *port)
  53 {
  54     pktout_t pktout = {0};
  55 
  56     pktout.dpid = dpid;
  57     pktout.port = -1;
  58 
  59     switch_t sw = {0};
  60     sw.dpid = dpid;
  61     ev_sw_get_xid(TOPO_MGMT_ID, &sw);
  62 
  63     pktout.xid = sw.xid;
  64     pktout.buffer_id = -1;
  65 
  66     pktout.total_len = 46;
  67 
  68     // LLDP
  69     lldp_chassis_id *chassis;  // mandatory
  70     lldp_port_id *portid;      // mandatory
  71     lldp_ttl *ttl;             // mandatory
  72     lldp_system_desc *desc;    // optional
  73     lldp_eol *eol;             // mandatory
  74 
  75     uint8_t multi[] = {0x01, 0x80, 0xc2, 0x00, 0x00, 0x0e};
  76     uint8_t mac[ETH_ALEN] = { *((uint8_t *)&dpid + 5),
  77         *((uint8_t *)&dpid + 4),
  78         *((uint8_t *)&dpid + 3),
  79         *((uint8_t *)&dpid + 2),
  80         *((uint8_t *)&dpid + 1),
  81         *((uint8_t *)&dpid + 0) };
  82 
  83     struct ether_header *eth_header = (struct ether_header *)pktout.data; // 14
  84 
  85     memmove(eth_header->ether_shost, mac, ETH_ALEN);
  86     memmove(eth_header->ether_dhost, multi, ETH_ALEN);
  87 
  88     eth_header->ether_type = htons(ETHERTYPE_LLDP); // 32
  89 
  90     // lldp_chassis_id tlv
  91     chassis = (lldp_chassis_id *)(eth_header + 1);
  92     chassis->hdr.type = LLDP_TLVT_CHASSISID;
  93     chassis->hdr.length = 7;
  94     chassis->subtype = LLDP_CHASSISID_SUBTYPE_MACADDR;
  95 
  96     memmove(chassis->data, &port->hw_addr, ETH_ALEN);
  97 
  98     // lldp_port_id tlv
  99     portid = (lldp_port_id *)((uint8_t *)chassis + chassis->hdr.length + 2);
 100     portid->hdr.type = LLDP_TLVT_PORTID;
 101     portid->hdr.length = 5;
 102     portid->subtype = LLDP_PORTID_SUBTYPE_COMPONENT;
 103 
 104     uint32_t port_data = htonl(port->port);
 105     memmove(portid->data, &port_data, 4);
 106 
 107     // lldp_time_to_live tlv
 108     ttl = (lldp_ttl *)((uint8_t *)portid + portid->hdr.length + 2);
 109     ttl->hdr.type = LLDP_TLVT_TTL;
 110     ttl->hdr.length = 2;
 111     ttl->ttl = htons(DEFAULT_TTL);
 112 
 113     // lldp_port_description tlv
 114     desc = (lldp_system_desc *)((uint8_t *)ttl + ttl->hdr.length + 2);
 115     desc->hdr.type = LLDP_TLVT_SYSTEM_DESC;
 116     desc->hdr.length = 8;
 117 
 118     uint64_t net_dpid = htonll(dpid);
 119     memmove(desc->data, &net_dpid, 8);
 120 
 121     // lldp_end_of_lldpdu tlv
 122     eol = (lldp_eol *)((uint8_t *)desc + desc->hdr.length + 2);
 123     eol->hdr.type = LLDP_TLVT_EOL;
 124     eol->hdr.length = 0;
 125 
 126     // flip bytes
 127     chassis->hdr.raw = htons(chassis->hdr.raw);
 128     portid->hdr.raw = htons(portid->hdr.raw);
 129     ttl->hdr.raw = htons(ttl->hdr.raw);
 130     desc->hdr.raw = htons(desc->hdr.raw);
 131     eol->hdr.raw = htons(eol->hdr.raw);
 132 
 133     // output
 134     pktout.num_actions = 1;
 135     pktout.action[0].type = ACTION_OUTPUT;
 136     pktout.action[0].port = port->port;
 137 
 138     ev_dp_send_packet(TOPO_MGMT_ID, &pktout);
 139 
 140     return 0;
 141 }

 147 static int discard_packet(const pktin_t *pktin)
 148 {

 163 /**
 164  * \brief Function to insert a new link
 165  * \param src_dpid Source datapath ID
 166  * \param src_port Source port
 167  * \param dst_dpid Destination datapath ID
 168  * \param dst_port Destination port
 169  */
 170 static int insert_link(uint64_t src_dpid, uint16_t src_port, uint64_t dst_dpid, uint16_t dst_port)
 171 {
 172     if (src_dpid == 0 || src_port == 0 || src_port > __MAX_NUM_PORTS)
 173         return 0;
 174 
 175     if (dst_dpid == 0 || dst_port == 0 || dst_port > __MAX_NUM_PORTS)
 176         return 0;
 177 
 178     pthread_rwlock_rdlock(&topo_lock);
 179 
 180     int i;
 181     for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 182         if (topo[i].dpid == src_dpid) { // check source dpid
 183             if (topo[i].port[src_port].port == src_port) { // check source port
 184                 if (topo[i].port[src_port].target_dpid == dst_dpid && topo[i].port[src_port].target_port == dst_port) {
 185                     break;
 186                 } else if (topo[i].port[src_port].target_dpid == 0 && topo[i].port[src_port].target_port == 0) {
 187                     pthread_rwlock_unlock(&topo_lock);
 188 
 189                     pthread_rwlock_wrlock(&topo_lock);
 190 
 191                     topo[i].port[src_port].target_dpid = dst_dpid;
 192                     topo[i].port[src_port].target_port = dst_port;
 193 
 194                     num_links++;
 195 
 196                     pthread_rwlock_unlock(&topo_lock);
 197 
 198                     port_t link = {0};
 199 
 200                     link.remote = FALSE;
 201 
 202                     link.dpid = src_dpid;
 203                     link.port = src_port;
 204                     link.target_dpid = dst_dpid;
 205                     link.target_port = dst_port;
 206 
 207                     ev_link_added(TOPO_MGMT_ID, &link);
 208 
 209                     return 1;
 210                 } else {
 211                     DEBUG("Inconsistent link {(%lu, %u) -> (%lu, %u)} -> {(%lu, %u) -> (%lu, %u)}\n",
 212                           src_dpid, src_port, topo[i].port[src_port].target_dpid, topo[i].port[src_port].target_port,
 213                           src_dpid, src_port, dst_dpid, dst_port);
 214                     break;
 215                 }
 216             }
 217         }
 218     }
 219 
 220     pthread_rwlock_unlock(&topo_lock);
 221 
 222     return 0;
 223 }

 227 /**
 228  * \brief The main function
 229  * \param activated The activation flag of this component
 230  * \param argc The number of arguments
 231  * \param argv Arguments
 232  */
 233 int topo_mgmt_main(int *activated, int argc, char **argv)
 234 {
 235     LOG_INFO(TOPO_MGMT_ID, "Init - Topology management");
 236 
 237     num_links = 0;
 238 
 239     topo = (topo_t *)MALLOC(sizeof(topo_t) * __DEFAULT_TABLE_SIZE);
 240     if (topo == NULL) {
 241         PERROR("malloc");
 242         return -1;
 243     }
 244 
 245     memset(topo, 0, sizeof(topo_t) * __DEFAULT_TABLE_SIZE);
 246 
 247     pthread_rwlock_init(&topo_lock, NULL);
 248 
 249     activate();
 250 
 251     while (*activated) {
 252         pthread_rwlock_rdlock(&topo_lock);
 253 
 254         int i;
 255         for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 256             if (topo[i].dpid > 0) { // check source dpid
 257                 if (topo[i].remote == TRUE) // check location
 258                     continue;
 259 
 260                 int j;
 261                 for (j=0; j<__MAX_NUM_PORTS; j++) {
 262                     if (topo[i].port[j].port > 0) { // check source port
 263                         send_lldp(topo[i].dpid, &topo[i].port[j]);
 264                     }
 265                 }
 266             }
 267         }
 268 
 269         pthread_rwlock_unlock(&topo_lock);
 270 
 271         waitsec(TOPO_MGMT_REQUEST_TIME, 0);
 272     }
 273 
 274     return 0;
 275 }

 294 /**
 295  * \brief Function to list up all detected links
 296  * \param cli The CLI pointer
 297  */
 298 static int topo_listup(cli_t *cli)
 299 {
 300     cli_print(cli, "<Link Lists>");
 301 
 302     pthread_rwlock_rdlock(&topo_lock);
 303 
 304     cli_print(cli, "  The total number of links: %d", num_links);
 305 
 306     int i, cnt = 0;
 307     for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 308         if (topo[i].dpid > 0) { // check source dpid
 309             int j;
 310             for (j=0; j<__MAX_NUM_PORTS; j++) {
 311                 if (topo[i].port[j].port > 0) { // check source port
 312                     if (topo[i].port[j].target_port > 0) { // check destination port
 313                         cli_print(cli, "  Link #%d (%s): {(%lu, %u) -> (%lu, %u)}, rx_pkts: %lu, rx_bytes: %lu, "
 314                                        "tx_pkts: %lu, tx_bytes: %lu", ++cnt, (topo[i].remote == TRUE) ? "Remote" : "Local",
 315                                   topo[i].dpid, topo[i].port[j].port, topo[i].port[j].target_dpid, topo[i].port[j].target_port,
 316                                   topo[i].port[j].rx_packets, topo[i].port[j].rx_bytes, topo[i].port[j].tx_packets, 
 317                                   topo[i].port[j].tx_bytes);
 318                     }
 319                 }
 320             }
 321         }
 322     }
 323 
 324     pthread_rwlock_unlock(&topo_lock);
 325 
 326     return 0;
 327 }

 392 /**
 393  * \brief The handler function
 394  * \param ev Read-only event
 395  * \param ev_out Read-write event (if this component has the write permission)
 396  */
 397 int topo_mgmt_handler(const event_t *ev, event_out_t *ev_out)
 398 {
 399     switch (ev->type) {
 400     case EV_DP_RECEIVE_PACKET:
 401         PRINT_EV("EV_DP_RECEIVE_PACKET\n");
 402         {
 403             const pktin_t *pktin = ev->pktin;
 404             struct ether_header *eth_header = (struct ether_header *)pktin->data;
 405 
 406             if (pktin->proto & PROTO_LLDP) {
 407                 uint64_t src_dpid = 0, dst_dpid = pktin->dpid;
 408                 uint32_t src_port = 0, dst_port = pktin->port;
 409 
 410                 struct tlv_header *tlv = (struct tlv_header *)(eth_header + 1);
 411                 tlv->raw = ntohs(tlv->raw);
 412                 while (tlv->type != LLDP_TLVT_EOL) {
 413                     switch (tlv->type) {
 414                     case LLDP_TLVT_PORTID:
 415                         {
 416                             lldp_port_id *port = (lldp_port_id *)tlv;
 417                             switch (port->subtype) {
 418                             case LLDP_PORTID_SUBTYPE_COMPONENT:
 419                                 memmove(&src_port, port->data, 4);
 420                                 src_port = ntohl(src_port);
 421                                 break;
 422                             default:
 423                                 break;
 424                             }
 425                         }
 426                         break;
 427                     case LLDP_TLVT_SYSTEM_DESC:
 428                         {
 429                             lldp_system_desc *desc = (lldp_system_desc *)tlv;
 430                             memmove(&src_dpid, desc->data, 8);
 431                             src_dpid = ntohll(src_dpid);
 432                         }
 433                         break;
 434                     default:
 435                         break;
 436                     }
 437 
 438                     // move to next tlv
 439                     tlv = (struct tlv_header *)((uint8_t *)tlv + sizeof(struct tlv_header) + tlv->length);
 440                     tlv->raw = ntohs(tlv->raw);
 441                 }
 442 
 443                 if (insert_link(src_dpid, src_port, dst_dpid, dst_port)) {
 444                     LOG_INFO(TOPO_MGMT_ID, "Detected a new link {(%lu, %u) -> (%lu, %u)}", src_dpid, src_port, dst_dpid, dst_port);
 445                 }
 446 
 447                 discard_packet(pktin);
 448 
 449                 return -1;
 450             }
 451         }
 452         break;
 453     case EV_SW_CONNECTED:
 454         PRINT_EV("EV_SW_CONNECTED\n");
 455         {
 456             const switch_t *sw = ev->sw;
 457 
 458             pthread_rwlock_wrlock(&topo_lock);
 459 
 460             int i;
 461             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 462                 if (topo[i].dpid == 0) {
 463                     topo[i].dpid = sw->dpid;
 464                     topo[i].remote = sw->remote;
 465                     break;
 466                 }
 467             }
 468 
 469             pthread_rwlock_unlock(&topo_lock);
 470         }
 471         break;
 472     case EV_SW_DISCONNECTED:
 473         PRINT_EV("EV_SW_DISCONNECTED\n");
 474         {
 475             const switch_t *sw = ev->sw;
 476 
 477             pthread_rwlock_wrlock(&topo_lock);
 478 
 479             int i;
 480             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 481                 if (topo[i].dpid == sw->dpid) { // check source dpid
 482                     int j;
 483                     for (j=0; j<__MAX_NUM_PORTS; j++) {
 484                         if (topo[i].port[j].port > 0) { // check source port
 485                             if (topo[i].port[j].target_port > 0) { // check destination port
 486                                 LOG_INFO(TOPO_MGMT_ID, "Deleted a link {(%lu, %u) -> (%lu, %u)}", topo[i].dpid, 
 487                                          topo[i].port[j].port, topo[i].port[j].target_dpid, topo[i].port[j].target_port);
 488 
 489                                 if (topo[i].remote == FALSE) {
 490                                     port_t link = {0};
 491 
 492                                     link.remote = FALSE;
 493 
 494                                     link.dpid = topo[i].dpid;
 495                                     link.port = topo[i].port[j].port;
 496                                     link.target_dpid = topo[i].port[j].target_dpid;
 497                                     link.target_port = topo[i].port[j].target_port;
 498 
 499                                     ev_link_deleted(TOPO_MGMT_ID, &link);
 500                                 }
 501 
 502                                 num_links--;
 503                             }
 504 
 505                             // reset port
 506                             memset(&topo[i].port[j], 0, sizeof(port_t));
 507                         }
 508                     }
 509 
 510                     // reset dpid
 511                     topo[i].dpid = 0;
 512                     topo[i].remote = FALSE;
 513 
 514                     break;
 515                 }
 516             }
 517 
 518             pthread_rwlock_unlock(&topo_lock);
 519         }
 520         break;
 521     case EV_DP_PORT_ADDED:
 522         PRINT_EV("EV_DP_PORT_ADDED\n");
 523         {
 524             const port_t *port = ev->port;
 525 
 526             pthread_rwlock_wrlock(&topo_lock);
 527 
 528             int i;
 529             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 530                 if (topo[i].dpid == port->dpid) { // check source dpid
 531                     if (port->port > 0) { // check source port
 532                         topo[i].port[port->port].port = port->port;
 533                         memmove(&topo[i].port[port->port].hw_addr, port->hw_addr, ETH_ALEN);
 534                     }
 535                     break;
 536                 }
 537             }
 538 
 539             pthread_rwlock_unlock(&topo_lock);
 540         }
 541         break;
 542     case EV_DP_PORT_DELETED:
 543         PRINT_EV("EV_DP_PORT_DELETED\n");
 544         {
 545             const port_t *port = ev->port;
 546 
 547             pthread_rwlock_wrlock(&topo_lock);
 548 
 549             int i;
 550             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 551                 if (topo[i].dpid == port->dpid) { // check source dpid
 552                     if (topo[i].port[port->port].port == port->port) { // check source port
 553                         if (topo[i].port[port->port].target_port > 0) { // check destination port
 554                             LOG_INFO(TOPO_MGMT_ID, "Deleted a link {(%lu, %u) -> (%lu, %u)}",
 555                                      topo[i].dpid, topo[i].port[port->port].port, 
 556                                      topo[i].port[port->port].target_dpid, topo[i].port[port->port].target_port);
 557 
 558                             if (topo[i].remote == FALSE) {
 559                                 port_t link = {0};
 560 
 561                                 link.remote = FALSE;
 562 
 563                                 link.dpid = port->dpid;
 564                                 link.port = port->port;
 565                                 link.target_dpid = topo[i].port[port->port].target_dpid;
 566                                 link.target_port = topo[i].port[port->port].target_port;
 567 
 568                                 ev_link_deleted(TOPO_MGMT_ID, &link);
 569                             }
 570 
 571                             num_links--;
 572                         }
 573 
 574                         memset(&topo[i].port[port->port], 0, sizeof(port_t));
 575 
 576                         break;
 577                     }
 578                 }
 579             }
 580 
 581             pthread_rwlock_unlock(&topo_lock);
 582         }
 583         break;
 584     case EV_DP_PORT_STATS:
 585         PRINT_EV("EV_DP_PORT_STATS\n");
 586         {
 587             const port_t *port = ev->port;
 588 
 589             pthread_rwlock_wrlock(&topo_lock);
 590 
 591             int i;
 592             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 593                 if (topo[i].dpid == port->dpid) { // check source dpid
 594                     if (topo[i].port[port->port].port == 0) { // check source port
 595                         if (topo[i].remote == FALSE) {
 596                             topo[i].port[port->port].port = port->port;
 597                         }
 598 
 599                         break;
 600                     } else {
 601                         topo[i].port[port->port].rx_packets = port->rx_packets - topo[i].port[port->port].old_rx_packets;
 602                         topo[i].port[port->port].rx_bytes = port->rx_bytes - topo[i].port[port->port].old_rx_bytes;
 603                         topo[i].port[port->port].tx_packets = port->tx_packets - topo[i].port[port->port].old_tx_packets;
 604                         topo[i].port[port->port].tx_bytes = port->tx_bytes - topo[i].port[port->port].old_tx_bytes;
 605 
 606                         topo[i].port[port->port].old_rx_packets = port->rx_packets;
 607                         topo[i].port[port->port].old_rx_bytes = port->rx_bytes;
 608                         topo[i].port[port->port].old_tx_packets = port->tx_packets;
 609                         topo[i].port[port->port].old_tx_bytes = port->tx_bytes;
 610 
 611                         DEBUG("[%lu, %u] - rx_pkts: %lu, rx_bytes: %lu, tx_pkts: %lu, tx_bytes: %lu\n",
 612                                port->dpid, port->port, topo[i].port[port->port].rx_packets,
 613                                topo[i].port[port->port].rx_bytes, topo[i].port[port->port].tx_packets,
 614                                topo[i].port[port->port].tx_bytes);
 615 
 616                         break;
 617                     }
 618                 }
 619             }
 620 
 621             pthread_rwlock_unlock(&topo_lock);
 622         }
 623         break;
 624     case EV_LINK_ADDED:
 625         PRINT_EV("EV_LINK_ADDED\n");
 626         {
 627             const port_t *link = ev->port;
 628 
 629             if (link->remote == FALSE)
 630                 break;
 631 
 632             pthread_rwlock_wrlock(&topo_lock);
 633 
 634             int i;
 635             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 636                 if (topo[i].dpid == link->dpid) { // check source dpid
 637                     if (topo[i].port[link->port].port == 0) { // check source port
 638                         topo[i].port[link->port].port = link->port;
 639 
 640                         topo[i].port[link->port].target_dpid = link->target_dpid;
 641                         topo[i].port[link->port].target_port = link->target_port;
 642 
 643                         num_links++;
 644 
 645                         LOG_INFO(TOPO_MGMT_ID, "Detected a new link {(%lu, %u) -> (%lu, %u)}", 
 646                                  link->dpid, link->port, link->target_dpid, link->target_port);
 647                     }
 648                     break;
 649                 }
 650             }
 651 
 652             pthread_rwlock_unlock(&topo_lock);
 653         }
 654         break;
 655     case EV_LINK_DELETED:
 656         PRINT_EV("EV_LINK_DELETED\n");
 657         {
 658             const port_t *link = ev->port;
 659 
 660             if (link->remote == FALSE)
 661                 break;
 662 
 663             pthread_rwlock_wrlock(&topo_lock);
 664 
 665             int i;
 666             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 667                 if (topo[i].dpid == link->dpid) { // check source dpid
 668                     if (topo[i].port[link->port].port == link->port) { // check source port
 669                         memset(&topo[i].port[link->port], 0, sizeof(port_t));
 670                         num_links--;
 671 
 672                         LOG_INFO(TOPO_MGMT_ID, "Deleted a link {(%lu, %u) -> (%lu, %u)}",
 673                                  link->dpid, link->port, link->target_dpid, link->target_port);
 674                     }
 675                     break;
 676                 }
 677             }
 678 
 679             pthread_rwlock_unlock(&topo_lock);
 680         }
 681         break;
 682     case EV_LINK_GET_INFO:
 683         PRINT_EV("EV_LINK_GET_INFO\n");
 684         {
 685             port_t *link = ev_out->port_data;
 686 
 687             link = (port_t *)MALLOC(sizeof(port_t) * num_links);
 688             if (link == NULL) {
 689                 break;
 690             }
 691 
 692             memset(link, 0, sizeof(port_t) * num_links);
 693 
 694             pthread_rwlock_rdlock(&topo_lock);
 695 
 696             int i, idx = 0;
 697             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 698                 if (topo[i].dpid == link->dpid) { // check source dpid
 699                     int j;
 700                     for (j=0; j<__MAX_NUM_PORTS; j++) {
 701                         if (topo[i].port[j].port > 0) { //check source port
 702                             if (topo[i].port[j].target_port > 0) { // check destination port
 703                                 link[idx].dpid = topo[i].dpid;
 704                                 link[idx].port = topo[i].port[j].port;
 705                                 link[idx].target_dpid = topo[i].port[j].target_dpid;
 706                                 link[idx].target_port = topo[i].port[j].target_port;
 707 
 708                                 link[idx].rx_packets = topo[i].port[j].rx_packets;
 709                                 link[idx].rx_bytes = topo[i].port[j].rx_bytes;
 710                                 link[idx].tx_packets = topo[i].port[j].tx_packets;
 711                                 link[idx].tx_bytes = topo[i].port[j].tx_bytes;
 712 
 713                                 if (idx == 0) {
 714                                     link[idx].prev = NULL;
 715 
 716                                     if (num_links > 1) {
 717                                         link[idx].next = &link[idx + 1];
 718                                     } else {
 719                                         link[idx].next = NULL;
 720                                     }
 721                                 } else {
 722                                     link[idx].prev = &link[idx - 1];
 723 
 724                                     if (idx < num_links - 1) {
 725                                         link[idx].next = &link[idx + 1];
 726                                     } else {
 727                                         link[idx].next = NULL;
 728                                     }
 729                                 }
 730                             }
 731 
 732                             idx++;
 733                         }
 734                     }
 735                 }
 736             }
 737 
 738             pthread_rwlock_unlock(&topo_lock);
 739         }
 740         break;
 741     case EV_LINK_GET_ALL_INFO:
 742         PRINT_EV("EV_LINK_GET_ALL_INFO\n");
 743         {
 744             port_t *link = ev_out->port_data;
 745 
 746             link = (port_t *)MALLOC(sizeof(port_t) * num_links);
 747             if (link == NULL) {
 748                 break;
 749             }
 750 
 751             memset(link, 0, sizeof(port_t) * num_links);
 752 
 753             pthread_rwlock_rdlock(&topo_lock);
 754 
 755             int i, idx = 0;
 756             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 757                 if (topo[i].dpid > 0) { // check source dpid
 758                     int j;
 759                     for (j=0; j<__MAX_NUM_PORTS; j++) {
 760                         if (topo[i].port[j].port > 0) { // check source port
 761                             if (topo[i].port[j].target_port > 0) { // check destination port
 762                                 link[idx].dpid = topo[i].dpid;
 763                                 link[idx].port = topo[i].port[j].port;
 764                                 link[idx].target_dpid = topo[i].port[j].target_dpid;
 765                                 link[idx].target_port = topo[i].port[j].target_port;
 766 
 767                                 link[idx].rx_packets = topo[i].port[j].rx_packets;
 768                                 link[idx].rx_bytes = topo[i].port[j].rx_bytes;
 769                                 link[idx].tx_packets = topo[i].port[j].tx_packets;
 770                                 link[idx].tx_bytes = topo[i].port[j].tx_bytes;
 771 
 772                                 if (idx == 0) {
 773                                     link[idx].prev = NULL;
 774 
 775                                     if (num_links > 1) {
 776                                         link[idx].next = &link[idx + 1];
 777                                     } else {
 778                                         link[idx].next = NULL;
 779                                     }
 780                                 } else {
 781                                     link[idx].prev = &link[idx - 1];
 782 
 783                                     if (idx < num_links - 1) {
 784                                         link[idx].next = &link[idx + 1];
 785                                     } else {
 786                                         link[idx].next = NULL;
 787                                     }
 788                                 }
 789                             }
 790 
 791                             idx++;
 792                         }
 793                     }
 794                 }
 795             }
 796 
 797             pthread_rwlock_unlock(&topo_lock);
 798         }
 799         break;
 800     default:
 801         break;
 802     }
 803 
 804     return 0;
 805 }
