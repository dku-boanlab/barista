  37 /* Idea for gt: either x > y, or y is much bigger (assume wrap) */
  38 #define GT_SEQ(x, y) ((gint32)((y) - (x)) < 0)
  39 #define LT_SEQ(x, y) ((gint32)((x) - (y)) < 0)
  40 #define GE_SEQ(x, y) ((gint32)((y) - (x)) <= 0)
  41 #define LE_SEQ(x, y) ((gint32)((x) - (y)) <= 0)
  42 #define EQ_SEQ(x, y) ((x) == (y))

  72 /*
  73  * Loop for dissecting PDUs within a TCP stream; assumes that a PDU
  74  * consists of a fixed-length chunk of data that contains enough information
  75  * to determine the length of the PDU, followed by rest of the PDU.
  76  *
  77  * The first three arguments are the arguments passed to the dissector
  78  * that calls this routine.
  79  *
  80  * "proto_desegment" is the dissector's flag controlling whether it should
  81  * desegment PDUs that cross TCP segment boundaries.
  82  *
  83  * "fixed_len" is the length of the fixed-length part of the PDU.
  84  *
  85  * "get_pdu_len()" is a routine called to get the length of the PDU from
  86  * the fixed-length part of the PDU; it's passed "pinfo", "tvb" and "offset".
  87  *
  88  * "dissect_pdu()" is the routine to dissect a PDU.
  89  */
  90 extern void
  91 tcp_dissect_pdus(tvbuff_t *tvb, packet_info *pinfo, proto_tree *tree,
  92                  gboolean proto_desegment, guint fixed_len,
  93                  guint (*get_pdu_len)(packet_info *, tvbuff_t *, int),
  94                  dissector_t dissect_pdu);

 132 typedef struct _tcp_flow_t {
 133         guint32 base_seq;       /* base seq number (used by relative sequence numbers)
 134                                  * or 0 if not yet known.
 135                                  */
 136         tcp_unacked_t *segments;
 137         guint32 lastack;        /* last seen ack */
 138         nstime_t lastacktime;   /* Time of the last ack packet */ 
 139         guint32 lastnondupack;  /* frame number of last seen non dupack */
 140         guint32 dupacknum;      /* dupack number */
 141         guint32 nextseq;        /* highest seen nextseq */
 142         guint32 nextseqframe;   /* frame number for segment with highest
 143                                  * sequence number
 144                                  */

 150         gint16  win_scale;      /* -1 is we dont know */

 155         guint16 flags;

 165 struct tcp_analysis {
 166         /* These two structs are managed based on comparing the source
 167          * and destination addresses and, if they're equal, comparing
 168          * the source and destination ports.
 169          *
 170          * If the source is greater than the destination, then stuff
 171          * sent from src is in ual1.
 172          *
 173          * If the source is less than the destination, then stuff
 174          * sent from src is in ual2.
 175          *
 176          * XXX - if the addresses and ports are equal, we don't guarantee
 177          * the behavior.
 178          */
 179         tcp_flow_t      flow1;
 180         tcp_flow_t      flow2;
 181 
 182         /* These pointers are set by get_tcp_conversation_data()
 183          * fwd point in the same direction as the current packet
 184          * and rev in the reverse direction
 185          */
 186         tcp_flow_t      *fwd;
 187         tcp_flow_t      *rev;
 188 
 189         /* This pointer is NULL   or points to a tcp_acked struct if this
 190          * packet has "interesting" properties such as being a KeepAlive or
 191          * similar 
 192          */
 193         struct tcp_acked *ta;
 194         /* This structure contains a tree containing all the various ta's
 195          * keyed by frame number.
 196          */
 197         emem_tree_t     *acked_table;
 198 
 199         /* Remember the timestamp of the first frame seen in this tcp
 200          * conversation to be able to calculate a relative time compared
 201          * to the start of this conversation
 202          */
 203         nstime_t        ts_first;
 204 
 205         /* Remember the timestamp of the frame that was last seen in this
 206          * tcp conversation to be able to calculate a delta time compared
 207          * to previous frame in this conversation
 208          */
 209         nstime_t        ts_prev;
 210 };
