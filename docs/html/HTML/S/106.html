 102 int init(struct oflops_context *ctx, char * config_str) {
 103   char *pos = NULL;
 104   char *param = config_str;
 105   char *value = NULL;
 106 
 107   printf("log initialized\n");
 108 
 109   //init measurement queue
 110   TAILQ_INIT(&head); 
 111   TAILQ_INIT(&stats_head); 
 112 
 113   while(*config_str == ' ') {
 114     config_str++;
 115   }
 116   param = config_str;
 117   while(1) {
 118     pos = index(param, ' ');
 119 
 120     if((pos == NULL)) {
 121       if (*param != '\0') {
 122         pos = param + strlen(param) + 1;
 123       } else
 124         break;
 125     }
 126     *pos='\0';
 127     pos++;
 128     value = index(param,'=');
 129     *value = '\0';
 130     value++;
 131     if(value != NULL) {
 132       if(strcmp(param, "flows") == 0) {
 133         flows = atoi(value);
 134         if(flows <= 0)
 135           perror_and_exit("Invalid flow number",1);
 136       } else if(strcmp(param, "network") == 0) {
 137         network = (char *)xmalloc(strlen(value) + 1);
 138         strcpy(network, value);
 139       } else if(strcmp(param, "pkt_size") == 0) {
 140         //parse int to get pkt size
 141         pkt_size = strtol(value, NULL, 0);
 142         if((pkt_size < MIN_PKT_SIZE) && (pkt_size > MAX_PKT_SIZE))  {
 143           perror_and_exit("Invalid packet size value", 1);
 144         }
 145       } else if(strcmp(param, "data_rate") == 0) {
 146         //parse int to get pkt size
 147         datarate = strtol(value, NULL, 0);
 148         if((datarate <= 0) || (datarate > 1010))  {
 149           perror_and_exit("Invalid data rate param(Values between 1 and 1010)", 1);
 150         }
 151       }  else if(strcmp(param, "probe_rate") == 0) {
 152         //parse int to get pkt size
 153         proberate = strtol(value, NULL, 0);
 154         if((proberate <= 0) || (proberate >= 1010)) {
 155           perror_and_exit("Invalid probe rate param(Value between 1 and 1010)", 1);
 156         }
 157       } else {
 158         fprintf(stderr, "Invalid parameter:%s\n", param);
 159       }
 160       param = pos;
 161     }
 162   } 
 163 
 164   //calculating interpacket gap
 165   data_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (datarate * mbits_to_bits);
 166   fprintf(stderr, "Sending data interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 167           (uint32_t)data_snd_interval, (uint32_t)pkt_size, (uint32_t)datarate);
 168   probe_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (proberate * mbits_to_bits);
 169   fprintf(stderr, "Sending probe interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 170           (uint32_t)probe_snd_interval, (uint32_t)pkt_size, (uint32_t)proberate);
 171 
 172   return 0;
 173 }

 176 int destroy(struct oflops_context *ctx) {
 177   char msg[1024];
 178   struct timeval now;
 179   FILE *out = fopen(logfile, "w");
 180   struct entry *np;
 181   struct stats_entry *stats_np;
 182   long long t[2][3], t_sq[2][3], mean, std;
 183   int count[2][3], min_id[2][3], max_id[2][3], delay;
 184   int ch, i, first=1;
 185   float loss;
 186 
 187   gettimeofday(&now, NULL);
 188   fprintf(stderr, "This is the destroy code of the module\n");
 189 
 190   for(ch = 0; ch < ctx->n_channels-1; ch++) {
 191     for (i=0;i<2;i++) {
 192       count[i][ch] = 0;
 193       min_id[i][ch] =  INT_MAX;
 194       max_id[i][ch] =  INT_MIN;
 195       t[i][ch] = 0;
 196       t_sq[i][ch] = 0;    
 197     }
 198   }
 199   
 200   for (np = head.tqh_first; np != NULL; np = np->entries.tqe_next) {
 201     if(fprintf(out, "%lu;%lu.%06lu;%lu.%06lu;%d\n", 
 202                (long unsigned int)np->id,  
 203                (long unsigned int)np->snd.tv_sec, 
 204                (long unsigned int)np->snd.tv_usec,
 205                (long unsigned int)np->rcv.tv_sec, 
 206                (long unsigned int)np->rcv.tv_usec,  np->ch) < 0)  
 207       perror_and_exit("fprintf fail", 1); 
 208 
 209     i = (time_diff(&stats_start,&np->snd)<=0)?0:1;
 210     ch = np->ch - 1;
 211     count[i][ch]++; 
 212     min_id[i][ch] = (np->id < min_id[i][ch])?np->id:min_id[i][ch];
 213     max_id[i][ch] = (np->id > max_id[i][ch])?np->id:max_id[i][ch];
 214     t[i][ch] += time_diff(&np->snd, &np->rcv);
 215     t_sq[i][ch] += time_diff(&np->snd, &np->rcv)*time_diff(&np->snd, &np->rcv);
 216     free(np);
 217   }
 218 
 219   for(ch = 0; ch < ctx->n_channels-1; ch++) {
 220     for(i = 0; i < 2; i++) {
 221       if(count[i][ch] == 0) continue;
 222       mean = t[i][ch]/count[i][ch];
 223       std = (t_sq[i][ch]/count[i][ch]) - mean*mean;
 224       std=(std >= 0)?sqrt(std):LONG_MAX;
 225       loss = (float)count[i][ch]/(float)(max_id[i][ch] - min_id[i][ch]);
 226       snprintf(msg, 1024, "statistics:port:%d.%d:%lld:%lld:%.4f:%d", 
 227                ctx->channels[ch + 1].of_port,i, mean, std, loss, count[i][ch]);
 228       oflops_log(now, GENERIC_MSG, msg);
 229     }
 230   }
 231 
 232   count[0][0] = 0;
 233   min_id[0][0] =  INT_MAX;
 234   max_id[0][0] =  INT_MIN;
 235   t[0][0] = 0;
 236   t_sq[0][0] = 0;  
 237   
 238   for (stats_np = stats_head.tqh_first; stats_np != NULL; stats_np = stats_np->entries.tqe_next) {
 239     delay = time_diff(&now, &stats_np->rcv);
 240     count[0][0]++;
 241     if(!first) {
 242       t[0][0] += delay;
 243       t_sq[0][0] += delay*delay;
 244     }
 245     first = 0;
 246     memcpy(&now,&stats_np->rcv, sizeof(struct timeval));
 247     free(stats_np);
 248   }
 249   if(count[0][0] > 0) {
 250       mean = t[0][0]/count[0][0];
 251       std = (t_sq[0][0]/count[0][0]) - mean*mean;
 252       std=(std >= 0)?sqrt(std):LONG_MAX;
 253       snprintf(msg, 1024, "stats_stats:%lld:%lld:%d", mean, std, count[0][0]);
 254       oflops_log(now, GENERIC_MSG, msg);
 255   } else {
 256       oflops_log(now, GENERIC_MSG, "stats_stats:fail");
 257   }
 258   return 0;
 259 }

 418 handle_snmp_event(struct oflops_context * ctx, struct snmp_event * se) {
 419   netsnmp_variable_list *vars;
 420   int i, len = 1024;
 421   char msg[1024], log[1024];
 422   struct timeval now;
 423 
 424   for(vars = se->pdu->variables; vars; vars = vars->next_variable)  {    
 425     snprint_value(msg, len, vars->name, vars->name_length, vars);
 426     if((vars->name_length == ctx->cpuOID_len) &&
 427        (memcmp(vars->name, ctx->cpuOID,  ctx->cpuOID_len * sizeof(oid)) == 0) ) {
 428       snprintf(log, len, "cpu : %s %%", msg);
 429       oflops_log(now, SNMP_MSG, log);
 430     } else {
 431       for(i=0;i<ctx->n_channels;i++) {
 432         if((vars->name_length == ctx->channels[i].inOID_len) &&
 433            (memcmp(vars->name, ctx->channels[i].inOID,  
 434                    ctx->channels[i].inOID_len * sizeof(oid)) == 0) ) {
 435           snprintf(log, len, "port %d : rx %s pkts",  
 436                    (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 437           oflops_log(now, SNMP_MSG, log);
 438           break;
 439         }
 440         
 441         if((vars->name_length == ctx->channels[i].outOID_len) &&
 442            (memcmp(vars->name, ctx->channels[i].outOID,  
 443                    ctx->channels[i].outOID_len * sizeof(oid))==0) ) {
 444           snprintf(log, len, "port %d : tx %s pkts",  
 445                    (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 446           oflops_log(now, SNMP_MSG, log);
 447           break;
 448         }
 449       } //for
 450     }// if cpu
 451   }
 452   return 0;
 453 }

 473 /** Handle pcap event.
 474  * @param ctx pointer to opaque context
 475  * @param pe pcap event
 476  * @param ch enumeration of channel that pcap event is triggered
 477  */
 478 int handle_pcap_event(struct oflops_context *ctx, struct pcap_event *pe,
 479                       oflops_channel_name ch) {
 480     if ( (ch == OFLOPS_DATA3) || (ch == OFLOPS_DATA2) || (ch == OFLOPS_DATA1)){
 481     struct pktgen_hdr *pktgen;
 482     pktgen = extract_pktgen_pkt((unsigned char *)pe->data, pe->pcaphdr.caplen, NULL);
 483     if(pktgen == NULL) //skip non IP packets
 484       return 0;
 485 
 486     struct entry *n1 = malloc(sizeof(struct entry));
 487     n1->snd.tv_sec = htonl(pktgen->tv_sec);
 488     n1->snd.tv_usec = htonl(pktgen->tv_usec);
 489     memcpy(&n1->rcv, &pe->pcaphdr.ts, sizeof(struct timeval));
 490     n1->id = htonl(pktgen->seq_num);
 491     n1->ch = ch;
 492     TAILQ_INSERT_TAIL(&head, n1, entries);
 493   }
 494   return 0;
 495 }

 536 int 
 537 dummy_parse_packet(void *data, int len) {
 538   // assume we have ethernet packets.
 539   // skip first bytes of the ether because they are
 540   // in the simple case of static length.
 541   struct ether_header *eth = (struct ether_header *) data;
 542   if (len < sizeof(struct ether_header))
 543     return -1;
 544   data += sizeof(struct ether_header);
 545   if(ntohs(eth->ether_type) != ETHERTYPE_IP) {
 546     return -1;
 547   }
 548   if (len < sizeof(struct iphdr))
 549     return -1;
 550   struct iphdr *ip_p = (struct iphdr *)data;
 551   if (len < 4*ip_p->ihl)
 552     return -1;
 553   data += 4*ip_p->ihl;
 554 
 555   if(ip_p->protocol != IPPROTO_UDP)
 556     return -1;
 557 
 558   if(len <  sizeof(struct tcphdr))
 559     return -1;
 560   struct udphdr *udp_p = (struct udphdr *)data;
 561   if (len < sizeof(struct udphdr))
 562     return -1;
 563 
 564   if((ntohs(udp_p->source) == 8080) && 
 565      (ntohs(udp_p->dest) == 8080))
 566     return 0;
 567   else 
 568     return -1;
 569 
 570 };

 581     struct ofp_stats_reply *ofpr = (struct ofp_stats_reply *)ofph;
 582     if(ntohs(ofpr->type) == OFPST_FLOW) {
 583       sprintf(msg, "%d", ntohl(ofph->xid));
 584       gettimeofday(&now, NULL);
 585       oflops_log(now, OFPT_STATS_REPLY_FLOW, msg);
 586       if((ntohs(ofpr->flags) & OFPSF_REPLY_MORE) == 0) {
 587         len = make_ofp_flow_get_stat(&b, trans_id++);
 588         res = oflops_send_of_mesg(ctx, b);
 589         free(b);
 590       }
 591       struct stats_entry *n1 = malloc(sizeof(struct stats_entry));
 592       memcpy(&n1->rcv, &now, sizeof(struct timeval));
 593       n1->id =  ntohl(ofph->xid);
 594       TAILQ_INSERT_TAIL(&stats_head, n1, entries); 
 595     }
 596   } else if (ofph->type == OFPT_ERROR) {
 597     err_p = (struct ofp_error_msg *)ofph;
 598     sprintf(msg, "OFPT_ERROR(type: %d, code: %d)", ntohs(err_p->type), ntohs(err_p->code));
 599     fprintf(stderr, "%s\n", msg);
 600     perror_and_exit(msg, 1);
 601   }
 602   return 0;
