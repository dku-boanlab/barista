  49 /* Internal function used to compare fields in flow. */
  50 static inline int
  51 flow_fields_match(const struct flow *a, const struct flow *b, uint32_t w,
  52                   uint32_t src_mask, uint32_t dst_mask)
  53 {
  54     return ((w & OFPFW_IN_PORT || a->in_port == b->in_port)
  55             && (w & OFPFW_DL_VLAN || a->dl_vlan == b->dl_vlan)
  56             && (w & OFPFW_DL_VLAN_PCP || a->dl_vlan_pcp == b->dl_vlan_pcp)
  57             && (w & OFPFW_DL_SRC || eth_addr_equals(a->dl_src, b->dl_src))
  58             && (w & OFPFW_DL_DST || eth_addr_equals(a->dl_dst, b->dl_dst))
  59             && (w & OFPFW_DL_TYPE || a->dl_type == b->dl_type)
  60             && (w & OFPFW_NW_TOS || a->nw_tos == b->nw_tos)
  61             && (w & OFPFW_NW_PROTO || a->nw_proto == b->nw_proto)
  62             && !((a->nw_src ^ b->nw_src) & src_mask)
  63             && !((a->nw_dst ^ b->nw_dst) & dst_mask)
  64             && (w & OFPFW_TP_SRC || a->tp_src == b->tp_src)
  65             && (w & OFPFW_TP_DST || a->tp_dst == b->tp_dst));
  66 }

  68 static uint32_t make_nw_mask(int n_wild_bits)
  69 {
  70     n_wild_bits &= (1u << OFPFW_NW_SRC_BITS) - 1;
  71     return n_wild_bits < 32 ? htonl(~((1u << n_wild_bits) - 1)) : 0;
  72 }

  74 /* Returns nonzero if 'a' and 'b' match, that is, if their fields are equal
  75  * modulo wildcards in 'b', zero otherwise. */
  76 inline int
  77 flow_matches_1wild(const struct sw_flow_key *a, const struct sw_flow_key *b)
  78 {
  79     return flow_fields_match(&a->flow, &b->flow, b->wildcards,
  80                              b->nw_src_mask, b->nw_dst_mask);
  81 }
  82 
  83 /* Returns nonzero if 'a' and 'b' match, that is, if their fields are equal
  84  * modulo wildcards in 'a' or 'b', zero otherwise. */
  85 inline int
  86 flow_matches_2wild(const struct sw_flow_key *a, const struct sw_flow_key *b)
  87 {
  88     return flow_fields_match(&a->flow, &b->flow, a->wildcards | b->wildcards,
  89                              a->nw_src_mask & b->nw_src_mask,
  90                              a->nw_dst_mask & b->nw_dst_mask);
  91 }
  92 
  93 /* Returns nonzero if 't' (the table entry's key) and 'd' (the key 
  94  * describing the match) match, that is, if their fields are 
  95  * equal modulo wildcards, zero otherwise.  If 'strict' is nonzero, the
  96  * wildcards must match in both 't_key' and 'd_key'.  Note that the
  97  * table's wildcards are ignored unless 'strict' is set. */
  98 int
  99 flow_matches_desc(const struct sw_flow_key *t, const struct sw_flow_key *d, 
 100         int strict)
 101 {
 102     if (strict && d->wildcards != t->wildcards) {
 103         return 0;
 104     }
 105     return flow_matches_1wild(t, d);
 106 }

 108 /* Returns nonzero if 't' (the table entry's key) and 'd' (the key
 109  * describing the match) match, that is, if their fields are
 110  * equal modulo 't' or 'd' wildcards, zero otherwise.  If 'strict' is nonzero, the
 111  * wildcards must match in both 't_key' and 'd_key'.  Note that the
 112  * table's wildcards are ignored unless 'strict' is set. */
 113 int
 114 flow_matches_2desc(const struct sw_flow_key *t, const struct sw_flow_key *d,
 115         int strict)
 116 {
 117     if (strict && d->wildcards != t->wildcards) {
 118         return 0;
 119     }
 120     return flow_matches_2wild(t, d);
 121 }

 123 void
 124 flow_extract_match(struct sw_flow_key* to, const struct ofp_match* from)
 125 {
 126     to->wildcards = ntohl(from->wildcards) & OFPFW_ALL;
 127     to->flow.dl_vlan_pcp = from->dl_vlan_pcp;
 128     to->flow.in_port = from->in_port;
 129     to->flow.dl_vlan = from->dl_vlan;
 130     memcpy(to->flow.dl_src, from->dl_src, ETH_ADDR_LEN);
 131     memcpy(to->flow.dl_dst, from->dl_dst, ETH_ADDR_LEN);
 132     to->flow.dl_type = from->dl_type;
 133 
 134     to->flow.nw_tos = to->flow.nw_proto = to->flow.nw_src = to->flow.nw_dst = 0;
 135     to->flow.tp_src = to->flow.tp_dst = 0;
 136     memset(to->flow.pad, 0, sizeof(to->flow.pad));
 137 
 138 #define OFPFW_TP (OFPFW_TP_SRC | OFPFW_TP_DST)
 139 #define OFPFW_NW (OFPFW_NW_TOS | OFPFW_NW_PROTO | OFPFW_NW_SRC_MASK | OFPFW_NW_DST_MASK)
 140     if (to->wildcards & OFPFW_DL_TYPE) {
 141         /* Can't sensibly match on network or transport headers if the
 142          * data link type is unknown. */
 143         to->wildcards |= OFPFW_NW | OFPFW_TP;
 144     } else if (from->dl_type == htons(ETH_TYPE_IP)) {
 145         to->flow.nw_tos   = from->nw_tos & 0xfc;
 146         to->flow.nw_proto = from->nw_proto;
 147         to->flow.nw_src   = from->nw_src;
 148         to->flow.nw_dst   = from->nw_dst;
 149 
 150         if (to->wildcards & OFPFW_NW_PROTO) {
 151             /* Can't sensibly match on transport headers if the network
 152              * protocol is unknown. */
 153             to->wildcards |= OFPFW_TP;
 154         } else if (from->nw_proto == IPPROTO_TCP 
 155                 || from->nw_proto == IPPROTO_UDP
 156                 || from->nw_proto == IPPROTO_ICMP) {
 157             to->flow.tp_src = from->tp_src;
 158             to->flow.tp_dst = from->tp_dst;
 159         } else {
 160             /* Transport layer fields are undefined.  Mark them as
 161              * exact-match to allow such flows to reside in table-hash,
 162              * instead of falling into table-linear. */
 163             to->wildcards &= ~OFPFW_TP;
 164         }
 165     } else if (from->dl_type == htons(ETH_TYPE_ARP)) {
 166         to->flow.nw_src   = from->nw_src;
 167         to->flow.nw_dst   = from->nw_dst;
 168         to->flow.nw_proto = from->nw_proto;
 169 
 170         /* Transport layer fields are undefined.  Mark them as
 171          * exact-match to allow such flows to reside in table-hash,
 172          * instead of falling into table-linear. */
 173         to->wildcards &= ~OFPFW_TP;
 174     } else {
 175         /* Network and transport layer fields are undefined.  Mark them
 176          * as exact-match to allow such flows to reside in table-hash,
 177          * instead of falling into table-linear. */
 178         to->wildcards &= ~(OFPFW_NW | OFPFW_TP);
 179     }
 180 
 181         /* We set these late because code above adjusts to->wildcards. */
 182         to->nw_src_mask = make_nw_mask(to->wildcards >> OFPFW_NW_SRC_SHIFT);
 183         to->nw_dst_mask = make_nw_mask(to->wildcards >> OFPFW_NW_DST_SHIFT);
 184 }

 186 /* Allocates and returns a new flow with room for 'actions_len' actions. 
 187  * Returns the new flow or a null pointer on failure. */
 188 struct sw_flow *
 189 flow_alloc(size_t actions_len)
 190 {
 191     struct sw_flow_actions *sfa;
 192     size_t size = sizeof *sfa + actions_len;
 193     struct sw_flow *flow = calloc(1, sizeof *flow);
 194     if (!flow)
 195         return NULL;
 196 
 197     sfa = calloc(1, size);
 198     if (!sfa) {
 199         free(flow);
 200         return NULL;
 201     }
 202     sfa->actions_len = actions_len;
 203     flow->sf_acts = sfa;
 204     return flow;
 205 }

 207 /* Setup the action on the flow, just after it was created with flow_alloc().
 208  * Jean II */
 209 void
 210 flow_setup_actions(struct sw_flow *                    flow,
 211                    const struct ofp_action_header *     actions,
 212                    int                                  actions_len)
 213 {
 214         /* Make sure we don't blow the allocation */
 215         if (actions_len > flow->sf_acts->actions_len)
 216                 ofp_fatal(0,
 217                           "flow_setup_actions: actions_len is too big (%d > %lu)",
 218                           actions_len, (unsigned long)flow->sf_acts->actions_len);
 219 
 220         flow->used = flow->created = time_msec();
 221         flow->sf_acts->actions_len = actions_len;
 222         flow->byte_count = 0;
 223         flow->packet_count = 0;
 224         memcpy(flow->sf_acts->actions, actions, actions_len);
 225 }

 232         return;

 238 /* Copies 'actions' into a newly allocated structure for use by 'flow'
 239  * and frees the structure that defined the previous actions. */
 240 void flow_replace_acts(struct sw_flow *flow, 
 241         const struct ofp_action_header *actions, size_t actions_len)
 242 {
 243     struct sw_flow_actions *sfa;
 244     int size = sizeof *sfa + actions_len;
 245 
 246     sfa = malloc(size);
 247     if (unlikely(!sfa))
 248         return;
 249 
 250     sfa->actions_len = actions_len;
 251     memcpy(sfa->actions, actions, actions_len);
 252 
 253     free(flow->sf_acts);
 254     flow->sf_acts = sfa;
 255 
 256     return;
 257 }

 291 bool flow_timeout(struct sw_flow *flow)
 292 {
 293     uint64_t now = time_msec();
 294     if (flow->idle_timeout != OFP_FLOW_PERMANENT
 295             && now > flow->used + flow->idle_timeout * 1000) {
 296         flow->reason = OFPRR_IDLE_TIMEOUT;
 297         return true;
 298     } else if (flow->hard_timeout != OFP_FLOW_PERMANENT
 299             && now > flow->created + flow->hard_timeout * 1000) {
 300         flow->reason = OFPRR_HARD_TIMEOUT;
 301         return true;
 302     } else {
 303         return false;
 304     }
 305 }

 307 /* Returns nonzero if 'flow' contains an output action to 'out_port' or
 308  * has the value OFPP_NONE. 'out_port' is in network-byte order. */
 309 int flow_has_out_port(struct sw_flow *flow, uint16_t out_port)
 310 {
 311     struct sw_flow_actions *sf_acts = flow->sf_acts;
 312     size_t actions_len = sf_acts->actions_len;
 313     uint8_t *p = (uint8_t *)sf_acts->actions;
 314 
 315     if (out_port == htons(OFPP_NONE))
 316         return 1;
 317 
 318     while (actions_len > 0) {
 319         struct ofp_action_header *ah = (struct ofp_action_header *)p;
 320         size_t len = ntohs(ah->len);
 321 
 322         if (ah->type == htons(OFPAT_OUTPUT)) {
 323             struct ofp_action_output *oa = (struct ofp_action_output *)p;
 324             if (oa->port == out_port) {
 325                 return 1;
 326             }
 327         }
 328         p += len;
 329         actions_len -= len;
 330     }
 331 
 332     return 0;
 333 }
