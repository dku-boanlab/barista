 335 void
 336 nf2_reset_card(struct nf2device *dev)
 337 {
 338         volatile unsigned int val;
 339 
 340         /* If we are operating on a NetFPGA enabled box, reset the card */
 341         readReg(dev, CPCI_REG_CTRL, (void *)&val);
 342         val |= 0x100;
 343         writeReg(dev, CPCI_REG_CTRL, val);
 344         DBG_VERBOSE("Reset the NetFPGA.\n");
 345         sleep(2);
 346 }

 348 void
 349 nf2_clear_watchdog(struct nf2device *dev)
 350 {
 351         volatile unsigned int enable_status;
 352 
 353 #ifndef NF2_WATCHDOG
 354         return;
 355 #endif
 356         log_watchdog_info(dev);
 357 
 358         readReg(dev, WDT_ENABLE_FLG_REG, (void *)&enable_status);
 359         enable_status &= 0x1;
 360 
 361         if (enable_status == WATCHDOG_DISABLE) {
 362                 enable_status = WATCHDOG_ENABLE;
 363                 writeReg(dev, WDT_ENABLE_FLG_REG, enable_status);
 364         }
 365 }

 367 /* Write a wildcard entry to the specified device and row. The row consists of
 368  * the actual entry, its mask that specifies wildcards, as well as the action(s)
 369  * to be taken if the row is matched
 370  */
 371 int
 372 nf2_write_of_wildcard(struct nf2device *dev, int row,
 373                       nf2_of_entry_wrap *entry, nf2_of_mask_wrap *mask,
 374                       nf2_of_action_wrap *action)
 375 {
 376         int i;
 377         unsigned int val;
 378 
 379         DBG_VERBOSE("** Begin wildcard entry write to row: %i\n", row);
 380         log_entry(entry);
 381         log_mask(mask);
 382         log_action(action);
 383         log_entry_raw(entry);
 384         log_mask_raw(mask);
 385         log_action_raw(action);
 386 
 387         for (i = 0; i < NF2_OF_ENTRY_WORD_LEN; ++i) {
 388                 writeReg(dev,
 389                                OPENFLOW_WILDCARD_LOOKUP_CMP_0_REG
 390                                + (4 * i), entry->raw[i]);
 391         }
 392 
 393         for (i = 0; i < NF2_OF_MASK_WORD_LEN; ++i) {
 394                 writeReg(dev,
 395                                OPENFLOW_WILDCARD_LOOKUP_CMP_MASK_0_REG
 396                                + (4 * i), mask->raw[i]);
 397         }
 398 
 399         for (i = 0; i < NF2_OF_ACTION_WORD_LEN; ++i) {
 400                 writeReg(dev,
 401                                OPENFLOW_WILDCARD_LOOKUP_ACTION_0_REG
 402                                + (4 * i), action->raw[i]);
 403         }
 404 
 405         // Reset the stats for the row
 406         val = 0;
 407         writeReg(dev,
 408                        OPENFLOW_WILDCARD_LOOKUP_BYTES_HIT_0_REG + (4 * row),
 409                        val);
 410         writeReg(dev, OPENFLOW_WILDCARD_LOOKUP_PKTS_HIT_0_REG + (4 * row),
 411                        val);
 412         writeReg(dev,
 413                        OPENFLOW_WILDCARD_LOOKUP_LAST_SEEN_TS_0_REG + (4 * row),
 414                        val);
 415         writeReg(dev, OPENFLOW_WILDCARD_LOOKUP_WRITE_ADDR_REG, row);
 416 
 417         DBG_VERBOSE("** End wildcard entry write to row: %i time(msec): %llu\n",
 418                     row, time_msec());
 419 
 420         return 0;
 421 }

 423 int
 424 nf2_write_of_exact(struct nf2device *dev, int row,
 425                    nf2_of_entry_wrap *entry, nf2_of_action_wrap *action)
 426 {
 427         int i;
 428         unsigned int val;
 429         unsigned int index = row << 7;
 430 
 431         DBG_VERBOSE("** Begin exact match entry write to row: %i\n", row);
 432         log_entry(entry);
 433         log_action(action);
 434         log_entry_raw(entry);
 435         log_action_raw(action);
 436 
 437         for (i = 0; i < NF2_OF_ENTRY_WORD_LEN; ++i) {
 438                 writeReg(dev, SRAM_BASE_ADDR + index
 439                                + (4 * i), entry->raw[i]);
 440         }
 441 
 442         // blank out the counters
 443         val = 0;
 444         for (i = 0; i < NF2_OF_EXACT_COUNTERS_WORD_LEN; ++i) {
 445                 writeReg(dev, SRAM_BASE_ADDR + index
 446                                + sizeof(nf2_of_entry_wrap)
 447                                + (4 * i), val);
 448         }
 449 
 450         // write the actions
 451         for (i = 0; i < NF2_OF_ACTION_WORD_LEN; ++i) {
 452                 writeReg(dev, SRAM_BASE_ADDR + index
 453                                + sizeof(nf2_of_entry_wrap)
 454                                + sizeof(nf2_of_exact_counters_wrap)
 455                                + (4 * i), action->raw[i]);
 456         }
 457 
 458         DBG_VERBOSE
 459             ("** End exact match entry write to row: %i time(msec): %llu\n",
 460              row, time_msec());
 461 
 462         return 0;
 463 }

 465 /* Write wildcard action(s) to the specified device and row. */
 466 int
 467 nf2_modify_write_of_wildcard(struct nf2device *dev, int row,
 468                              nf2_of_entry_wrap *entry, nf2_of_mask_wrap *mask,
 469                              nf2_of_action_wrap *action)
 470 {
 471         int i;
 472         unsigned int bytes_reg_val;
 473         unsigned int pkts_reg_val;
 474         unsigned int last_reg_val;
 475 
 476         DBG_VERBOSE("** Begin wildcard modified action write to row: %i\n",
 477                     row);
 478         log_entry(entry);
 479         log_mask(mask);
 480         log_action(action);
 481         log_entry_raw(entry);
 482         log_mask_raw(mask);
 483         log_action_raw(action);
 484 
 485         writeReg(dev, OPENFLOW_WILDCARD_LOOKUP_READ_ADDR_REG, row);
 486         readReg(dev,
 487                       OPENFLOW_WILDCARD_LOOKUP_BYTES_HIT_0_REG + (4 * row),
 488                       &bytes_reg_val);
 489         readReg(dev,
 490                       OPENFLOW_WILDCARD_LOOKUP_PKTS_HIT_0_REG + (4 * row),
 491                       &pkts_reg_val);
 492         readReg(dev,
 493                       OPENFLOW_WILDCARD_LOOKUP_LAST_SEEN_TS_0_REG + (4 * row),
 494                       &last_reg_val);
 495 
 496         for (i = 0; i < NF2_OF_ENTRY_WORD_LEN; ++i) {
 497                 writeReg(dev, OPENFLOW_WILDCARD_LOOKUP_CMP_0_REG
 498                                + (4 * i), entry->raw[i]);
 499         }
 500 
 501         for (i = 0; i < NF2_OF_MASK_WORD_LEN; ++i) {
 502                 writeReg(dev, OPENFLOW_WILDCARD_LOOKUP_CMP_MASK_0_REG
 503                                + (4 * i), mask->raw[i]);
 504         }
 505 
 506         for (i = 0; i < NF2_OF_ACTION_WORD_LEN; ++i) {
 507                 writeReg(dev, OPENFLOW_WILDCARD_LOOKUP_ACTION_0_REG
 508                                + (4 * i), action->raw[i]);
 509         }
 510 
 511         writeReg(dev,
 512                        OPENFLOW_WILDCARD_LOOKUP_BYTES_HIT_0_REG + (4 * row),
 513                        bytes_reg_val);
 514         writeReg(dev, OPENFLOW_WILDCARD_LOOKUP_PKTS_HIT_0_REG + (4 * row),
 515                        pkts_reg_val);
 516         writeReg(dev,
 517                        OPENFLOW_WILDCARD_LOOKUP_LAST_SEEN_TS_0_REG + (4 * row),
 518                        last_reg_val);
 519         writeReg(dev, OPENFLOW_WILDCARD_LOOKUP_WRITE_ADDR_REG, row);
 520 
 521         DBG_VERBOSE
 522             ("** End wildcard modified action write to row: %i time(msec): %llu\n",
 523              row, time_msec());
 524         DBG_VERBOSE("   Bytes hit count: %d\n", bytes_reg_val);
 525         DBG_VERBOSE("   Pkts  hit count: %d\n", pkts_reg_val);
 526         DBG_VERBOSE("   Last seen      : %d\n", last_reg_val);
 527 
 528         return 0;
 529 }

 558 unsigned int
 559 nf2_get_exact_packet_count(struct nf2device *dev, int row)
 560 {
 561         unsigned int val = 0;
 562         unsigned int index = 0;
 563 
 564         /* TODO: Need to scrape data from all 4 registers
 565          * in the case of a wildcarded source port and
 566          * forward all action type
 567          */
 568         nf2_of_exact_counters_wrap counters;
 569         memset(&counters, 0, sizeof(nf2_of_exact_counters_wrap));
 570 
 571         // build the index to our counters
 572         index = row << 7;
 573 
 574         // Read the first word into our struct, to not disturb the byte count
 575         readReg(dev,
 576                       SRAM_BASE_ADDR + index + sizeof(nf2_of_entry_wrap),
 577                       &counters.raw[0]);
 578         val = counters.counters.pkt_count;
 579 
 580         DBG_VERBOSE
 581             ("** Exact match packet count(delta) row: %i count: %i time(msec): %llu\n",
 582              row, val, time_msec());
 583 
 584         return val;
 585 }

 588 nf2_get_exact_byte_count(struct nf2device *dev, int row)
 589 {
 590         unsigned int val = 0;
 591         unsigned int index = 0;
 592 
 593         /* TODO: Need to scrape data from all 4 registers
 594          * in the case of a wildcarded source port and
 595          * forward all action type
 596          */
 597         nf2_of_exact_counters_wrap counters;
 598         memset(&counters, 0, sizeof(nf2_of_exact_counters_wrap));
 599 
 600         // build the index to our counters
 601         index = row << 7;
 602 
 603         // Read the second word into our struct, to not disturb the packet count
 604         readReg(dev, SRAM_BASE_ADDR + index +
 605                       sizeof(nf2_of_entry_wrap) + 4, &counters.raw[1]);
 606         val = counters.counters.byte_count;
 607 
 608         DBG_VERBOSE
 609             ("** Exact match byte count(delta) row: %i count: %i time(msec): %llu\n",
 610              row, val, time_msec());
 611 
 612         return val;
 613 }

 647 unsigned long int
 648 nf2_get_matched_count(struct nf2device *dev)
 649 {
 650         unsigned int val_wild = 0;
 651         unsigned int val_exact = 0;
 652 
 653         readReg(dev, OPENFLOW_LOOKUP_WILDCARD_HITS_REG, &val_wild);
 654         readReg(dev, OPENFLOW_LOOKUP_EXACT_HITS_REG, &val_exact);
 655 
 656         DBG_VERBOSE("** Wildcard Matched count: %i time(msec): %llu\n",
 657                     val_wild, time_msec());
 658         DBG_VERBOSE("** Exact Matched count: %i time(msec): %llu\n",
 659                     val_exact, time_msec());
 660 
 661         return ((unsigned long int)(val_wild + val_exact));
 662 }

 803 int
 804 nf2_get_port_info(struct nf2device *dev, int nf_port,
 805                   struct nf2_port_info *nf2portinfo)
 806 {
 807         struct nf2_all_ports_info_addr *nf2addr;
 808 
 809         if ((nf_port >= NF2_PORT_NUM) || (nf_port < 0)) {
 810                 DBG_ERROR("Illegal port number\n");
 811                 return 1;
 812         }
 813 
 814         nf2addr = calloc(1, sizeof(struct nf2_all_ports_info_addr));
 815         if (nf2addr == NULL) {
 816                 DBG_ERROR("Could not allocate memory for port information gathering\n");
 817                 return 1;
 818         }
 819         nf2_get_all_ports_info_addr(nf2addr);
 820 
 821         readReg(dev, nf2addr->rx_q_num_pkts_stored_reg[nf_port],
 822                       &(nf2portinfo->rx_q_num_pkts_stored));
 823         readReg(dev, nf2addr->rx_q_num_pkts_dropped_full_reg[nf_port],
 824                       &(nf2portinfo->rx_q_num_pkts_dropped_full));
 825         readReg(dev, nf2addr->rx_q_num_pkts_dropped_bad_reg[nf_port],
 826                       &(nf2portinfo->rx_q_num_pkts_dropped_bad));
 827         readReg(dev, nf2addr->rx_q_num_words_pushed_reg[nf_port],
 828                       &(nf2portinfo->rx_q_num_words_pushed));
 829         readReg(dev, nf2addr->rx_q_num_bytes_pushed_reg[nf_port],
 830                       &(nf2portinfo->rx_q_num_bytes_pushed));
 831         readReg(dev, nf2addr->rx_q_num_pkts_dequeued_reg[nf_port],
 832                       &(nf2portinfo->rx_q_num_pkts_dequeued));
 833         readReg(dev, nf2addr->rx_q_num_pkts_in_queue_reg[nf_port],
 834                       &(nf2portinfo->rx_q_num_pkts_in_queue));
 835         readReg(dev, nf2addr->tx_q_num_pkts_in_queue_reg[nf_port],
 836                       &(nf2portinfo->tx_q_num_pkts_in_queue));
 837         readReg(dev, nf2addr->tx_q_num_pkts_sent_reg[nf_port],
 838                       &(nf2portinfo->tx_q_num_pkts_sent));
 839         readReg(dev, nf2addr->tx_q_num_words_pushed_reg[nf_port],
 840                       &(nf2portinfo->tx_q_num_words_pushed));
 841         readReg(dev, nf2addr->tx_q_num_bytes_pushed_reg[nf_port],
 842                       &(nf2portinfo->tx_q_num_bytes_pushed));
 843         readReg(dev, nf2addr->tx_q_num_pkts_enqueued_reg[nf_port],
 844                       &(nf2portinfo->tx_q_num_pkts_enqueued));
 845         free(nf2addr);
 846         return 0;
 847 }
