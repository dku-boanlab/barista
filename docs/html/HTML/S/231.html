  81 struct nf2device *
  82 nf2_get_net_device(void)
  83 {
  84         struct nf2device *dev;
  85         dev = calloc(1, sizeof(struct nf2device));
  86         dev->device_name = DEFAULT_IFACE;
  87 
  88         if (iface_chk_done) {
  89                 dev->net_iface = net_iface;
  90         } else {
  91                 if (check_iface(dev)) {
  92                         iface_chk_done = 0;
  93                         return NULL;
  94                 } else {
  95                         iface_chk_done = 1;
  96                         net_iface = dev->net_iface;
  97                 }
  98         }
  99 
 100         if (openDescriptor(dev)) {
 101                 return NULL;
 102         }
 103         return dev;
 104 }

 117 /* Checks to see if the actions requested by the flow are capable of being
 118  * done in the NF2 hardware. Returns 1 if yes, 0 for no.
 119  */
 120 int
 121 nf2_are_actions_supported(struct sw_flow *flow)
 122 {
 123         struct sw_flow_actions *sfa;
 124         size_t actions_len;
 125         uint8_t *p;
 126 
 127         sfa = flow->sf_acts;
 128         actions_len = sfa->actions_len;
 129         p = (uint8_t *)sfa->actions;
 130 
 131         while (actions_len > 0) {
 132                 struct ofp_action_header *ah = (struct ofp_action_header *)p;
 133                 struct ofp_action_output *oa = (struct ofp_action_output *)p;
 134                 size_t len = ntohs(ah->len);
 135 
 136                 DBG_VERBOSE("Action Support Chk: Len of this action: %i\n",
 137                             len);
 138                 DBG_VERBOSE("Action Support Chk: Len of actions    : %i\n",
 139                             actions_len);
 140 
 141                 // All the modify actions are supported.
 142                 // Each of them can be specified once otherwise overwritten.
 143                 // Output action happens last.
 144                 if (!(ntohs(ah->type) == OFPAT_OUTPUT
 145                       || ntohs(ah->type) == OFPAT_SET_DL_SRC
 146                       || ntohs(ah->type) == OFPAT_SET_DL_DST
 147 
 148                       || ntohs(ah->type) == OFPAT_SET_NW_SRC
 149                       || ntohs(ah->type) == OFPAT_SET_NW_DST
 150 
 151                       || ntohs(ah->type) == OFPAT_SET_NW_TOS
 152 
 153                       || ntohs(ah->type) == OFPAT_SET_TP_SRC
 154                       || ntohs(ah->type) == OFPAT_SET_TP_DST
 155 
 156                       || ntohs(ah->type) == OFPAT_SET_VLAN_VID
 157                       || ntohs(ah->type) == OFPAT_SET_VLAN_PCP
 158                       || ntohs(ah->type) == OFPAT_STRIP_VLAN)) {
 159                         DBG_VERBOSE
 160                                 ("Flow action type %#0x not supported in hardware\n",
 161                                  ntohs(ah->type));
 162                         return 0;
 163                 }
 164                 // Only support ports 1-4(incl. IN_PORT), ALL, FLOOD.
 165                 // Let CONTROLLER/LOCAL fall through
 166                 if ((ntohs(ah->type) == OFPAT_OUTPUT)
 167                     && (!((ntohs(oa->port) >= PORT_BASE)
 168                           && (ntohs(oa->port) <= MAX_IFACE))
 169                         && !(ntohs(oa->port) == OFPP_ALL)
 170                         && !(ntohs(oa->port) == OFPP_FLOOD)
 171                         && !(ntohs(oa->port) == OFPP_IN_PORT))) {
 172                         DBG_VERBOSE
 173                                 ("Flow action output port %#0x is not supported in hardware\n",
 174                                  ntohs(oa->port));
 175                         return 0;
 176                 }
 177                 p += len;
 178                 actions_len -= len;
 179         }
 180 
 181         return 1;
 182 }

 298 /* Setup the wildcard table by adding static flows that will handle
 299  * misses by sending them up to the cpu ports, and handle packets coming
 300  * back down from the cpu by sending them out the corresponding port.
 301  */
 302 int
 303 nf2_write_static_wildcard(void)
 304 {
 305         nf2_of_entry_wrap entry;
 306         nf2_of_mask_wrap mask;
 307         nf2_of_action_wrap action;
 308         int i;
 309         struct nf2device *dev;
 310 
 311         dev = nf2_get_net_device();
 312         if (dev == NULL)
 313                 return 1;
 314 
 315         memset(&entry, 0x00, sizeof(nf2_of_entry_wrap));
 316         memset(&mask, 0xFF, sizeof(nf2_of_mask_wrap));
 317         // Only non-wildcard section is the source port
 318         mask.entry.src_port = 0;
 319         memset(&action, 0, sizeof(nf2_of_action_wrap));
 320 
 321         // write the catch all entries to send to the cpu
 322         for (i = 0; i < 4; ++i) {
 323                 entry.entry.src_port = i * 2;
 324                 action.action.forward_bitmask = 0x1 << ((i * 2) + 1);
 325                 nf2_write_of_wildcard(dev, (OPENFLOW_WILDCARD_TABLE_SIZE - 4)
 326                                       + i, &entry, &mask, &action);
 327         }
 328 
 329         // write the entries to send out packets coming from the cpu
 330         for (i = 0; i < 4; ++i) {
 331                 entry.entry.src_port = (i * 2) + 1;
 332                 action.action.forward_bitmask = 0x1 << (i * 2);
 333                 nf2_write_of_wildcard(dev, (OPENFLOW_WILDCARD_TABLE_SIZE - 8)
 334                                       + i, &entry, &mask, &action);
 335         }
 336 
 337         nf2_free_net_device(dev);
 338         return 0;
 339 }

 341 /* Populate a nf2_of_entry_wrap with entries from a struct sw_flow. */
 342 void
 343 nf2_populate_of_entry(nf2_of_entry_wrap *key, struct sw_flow *flow)
 344 {
 345         int vlan_vid;
 346         int vlan_pcp;
 347         int i;
 348 
 349         key->entry.transp_dst = ntohs(flow->key.flow.tp_dst);
 350         key->entry.transp_src = ntohs(flow->key.flow.tp_src);
 351         key->entry.ip_proto = flow->key.flow.nw_proto;
 352         key->entry.ip_dst = ntohl(flow->key.flow.nw_dst);
 353         key->entry.ip_src = ntohl(flow->key.flow.nw_src);
 354         key->entry.eth_type = ntohs(flow->key.flow.dl_type);
 355         // Blame Jad for applying endian'ness to character arrays
 356         for (i = 0; i < 6; ++i) {
 357                 key->entry.eth_dst[i] = flow->key.flow.dl_dst[5 - i];
 358         }
 359         for (i = 0; i < 6; ++i) {
 360                 key->entry.eth_src[i] = flow->key.flow.dl_src[5 - i];
 361         }
 362         key->entry.src_port = (ntohs(flow->key.flow.in_port) - PORT_BASE) * 2;
 363         key->entry.ip_tos = TOS_BITMASK & flow->key.flow.nw_tos;
 364         if (ntohs(flow->key.flow.dl_vlan) == 0xffff) {
 365                 key->entry.vlan_id = 0xffff;
 366         } else {
 367                 vlan_vid = VID_BITMASK & ntohs(flow->key.flow.dl_vlan);
 368                 vlan_pcp = PCP_BITMASK
 369                         & ((uint16_t)(flow->key.flow.dl_vlan_pcp) << PCP_BITSHIFT);
 370                 key->entry.vlan_id = vlan_pcp | vlan_vid;
 371         }
 372 }

 374 static uint32_t
 375 make_nw_wildcard(int n_wild_bits)
 376 {
 377         n_wild_bits &= (1u << OFPFW_NW_SRC_BITS) - 1;
 378         return n_wild_bits < 32 ? ((1u << n_wild_bits) - 1) : 0xFFFFFFFF;
 379 }

 381 /* Populate a nf2_of_mask_wrap with entries from a struct sw_flow's wildcards. */
 382 void
 383 nf2_populate_of_mask(nf2_of_mask_wrap *mask, struct sw_flow *flow)
 384 {
 385         int vlan_vid = 0;
 386         int vlan_pcp = 0;
 387         int i;
 388 
 389         if (OFPFW_IN_PORT & flow->key.wildcards) {
 390                 mask->entry.src_port = 0xFF;
 391         }
 392         if (OFPFW_DL_VLAN & flow->key.wildcards) {
 393                 vlan_vid = VID_BITMASK;
 394         }
 395         if (OFPFW_DL_VLAN_PCP & flow->key.wildcards) {
 396                 vlan_pcp = 0xF000;
 397         }
 398         mask->entry.vlan_id = vlan_pcp | vlan_vid;
 399         if (OFPFW_DL_SRC & flow->key.wildcards) {
 400                 for (i = 0; i < 6; ++i) {
 401                         mask->entry.eth_src[i] = 0xFF;
 402                 }
 403         }
 404         if (OFPFW_DL_DST & flow->key.wildcards) {
 405                 for (i = 0; i < 6; ++i) {
 406                         mask->entry.eth_dst[i] = 0xFF;
 407                 }
 408         }
 409         if (OFPFW_DL_TYPE & flow->key.wildcards)
 410                 mask->entry.eth_type = 0xFFFF;
 411         if ((OFPFW_NW_SRC_ALL & flow->key.wildcards)
 412             || (OFPFW_NW_SRC_MASK & flow->key.wildcards))
 413                 mask->entry.ip_src = make_nw_wildcard
 414                         (flow->key.wildcards >> OFPFW_NW_SRC_SHIFT);
 415         if ((OFPFW_NW_DST_ALL & flow->key.wildcards)
 416             || (OFPFW_NW_DST_MASK & flow->key.wildcards))
 417                 mask->entry.ip_dst = make_nw_wildcard
 418                         (flow->key.wildcards >> OFPFW_NW_DST_SHIFT);
 419         if (OFPFW_NW_PROTO & flow->key.wildcards)
 420                 mask->entry.ip_proto = 0xFF;
 421         if (OFPFW_TP_SRC & flow->key.wildcards)
 422                 mask->entry.transp_src = 0xFFFF;
 423         if (OFPFW_TP_DST & flow->key.wildcards)
 424                 mask->entry.transp_dst = 0xFFFF;
 425         if (OFPFW_NW_TOS & flow->key.wildcards)
 426                 mask->entry.ip_tos = TOS_BITMASK;
 427 
 428         mask->entry.pad = 0x00;
 429 }

 431 static void
 432 populate_action_output(nf2_of_action_wrap *action, nf2_of_entry_wrap *entry,
 433                        uint8_t *flowact)
 434 {
 435         uint16_t port = 0;
 436         struct ofp_action_output *actout = (struct ofp_action_output *)flowact;
 437         int i;
 438 
 439         port = ntohs(actout->port);
 440         DBG_VERBOSE("Action Type: %i Output Port: %i\n",
 441                     ntohs(actout->type), port);
 442 
 443         if ((port >= PORT_BASE) && (port <= MAX_IFACE)) {
 444                 // Bitmask for output port(s), evens are phys odds cpu
 445                 action->action.forward_bitmask
 446                         |= (1 << ((port - PORT_BASE) * 2));
 447                 DBG_VERBOSE("Output Port: %i Forward Bitmask: %x\n",
 448                             port, action->action.forward_bitmask);
 449         } else if (port == OFPP_IN_PORT) {
 450                 // Send out to input port
 451                 action->action.forward_bitmask
 452                         |= (1 << (entry->entry.src_port));
 453                 DBG_VERBOSE("Output Port = Input Port  Forward Bitmask: %x\n",
 454                             action->action.forward_bitmask);
 455         } else if (port == OFPP_ALL || port == OFPP_FLOOD) {
 456                 // Send out all ports except the source
 457                 for (i = 0; i < 4; ++i) {
 458                         if ((i * 2) != entry->entry.src_port) {
 459                                 // Bitmask for output port(s), evens are
 460                                 // phys odds cpu
 461                                 action->action.forward_bitmask
 462                                         |= (1 << (i * 2));
 463                                 DBG_VERBOSE
 464                                         ("Output Port: %i Forward Bitmask: %x\n",
 465                                          port, action->action.forward_bitmask);
 466                         }
 467                 }
 468         }
 469 }

 546         struct ofp_action_vlan_pcp *actvlan = (struct ofp_action_vlan_pcp *)flowact;
 547         action->action.vlan_pcp = actvlan->vlan_pcp;
 548         action->action.nf2_action_flag |= (1 << OFPAT_SET_VLAN_PCP);
 549 }

 551 static void
 552 populate_action_strip_vlan(nf2_of_action_wrap *action)
 553 {
 554         action->action.nf2_action_flag |= (1 << OFPAT_STRIP_VLAN);
 555 }

 557 /* Populate an nf2_of_action_wrap. */
 558 void
 559 nf2_populate_of_action(nf2_of_action_wrap *action, nf2_of_entry_wrap *entry,
 560                        struct sw_flow *flow)
 561 {
 562         struct sw_flow_actions *sfa;
 563         size_t actions_len;
 564         uint8_t *p;
 565 
 566         sfa = flow->sf_acts;
 567         actions_len = sfa->actions_len;
 568         p = (uint8_t *)sfa->actions;
 569 
 570         // zero it out for now
 571         memset(action, 0, sizeof(nf2_of_action_wrap));
 572         action->action.nf2_action_flag = 0;
 573 
 574         while (actions_len > 0) {
 575                 struct ofp_action_header *acth = (struct ofp_action_header *)p;
 576                 size_t len = ntohs(acth->len);
 577 
 578                 DBG_VERBOSE("Action Populate: Len of this action: %i\n", len);
 579                 DBG_VERBOSE("Action Populate: Len of actions    : %i\n",
 580                             actions_len);
 581 
 582                 if (acth->type == htons(OFPAT_OUTPUT)) {
 583                         populate_action_output(action, entry, p);
 584                 } else if (acth->type == htons(OFPAT_SET_DL_SRC)) {
 585                         populate_action_set_dl_src(action, p);
 586                 } else if (acth->type == htons(OFPAT_SET_DL_DST)) {
 587                         populate_action_set_dl_dst(action, p);
 588                 } else if (acth->type == htons(OFPAT_SET_NW_SRC)) {
 589                         populate_action_set_nw_src(action, p);
 590                 } else if (acth->type == htons(OFPAT_SET_NW_DST)) {
 591                         populate_action_set_nw_dst(action, p);
 592                 } else if (acth->type == htons(OFPAT_SET_NW_TOS)) {
 593                         populate_action_set_nw_tos(action, p);
 594                 } else if (acth->type == htons(OFPAT_SET_TP_SRC)) {
 595                         populate_action_set_tp_src(action, p);
 596                 } else if (acth->type == htons(OFPAT_SET_TP_DST)) {
 597                         populate_action_set_tp_dst(action, p);
 598                 } else if (acth->type == htons(OFPAT_SET_VLAN_VID)) {
 599                         populate_action_set_vlan_vid(action, p);
 600                 } else if (acth->type == htons(OFPAT_SET_VLAN_PCP)) {
 601                         populate_action_set_vlan_pcp(action, p);
 602                 } else if (acth->type == htons(OFPAT_STRIP_VLAN)) {
 603                         populate_action_strip_vlan(action);
 604                 }
 605 
 606                 p += len;
 607                 actions_len -= len;
 608         }
 609 }

 611 /* Add a free hardware entry back to the exact pool. */
 612 void
 613 nf2_add_free_exact(struct nf2_flow *sfw)
 614 {
 615         // clear the node entry
 616         list_init(&sfw->node);
 617 
 618         // Critical section, adding to the actual list
 619         exact_free_list[sfw->pos] = sfw;
 620 }

 622 /* Add a free hardware entry back to the wildcard pool. */
 623 void
 624 nf2_add_free_wildcard(struct nf2_flow *sfw)
 625 {
 626         // clear the hw values
 627         sfw->hw_packet_count = 0;
 628         sfw->hw_byte_count = 0;
 629 
 630         // Critical section, adding to the actual list
 631         list_insert(&wildcard_free_list, &sfw->node);
 632 }

 634 /* Hashes the entry to find where it should exist in the exact table
 635  * returns NULL on failure
 636  */
 637 static struct nf2_flow *
 638 get_free_exact(nf2_of_entry_wrap *entry)
 639 {
 640         unsigned int poly1 = 0x04C11DB7;
 641         unsigned int poly2 = 0x1EDC6F41;
 642         struct nf2_flow *sfw = NULL;
 643         unsigned int hash = 0x0;
 644         unsigned int index = 0x0;
 645         struct crc32 crc;
 646 
 647         crc32_init(&crc, poly1);
 648         hash = crc32_calculate(&crc, entry, sizeof(nf2_of_entry_wrap));
 649 
 650         // the bottom 15 bits of hash == the index into the table
 651         index = 0x7FFF & hash;
 652 
 653         // if this index is free, grab it
 654         sfw = exact_free_list[index];
 655         exact_free_list[index] = NULL;
 656 
 657         if (sfw != NULL) {
 658                 return sfw;
 659         }
 660         // try the second index
 661         crc32_init(&crc, poly2);
 662         hash = crc32_calculate(&crc, entry, sizeof(nf2_of_entry_wrap));
 663         // the bottom 15 bits of hash == the index into the table
 664         index = 0x7FFF & hash;
 665 
 666         // if this index is free, grab it
 667         sfw = exact_free_list[index];
 668         exact_free_list[index] = NULL;
 669 
 670         // return whether its good or not
 671         return sfw;
 672 }

 674 /* Get the first free position in the wildcard hardware table
 675  * to write into.
 676  */
 677 static struct nf2_flow *
 678 get_free_wildcard(void)
 679 {
 680         struct nf2_flow *sfw = NULL;
 681 
 682         // Critical section, pulling the first available from the list
 683         if (list_is_empty(&wildcard_free_list)) {
 684                 // empty :(
 685                 sfw = NULL;
 686         } else {
 687                 sfw = CONTAINER_OF(list_front(&wildcard_free_list), struct nf2_flow, node);
 688                 list_remove(&sfw->node);
 689                 list_init(&sfw->node);
 690         }
 691 
 692         return sfw;
 693 }

 695 /* Retrieves the type of table this flow should go into. */
 696 int
 697 nf2_get_table_type(struct sw_flow *flow)
 698 {
 699         if (flow->key.wildcards != 0) {
 700                 DBG_VERBOSE("--- TABLE TYPE: WILDCARD ---\n");
 701                 return NF2_TABLE_WILDCARD;
 702         } else {
 703                 DBG_VERBOSE("--- TABLE TYPE: EXACT ---\n");
 704                 return NF2_TABLE_EXACT;
 705         }
 706 }

 708 /* Returns 1 if this flow contains an action outputting to all ports except
 709  * input port, 0 otherwise. We support OFPP_ALL and OFPP_FLOOD actions, however
 710  * since we do not perform the spanning tree protocol (STP) then OFPP_FLOOD is
 711  * equivalent to OFPP_ALL.
 712  */
 713 static int
 714 is_action_forward_all(struct sw_flow *flow)
 715 {
 716         struct sw_flow_actions *sfa;
 717         size_t actions_len;
 718         uint8_t *p;
 719 
 720         sfa = flow->sf_acts;
 721         actions_len = sfa->actions_len;
 722         p = (uint8_t *)sfa->actions;
 723 
 724         while (actions_len > 0) {
 725                 struct ofp_action_header *ah = (struct ofp_action_header *)p;
 726                 struct ofp_action_output *oa = (struct ofp_action_output *)p;
 727                 size_t len = ntohs(ah->len);
 728 
 729                 DBG_VERBOSE("Fwd Action Chk: Action type: %x\n",
 730                             ntohs(ah->type));
 731                 DBG_VERBOSE("Fwd Action Chk: Output port: %x\n",
 732                             ntohs(oa->port));
 733                 DBG_VERBOSE("Fwd Action Chk: Len of this action: %i\n", len);
 734                 DBG_VERBOSE("Fwd Action Chk: Len of actions    : %i\n",
 735                             actions_len);
 736                 // Currently only support the output port(s) action
 737                 if (ntohs(ah->type) == OFPAT_OUTPUT
 738                     && (ntohs(oa->port) == OFPP_ALL
 739                         || ntohs(oa->port) == OFPP_FLOOD)) {
 740                         return 1;
 741                 }
 742                 p += len;
 743                 actions_len -= len;
 744         }
 745 
 746         return 0;
 747 }

 749 /* Attempts to build and write the flow to hardware.
 750  * Returns 0 on success, 1 on failure.
 751  */
 752 int
 753 nf2_build_and_write_flow(struct sw_flow *flow)
 754 {
 755         struct nf2_flow *sfw = NULL;
 756         struct nf2_flow *sfw_next = NULL;
 757         struct nf2device *dev;
 758         int num_entries = 0;
 759         int i, table_type;
 760         nf2_of_entry_wrap key;
 761         nf2_of_mask_wrap mask;
 762         nf2_of_action_wrap action;
 763 
 764         memset(&key, 0, sizeof(nf2_of_entry_wrap));
 765         memset(&mask, 0, sizeof(nf2_of_mask_wrap));
 766         memset(&action, 0, sizeof(nf2_of_action_wrap));
 767 
 768         dev = nf2_get_net_device();
 769         if (dev == NULL) {
 770                 return 1;
 771         }
 772 
 773         table_type = nf2_get_table_type(flow);
 774         switch (table_type) {
 775         default:
 776                 break;
 777 
 778         case NF2_TABLE_EXACT:
 779                 DBG_VERBOSE("---Exact Entry---\n");
 780                 nf2_populate_of_entry(&key, flow);
 781                 nf2_populate_of_action(&action, &key, flow);
 782                 sfw = get_free_exact(&key);
 783                 if (sfw == NULL) {
 784                         DBG_VERBOSE
 785                                 ("Collision getting free exact match entry\n");
 786                         // collision
 787                         nf2_free_net_device(dev);
 788                         return 1;
 789                 }
 790                 // set the active bit on this entry
 791                 key.entry.pad = 0x80;
 792                 nf2_write_of_exact(dev, sfw->pos, &key, &action);
 793                 flow->private = (void *)sfw;
 794                 break;
 795 
 796         case NF2_TABLE_WILDCARD:
 797                 DBG_VERBOSE("---Wildcard Entry---\n");
 798                 // if action is all out and source port is wildcarded
 799                 if ((is_action_forward_all(flow)) &&
 800                     (flow->key.wildcards & OFPFW_IN_PORT)) {
 801                         DBG_VERBOSE("Grab four wildcard tables\n");
 802                         if (!(sfw = get_free_wildcard())) {
 803                                 DBG_VERBOSE("No free wildcard entries found.");
 804                                 // no free entries
 805                                 nf2_free_net_device(dev);
 806                                 return 1;
 807                         }
 808                         // try to get 3 more positions
 809                         for (i = 0; i < 3; ++i) {
 810                                 if (!(sfw_next = get_free_wildcard())) {
 811                                         break;
 812                                 }
 813                                 list_insert(&sfw->node, &sfw_next->node);
 814                                 ++num_entries;
 815                         }
 816 
 817                         if (num_entries < 3) {
 818                                 // failed to get enough entries, return them and exit
 819                                 nf2_delete_private((void *)sfw);
 820                                 nf2_free_net_device(dev);
 821                                 return 1;
 822                         }
 823 
 824                         nf2_populate_of_entry(&key, flow);
 825                         nf2_populate_of_mask(&mask, flow);
 826 
 827                         // set first entry's src port to 0, remove wildcard mask on src
 828                         key.entry.src_port = 0;
 829                         mask.entry.src_port = 0;
 830                         nf2_populate_of_action(&action, &key, flow);
 831                         nf2_write_of_wildcard(dev, sfw->pos, &key, &mask,
 832                                               &action);
 833 
 834                         i = 1;
 835                         sfw_next = CONTAINER_OF(list_front(&sfw->node),
 836                                               struct nf2_flow, node);
 837                         // walk through and write the remaining 3 entries
 838                         while (sfw_next != sfw) {
 839                                 key.entry.src_port = i * 2;
 840                                 nf2_populate_of_action(&action, &key, flow);
 841                                 nf2_write_of_wildcard(dev, sfw_next->pos, &key,
 842                                                       &mask, &action);
 843                                 sfw_next = CONTAINER_OF(list_front(&sfw_next->node),
 844                                                       struct nf2_flow, node);
 845                                 ++i;
 846                         }
 847                         flow->private = (void *)sfw;
 848                 } else {
 849                         /* Get a free position here, and write to it */
 850                         if ((sfw = get_free_wildcard())) {
 851                                 nf2_populate_of_entry(&key, flow);
 852                                 nf2_populate_of_mask(&mask, flow);
 853                                 nf2_populate_of_action(&action, &key, flow);
 854                                 if (nf2_write_of_wildcard
 855                                     (dev, sfw->pos, &key, &mask, &action)) {
 856                                         // failure writing to hardware
 857                                         nf2_add_free_wildcard(sfw);
 858                                         DBG_VERBOSE
 859                                                 ("Failure writing to hardware\n");
 860                                         nf2_free_net_device(dev);
 861                                         return 1;
 862                                 } else {
 863                                         // success writing to hardware, store the position
 864                                         flow->private = (void *)sfw;
 865                                 }
 866                         } else {
 867                                 // hardware is full, return 0
 868                                 DBG_VERBOSE("No free wildcard entries found.");
 869                                 nf2_free_net_device(dev);
 870                                 return 1;
 871                         }
 872                 }
 873                 break;
 874         }
 875 
 876         nf2_free_net_device(dev);
 877         return 0;
 878 }
 879 
 880 void
 881 nf2_delete_private(void *private)
 882 {
 883         struct nf2_flow *sfw = (struct nf2_flow *)private;
 884         struct nf2_flow *sfw_next;
 885         struct list *next;
 886 
 887         switch (sfw->type) {
 888         default:
 889                 break;
 890 
 891         case NF2_TABLE_EXACT:
 892                 nf2_clear_of_exact(sfw->pos);
 893                 nf2_add_free_exact(sfw);
 894                 break;
 895 
 896         case NF2_TABLE_WILDCARD:
 897                 while (!list_is_empty(&sfw->node)) {
 898                         next = sfw->node.next;
 899                         sfw_next = CONTAINER_OF(list_front(&sfw->node), struct nf2_flow, node);
 900                         list_remove(&sfw_next->node);
 901                         list_init(&sfw_next->node);
 902                         // Immediately zero out the entry in hardware
 903                         nf2_clear_of_wildcard(sfw_next->pos);
 904                         // add it back to the pool
 905                         nf2_add_free_wildcard(sfw_next);
 906                 }
 907                 // zero the core entry
 908                 nf2_clear_of_wildcard(sfw->pos);
 909                 // add back the core entry
 910                 nf2_add_free_wildcard(sfw);
 911                 break;
 912         }
 913 }

 915 int
 916 nf2_modify_acts(struct sw_flow *flow)
 917 {
 918         struct nf2_flow *sfw = (struct nf2_flow *)flow->private;
 919         struct nf2device *dev;
 920         nf2_of_entry_wrap key;
 921         nf2_of_mask_wrap mask;
 922         nf2_of_action_wrap action;
 923 
 924         memset(&key, 0, sizeof(nf2_of_entry_wrap));
 925         memset(&mask, 0, sizeof(nf2_of_mask_wrap));
 926         memset(&action, 0, sizeof(nf2_of_action_wrap));
 927 
 928         dev = nf2_get_net_device();
 929         if (dev == NULL)
 930                 return 0;
 931 
 932         switch (sfw->type) {
 933         default:
 934                 break;
 935 
 936         case NF2_TABLE_EXACT:
 937                 nf2_populate_of_entry(&key, flow);
 938                 nf2_populate_of_action(&action, &key, flow);
 939                 key.entry.pad = 0x80;
 940                 nf2_modify_write_of_exact(dev, sfw->pos, &action);
 941                 break;
 942 
 943         case NF2_TABLE_WILDCARD:
 944                 if (flow->key.wildcards & OFPFW_IN_PORT) {
 945                         nf2_free_net_device(dev);
 946                         return 0;
 947                 }
 948                 nf2_populate_of_entry(&key, flow);
 949                 nf2_populate_of_mask(&mask, flow);
 950                 nf2_populate_of_action(&action, &key, flow);
 951                 nf2_modify_write_of_wildcard(dev, sfw->pos,
 952                                              &key, &mask, &action);
 953                 break;
 954         }
 955 
 956         nf2_free_net_device(dev);
 957         return 1;
 958 }

 960 uint64_t
 961 nf2_get_packet_count(struct nf2device *dev, struct nf2_flow *sfw)
 962 {
 963         uint32_t count = 0;
 964         uint32_t hw_count = 0;
 965         uint64_t total = 0;
 966         struct nf2_flow *sfw_next = NULL;
 967 
 968         switch (sfw->type) {
 969         default:
 970                 break;
 971 
 972         case NF2_TABLE_EXACT:
 973                 // Get delta value
 974                 total = nf2_get_exact_packet_count(dev, sfw->pos);
 975                 break;
 976 
 977         case NF2_TABLE_WILDCARD:
 978                 sfw_next = sfw;
 979                 do {
 980                         // Get sum value
 981                         hw_count = nf2_get_wildcard_packet_count(dev,
 982                                                                  sfw_next->pos);
 983                         if (hw_count >= sfw_next->hw_packet_count) {
 984                                 count = hw_count - sfw_next->hw_packet_count;
 985                                 sfw_next->hw_packet_count = hw_count;
 986                         } else {
 987                                 // wrapping occurred
 988                                 count = (MAX_INT_32 - sfw_next->hw_packet_count)
 989                                         + hw_count;
 990                                 sfw_next->hw_packet_count = hw_count;
 991                         }
 992                         total += count;
 993 
 994                         if(!list_is_empty(&sfw_next->node)){
 995                                 sfw_next = CONTAINER_OF(list_front(&sfw_next->node),
 996                                                       struct nf2_flow, node);
 997                         }
 998                 } while (sfw_next != sfw);
 999                 break;
1000         }
1001 
1002         return total;
1003 }
