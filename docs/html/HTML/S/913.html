 132 /* Inserts 'node', with the given 'hash', into 'hmap'.  'hmap' is never
 133  * expanded automatically. */
 134 static inline void
 135 hmap_insert_fast(struct hmap *hmap, struct hmap_node *node, size_t hash)
 136 {
 137     struct hmap_node **bucket = &hmap->buckets[hash & hmap->mask];
 138     node->hash = hash;
 139     node->next = *bucket;
 140     *bucket = node;
 141     hmap->n++;
 142 }
 143 
 144 /* Inserts 'node', with the given 'hash', into 'hmap', and expands 'hmap' if
 145  * necessary to optimize search performance. */
 146 static inline void
 147 hmap_insert(struct hmap *hmap, struct hmap_node *node, size_t hash)
 148 {
 149     hmap_insert_fast(hmap, node, hash);
 150     if (hmap->n / 2 > hmap->mask) {
 151         hmap_expand(hmap);
 152     }
 153 }
 154 
 155 /* Removes 'node' from 'hmap'.  Does not shrink the hash table; call
 156  * hmap_shrink() directly if desired. */
 157 static inline void
 158 hmap_remove(struct hmap *hmap, struct hmap_node *node)
 159 {
 160     struct hmap_node **bucket = &hmap->buckets[node->hash & hmap->mask];
 161     while (*bucket != node) {
 162         bucket = &(*bucket)->next;
 163     }
 164     *bucket = node->next;
 165     hmap->n--;
 166 }

 168 static inline struct hmap_node *
 169 hmap_next_with_hash__(const struct hmap_node *node, size_t hash)
 170 {
 171     while (node != NULL && node->hash != hash) {
 172         node = node->next;
 173     }
 174     return (struct hmap_node *) node;
 175 }

 177 /* Returns the first node in 'hmap' with the given 'hash', or a null pointer if
 178  * no nodes have that hash value. */
 179 static inline struct hmap_node *
 180 hmap_first_with_hash(const struct hmap *hmap, size_t hash)
 181 {
 182     return hmap_next_with_hash__(hmap->buckets[hash & hmap->mask], hash);
 183 }

 185 /* Returns the next node in the same hash map as 'node' with the same hash
 186  * value, or a null pointer if no more nodes have that hash value.
 187  *
 188  * If the hash map has been reallocated since 'node' was visited, some nodes
 189  * may be skipped; if new nodes with the same hash value have been added, they
 190  * will be skipped.  (Removing 'node' from the hash map does not prevent
 191  * calling this function, since node->next is preserved, although freeing
 192  * 'node' of course does.) */
 193 static inline struct hmap_node *
 194 hmap_next_with_hash(const struct hmap_node *node)
 195 {
 196     return hmap_next_with_hash__(node->next, node->hash);
 197 }
 198 
 199 static inline struct hmap_node *
 200 hmap_next__(const struct hmap *hmap, size_t start)
 201 {
 202     size_t i;
 203     for (i = start; i <= hmap->mask; i++) {
 204         struct hmap_node *node = hmap->buckets[i];
 205         if (node) {
 206             return node;
 207         }
 208     }
 209     return NULL;
 210 }

 220 /* Returns the next node in 'hmap' following 'node', in arbitrary order, or a
 221  * null pointer if 'node' is the last node in 'hmap'.
 222  *
 223  * If the hash map has been reallocated since 'node' was visited, some nodes
 224  * may be skipped or visited twice.  (Removing 'node' from the hash map does
 225  * not prevent calling this function, since node->next is preserved, although
 226  * freeing 'node' of course does.) */
 227 static inline struct hmap_node *
 228 hmap_next(const struct hmap *hmap, const struct hmap_node *node)
 229 {
 230     return (node->next
 231             ? node->next
 232             : hmap_next__(hmap, (node->hash & hmap->mask) + 1));
 233 }
