 211 /*
 212  * Check the iface name to make sure we can find the interface
 213  */
 214 int check_iface(struct nf2device *nf2)
 215 {
 216         struct stat buf;
 217         char filename[PATHLEN];
 218 
 219         /* See if we can find the interface name as a network device */
 220 
 221         /* Test the length first of all */
 222         if (strlen(nf2->device_name) > IFNAMSIZ)
 223         {
 224                 fprintf(stderr, "Interface name is too long: %s\n", nf2->device_name);
 225                 return -1;
 226         }
 227 
 228         /* Check for /sys/class/net/iface_name */
 229         strcpy(filename, "/sys/class/net/");
 230         strcat(filename, nf2->device_name);
 231         if (stat(filename, &buf) == 0)
 232         {
 233                 fprintf(stderr, "Found net device: %s\n", nf2->device_name);
 234                 nf2->net_iface = 1;
 235                 return 0;
 236         }
 237 
 238         /* Check for /dev/iface_name */
 239         strcpy(filename, "/dev/");
 240         strcat(filename, nf2->device_name);
 241         if (stat(filename, &buf) == 0)
 242         {
 243                 fprintf(stderr, "Found dev device: %s\n", nf2->device_name);
 244                 nf2->net_iface = 0;
 245                 return 0;
 246         }
 247 
 248         fprintf(stderr, "Can't find device: %s\n", nf2->device_name);
 249         return -1;
 250 }

 253  * Open the descriptor associated with the device name
 254  */
 255 int openDescriptor(struct nf2device *nf2)
 256 {
 257         struct ifreq ifreq;
 258         char filename[PATHLEN];
 259         struct sockaddr_in address;
 260         int i;
 261         struct sockaddr_in *sin = (struct sockaddr_in *) &ifreq.ifr_addr;
 262         int found = 0;
 263 
 264         if (nf2->net_iface)
 265         {
 266                 /* Open a network socket */
 267                 nf2->fd = socket(AF_INET, SOCK_DGRAM, 0);
 268                 if (nf2->fd == -1)
 269                 {
 270                         perror("socket: creating socket");
 271                         return -1;
 272                 }
 273                 else
 274                 {
 275                         /* Root can bind to a network interface.
 276                            Non-root has to bind to a network address. */
 277                         if (geteuid() == 0)
 278                         {
 279                                 strncpy(ifreq.ifr_ifrn.ifrn_name, nf2->device_name, IFNAMSIZ);
 280                                 if (setsockopt(nf2->fd, SOL_SOCKET, SO_BINDTODEVICE,
 281                                         (char *)&ifreq, sizeof(ifreq)) < 0) {
 282                                         perror("setsockopt: setting SO_BINDTODEVICE");
 283                                         return -1;
 284                                 }
 285 
 286                         }
 287                 }
 288         }
 289         else
 290         {
 291                 strcpy(filename, "/dev/");
 292                 strcat(filename, nf2->device_name);
 293                 nf2->fd = fileno(fopen(filename, "w+"));
 294                 if (nf2->fd == -1)
 295                 {
 296                         perror("fileno: creating descriptor");
 297                         return -1;
 298                 }
 299         }
 300 
 301         return 0;
 302 }

 325 /*
 326  *  Read the version info from the Virtex and CPCI
 327  */
 328 void nf2_read_info(struct nf2device *nf2)
 329 {
 330         int i;
 331         int md5_good_v1 = 1;
 332         int md5_good_v2 = 1;
 333         unsigned md5[MD5_LEN];
 334         unsigned cpci_id;
 335         unsigned nf2_cpci_id;
 336 
 337         // Copy the device info
 338         prev_dev = *nf2;
 339 
 340         // Read the CPCI version/revision
 341         readReg(nf2, CPCI_ID_REG, &cpci_id);
 342         cpci_version = cpci_id & 0xffffff;
 343         cpci_revision = cpci_id >> 24;
 344         have_version_info = 1;
 345 
 346         // Check if the Virtex is programmed
 347         virtex_programmed = isVirtexProgrammed(nf2);
 348 
 349         // Clear the Virtex-related variables
 350         nf2_dev_id_module_version = -1;
 351         nf2device_id = -1;
 352         nf2_version = -1;
 353         nf2_cpci_version = -1;
 354         nf2_cpci_revision = -1;
 355         nf2_proj_name_v1[0] = '\0';
 356         nf2_proj_dir[0] = '\0';
 357         nf2_proj_name[0] = '\0';
 358         nf2_proj_desc[0] = '\0';
 359 
 360         // Verify the MD5 checksum of the device ID block
 361         for  (i = 0; i < MD5_LEN; i++) {
 362                 readReg(nf2, DEV_ID_MD5_0_REG + i * 4, &md5[i]);
 363         }
 364 
 365         md5_good_v1 &= md5[0] == DEV_ID_MD5_VALUE_V1_0;
 366         md5_good_v1 &= md5[1] == DEV_ID_MD5_VALUE_V1_1;
 367         md5_good_v1 &= md5[2] == DEV_ID_MD5_VALUE_V1_2;
 368         md5_good_v1 &= md5[3] == DEV_ID_MD5_VALUE_V1_3;
 369 
 370         md5_good_v2 &= md5[0] == DEV_ID_MD5_VALUE_V2_0;
 371         md5_good_v2 &= md5[1] == DEV_ID_MD5_VALUE_V2_1;
 372         md5_good_v2 &= md5[2] == DEV_ID_MD5_VALUE_V2_2;
 373         md5_good_v2 &= md5[3] == DEV_ID_MD5_VALUE_V2_3;
 374 
 375         // Process only if the MD5 sum is good
 376         if (md5_good_v1 || md5_good_v2) {
 377                 // Read the version and revision
 378                 readReg(nf2, DEV_ID_DEVICE_ID_REG, &nf2device_id);
 379                 readReg(nf2, DEV_ID_VERSION_REG, &nf2_version);
 380                 readReg(nf2, DEV_ID_CPCI_ID_REG, &nf2_cpci_id);
 381                 nf2_cpci_version = nf2_cpci_id & 0xffffff;
 382                 nf2_cpci_revision = nf2_cpci_id >> 24;
 383         }
 384 
 385         if (md5_good_v1) {
 386                 nf2_dev_id_module_version = 1;
 387                 readStr(nf2, DEV_ID_PROJ_DIR_0_REG, DEV_ID_PROJ_NAME_BYTE_LEN_V1, nf2_proj_name_v1);
 388         }
 389 
 390         if (md5_good_v2) {
 391                 nf2_dev_id_module_version = 2;
 392                 readStr(nf2, DEV_ID_PROJ_DIR_0_REG, DEV_ID_PROJ_DIR_BYTE_LEN, nf2_proj_dir);
 393                 readStr(nf2, DEV_ID_PROJ_NAME_0_REG, DEV_ID_PROJ_NAME_BYTE_LEN, nf2_proj_name);
 394                 readStr(nf2, DEV_ID_PROJ_DESC_0_REG, DEV_ID_PROJ_DESC_BYTE_LEN, nf2_proj_desc);
 395         }
 396 }

 398 /*
 399  * Read a string from the NetFPGA
 400  *
 401  * Ensure that the string is null-terminated
 402  *
 403  * Note: the length *must* be an integral number of words or the
 404  * final bytes will be truncated.
 405  */
 406 static void readStr(struct nf2device *nf2, unsigned regStart, unsigned len, char *dst)
 407 {
 408         int i;
 409 
 410         // Read the string
 411         for (i = 0; i < len / 4; i++)
 412         {
 413                 readReg(nf2, regStart + i * 4, (unsigned *)(dst + i * 4));
 414 
 415                 // Perform byte swapping if necessary
 416                 *(unsigned *)(dst + i * 4) = ntohl(*(unsigned *)(dst + i * 4));
 417         }
 418 
 419         // Ensure that the string is null-terminated
 420         dst[(len / 4) * 4 - 1] = '\0';
 421         //dst[len - 1] = '\0';
 422 }

 651 /*
 652  * Check if a particular bitfile is downloaded and whether it's the correct version.
 653  */
 654 int checkVirtexBitfile(struct nf2device *nf2, char *projDir,
 655     int minVerMajor, int minVerMinor, int minVerRev,
 656     int maxVerMajor, int maxVerMinor, int maxVerRev) {
 657 
 658     int minVer, maxVer, virtexVer;
 659 
 660     // Ensure that we've read the device info
 661     prepDeviceInfo(nf2);
 662 
 663     // Check if the Virtex is programmed
 664     if (!virtex_programmed) {
 665         sprintf(nf2_version_err, "Error: Virtex is not programmed");
 666         return 0;
 667     }
 668 
 669     // Calculate the version numbers as necessary
 670     minVer = 0;
 671     if (minVerMajor != VERSION_ANY) {
 672         minVer += minVerMajor;
 673     }
 674     minVer <<= 8;
 675     if (minVerMinor != VERSION_ANY) {
 676         minVer += minVerMinor;
 677     }
 678     minVer <<= 8;
 679     if (minVerRev != VERSION_ANY) {
 680         minVer += minVerRev;
 681     }
 682 
 683     maxVer = 0;
 684     maxVer += (maxVerMajor != VERSION_ANY) ? maxVerMajor : 255;
 685     maxVer <<= 8;
 686     maxVer += (maxVerMinor != VERSION_ANY) ? maxVerMinor : 255;
 687     maxVer <<= 8;
 688     maxVer += (maxVerRev != VERSION_ANY) ? maxVerRev : 255;
 689 
 690     virtexVer = (getDeviceMajor(nf2) << 16) |
 691                  (getDeviceMinor(nf2) << 8) |
 692                  getDeviceRevision(nf2);
 693 
 694     // Check the device name
 695     const char *virtexProjDir;
 696     const char *virtexProjName;
 697     if (nf2_dev_id_module_version >= 2) {
 698         virtexProjDir = getProjDir(nf2);
 699         virtexProjName = getProjName(nf2);
 700     }
 701     else {
 702         virtexProjDir = getProjName(nf2);
 703         virtexProjName = PROJ_UNKNOWN;
 704     }
 705 
 706     if (strcmp(virtexProjDir, projDir) != 0) {
 707         sprintf(nf2_version_err, "Error: Incorrect bitfile loaded. Found '%s' (%s), expecting: '%s'",
 708                 virtexProjDir, virtexProjName, projDir);
 709         return 0;
 710     }
 711 
 712     // Check the version number
 713     if (virtexVer < minVer || virtexVer > maxVer) {
 714         // Work out equality etc
 715         int hasMin = (minVerMajor != VERSION_ANY) ||
 716                      (minVerMinor != VERSION_ANY) ||
 717                      (minVerRev != VERSION_ANY);
 718         int hasMax = (maxVerMajor != VERSION_ANY) ||
 719                      (maxVerMinor != VERSION_ANY) ||
 720                      (maxVerRev != VERSION_ANY);
 721         int verMajorEqual = ((minVerMajor != VERSION_ANY) ? minVerMajor : -1) ==
 722                             ((maxVerMajor != VERSION_ANY) ? maxVerMajor : -1);
 723         int verMinorEqual = ((minVerMinor != VERSION_ANY) ? minVerMinor : -1) ==
 724                             ((maxVerMinor != VERSION_ANY) ? maxVerMinor : -1);
 725         int verRevEqual = ((minVerRev != VERSION_ANY) ? minVerRev : -1) ==
 726                           ((maxVerRev != VERSION_ANY) ? maxVerRev : -1);
 727         int minMaxEqual = verMajorEqual && verMinorEqual && verRevEqual;
 728 
 729         // Generate strings for min and max
 730         char minStr[MAX_VER_LEN];
 731         if (minVerRev != VERSION_ANY)
 732             sprintf(minStr, "%d.%d.%d", minVerMajor, minVerMinor, minVerRev);
 733         else if (minVerMinor != VERSION_ANY)
 734             sprintf(minStr, "%d.%d.x", minVerMajor, minVerMinor);
 735         else if (minVerMajor != VERSION_ANY)
 736             sprintf(minStr, "%d.x.x", minVerMajor);
 737         else
 738             sprintf(minStr, "x.x.x");
 739 
 740         char maxStr[MAX_VER_LEN];
 741         if (maxVerRev != VERSION_ANY)
 742             sprintf(maxStr, "%d.%d.%d", maxVerMajor, maxVerMinor, maxVerRev);
 743         else if (maxVerMinor != VERSION_ANY)
 744             sprintf(maxStr, "%d.%d.x", maxVerMajor, maxVerMinor);
 745         else if (maxVerMajor != VERSION_ANY)
 746             sprintf(maxStr, "%d.x.x", maxVerMajor);
 747         else
 748             sprintf(maxStr, "x.x.x");
 749 
 750         char virtexVerStr[MAX_VER_LEN];
 751         sprintf(virtexVerStr, "%d.%d.%d",
 752                 getDeviceMajor(nf2),
 753                 getDeviceMinor(nf2),
 754                 getDeviceRevision(nf2));
 755 
 756         if (minMaxEqual)
 757             sprintf(nf2_version_err,
 758                     "Error: Incorrect version for bitfile: '%s' (%s).  Expecting: %s   Active: %s",
 759                     projDir, virtexProjName, minStr, virtexVerStr);
 760         else if (hasMin && hasMax)
 761             sprintf(nf2_version_err,
 762                     "Error: Incorrect version for bitfile: '%s' (%s).  Expecting: %s -- %s   Active: %s",
 763                     projDir, virtexProjName, minStr, maxStr, virtexVerStr);
 764         else if (hasMin)
 765             sprintf(nf2_version_err,
 766                     "Error: Incorrect version for bitfile: '%s' (%s).  Expecting: > %s   Active: %s",
 767                     projDir, virtexProjName, minStr, virtexVerStr);
 768         else
 769             sprintf(nf2_version_err,
 770                     "Error: Incorrect version for bitfile: '%s' (%s).  Expecting: < %s   Active: %s",
 771                     projDir, virtexProjName, maxStr, virtexVerStr);
 772         return 0;
 773     }
 774 
 775     sprintf(nf2_version_err, "GOOD");
 776     return 1;
 777 }
