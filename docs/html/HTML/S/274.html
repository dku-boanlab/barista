  64 static int
  65 disp_dev_info(char *page)
  66 {
  67         struct net_device *netdev;
  68         struct nf2_device_info *nf2devinfo;
  69         int len = 0;
  70 
  71         netdev = nf2_get_net_device();
  72         if (netdev == NULL)
  73                 return 0;
  74         nf2devinfo = nf2_get_device_info(netdev);
  75         if (nf2devinfo == NULL) {
  76                 nf2_free_net_device(netdev);
  77                 return 0;
  78         }
  79         nf2_free_net_device(netdev);
  80 
  81         len += sprintf(page + len,
  82                        "NetFPGA: "
  83                        "design name: %s, device ID: %d, device revision: %d\n",
  84                        nf2devinfo->nf2_device_str,
  85                        nf2devinfo->nf2_device_id, nf2devinfo->nf2_device_rev);
  86         len += sprintf(page + len, "\n");
  87 
  88         return len;

 127                                "  Output queue: %u/%u (current/queued)\n",

 131                                "  %u packets output, "

 135                                nf2portinfo->port[i].tx_q_num_bytes_pushed);

 199 static int
 200 proc_read(char *page, char **start, off_t offset, int count, int *eof,
 201           void *data)
 202 {
 203         int len = 0;
 204         int buf_pos = 1;
 205 
 206         if (down_interruptible(&proc_sem))
 207                 return -ERESTARTSYS;
 208 
 209         if (buf_pos != 0) {
 210                 len += disp_dev_info(page + len);
 211                 len += disp_port_info(page + len);
 212                 len += disp_match_info(page + len);
 213                 len += disp_watchdog_info(page + len);
 214                 buf_pos = 0;
 215         }
 216         up(&proc_sem);
 217 
 218         *eof = 1;
 219         return len;
 220 }

 222 void
 223 nf2_create_procfs(void)
 224 {
 225         struct proc_dir_entry *entry;
 226 
 227         entry = create_proc_entry(NF2_PROCFS_NAME,
 228                                   S_IFREG | S_IRUGO | S_IWUGO, NULL);
 229         if (entry == NULL)
 230                 return;
 231 
 232         entry->read_proc = proc_read;
 233         sema_init(&proc_sem, 1);
 234 }
