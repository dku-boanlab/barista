  34 /* Individual switching tables.  Generally grouped together in a chain (see
  35  * chain.h). */

  40 #include <stddef.h>

  43 struct datapath; /* Forward declaration for delete operation */
  44 struct sw_flow;
  45 struct sw_flow_key;
  46 struct ofp_action_header;
  47 struct list;

  49 /* Table statistics. */
  50 struct sw_table_stats {
  51     const char *name;            /* Human-readable name. */
  52     uint32_t wildcards;          /* Bitmap of OFPFW_* wildcards that are
  53                                     supported by the table. */
  54     unsigned int n_flows;        /* Number of active flows. */
  55     unsigned int max_flows;      /* Flow capacity. */
  56     unsigned long int n_lookup;  /* Number of packets looked up. */
  57     unsigned long int n_matched; /* Number of packets that have hit. */
  58 };
  59 
  60 /* Position within an iteration of a sw_table.
  61  *
  62  * The contents are private to the table implementation, except that a position
  63  * initialized to all-zero-bits represents the start of a table. */
  64 struct sw_table_position {
  65     unsigned long private[4];
  66 };
  67 
  68 /* A single table of flows.  */
  69 struct sw_table {
  70     /* The number of packets that have been looked up and matched,
  71      * respecitvely.  To make these 100% accurate, they should be atomic.  
  72      * However, we're primarily concerned about speed. */
  73     unsigned long long n_lookup;
  74     unsigned long long n_matched;
  75 
  76     /* Searches 'table' for a flow matching 'key', which must not have any
  77      * wildcard fields.  Returns the flow if successful, a null pointer
  78      * otherwise. */
  79     struct sw_flow *(*lookup)(struct sw_table *table,
  80                               const struct sw_flow_key *key);
  81 
  82     /* Inserts 'flow' into 'table', replacing any duplicate flow.  Returns
  83      * 0 if successful or a negative error.  Error can be due to an
  84      * over-capacity table or because the flow is not one of the kind that
  85      * the table accepts.
  86      *
  87      * If successful, 'flow' becomes owned by 'table', otherwise it is
  88      * retained by the caller. */
  89     int (*insert)(struct sw_table *table, struct sw_flow *flow);
  90 
  91     /* Modifies the actions in 'table' that match 'key'.  If 'strict'
  92      * set, wildcards and priority must match.  Returns the number of flows
  93      * that were modified. */
  94     int (*modify)(struct sw_table *table, const struct sw_flow_key *key,
  95             uint16_t priority, int strict,
  96             const struct ofp_action_header *actions, size_t actions_len);
  97 
  98     /* Checks whether 'table' has an equal priotiry  entry of thethat conflicts
  99      * with 'key'.  If 'strict' is set, wildcards must match.
 100      * Returns the number of flows that were modified. */
 101     int (*has_conflict)(struct sw_table *table, const struct sw_flow_key *key,
 102                         uint16_t priority, int strict);
 103 
 104     /* Deletes from 'table' any and all flows that match 'key' from
 105      * 'table'.  If 'out_port' is not OFPP_NONE, then matching entries
 106      * must have that port as an argument for an output action.  If 
 107      * 'strict' is set, wildcards and priority must match.  Returns the
 108      * number of flows that were deleted. */
 109     int (*delete)(struct datapath *dp, struct sw_table *table, 
 110                   const struct sw_flow_key *key, 
 111                   uint16_t out_port, uint16_t priority, int strict);
 112 
 113     /* Performs timeout processing on all the flow entries in 'table'.
 114      * Appends all the flow entries removed from 'table' to 'deleted' for the
 115      * caller to free. */
 116     void (*timeout)(struct sw_table *table, struct list *deleted);
 117 
 118     /* Destroys 'table', which must not have any users. */
 119     void (*destroy)(struct sw_table *table);
 120 
 121     /* Iterates through the flow entries in 'table', passing each one
 122      * matches 'key' and output port 'out_port' to 'callback'.  The 
 123      * callback function should return 0 to continue iteration or a 
 124      * nonzero error code to stop.  The iterator function returns either 
 125      * 0 if the table iteration completed or the value returned by the 
 126      * callback function otherwise.
 127      *
 128      * The iteration starts at 'position', which may be initialized to
 129      * all-zero-bits to iterate from the beginning of the table.  If the
 130      * iteration terminates due to an error from the callback function,
 131      * 'position' is updated to a value that can be passed back to the
 132      * iterator function to resume iteration later with the following
 133      * flow. */
 134     int (*iterate)(struct sw_table *table,
 135                const struct sw_flow_key *key, uint16_t out_port,
 136                struct sw_table_position *position,
 137                int (*callback)(struct sw_flow *flow, void *private),
 138                void *private);
 139 
 140     /* Dumps statistics for 'table' into 'stats'. */
 141     void (*stats)(struct sw_table *table, struct sw_table_stats *stats);
 142 };
