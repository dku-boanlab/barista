  28 /*
  29  * Type of time-stamp shown in the summary display.

  31 typedef enum {
  32         TS_RELATIVE,            /* Since start of capture */
  33         TS_ABSOLUTE,
  34         TS_ABSOLUTE_WITH_DATE,
  35         TS_DELTA,               /* Since previous captured packet */
  36         TS_DELTA_DIS,           /* Since previous displayed packet */
  37         TS_EPOCH,               /* Seconds (and fractions) since epoch */
  38 
  39 /*
  40  * Special value used for the command-line setting in Wireshark, to indicate

  53         TS_PREC_FIXED_NSEC,     /* recent and internal */
  54         TS_PREC_AUTO_SEC,       /* internal */
  55         TS_PREC_AUTO_DSEC,      /* internal */
  56         TS_PREC_AUTO_CSEC,      /* internal */
  57         TS_PREC_AUTO_MSEC,      /* internal */
  58         TS_PREC_AUTO_USEC,      /* internal */
  59         TS_PREC_AUTO_NSEC       /* internal */
  60 } ts_precision;
