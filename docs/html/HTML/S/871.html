  65 /* Initializes the signals subsystem (if it is not already initialized).  Calls
  66  * exit() if initialization fails.
  67  *
  68  * Calling this function is optional; it will be called automatically by
  69  * signal_start() if necessary.  Calling it explicitly allows the client to
  70  * prevent the process from exiting at an unexpected time. */
  71 void
  72 signal_init(void)
  73 {
  74     static bool inited;
  75     if (!inited) {
  76         inited = true;
  77         if (pipe(fds)) {
  78             ofp_fatal(errno, "could not create pipe");
  79         }
  80         set_nonblocking(fds[0]);
  81         set_nonblocking(fds[1]);
  82     }
  83 }

  85 /* Sets up a handler for 'signr' and returns a structure that represents it.
  86  *
  87  * Only one handler for a given signal may be registered at a time. */
  88 struct signal *
  89 signal_register(int signr)
  90 {
  91     struct sigaction sa;
  92     struct signal *s;
  93 
  94     signal_init();
  95 
  96     /* Set up signal handler. */
  97     assert(signr >= 1 && signr < N_SIGNALS);
  98     memset(&sa, 0, sizeof sa);
  99     sa.sa_handler = signal_handler;
 100     sigemptyset(&sa.sa_mask);
 101     sa.sa_flags = SA_RESTART;
 102     if (sigaction(signr, &sa, NULL)) {
 103         ofp_fatal(errno, "sigaction(%d) failed", signr);
 104     }
 105 
 106     /* Return structure. */
 107     s = xmalloc(sizeof *s);
 108     s->signr = signr;
 109     return s;
 110 }

 112 /* Returns true if signal 's' has been received since the last call to this
 113  * function with argument 's'. */
 114 bool
 115 signal_poll(struct signal *s)
 116 {
 117     char buf[_POSIX_PIPE_BUF];
 118     read(fds[0], buf, sizeof buf);
 119     if (signaled[s->signr]) {
 120         signaled[s->signr] = 0;
 121         return true;
 122     }
 123     return false;
 124 }
 125 
 126 /* Causes the next call to poll_block() to wake up when signal_poll(s) would
 127  * return true. */
 128 void
 129 signal_wait(struct signal *s)
 130 {
 131     if (signaled[s->signr]) {
 132         poll_immediate_wake();
 133     } else {
 134         poll_fd_wait(fds[0], POLLIN);
 135     }
 136 }

 138 static void
 139 signal_handler(int signr)
 140 {
 141     if (signr >= 1 && signr < N_SIGNALS) {
 142         write(fds[1], "", 1);
 143         signaled[signr] = true;
 144     }
