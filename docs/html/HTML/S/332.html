  27 /* Identification data for a flow.
  28  * Network byte order except for the "wildcards" field.
  29  * Ordered to make bytewise comparisons (e.g. with memcmp()) fail quickly and
  30  * to keep the amount of padding to a minimum.
  31  * If you change the ordering of fields here, change flow_keys_equal() to
  32  * compare the proper fields.
  33  */
  34 struct sw_flow_key {
  35         uint32_t nw_src;        /* IP source address. */
  36         uint32_t nw_dst;        /* IP destination address. */
  37         uint16_t in_port;       /* Input switch port */
  38         uint16_t dl_vlan;       /* Input VLAN id. */
  39         uint16_t dl_type;       /* Ethernet frame type. */
  40         uint16_t tp_src;        /* TCP/UDP source port. */
  41         uint16_t tp_dst;        /* TCP/UDP destination port. */
  42         uint8_t dl_src[ETH_ALEN]; /* Ethernet source address. */
  43         uint8_t dl_dst[ETH_ALEN]; /* Ethernet destination address. */
  44         uint8_t dl_vlan_pcp;    /* Input VLAN priority. */
  45         uint8_t nw_tos;         /* IPv4 DSCP */
  46         uint8_t nw_proto;       /* IP protocol. */
  47         uint8_t pad[3];
  48         uint32_t wildcards;     /* Wildcard fields (host byte order). */
  49         uint32_t nw_src_mask;   /* 1-bit in each significant nw_src bit. */
  50         uint32_t nw_dst_mask;   /* 1-bit in each significant nw_dst bit. */

  58 /* Compare two sw_flow_keys and return true if they are the same flow, false

 140 static inline int iphdr_ok(struct sk_buff *skb)
 141 {
 142         int nh_ofs = skb_network_offset(skb);
 143         if (skb->len >= nh_ofs + sizeof(struct iphdr)) {
 144                 int ip_len = ip_hdrlen(skb);
 145                 return (ip_len >= sizeof(struct iphdr)
 146                         && pskb_may_pull(skb, nh_ofs + ip_len));
 147         }
 148         return 0;
 149 }
 150 
 151 static inline int tcphdr_ok(struct sk_buff *skb)
 152 {
 153         int th_ofs = skb_transport_offset(skb);
 154         if (pskb_may_pull(skb, th_ofs + sizeof(struct tcphdr))) {
 155                 int tcp_len = tcp_hdrlen(skb);
 156                 return (tcp_len >= sizeof(struct tcphdr)
 157                         && skb->len >= th_ofs + tcp_len);
 158         }
 159         return 0;
 160 }

 182 static inline void flow_used(struct sw_flow *flow, struct sk_buff *skb) 
 183 {
 184         unsigned long flags;
 185 
 186         flow->used = get_jiffies_64();
 187 
 188         spin_lock_irqsave(&flow->lock, flags);
 189         flow->packet_count++;
 190         flow->byte_count += skb->len;
 191         spin_unlock_irqrestore(&flow->lock, flags);
 192 }
