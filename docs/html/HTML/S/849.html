  77  *
  78  * See the large comment in rconn.h for more information. */
  79 struct rconn {
  80     enum state state;
  81     time_t state_entered;
  82 
  83     struct vconn *vconn;
  84     char *name;
  85     bool reliable;
  86 
  87     struct ofp_queue txq;
  88 
  89     int backoff;
  90     int max_backoff;
  91     time_t backoff_deadline;
  92     time_t last_received;
  93     time_t last_connected;
  94     unsigned int packets_sent;
  95     unsigned int seqno;
  96 
  97     /* In S_ACTIVE and S_IDLE, probably_admitted reports whether we believe
  98      * that the peer has made a (positive) admission control decision on our
  99      * connection.  If we have not yet been (probably) admitted, then the
 100      * connection does not reset the timer used for deciding whether the switch
 101      * should go into fail-open mode.
 102      *
 103      * last_admitted reports the last time we believe such a positive admission
 104      * control decision was made. */
 105     bool probably_admitted;
 106     time_t last_admitted;
 107 
 108     /* These values are simply for statistics reporting, not used directly by
 109      * anything internal to the rconn (or the secchan for that matter). */
 110     unsigned int packets_received;
 111     unsigned int n_attempted_connections, n_successful_connections;
 112     time_t creation_time;
 113     unsigned long int total_time_connected;
 114 
 115     /* If we can't connect to the peer, it could be for any number of reasons.
 116      * Usually, one would assume it is because the peer is not running or
 117      * because the network is partitioned.  But it could also be because the
 118      * network topology has changed, in which case the upper layer will need to
 119      * reassess it (in particular, obtain a new IP address via DHCP and find
 120      * the new location of the controller).  We set this flag when we suspect
 121      * that this could be the case. */
 122     bool questionable_connectivity;
 123     time_t last_questioned;
 124 
 125     /* Throughout this file, "probe" is shorthand for "inactivity probe".
 126      * When nothing has been received from the peer for a while, we send out
 127      * an echo request as an inactivity probe packet.  We should receive back
 128      * a response. */

 174 /* Creates and returns a new rconn.
 175  *
 176  * 'probe_interval' is a number of seconds.  If the interval passes once
 177  * without an OpenFlow message being received from the peer, the rconn sends
 178  * out an "echo request" message.  If the interval passes again without a
 179  * message being received, the rconn disconnects and re-connects to the peer.
 180  * Setting 'probe_interval' to 0 disables this behavior.
 181  *
 182  * 'max_backoff' is the maximum number of seconds between attempts to connect
 183  * to the peer.  The actual interval starts at 1 second and doubles on each
 184  * failure until it reaches 'max_backoff'.  If 0 is specified, the default of
 185  * 60 seconds is used. */
 186 struct rconn *
 187 rconn_create(int probe_interval, int max_backoff)
 188 {
 189     struct rconn *rc = xcalloc(1, sizeof *rc);
 190 
 191     rc->state = S_VOID;
 192     rc->state_entered = time_now();
 193 
 194     rc->vconn = NULL;
 195     rc->name = xstrdup("void");
 196     rc->reliable = false;
 197 
 198     queue_init(&rc->txq);
 199 
 200     rc->backoff = 0;
 201     rc->max_backoff = max_backoff ? max_backoff : 60;
 202     rc->backoff_deadline = TIME_MIN;
 203     rc->last_received = time_now();
 204     rc->last_connected = time_now();
 205     rc->seqno = 0;
 206 
 207     rc->packets_sent = 0;
 208 
 209     rc->probably_admitted = false;
 210     rc->last_admitted = time_now();
 211 
 212     rc->packets_received = 0;
 213     rc->n_attempted_connections = 0;
 214     rc->n_successful_connections = 0;
 215     rc->creation_time = time_now();
 216     rc->total_time_connected = 0;
 217 
 218     rc->questionable_connectivity = false;
 219     rc->last_questioned = time_now();
 220 
 221     rc->probe_interval = probe_interval ? MAX(1, probe_interval) : 0;
 222 
 223     rc->n_monitors = 0;
 224 
 225     memset(&rc->ofps_rcvd, 0, sizeof(rc->ofps_rcvd));
 226     memset(&rc->ofps_sent, 0, sizeof(rc->ofps_sent));
 227 
 228     rc->idle_echo_xid = 0;
 229 
 230     return rc;
 231 }

 257 void
 258 rconn_disconnect(struct rconn *rc)
 259 {
 260     if (rc->state != S_VOID) {
 261         if (rc->vconn) {
 262             vconn_close(rc->vconn);
 263             rc->vconn = NULL;
 264         }
 265         free(rc->name);
 266         rc->name = xstrdup("void");
 267         rc->reliable = false;
 268 
 269         rc->backoff = 0;
 270         rc->backoff_deadline = TIME_MIN;
 271 
 272         state_transition(rc, S_VOID);
 273     }

 278 rconn_destroy(struct rconn *rc)
 279 {
 280     if (rc) {
 281         size_t i;
 282

 284         vconn_close(rc->vconn);
 285         flush_queue(rc);

 306 static int
 307 reconnect(struct rconn *rc)
 308 {
 309     int retval;
 310 
 311     VLOG_INFO("%s: connecting...", rc->name);
 312     rc->n_attempted_connections++;
 313     retval = vconn_open(rc->name, OFP_VERSION, &rc->vconn);
 314     if (!retval) {
 315         if (!vconn_is_reconnectable(rc->vconn)) {
 316             rc->reliable = false;
 317         }
 318         rc->backoff_deadline = time_now() + rc->backoff;
 319         state_transition(rc, S_CONNECTING);
 320     } else {
 321         VLOG_WARN("%s: connection failed (%s)", rc->name, strerror(retval));
 322         rc->backoff_deadline = TIME_MAX; /* Prevent resetting backoff. */
 323         disconnect(rc, 0);
 324     }
 325     return retval;
 326 }

 348 static void
 349 run_CONNECTING(struct rconn *rc)
 350 {
 351     int retval = vconn_connect(rc->vconn);
 352     if (!retval) {
 353         VLOG_INFO("%s: connected", rc->name);
 354         rc->n_successful_connections++;
 355         state_transition(rc, S_ACTIVE);
 356         rc->last_connected = rc->state_entered;
 357     } else if (retval != EAGAIN) {
 358         VLOG_INFO("%s: connection failed (%s)", rc->name, strerror(retval));
 359         disconnect(rc, retval);
 360     } else if (timed_out(rc)) {
 361         VLOG_INFO("%s: connection timed out", rc->name);
 362         rc->backoff_deadline = TIME_MAX; /* Prevent resetting backoff. */
 363         disconnect(rc, 0);
 364     }
 365 }

 367 static void
 368 do_tx_work(struct rconn *rc)
 369 {
 370     if (!rc->txq.n) {
 371         return;
 372     }
 373     while (rc->txq.n > 0) {
 374         int error = try_send(rc);
 375         if (error) {
 376             break;
 377         }
 378     }
 379     if (!rc->txq.n) {
 380         poll_immediate_wake();
 381     }
 382 }

 384 static unsigned int
 385 timeout_ACTIVE(const struct rconn *rc)
 386 {
 387     if (rc->probe_interval) {
 388         unsigned int base = MAX(rc->last_received, rc->state_entered);
 389         unsigned int arg = base + rc->probe_interval - rc->state_entered;
 390         return arg;
 391     }
 392     return UINT_MAX;
 393 }

 395 static void
 396 run_ACTIVE(struct rconn *rc)
 397 {
 398     if (timed_out(rc)) {
 399         unsigned int base = MAX(rc->last_received, rc->state_entered);
 400         VLOG_DBG("%s: idle %u seconds, sending inactivity probe",
 401                  rc->name, (unsigned int) (time_now() - base));
 402 
 403         /* Ordering is important here: rconn_send() can transition to BACKOFF,
 404          * and we don't want to transition back to IDLE if so, because then we
 405          * can end up queuing a packet with vconn == NULL and then *boom*. */
 406         state_transition(rc, S_IDLE);
 407         rconn_send(rc, make_echo_request(), NULL);
 408         return;
 409     }
 410 
 411     do_tx_work(rc);
 412 }

 421 run_IDLE(struct rconn *rc)
 422 {

 434 /* Performs whatever activities are necessary to maintain 'rc': if 'rc' is
 435  * disconnected, attempts to (re)connect, backing off as necessary; if 'rc' is
 436  * connected, attempts to send packets in the send queue, if any. */
 437 void
 438 rconn_run(struct rconn *rc)
 439 {
 440     int old_state;
 441     do {
 442         old_state = rc->state;
 443         switch (rc->state) {
 444 #define STATE(NAME, VALUE) case S_##NAME: run_##NAME(rc); break;
 445             STATES
 446 #undef STATE
 447         default:
 448             NOT_REACHED();
 449         }
 450     } while (rc->state != old_state);
 451 }

 453 /* Causes the next call to poll_block() to wake up when rconn_run() should be
 454  * called on 'rc'. */
 455 void
 456 rconn_run_wait(struct rconn *rc)
 457 {
 458     unsigned int timeo = timeout(rc);
 459     if (timeo != UINT_MAX) {
 460         unsigned int expires = sat_add(rc->state_entered, timeo);
 461         unsigned int remaining = sat_sub(expires, time_now());
 462         poll_timer_wait(sat_mul(remaining, 1000));
 463     }
 464 
 465     if ((rc->state & (S_ACTIVE | S_IDLE)) && rc->txq.n) {
 466         vconn_wait(rc->vconn, WAIT_SEND);
 467     }
 468 }

 470 /* Attempts to receive a packet from 'rc'.  If successful, returns the packet;
 471  * otherwise, returns a null pointer.  The caller is responsible for freeing
 472  * the packet (with ofpbuf_delete()). */
 473 struct ofpbuf *
 474 rconn_recv(struct rconn *rc)
 475 {
 476     if (rc->state & (S_ACTIVE | S_IDLE)) {
 477         struct ofpbuf *buffer;
 478         int error = vconn_recv(rc->vconn, &buffer);
 479         if (!error) {
 480             struct ofp_header *h = buffer->data;
 481             copy_to_monitor(rc, buffer);
 482             if (is_admitted_msg(buffer)
 483                 || time_now() - rc->last_connected >= 30) {
 484                 rc->probably_admitted = true;
 485                 rc->last_admitted = time_now();
 486             }
 487             rc->last_received = time_now();
 488             rc->packets_received++;
 489             ofpstat_inc_protocol_stat(&rc->ofps_rcvd, h);
 490             if (rc->state == S_IDLE) {
 491                 /* Check liveliness of a peer. */
 492                 if (h->type == OFPT_ECHO_REPLY) {
 493                     if (rc->idle_echo_xid == 0) {
 494                         state_transition(rc, S_ACTIVE);
 495                     } else {
 496                         if (rc->idle_echo_xid == h->xid)
 497                             state_transition(rc, S_ACTIVE);
 498                         rc->idle_echo_xid = 0;
 499                     }
 500                 } else {
 501                     state_transition(rc, S_ACTIVE);
 502                 }
 503             }
 504             return buffer;
 505         } else if (error != EAGAIN) {
 506             disconnect(rc, error);
 507         }
 508     }
 509     return NULL;
 510 }

 522 /* Sends 'b' on 'rc'.  Returns 0 if successful (in which case 'b' is
 523  * destroyed), or ENOTCONN if 'rc' is not currently connected (in which case
 524  * the caller retains ownership of 'b').
 525  *
 526  * If 'n_queued' is non-null, then '*n_queued' will be incremented while the
 527  * packet is in flight, then decremented when it has been sent (or discarded
 528  * due to disconnection).  Because 'b' may be sent (or discarded) before this
 529  * function returns, the caller may not be able to observe any change in
 530  * '*n_queued'.
 531  *
 532  * There is no rconn_send_wait() function: an rconn has a send queue that it
 533  * takes care of sending if you call rconn_run(), which will have the side
 534  * effect of waking up poll_block(). */
 535 int
 536 rconn_send(struct rconn *rc, struct ofpbuf *b, int *n_queued)
 537 {
 538     if (rconn_is_connected(rc)) {
 539         copy_to_monitor(rc, b);
 540         b->private = n_queued;
 541         if (n_queued) {
 542             ++*n_queued;
 543         }
 544         queue_push_tail(&rc->txq, b);
 545 
 546         /* If the queue was empty before we added 'b', try to send some
 547          * packets.  (But if the queue had packets in it, it's because the
 548          * vconn is backlogged and there's no point in stuffing more into it
 549          * now.  We'll get back to that in rconn_run().) */
 550         if (rc->txq.n == 1) {
 551             try_send(rc);
 552         }
 553         return 0;
 554     } else {
 555         return ENOTCONN;
 556     }
 557 }

 559 /* Sends 'b' on 'rc'.  Increments '*n_queued' while the packet is in flight; it
 560  * will be decremented when it has been sent (or discarded due to
 561  * disconnection).  Returns 0 if successful, EAGAIN if '*n_queued' is already
 562  * at least as large as 'queue_limit', or ENOTCONN if 'rc' is not currently
 563  * connected.  Regardless of return value, 'b' is destroyed.
 564  *
 565  * Because 'b' may be sent (or discarded) before this function returns, the
 566  * caller may not be able to observe any change in '*n_queued'.
 567  *
 568  * There is no rconn_send_wait() function: an rconn has a send queue that it
 569  * takes care of sending if you call rconn_run(), which will have the side
 570  * effect of waking up poll_block(). */
 571 int
 572 rconn_send_with_limit(struct rconn *rc, struct ofpbuf *b,
 573                       int *n_queued, int queue_limit)
 574 {
 575     int retval;
 576     retval = *n_queued >= queue_limit ? EAGAIN : rconn_send(rc, b, n_queued);
 577     if (retval) {
 578         ofpbuf_delete(b);
 579     }
 580     return retval;
 581 }

 741 /* Returns protocol statistical information. */
 742 void
 743 rconn_update_protocol_stat(struct rconn *rconn,
 744                            struct ofpstat *ofps_rcvd,
 745                            struct ofpstat *ofps_sent)
 746 {
 747     struct vconn *vconn = rconn->vconn;
 748 
 749     if (vconn != NULL) {
 750         rconn->ofps_rcvd.ofps_total += vconn->ofps_rcvd.ofps_total;
 751         vconn->ofps_rcvd.ofps_total = 0;
 752         rconn->ofps_rcvd.ofps_hello += vconn->ofps_rcvd.ofps_hello;
 753         vconn->ofps_rcvd.ofps_hello = 0;
 754     }
 755     *ofps_rcvd = rconn->ofps_rcvd;
 756 
 757     if (vconn != NULL) {
 758         rconn->ofps_sent.ofps_total += vconn->ofps_sent.ofps_total;
 759         vconn->ofps_sent.ofps_total = 0;
 760         rconn->ofps_sent.ofps_hello += vconn->ofps_sent.ofps_hello;
 761         vconn->ofps_sent.ofps_hello = 0;
 762         rconn->ofps_sent.ofps_error += vconn->ofps_sent.ofps_error;
 763         vconn->ofps_sent.ofps_error = 0;
 764         rconn->ofps_sent.ofps_error_type.hello_fail
 765             += vconn->ofps_sent.ofps_error_type.hello_fail;
 766         vconn->ofps_sent.ofps_error_type.hello_fail = 0;
 767         rconn->ofps_sent.ofps_error_code.hf_incompat
 768             += vconn->ofps_sent.ofps_error_code.hf_incompat;
 769         vconn->ofps_sent.ofps_error_code.hf_incompat = 0;
 770     }
 771     *ofps_sent = rconn->ofps_sent;
 772 }

 774 /* Tries to send a packet from 'rc''s send buffer.  Returns 0 if successful,
 775  * otherwise a positive errno value. */
 776 static int
 777 try_send(struct rconn *rc)
 778 {
 779     int retval = 0;
 780     struct ofpbuf *next = rc->txq.head->next;
 781     struct ofp_header *h = rc->txq.head->data;
 782     int *n_queued = rc->txq.head->private;
 783     ofpstat_inc_protocol_stat(&rc->ofps_sent, h);
 784     rc->idle_echo_xid = h->xid;
 785     retval = vconn_send(rc->vconn, rc->txq.head);
 786     if (retval) {
 787         rc->idle_echo_xid = 0;
 788         if (retval != EAGAIN) {
 789             disconnect(rc, retval);
 790         }
 791         return retval;
 792     }
 793     rc->packets_sent++;
 794     if (n_queued) {
 795         --*n_queued;
 796     }
 797     queue_advance_head(&rc->txq, next);
 798     return 0;
 799 }

 801 /* Disconnects 'rc'.  'error' is used only for logging purposes.  If it is
 802  * nonzero, then it should be EOF to indicate the connection was closed by the
 803  * peer in a normal fashion or a positive errno value. */
 804 static void
 805 disconnect(struct rconn *rc, int error)
 806 {
 807     if (rc->reliable) {
 808         time_t now = time_now();
 809 
 810         if (rc->state & (S_CONNECTING | S_ACTIVE | S_IDLE)) {
 811             if (error > 0) {
 812                 VLOG_WARN("%s: connection dropped (%s)",
 813                           rc->name, strerror(error));
 814             } else if (error == EOF) {
 815                 if (rc->reliable) {
 816                     VLOG_INFO("%s: connection closed by peer", rc->name);
 817                 }
 818             } else {
 819                 VLOG_INFO("%s: connection dropped", rc->name);
 820             }
 821             vconn_close(rc->vconn);
 822             rc->vconn = NULL;
 823             flush_queue(rc);
 824         }
 825 
 826         if (now >= rc->backoff_deadline) {
 827             rc->backoff = 1;
 828         } else {
 829             rc->backoff = MIN(rc->max_backoff, MAX(1, 2 * rc->backoff));
 830             VLOG_INFO("%s: waiting %d seconds before reconnect",
 831                       rc->name, rc->backoff);
 832         }
 833         rc->backoff_deadline = now + rc->backoff;
 834         state_transition(rc, S_BACKOFF);
 835         if (now - rc->last_connected > 60) {
 836             question_connectivity(rc);
 837         }
 838     } else {
 839         rconn_disconnect(rc);
 840     }
 841 }
 842 
 843 /* Drops all the packets from 'rc''s send queue and decrements their queue
 844  * counts. */
 845 static void
 846 flush_queue(struct rconn *rc)
 847 {
 848     if (!rc->txq.n) {
 849         return;
 850     }
 851     while (rc->txq.n > 0) {
 852         struct ofpbuf *b = queue_pop_head(&rc->txq);
 853         int *n_queued = b->private;
 854         if (n_queued) {
 855             --*n_queued;
 856         }
 857         ofpbuf_delete(b);
 858     }
 859     poll_immediate_wake();
 860 }

 871     switch (rc->state) {
 872 #define STATE(NAME, VALUE) case S_##NAME: return timeout_##NAME(rc);
 873         STATES
 874 #undef STATE

 877     }

 886 static void
 887 state_transition(struct rconn *rc, enum state state)
 888 {
 889     rc->seqno += (rc->state == S_ACTIVE) != (state == S_ACTIVE);
 890     if (is_connected_state(state) && !is_connected_state(rc->state)) {
 891         rc->probably_admitted = false;
 892     }
 893     if (rconn_is_connected(rc)) {
 894         rc->total_time_connected += elapsed_in_this_state(rc);
 895     }
 896     VLOG_DBG("%s: entering %s", rc->name, state_name(state));
 897     rc->state = state;
 898     rc->state_entered = time_now();
 899 }

 901 static void

 903 {
 904     time_t now = time_now();

 906         rc->questionable_connectivity = true;
 907         rc->last_questioned = now;

 911 static void
 912 copy_to_monitor(struct rconn *rc, const struct ofpbuf *b)
 913 {
 914     struct ofpbuf *clone = NULL;
 915     int retval;
 916     size_t i;
 917 
 918     for (i = 0; i < rc->n_monitors; ) {
 919         struct vconn *vconn = rc->monitors[i];
 920 
 921         if (!clone) {
 922             clone = ofpbuf_clone(b);
 923         }
 924         retval = vconn_send(vconn, clone);
 925         if (!retval) {
 926             clone = NULL;
 927         } else if (retval != EAGAIN) {
 928             VLOG_DBG("%s: closing monitor connection to %s: %s",
 929                      rconn_get_name(rc), vconn_get_name(vconn),
 930                      strerror(retval));
 931             rc->monitors[i] = rc->monitors[--rc->n_monitors];
 932             continue;
 933         }
 934         i++;
 935     }
 936     ofpbuf_delete(clone);
 937 }

 945 static bool
 946 is_admitted_msg(const struct ofpbuf *b)
 947 {
 948     struct ofp_header *oh = b->data;
 949     uint8_t type = oh->type;
 950     return !(type < 32
 951              && (1u << type) & ((1u << OFPT_HELLO) |
 952                                 (1u << OFPT_ERROR) |
 953                                 (1u << OFPT_ECHO_REQUEST) |
 954                                 (1u << OFPT_ECHO_REPLY) |
 955                                 (1u << OFPT_VENDOR) |
 956                                 (1u << OFPT_FEATURES_REQUEST) |
 957                                 (1u << OFPT_FEATURES_REPLY) |
 958                                 (1u << OFPT_GET_CONFIG_REQUEST) |
 959                                 (1u << OFPT_GET_CONFIG_REPLY) |
 960                                 (1u << OFPT_SET_CONFIG)));
 961 }
