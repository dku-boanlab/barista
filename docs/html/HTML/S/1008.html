  55 /**
  56  * \brief Function to convert an event string to an event ID
  57  * \param event Event string
  58  * \return Event ID
  59  */
  60 static int ev_type(const char *event)
  61 {
  62     int i;
  63     for (i=0; i<EV_NUM_EVENTS+1; i++) {
  64         if (strcmp(event, ev_str[i]) == 0)
  65             return i;
  66     }
  67     return EV_NUM_EVENTS;
  68 }

  72 /**
  73  * \brief Function to load component configurations
  74  * \param conf_file Component configuration file
  75  */
  76 static int config_load(char *conf_file)
  77 {
  78     json_t *json;
  79     json_error_t error;
  80 
  81     LOG_INFO(CAC_ID, "Component configuration file: %s", conf_file);
  82 
  83     if (access(conf_file, F_OK)) {
  84         LOG_ERROR(CAC_ID, "No configuration file whose name is '%s'", conf_file);
  85         return -1;
  86     }
  87 
  88     char *raw = str_read(conf_file);
  89     char *conf = str_preproc(raw);
  90 
  91     json = json_loads(conf, 0, &error);
  92     if (!json) {
  93         PERROR("json_loads");
  94         return -1;
  95     } else if (!json_is_array(json)) {
  96         PERROR("json_is_array");
  97         json_decref(json);
  98         return -1;
  99     }
 100 
 101     int i;
 102     for (i=0; i<json_array_size(json); i++) {
 103         json_t *data = json_array_get(json, i);
 104 
 105         char name[__CONF_WORD_LEN] = {0};
 106         json_t *j_name = json_object_get(data, "name");
 107         if (json_is_string(j_name)) {
 108             strcpy(name, json_string_value(j_name));
 109         }
 110 
 111         // set the component name
 112         if (strlen(name) == 0) {
 113             PRINTF("no component name\n");
 114             return -1;
 115         } else {
 116             strcpy(component[num_components].name, name);
 117             component[num_components].id = hash_func((uint32_t *)&component[num_components].name, __HASHING_NAME_LENGTH);
 118         }
 119 
 120         // set inbound events
 121         json_t *events = json_object_get(data, "inbounds");
 122         if (json_is_array(events)) {
 123             int j;
 124             for (j=0; j<json_array_size(events); j++) {
 125                 json_t *event = json_array_get(events, j);
 126 
 127                 if (ev_type(json_string_value(event)) == EV_NUM_EVENTS) {
 128                     PRINTF("wrong event name\n");
 129                     return -1;
 130                 }
 131             }
 132 
 133             for (j=0; j<json_array_size(events); j++) {
 134                 json_t *event = json_array_get(events, j);
 135 
 136                 if (ev_type(json_string_value(event)) == EV_NONE) {
 137                     break;
 138                 } else if (ev_type(json_string_value(event)) == EV_ALL_UPSTREAM) {
 139                     int k;
 140                     for (k=EV_NONE+1; k<EV_ALL_UPSTREAM; k++) {
 141                         component[num_components].in_list[component[num_components].in_num] = k;
 142                         component[num_components].in_num++;
 143                     }
 144                 } else if (ev_type(json_string_value(event)) == EV_ALL_DOWNSTREAM) {
 145                     int k;
 146                     for (k=EV_ALL_UPSTREAM+1; k<EV_ALL_DOWNSTREAM; k++) {
 147                         component[num_components].in_list[component[num_components].in_num] = k;
 148                         component[num_components].in_num++;
 149                     }
 150                 } else if (ev_type(json_string_value(event)) == EV_WRT_INTSTREAM) {
 151                     int k;
 152                     for (k=EV_ALL_DOWNSTREAM+1; k<EV_WRT_INTSTREAM; k++) {
 153                         component[num_components].in_list[component[num_components].in_num] = k;
 154                         component[num_components].in_num++;
 155                     }
 156                 } else if (ev_type(json_string_value(event)) == EV_ALL_INTSTREAM) {
 157                     int k;
 158                     for (k=EV_WRT_INTSTREAM+1; k<EV_ALL_INTSTREAM; k++) {
 159                         component[num_components].in_list[component[num_components].in_num] = k;
 160                         component[num_components].in_num++;
 161                     }
 162                 } else {
 163                     component[num_components].in_list[component[num_components].in_num] = ev_type(json_string_value(event));
 164                     component[num_components].in_num++;
 165                 }
 166             }
 167         }
 168 
 169         // set outbound events
 170         json_t *out_events = json_object_get(data, "outbounds");
 171         if (json_is_array(out_events)) {
 172             int j;
 173             for (j=0; j<json_array_size(out_events); j++) {
 174                 json_t *out_event = json_array_get(out_events, j);
 175 
 176                 if (ev_type(json_string_value(out_event)) == EV_NUM_EVENTS) {
 177                     PRINTF("wrong event name\n");
 178                     return -1;
 179                 }
 180             }
 181 
 182             for (j=0; j<json_array_size(out_events); j++) {
 183                 json_t *out_event = json_array_get(out_events, j);
 184 
 185                 component[num_components].out_list[component[num_components].out_num] = ev_type(json_string_value(out_event));
 186                 component[num_components].out_num++;
 187             }
 188         }
 189 
 190         num_components++;
 191     }
 192 
 193     json_decref(json);
 194 
 195     return 0;
 196 }

 200 /**
 201  * \brief Function to verity the permission of a component
 202  * \param id Component ID
 203  * \param type Event type
 204  */
 205 static int verify_component(uint32_t id, int type)
 206 {
 207     int i;
 208     for (i=0; i<num_components; i++) {
 209         if (component[i].id == id) {
 210             int j;
 211             for (j=0; j<component[i].out_num; j++)
 212                 if (component[i].out_list[j] == type)
 213                     return 0;
 214             return 1;
 215         }
 216     }
 217     return -1;
 218 }

 223  * \brief The main function

 228 int cac_main(int *activated, int argc, char **argv)
 229 {
 230     LOG_INFO(CAC_ID, "Init - Component access control");
 231 
 232     num_components = 0;
 233 
 234     component = (component_t *)CALLOC(__MAX_COMPONENTS, sizeof(component_t));
 235     if (component == NULL) {
 236         PERROR("calloc");
 237         return -1;
 238     }
 239 
 240     if (config_load(compnt_file) < 0)
 241         return -1;
 242 
 243     activate();
 244 
 245     return 0;
 246 }

 273 /**
 274  * \brief The handler function
 275  * \param ev Read-only event
 276  * \param ev_out Read-write event (if this component has the write permission)
 277  */
 278 int cac_handler(const event_t *ev, event_out_t *ev_out)
 279 {
 280     int res = verify_component(ev->id, ev->type);
 281     if (res < 0) {
 282         LOG_WARN(CAC_ID, "Wrong component id (caller: %u, event: %s)", ev->id, ev_str[ev->type]);
 283         return -1;
 284     } else if (res > 0) {
 285         int i;
 286         for (i=0; i<num_components; i++) {
 287             if (component[i].id == ev->id) {
 288                 LOG_WARN(CAC_ID, "Unexcepted access (caller: %s, event: %s)", component[i].name, ev_str[ev->type]);
 289                 break;
 290             }
 291         }
 292         return -1;
 293     }
 294 
 295     return 0;
 296 }
