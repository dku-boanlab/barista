  66 static struct netlink_vconn *
  67 netlink_vconn_cast(struct vconn *vconn) 
  68 {
  69     vconn_assert_class(vconn, &netlink_vconn_class);
  70     return CONTAINER_OF(vconn, struct netlink_vconn, vconn); 
  71 }

  73 static int
  74 netlink_open(const char *name, char *suffix, struct vconn **vconnp)
  75 {
  76     struct netlink_vconn *netlink;
  77     int subscribe;
  78     int dp_idx;
  79     int retval;
  80 
  81     subscribe = 1;
  82     if (sscanf(suffix, "%d:%d", &dp_idx, &subscribe) < 1) {
  83         ofp_error(0, "%s: syntax error", name);
  84         return EAFNOSUPPORT;
  85     }
  86 
  87     netlink = xmalloc(sizeof *netlink);
  88     vconn_init(&netlink->vconn, &netlink_vconn_class, 0, 0, name, true);
  89     retval = dpif_open(subscribe ? dp_idx : -1, &netlink->dp);
  90     netlink->dp_idx = dp_idx;
  91     if (retval) {
  92         free(netlink);
  93         *vconnp = NULL;
  94         return retval;
  95     }
  96     *vconnp = &netlink->vconn;
  97     return 0;
  98 }

 100 static void
 101 netlink_close(struct vconn *vconn) 
 102 {
 103     struct netlink_vconn *netlink = netlink_vconn_cast(vconn);
 104     dpif_close(&netlink->dp);
 105     free(netlink);
 106 }

 108 static int
 109 netlink_recv(struct vconn *vconn, struct ofpbuf **bufferp)
 110 {
 111     struct netlink_vconn *netlink = netlink_vconn_cast(vconn);
 112     return dpif_recv_openflow(&netlink->dp, netlink->dp_idx, bufferp, false);
 113 }
 114 
 115 static int
 116 netlink_send(struct vconn *vconn, struct ofpbuf *buffer) 
 117 {
 118     struct netlink_vconn *netlink = netlink_vconn_cast(vconn);
 119     int retval = dpif_send_openflow(&netlink->dp, netlink->dp_idx, buffer);
 120     if (!retval) {
 121         ofpbuf_delete(buffer);
 122     }
 123     return retval;
 124 }

 126 static void
 127 netlink_wait(struct vconn *vconn, enum vconn_wait_type wait) 
 128 {
 129     struct netlink_vconn *netlink = netlink_vconn_cast(vconn);
 130     short int events = 0;
 131     switch (wait) {
 132     case WAIT_CONNECT:
 133         NOT_REACHED();
 134 
 135     case WAIT_RECV:
 136         events = POLLIN;
 137         break;
 138 
 139     case WAIT_SEND:
 140         events = 0;
 141         break;
 142 
 143     default:
 144         NOT_REACHED();
 145     }
 146     nl_sock_wait(netlink->dp.sock, events);
 147 }
