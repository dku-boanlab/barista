 225 /**
 226  * \brief Function to process meta app events
 227  * \return NULL
 228  */
 229 static void *meta_app_events(void *null)
 230 {
 231     int *num_events = av_ctx->num_app_events;
 232     meta_event_t *meta = av_ctx->meta_app_event;
 233 
 234     while (app_event_is_set) {
 235         int av_id;
 236         for (av_id=0; av_id<__MAX_APP_EVENTS; av_id++) {
 237             int num_event = num_events[av_id];
 238 
 239             int j;
 240             for (j=0; j<__MAX_META_EVENTS; j++) {
 241                 if (meta[j].event == av_id) {
 242                     switch (meta[j].condition) {
 243                     case META_GT: // >
 244                         if (num_event > meta[j].threshold) {
 245                             //char **args = split_line(meta[j].cmd);
 246                             //execute(args);
 247                             //FREE(args);
 248                         }
 249                         break;
 250                     case META_GTE: // >=
 251                         if (num_event >= meta[j].threshold) {
 252                             //char **args = split_line(meta[j].cmd);
 253                             //execute(args);
 254                             //FREE(args);
 255                         }
 256                         break;
 257                     case META_LT: // <
 258                         if (num_event < meta[j].threshold) {
 259                             //char **args = split_line(meta[j].cmd);
 260                             //execute(args);
 261                             //FREE(args);
 262                         }
 263                         break;
 264                     case META_LTE: // <=
 265                         if (num_event <= meta[j].threshold) {
 266                             //char **args = split_line(meta[j].cmd);
 267                             //execute(args);
 268                             //FREE(args);
 269                         }
 270                         break;
 271                     case META_EQ: // ==
 272                         if (num_event == meta[j].threshold) {
 273                             //char **args = split_line(meta[j].cmd);
 274                             //execute(args);
 275                             //FREE(args);
 276                         }
 277                         break;
 278                     default:
 279                         break;
 280                     }
 281                 }
 282             }
 283 
 284             num_events[av_id] = 0;
 285         }
 286 
 287         waitsec(1, 0);
 288     }
 289 
 290     return NULL;
 291 }

 295 /**
 296  * \brief Function to process requests from applications
 297  * \return NULL
 298  */
 299 static void *reply_app_events(void *null)
 300 {
 301     void *recv = zmq_socket(av_rep_ctx, ZMQ_REP);
 302     zmq_connect(recv, "inproc://reply_workers");
 303 
 304     while (av_worker_on) {
 305         msg_t msg = {0};
 306 
 307         zmq_recv(recv, &msg, sizeof(msg_t), 0);
 308 
 309         switch (msg.type) {
 310 
 311         // request-reply events
 312 
 313         case AV_SW_GET_INFO:
 314             sw_rw_raise(msg.id, AV_SW_GET_INFO, sizeof(switch_t), (switch_t *)msg.data);
 315             break;
 316         case AV_SW_GET_ALL_INFO:
 317             sw_rw_raise(msg.id, AV_SW_GET_ALL_INFO, sizeof(switch_t), (switch_t *)msg.data);
 318             break;
 319         case AV_HOST_GET_INFO:
 320             host_rw_raise(msg.id, AV_HOST_GET_INFO, sizeof(host_t), (host_t *)msg.data);
 321             break;
 322         case AV_HOST_GET_ALL_INFO:
 323             host_rw_raise(msg.id, AV_HOST_GET_ALL_INFO, sizeof(host_t), (host_t *)msg.data);
 324             break;
 325         case AV_LINK_GET_INFO:
 326             port_rw_raise(msg.id, AV_LINK_GET_INFO, sizeof(port_t), (port_t *)msg.data);
 327             break;
 328         case AV_LINK_GET_ALL_INFO:
 329             port_rw_raise(msg.id, AV_LINK_GET_ALL_INFO, sizeof(port_t), (port_t *)msg.data);
 330             break;
 331         case AV_FLOW_GET_INFO:
 332             flow_rw_raise(msg.id, AV_FLOW_GET_INFO, sizeof(flow_t), (flow_t *)msg.data);
 333             break;
 334         case AV_FLOW_GET_ALL_INFO:
 335             flow_rw_raise(msg.id, AV_FLOW_GET_ALL_INFO, sizeof(flow_t), (flow_t *)msg.data);
 336             break;
 337 
 338         default:
 339             break;
 340         }
 341 
 342         zmq_send(recv, &msg, sizeof(msg_t), 0);
 343     }
 344 
 345     zmq_close(recv);
 346 
 347     return NULL;
 348 }

 361 /**
 362  * \brief Function to receive app events from external applications
 363  * \return NULL
 364  */
 365 static void *receive_app_events(void *null)
 366 {
 367     while (av_worker_on) {
 368         msg_t msg = {0};
 369 
 370         zmq_recv(av_pull_sock, &msg, sizeof(msg_t), 0);
 371 
 372         switch (msg.type) {
 373 
 374         // upstream events
 375 
 376         case AV_DP_RECEIVE_PACKET:
 377             pktin_av_raise(msg.id, AV_DP_RECEIVE_PACKET, sizeof(pktin_t), (const pktin_t *)msg.data);
 378             break;
 379         case AV_DP_FLOW_EXPIRED:
 380             flow_av_raise(msg.id, AV_DP_FLOW_EXPIRED, sizeof(flow_t), (const flow_t *)msg.data);
 381             break;
 382         case AV_DP_FLOW_DELETED:
 383             flow_av_raise(msg.id, AV_DP_FLOW_DELETED, sizeof(flow_t), (const flow_t *)msg.data);
 384             break;
 385         case AV_DP_PORT_ADDED:
 386             port_av_raise(msg.id, AV_DP_PORT_ADDED, sizeof(port_t), (const port_t *)msg.data);
 387             break;
 388         case AV_DP_PORT_MODIFIED:
 389             port_av_raise(msg.id, AV_DP_PORT_MODIFIED, sizeof(port_t), (const port_t *)msg.data);
 390             break;
 391         case AV_DP_PORT_DELETED:
 392             port_av_raise(msg.id, AV_DP_PORT_DELETED, sizeof(port_t), (const port_t *)msg.data);
 393             break;
 394 
 395         // downstream events
 396 
 397         case AV_DP_SEND_PACKET:
 398             pktout_av_raise(msg.id, AV_DP_SEND_PACKET, sizeof(pktout_t), (const pktout_t *)msg.data);
 399             break;
 400         case AV_DP_INSERT_FLOW:
 401             flow_av_raise(msg.id, AV_DP_INSERT_FLOW, sizeof(flow_t), (const flow_t *)msg.data);
 402             break;
 403         case AV_DP_MODIFY_FLOW:
 404             flow_av_raise(msg.id, AV_DP_MODIFY_FLOW, sizeof(flow_t), (const flow_t *)msg.data);
 405             break;
 406         case AV_DP_DELETE_FLOW:
 407             flow_av_raise(msg.id, AV_DP_DELETE_FLOW, sizeof(flow_t), (const flow_t *)msg.data);
 408             break;
 409 
 410         // log events
 411 
 412         case AV_LOG_DEBUG:
 413             log_av_raise(msg.id, AV_LOG_DEBUG, strlen((const char *)msg.data), (const char *)msg.data);
 414             break;
 415         case AV_LOG_INFO:
 416             log_av_raise(msg.id, AV_LOG_INFO, strlen((const char *)msg.data), (const char *)msg.data);
 417             break;
 418         case AV_LOG_WARN:
 419             log_av_raise(msg.id, AV_LOG_WARN, strlen((const char *)msg.data), (const char *)msg.data);
 420             break;
 421         case AV_LOG_ERROR:
 422             log_av_raise(msg.id, AV_LOG_ERROR, strlen((const char *)msg.data), (const char *)msg.data);
 423             break;
 424         case AV_LOG_FATAL:
 425             log_av_raise(msg.id, AV_LOG_FATAL, strlen((const char *)msg.data), (const char *)msg.data);
 426             break;
 427 
 428         default:
 429             break;
 430         }
 431     }
 432 
 433     return NULL;
 434 }

 456 /**
 457  * \brief Function to initialize the app event handler
 458  * \param ctx The context of the Barista NOS
 459  */
 460 int app_event_init(ctx_t *ctx)
 461 {
 462     if (app_event_is_set == FALSE) {
 463         av_ctx = ctx;
 464         app_event_is_set = TRUE;
 465     } else {
 466         return -1;
 467     }
 468 
 469     pthread_t thread;
 470     if (pthread_create(&thread, NULL, &meta_app_events, NULL) < 0) {
 471         PERROR("pthread_create");
 472         return -1;
 473     }
 474 
 475     av_worker_on = TRUE;
 476 
 477     // pull (outside)
 478 
 479     av_pull_ctx = zmq_ctx_new();
 480     av_pull_sock = zmq_socket(av_pull_ctx, ZMQ_PULL);
 481 
 482     if (access("../tmp/ext_app_event_handler", F_OK) != -1) {
 483         remove("../tmp/ext_app_event_handler");
 484     }
 485 
 486     if (zmq_bind(av_pull_sock, "ipc://../tmp/ext_app_event_handler")) {
 487         PERROR("zmq_bind");
 488         return -1;
 489     }
 490 
 491     int i;
 492     for (i=0; i<__NUM_THREADS; i++) {
 493         if (pthread_create(&thread, NULL, &receive_app_events, NULL) < 0) {
 494             PERROR("pthread_create");
 495             return -1;
 496         }
 497     }
 498 
 499     // reply
 500 
 501     if (access("../tmp/app_req_handler", F_OK) != -1) {
 502         remove("../tmp/app_req_handler");
 503     }
 504 
 505     av_rep_ctx = zmq_ctx_new();
 506     av_rep_app = zmq_socket(av_rep_ctx, ZMQ_ROUTER);
 507     if (zmq_bind(av_rep_app, "ipc://../tmp/app_req_handler")) {
 508         PERROR("zmq_bind");
 509         return -1;
 510     }
 511 
 512     av_rep_work = zmq_socket(av_rep_ctx, ZMQ_DEALER);
 513     if (zmq_bind(av_rep_work, "inproc://reply_workers")) {
 514         PERROR("zmq_bind");
 515         return -1;
 516     }
 517 
 518     for (i=0; i<__NUM_THREADS; i++) {
 519         if (pthread_create(&thread, NULL, &reply_app_events, NULL) < 0) {
 520             PERROR("pthread_create");
 521             return -1;
 522         }
 523     }
 524 
 525     if (pthread_create(&thread, NULL, &reply_proxy, ctx) < 0) {
 526         PERROR("pthread_create");
 527         return -1;
 528     }
 529 
 530     return 0;
 531 }
