  45 struct sw_table_hash {
  46     struct sw_table swt;
  47     struct crc32 crc32;
  48     unsigned int n_flows;
  49     unsigned int bucket_mask; /* Number of buckets minus 1. */
  50     struct sw_flow **buckets;
  51 };

  53 static struct sw_flow **find_bucket(struct sw_table *swt,
  54                                     const struct sw_flow_key *key)
  55 {
  56     struct sw_table_hash *th = (struct sw_table_hash *) swt;
  57     unsigned int crc = crc32_calculate(&th->crc32, key, 
  58             offsetof(struct sw_flow_key, wildcards));
  59     return &th->buckets[crc & th->bucket_mask];
  60 }

  62 static struct sw_flow *table_hash_lookup(struct sw_table *swt,
  63                                          const struct sw_flow_key *key)
  64 {
  65     struct sw_flow *flow = *find_bucket(swt, key);
  66     return flow && !flow_compare(&flow->key.flow, &key->flow) ? flow : NULL;
  67 }

  69 static int table_hash_insert(struct sw_table *swt, struct sw_flow *flow)
  70 {
  71     struct sw_table_hash *th = (struct sw_table_hash *) swt;
  72     struct sw_flow **bucket;
  73     int retval;
  74 
  75     if (flow->key.wildcards != 0)
  76         return 0;
  77 
  78     bucket = find_bucket(swt, &flow->key);
  79     if (*bucket == NULL) {
  80         th->n_flows++;
  81         *bucket = flow;
  82         retval = 1;
  83     } else {
  84         struct sw_flow *old_flow = *bucket;
  85         if (!flow_compare(&old_flow->key.flow, &flow->key.flow)) {
  86             *bucket = flow;
  87             flow_free(old_flow);
  88             retval = 1;
  89         } else {
  90             retval = 0;
  91         }
  92     }
  93     return retval;
  94 }

  96 static int table_hash_modify(struct sw_table *swt, 
  97         const struct sw_flow_key *key, uint16_t priority, int strict,
  98         const struct ofp_action_header *actions, size_t actions_len) 
  99 {
 100     struct sw_table_hash *th = (struct sw_table_hash *) swt;
 101     unsigned int count = 0;
 102 
 103     if (key->wildcards == 0) {
 104         struct sw_flow **bucket = find_bucket(swt, key);
 105         struct sw_flow *flow = *bucket;
 106         if (flow && flow_matches_desc(&flow->key, key, strict)
 107                 && (!strict || (flow->priority == priority))) {
 108             flow_replace_acts(flow, actions, actions_len);
 109             count = 1;
 110         }
 111     } else {
 112         unsigned int i;
 113 
 114         for (i = 0; i <= th->bucket_mask; i++) {
 115             struct sw_flow **bucket = &th->buckets[i];
 116             struct sw_flow *flow = *bucket;
 117             if (flow && flow_matches_desc(&flow->key, key, strict)
 118                     && (!strict || (flow->priority == priority))) {
 119                 flow_replace_acts(flow, actions, actions_len);
 120                 count++;
 121             }
 122         }
 123     }
 124     return count;
 125 }

 127 static int table_hash_has_conflict(struct sw_table *swt,
 128                                    const struct sw_flow_key *key,
 129                                    uint16_t priority, int strict)
 130 {
 131     struct sw_table_hash *th = (struct sw_table_hash *) swt;
 132 
 133     if (key->wildcards == 0) {
 134         struct sw_flow **bucket = find_bucket(swt, key);
 135         struct sw_flow *flow = *bucket;
 136         if (flow && flow_matches_2desc(&flow->key, key,strict)
 137                 && (flow->priority == priority)) {
 138             return true;
 139         }
 140     } else {
 141         unsigned int i;
 142 
 143         for (i = 0; i <= th->bucket_mask; i++) {
 144             struct sw_flow **bucket = &th->buckets[i];
 145             struct sw_flow *flow = *bucket;
 146             if (flow && flow_matches_2desc(&flow->key, key, strict)
 147                     && (flow->priority == priority)) {
 148                 return true;
 149             }
 150         }
 151     }
 152     return false;
 153 }

 163 /* Returns number of deleted flows.  We ignore the priority
 164  * argument, since all exact-match entries are the same (highest)
 165  * priority. */
 166 static int table_hash_delete(struct datapath *dp, struct sw_table *swt,
 167                              const struct sw_flow_key *key, 
 168                              uint16_t out_port,
 169                              uint16_t priority UNUSED, int strict)
 170 {
 171     struct sw_table_hash *th = (struct sw_table_hash *) swt;
 172     unsigned int count = 0;
 173 
 174     if (key->wildcards == 0) {
 175         struct sw_flow **bucket = find_bucket(swt, key);
 176         struct sw_flow *flow = *bucket;
 177         if (flow && !flow_compare(&flow->key.flow, &key->flow)
 178                 && flow_has_out_port(flow, out_port)) {
 179             dp_send_flow_end(dp, flow, OFPRR_DELETE);
 180             do_delete(bucket);
 181             count = 1;
 182         }
 183     } else {
 184         unsigned int i;
 185 
 186         for (i = 0; i <= th->bucket_mask; i++) {
 187             struct sw_flow **bucket = &th->buckets[i];
 188             struct sw_flow *flow = *bucket;
 189             if (flow && flow_matches_desc(&flow->key, key, strict)
 190                     && flow_has_out_port(flow, out_port)) {
 191                 dp_send_flow_end(dp, flow, OFPRR_DELETE);
 192                 do_delete(bucket);
 193                 count++;
 194             }
 195         }
 196     }
 197     th->n_flows -= count;
 198     return count;
 199 }

 201 static void table_hash_timeout(struct sw_table *swt, struct list *deleted)
 202 {
 203     struct sw_table_hash *th = (struct sw_table_hash *) swt;
 204     unsigned int i;
 205 
 206     for (i = 0; i <= th->bucket_mask; i++) {
 207         struct sw_flow **bucket = &th->buckets[i];
 208         struct sw_flow *flow = *bucket;
 209         if (flow && flow_timeout(flow)) {
 210             list_push_back(deleted, &flow->node);
 211             *bucket = NULL;
 212             th->n_flows--;
 213         }
 214     }
 215 }

 217 static void table_hash_destroy(struct sw_table *swt)
 218 {
 219     struct sw_table_hash *th = (struct sw_table_hash *) swt;
 220     unsigned int i;
 221     for (i = 0; i <= th->bucket_mask; i++) {
 222         if (th->buckets[i]) {

 230 static int table_hash_iterate(struct sw_table *swt,
 231                               const struct sw_flow_key *key, uint16_t out_port,
 232                               struct sw_table_position *position,
 233                               int (*callback)(struct sw_flow *, void *private),
 234                               void *private) 
 235 {
 236     struct sw_table_hash *th = (struct sw_table_hash *) swt;
 237 
 238     if (position->private[0] > th->bucket_mask)
 239         return 0;
 240 
 241     if (key->wildcards == 0) {
 242         struct sw_flow *flow = table_hash_lookup(swt, key);
 243         position->private[0] = -1;
 244         if (!flow || !flow_has_out_port(flow, out_port)) {
 245             return 0;
 246         }
 247         return callback(flow, private);
 248     } else {
 249         int i;
 250 
 251         for (i = position->private[0]; i <= th->bucket_mask; i++) {
 252             struct sw_flow *flow = th->buckets[i];
 253             if (flow && flow_matches_1wild(&flow->key, key)
 254                     && flow_has_out_port(flow, out_port)) {
 255                 int error = callback(flow, private);
 256                 if (error) {
 257                     position->private[0] = i + 1;
 258                     return error;
 259                 }
 260             }
 261         }
 262         return 0;
 263     }
 264 }

 278 struct sw_table *table_hash_create(unsigned int polynomial,
 279                                    unsigned int n_buckets)
 280 {
 281     struct sw_table_hash *th;
 282     struct sw_table *swt;
 283 
 284     th = malloc(sizeof *th);
 285     if (th == NULL)
 286         return NULL;
 287     memset(th, '\0', sizeof *th);
 288 
 289     assert(!(n_buckets & (n_buckets - 1)));
 290     th->buckets = calloc(n_buckets, sizeof *th->buckets);
 291     if (th->buckets == NULL) {
 292         printf("failed to allocate %u buckets\n", n_buckets);
 293         free(th);
 294         return NULL;
 295     }
 296     th->n_flows = 0;
 297     th->bucket_mask = n_buckets - 1;
 298 
 299     swt = &th->swt;
 300     swt->lookup = table_hash_lookup;
 301     swt->insert = table_hash_insert;
 302     swt->modify = table_hash_modify;
 303     swt->has_conflict = table_hash_has_conflict;
 304     swt->delete = table_hash_delete;
 305     swt->timeout = table_hash_timeout;
 306     swt->destroy = table_hash_destroy;
 307     swt->iterate = table_hash_iterate;
 308     swt->stats = table_hash_stats;
 309 
 310     crc32_init(&th->crc32, polynomial);
 311 
 312     return swt;
 313 }

 322 static struct sw_flow *table_hash2_lookup(struct sw_table *swt,
 323                                           const struct sw_flow_key *key)
 324 {
 325     struct sw_table_hash2 *t2 = (struct sw_table_hash2 *) swt;
 326     int i;
 327         
 328     for (i = 0; i < 2; i++) {
 329         struct sw_flow *flow = *find_bucket(t2->subtable[i], key);
 330         if (flow && !flow_compare(&flow->key.flow, &key->flow))
 331             return flow;
 332     }
 333     return NULL;
 334 }

 336 static int table_hash2_insert(struct sw_table *swt, struct sw_flow *flow)
 337 {
 338     struct sw_table_hash2 *t2 = (struct sw_table_hash2 *) swt;
 339 
 340     if (table_hash_insert(t2->subtable[0], flow))
 341         return 1;
 342     return table_hash_insert(t2->subtable[1], flow);
 343 }

 392 static int table_hash2_iterate(struct sw_table *swt,
 393                                const struct sw_flow_key *key,
 394                                uint16_t out_port,
 395                                struct sw_table_position *position,
 396                                int (*callback)(struct sw_flow *, void *),
 397                                void *private)

 399     struct sw_table_hash2 *t2 = (struct sw_table_hash2 *) swt;

 401

 403         int error = table_hash_iterate(t2->subtable[i], key, out_port,

 459     swt->iterate = table_hash2_iterate;
 460     swt->stats = table_hash2_stats;
