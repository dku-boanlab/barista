 239 /**
 240  * \brief Function to process FEATURES_REPLY messages (only for sw_mgmt)
 241  * \param msg FEATURE_REPLY message
 242  */
 243 static int ofp10_features_reply(const raw_msg_t *msg)
 244 {
 245     struct ofp_switch_features *reply = (struct ofp_switch_features *)msg->data;
 246 
 247     switch_t sw = {0};
 248 
 249     sw.dpid = ntohll(reply->datapath_id);
 250 
 251     sw.fd = msg->fd;
 252     sw.xid = ntohl(reply->header.xid) + 1;
 253 
 254     sw.remote = FALSE;
 255     sw.version = OFP_VERSION;
 256 
 257     sw.n_tables = reply->n_tables;
 258     sw.n_buffers = ntohl(reply->n_buffers);
 259 
 260     uint32_t cap = ntohl(reply->capabilities);
 261     uint32_t new_cap = 0;
 262 
 263     if (cap & OFPC_FLOW_STATS) new_cap |= SC_FLOW_STATS;
 264     if (cap & OFPC_TABLE_STATS) new_cap |= SC_TABLE_STATS;
 265     if (cap & OFPC_PORT_STATS) new_cap |= SC_PORT_STATS;
 266     if (cap & OFPC_QUEUE_STATS) new_cap |= SC_QUEUE_STATS;
 267 
 268     sw.capabilities = new_cap;
 269 
 270     uint32_t actions = ntohl(reply->actions);
 271     uint32_t new_actions = 0;
 272 
 273     if (actions & 1) new_actions |= ACTION_OUTPUT;
 274     if (actions & 2) new_actions |= ACTION_SET_VLAN_VID;
 275     if (actions & 4) new_actions |= ACTION_SET_VLAN_PCP;
 276     if (actions & 8) new_actions |= ACTION_STRIP_VLAN;
 277     if (actions & 16) new_actions |= ACTION_SET_SRC_MAC;
 278     if (actions & 32) new_actions |= ACTION_SET_DST_MAC;
 279     if (actions & 64) new_actions |= ACTION_SET_SRC_IP;
 280     if (actions & 128) new_actions |= ACTION_SET_DST_IP;
 281     if (actions & 256) new_actions |= ACTION_SET_IP_TOS;
 282     if (actions & 512) new_actions |= ACTION_SET_SRC_PORT;
 283     if (actions & 1024) new_actions |= ACTION_SET_DST_PORT;
 284 
 285     sw.actions = actions;
 286 
 287     ev_sw_update_config(OFP_ID, &sw);
 288 
 289     return 0;
 290 }

 445 /**
 446  * \brief Function to process PACKET_IN messages
 447  * \param msg PACKET_IN message
 448  */
 449 static int ofp10_packet_in(const raw_msg_t *msg)
 450 {
 451     struct ofp_packet_in *in = (struct ofp_packet_in *)msg->data;
 452 
 453     uint32_t in_port = ntohs(in->in_port);
 454 
 455     pktin_t pktin = {0};
 456 
 457     switch_t sw = {0};
 458     sw.fd = msg->fd;
 459     ev_sw_get_dpid(OFP_ID, &sw);
 460 
 461     pktin.dpid = sw.dpid;
 462     pktin.port = in_port;
 463 
 464     pktin.xid = ntohl(in->header.xid);
 465     pktin.buffer_id = ntohl(in->buffer_id);
 466 
 467     pktin.version = OFP_VERSION;
 468     pktin.reason = in->reason;
 469 
 470     if (in_port > __MAX_NUM_PORTS) {
 471         discard_packet(&pktin);
 472         return -1;
 473     }
 474 
 475     pktin.total_len = ntohs(in->total_len);
 476     uint8_t *data = (uint8_t *)msg->data + sizeof(struct ofp_packet_in) - 2;
 477     memmove(pktin.data, data, MIN(pktin.total_len, __MAX_PKT_SIZE));
 478 
 479     struct ether_header *eth_header = (struct ether_header *)data;
 480     uint16_t ether_type = ntohs(eth_header->ether_type);
 481 
 482     int base = sizeof(struct ether_header);
 483 
 484     if (ether_type == 0x8100) { // VLAN
 485         struct ether_vlan_header *eth_vlan_header = (struct ether_vlan_header *)data;
 486 
 487         base = sizeof(struct ether_vlan_header);
 488         ether_type = ntohs(eth_vlan_header->ether_type);
 489 
 490         pktin.proto |= PROTO_VLAN;
 491 
 492         pktin.vlan_id = ntohs(eth_vlan_header->tci) & 0xfff;
 493         pktin.vlan_pcp = ntohs(eth_vlan_header->tci) >> 13;
 494     }
 495 
 496     memmove(pktin.src_mac, eth_header->ether_shost, ETH_ALEN);
 497     memmove(pktin.dst_mac, eth_header->ether_dhost, ETH_ALEN);
 498 
 499     switch (ether_type) {
 500     case 0x0800: // IPv4
 501         {
 502             struct iphdr *ip_header = (struct iphdr *)(data + base);
 503 
 504             pktin.proto |= PROTO_IPV4;
 505 
 506             pktin.src_ip = ip_header->saddr;
 507             pktin.dst_ip = ip_header->daddr;
 508 
 509             if (ip_header->protocol == IPPROTO_ICMP) {
 510                 struct icmphdr *icmp_header = (struct icmphdr *)((uint8_t *)ip_header + sizeof(struct iphdr));
 511 
 512                 pktin.proto |= PROTO_ICMP;
 513 
 514                 pktin.type = icmp_header->type;
 515                 pktin.code = icmp_header->code;
 516             } else if (ip_header->protocol == IPPROTO_TCP) {
 517                 struct tcphdr *tcp_header = (struct tcphdr *)((uint8_t *)ip_header + sizeof(struct iphdr));
 518 
 519                 pktin.proto |= PROTO_TCP;
 520 
 521                 pktin.src_port = ntohs(tcp_header->source);
 522                 pktin.dst_port = ntohs(tcp_header->dest);
 523             } else if (ip_header->protocol == IPPROTO_UDP) {
 524                 struct tcphdr *tcp_header = (struct tcphdr *)((uint8_t *)ip_header + sizeof(struct iphdr));
 525 
 526                 pktin.proto |= PROTO_UDP;
 527 
 528                 pktin.src_port = ntohs(tcp_header->source);
 529                 pktin.dst_port = ntohs(tcp_header->dest);
 530             } else { // IPv4
 531                 // no pktin.src_port and pktin.dst_port
 532             }
 533         }
 534         break;
 535     case 0x0806: // ARP
 536         {
 537             struct arphdr *arp_header = (struct arphdr *)(data + base);
 538 
 539             pktin.proto |= PROTO_ARP;
 540 
 541             uint32_t *src_ip = (uint32_t *)arp_header->arp_spa;
 542             uint32_t *dst_ip = (uint32_t *)arp_header->arp_tpa;
 543 
 544             pktin.src_ip = *src_ip;
 545             pktin.dst_ip = *dst_ip;
 546 
 547             pktin.opcode = ntohs(arp_header->ar_op);
 548 
 549             // no pktin.dst_port
 550         }
 551         break;
 552     case 0x88cc: // LLDP
 553         {
 554             pktin.proto |= PROTO_LLDP;
 555 
 556             // no pktin.src_ip, pktin.dst_ip, pktin.src_port, pktin.dst_port
 557         }
 558         break;
 559     default:
 560         {
 561             pktin.proto |= PROTO_UNKNOWN;
 562 
 563             discard_packet(&pktin);
 564 
 565             return -1;
 566         }
 567         break;
 568     }
 569 
 570     ev_dp_receive_packet(OFP_ID, &pktin);
 571 
 572     return 0;
 573 }

 575 /**
 576  * \brief Function to process FLOW_REMOVED messages
 577  * \param msg FLOW_REMOVED message
 578  */
 579 static int ofp10_flow_removed(const raw_msg_t *msg)
 580 {
 581     struct ofp_flow_removed *removed = (struct ofp_flow_removed *)msg->data;
 582 
 583     flow_t flow = {0};
 584 
 585     switch_t sw = {0};
 586     sw.fd = msg->fd;
 587     ev_sw_get_dpid(OFP_ID, &sw);
 588 
 589     flow.dpid = sw.dpid;
 590     flow.port = ntohs(removed->match.in_port);
 591 
 592     flow.xid = ntohl(removed->header.xid);
 593     flow.cookie = ntohll(removed->cookie);
 594     flow.priority = ntohs(removed->priority);
 595 
 596     flow.idle_timeout = ntohs(removed->idle_timeout);
 597 
 598     flow.remote = FALSE;
 599     flow.version = OFP_VERSION;
 600 
 601     // begin of match
 602 
 603     uint32_t wildcards = ntohl(removed->match.wildcards);
 604     uint32_t new_wildcards = 0;
 605 
 606     if (wildcards) {
 607         if (wildcards & OFPFW_IN_PORT) new_wildcards |= WCARD_PORT;
 608         if (wildcards & OFPFW_DL_VLAN) new_wildcards |= WCARD_VLAN;
 609         if (wildcards & OFPFW_DL_SRC) new_wildcards |= WCARD_SRC_MAC;
 610         if (wildcards & OFPFW_DL_DST) new_wildcards |= WCARD_DST_MAC;
 611         if (wildcards & OFPFW_DL_TYPE) new_wildcards |= WCARD_ETH_TYPE;
 612         if (wildcards & OFPFW_NW_PROTO) new_wildcards |= WCARD_PROTO;
