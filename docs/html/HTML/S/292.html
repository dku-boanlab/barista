  33 /* A single table of flows.
  34  *
  35  * All functions, except destroy, must be called holding the
  36  * rcu_read_lock.  destroy must be fully serialized.
  37  */
  38 struct sw_table {
  39         /* The number of packets that have been looked up and matched,
  40          * respecitvely.  To make these 100% accurate, they should be atomic.  
  41          * However, we're primarily concerned about speed. */
  42         unsigned long long n_lookup;
  43         unsigned long long n_matched;
  44 
  45         /* Searches 'table' for a flow matching 'key', which must not have any
  46          * wildcard fields.  Returns the flow if successful, a null pointer
  47          * otherwise. */
  48         struct sw_flow *(*lookup)(struct sw_table *table,
  49                         const struct sw_flow_key *key);
  50 
  51         /* Inserts 'flow' into 'table', replacing any duplicate flow.  Returns
  52          * 0 if successful or a negative error.  Error can be due to an
  53          * over-capacity table or because the flow is not one of the kind that
  54          * the table accepts.
  55          *
  56          * If successful, 'flow' becomes owned by 'table', otherwise it is
  57          * retained by the caller. */
  58         int (*insert)(struct sw_table *table, struct sw_flow *flow);
  59 
  60         /* Modifies the actions in 'table' that match 'key'.  If 'strict'
  61          * set, wildcards and priority must match.  Returns the number of flows 
  62          * that were modified. */
  63         int (*modify)(struct sw_table *table, const struct sw_flow_key *key,
  64                         uint16_t priority, int strict,
  65                         const struct ofp_action_header *actions, size_t actions_len);
  66 
  67         /* Checks whether 'table' has an equal priotiry  entry of thethat conflicts
  68          * with 'key'.  If 'strict' is set, wildcards must match.
  69          * Returns the number of flows that were modified. */
  70         int (*has_conflict)(struct sw_table *table, const struct sw_flow_key *key,
  71                             uint16_t priority, int strict);
  72 
  73         /* Deletes from 'table' any and all flows that match 'key' from
  74          * 'table'.  If 'out_port' is not OFPP_NONE, then matching entries
  75          * must have that port as an argument for an output action.  If 
  76          * 'strict' is set, wildcards and priority must match.  Returns the 
  77          * number of flows that were deleted. */
  78         int (*delete)(struct datapath *dp, struct sw_table *table, 
  79                         const struct sw_flow_key *key, 
  80                         uint16_t out_port, uint16_t priority, int strict);
  81 
  82         /* Performs timeout processing on all the flow entries in 'table'.
  83          * Returns the number of flow entries deleted through expiration. */
  84         int (*timeout)(struct datapath *dp, struct sw_table *table);
  85 
  86         /* Destroys 'table', which must not have any users. */
  87         void (*destroy)(struct sw_table *table);
  88 
  89         /* Iterates through the flow entries in 'table', passing each one
  90          * matches 'key' and output port 'out_port' to 'callback'.  The 
  91          * callback function should return 0 to continue iteration or a 
  92          * nonzero error code to stop.  The iterator function returns either 
  93          * 0 if the table iteration completed or the value returned by the 
  94          * callback function otherwise.
  95          *
  96          * The iteration starts at 'position', which may be initialized to
  97          * all-zero-bits to iterate from the beginning of the table.  If the
  98          * iteration terminates due to an error from the callback function,
  99          * 'position' is updated to a value that can be passed back to the
 100          * iterator function to continue iteration later from the same position
 101          * that caused the error (assuming that that flow entry has not been
 102          * deleted in the meantime). */
 103         int (*iterate)(struct sw_table *table,
 104                        const struct sw_flow_key *key, uint16_t out_port,
 105                        struct sw_table_position *position,
 106                        int (*callback)(struct sw_flow *flow, void *private),
 107                        void *private);
 108 
 109         /* Dumps statistics for 'table' into 'stats'. */
 110         void (*stats)(struct sw_table *table, struct sw_table_stats *stats);
 111 };
