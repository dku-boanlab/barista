 384 /* The match type indicates the match structure (set of fields that compose the

 396 /* Fields to match against flows */
 397 struct ofp_match {
 398     uint16_t type;             /* One of OFPMT_* */
 399     uint16_t length;           /* Length of ofp_match (excluding padding) */
 400     /* Followed by:
 401      *   - Exactly (length - 4) (possibly 0) bytes containing OXM TLVs, then
 402      *   - Exactly ((length + 7)/8*8 - length) (between 0 and 7) bytes of
 403      *     all-zero bytes
 404      * In summary, ofp_match is padded as needed, to make its overall size
 405      * a multiple of 8, to preserve alignment in structures using it.
 406      */
 407     uint8_t oxm_fields[0];     /* 0 or more OXM match fields */
 408     uint8_t pad[4];            /* Zero bytes - see above for sizing */
 409 };

 412 /* Components of a OXM TLV header.
 413  * Those macros are not valid for the experimenter class, macros for the
 414  * experimenter class will depend on the experimenter header used. */
 415 #define OXM_HEADER__(CLASS, FIELD, HASMASK, LENGTH) \
 416     (((CLASS) << 16) | ((FIELD) << 9) | ((HASMASK) << 8) | (LENGTH))
 417 #define OXM_HEADER(CLASS, FIELD, LENGTH) \
 418     OXM_HEADER__(CLASS, FIELD, 0, LENGTH)
 419 #define OXM_HEADER_W(CLASS, FIELD, LENGTH) \
 420     OXM_HEADER__(CLASS, FIELD, 1, (LENGTH) * 2)
 421 #define OXM_CLASS(HEADER) ((HEADER) >> 16)
 422 #define OXM_FIELD(HEADER) (((HEADER) >> 9) & 0x7f)
 423 #define OXM_TYPE(HEADER) (((HEADER) >> 9) & 0x7fffff)
 424 #define OXM_HASMASK(HEADER) (((HEADER) >> 8) & 1)
 425 #define OXM_LENGTH(HEADER) ((HEADER) & 0xff)
 426 
 427 #define OXM_MAKE_WILD_HEADER(HEADER) \
 428         OXM_HEADER_W(OXM_CLASS(HEADER), OXM_FIELD(HEADER), OXM_LENGTH(HEADER))
