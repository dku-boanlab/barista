  37 /* This is the public domain lookup3 hash by Bob Jenkins from
  38  * http://burtleburtle.net/bob/c/lookup3.c, modified for style. */
  39 #define rot(x, k) (((x) << (k)) | ((x) >> (32 - (k))))
  40 
  41 #define mix(a, b, c)                            \
  42     do {                                        \
  43       a -= c; a ^= rot(c,  4); c += b;          \
  44       b -= a; b ^= rot(a,  6); a += c;          \
  45       c -= b; c ^= rot(b,  8); b += a;          \
  46       a -= c; a ^= rot(c, 16); c += b;          \
  47       b -= a; b ^= rot(a, 19); a += c;          \
  48       c -= b; c ^= rot(b,  4); b += a;          \
  49     } while (0)
  50 
  51 #define final(a, b, c)                          \
  52     do {                                        \
  53       c ^= b; c -= rot(b, 14);                  \
  54       a ^= c; a -= rot(c, 11);                  \
  55       b ^= a; b -= rot(a, 25);                  \
  56       c ^= b; c -= rot(b, 16);                  \
  57       a ^= c; a -= rot(c,  4);                  \
  58       b ^= a; b -= rot(a, 14);                  \
  59       c ^= b; c -= rot(b, 24);                  \
  60     } while (0)

  62 /* Returns the hash of the 'n' 32-bit words at 'p', starting from 'basis'.
  63  * 'p' must be properly aligned. */
  64 uint32_t
  65 hash_words(const uint32_t *p, size_t n, uint32_t basis)
  66 {
  67     uint32_t a, b, c;
  68 
  69     a = b = c = 0xdeadbeef + (((uint32_t) n) << 2) + basis;
  70 
  71     while (n > 3) {
  72         a += p[0];
  73         b += p[1];
  74         c += p[2];
  75         mix(a, b, c);
  76         n -= 3;
  77         p += 3;
  78     }
  79 
  80     switch (n) {
  81     case 3:
  82         c += p[2];
  83         /* fall through */
  84     case 2:
  85         b += p[1];
  86         /* fall through */
  87     case 1:
  88         a += p[0];
  89         final(a, b, c);
  90         /* fall through */
  91     case 0:
  92         break;
  93     }
  94     return c;
  95 }

  97 /* Returns the hash of the 'n' bytes at 'p', starting from 'basis'. */
  98 uint32_t
  99 hash_bytes(const void *p_, size_t n, uint32_t basis)
 100 {
 101     const uint8_t *p = p_;
 102     uint32_t a, b, c;
 103     uint32_t tmp[3];
 104 
 105     a = b = c = 0xdeadbeef + n + basis;
 106 
 107     while (n >= sizeof tmp) {
 108         memcpy(tmp, p, sizeof tmp);
 109         a += tmp[0];
 110         b += tmp[1];
 111         c += tmp[2];
 112         mix(a, b, c);
 113         n -= sizeof tmp;
 114         p += sizeof tmp;
 115     }
 116 
 117     if (n) {
 118         tmp[0] = tmp[1] = tmp[2] = 0;
 119         memcpy(tmp, p, n);
 120         a += tmp[0];
 121         b += tmp[1];
 122         c += tmp[2];
 123         final(a, b, c);
 124     }
 125 
 126     return c;
 127 }
