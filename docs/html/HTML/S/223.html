 193 /*
 194  * Check the iface name to make sure we can find the interface
 195  */
 196 int check_iface(struct nf2device *nf2)
 197 {
 198         struct stat buf;
 199         char filename[PATHLEN];
 200 
 201         /* See if we can find the interface name as a network device */
 202 
 203         /* Test the length first of all */
 204         if (strlen(nf2->device_name) > IFNAMSIZ)
 205         {
 206                 fprintf(stderr, "Interface name is too long: %s\n", nf2->device_name);
 207                 return -1;
 208         }
 209 
 210         /* Check for /sys/class/net/iface_name */
 211         strcpy(filename, "/sys/class/net/");
 212         strcat(filename, nf2->device_name);
 213         if (stat(filename, &buf) == 0)
 214         {
 215                 nf2->net_iface = 1;
 216                 return 0;
 217         }
 218 
 219         /* Check for /dev/iface_name */
 220         strcpy(filename, "/dev/");
 221         strcat(filename, nf2->device_name);
 222         if (stat(filename, &buf) == 0)
 223         {
 224                 nf2->net_iface = 0;
 225                 return 0;
 226         }
 227 
 228         fprintf(stderr, "Can't find device: %s\n", nf2->device_name);
 229         return -1;
 230 }

 232 /*
 233  * Open the descriptor associated with the device name
 234  */
 235 int openDescriptor(struct nf2device *nf2)
 236 {
 237         struct ifreq ifreq;
 238         char filename[PATHLEN];
 239         struct sockaddr_in address;
 240         int i;
 241         struct sockaddr_in *sin = (struct sockaddr_in *) &ifreq.ifr_addr;
 242         int found = 0;
 243 
 244         if (nf2->net_iface)
 245         {
 246                 /* Open a network socket */
 247                 nf2->fd = socket(AF_INET, SOCK_DGRAM, 0);
 248                 if (nf2->fd == -1)
 249                 {
 250                         perror("socket: creating socket");
 251                         return -1;
 252                 }
 253                 else
 254                 {
 255                         /* Root can bind to a network interface.
 256                            Non-root has to bind to a network address. */
 257                         if (geteuid() == 0)
 258                         {
 259                                 strncpy(ifreq.ifr_ifrn.ifrn_name, nf2->device_name, IFNAMSIZ);
 260                                 if (setsockopt(nf2->fd, SOL_SOCKET, SO_BINDTODEVICE,
 261                                         (char *)&ifreq, sizeof(ifreq)) < 0) {
 262                                         perror("setsockopt: setting SO_BINDTODEVICE");
 263                                         return -1;
 264                                 }
 265 
 266                         }
 267                         else
 268                         {
 269                                 /* Attempt to find the IP address for the interface */
 270                                 for (i = 1; ; i++)
 271                                 {
 272                                         /* Find interface number i*/
 273                                         ifreq.ifr_ifindex = i;
 274                                         if (ioctl (nf2->fd, SIOCGIFNAME, &ifreq) < 0)
 275                                                 break;
 276 
 277                                         /* Check if we've found the correct interface */
 278                                         if (strcmp(ifreq.ifr_name, nf2->device_name) != 0)
 279                                                 continue;
 280 
 281                                         /* If we get to here we've found the IP */
 282                                         found = 1;
 283                                         break;
 284                                 }
 285 
 286                                 /* Verify that we found the interface */
 287                                 if (!found)
 288                                 {
 289                                         fprintf(stderr, "Can't find device: %s\n", nf2->device_name);
 290                                         return -1;
 291                                 }
 292 
 293                                 /* Attempt to get the IP address associated with the interface */
 294                                 if (ioctl (nf2->fd, SIOCGIFADDR, &ifreq) < 0)
 295                                 {
 296                                         perror("ioctl: calling SIOCGIFADDR");
 297 
 298                                         fprintf(stderr, "Unable to find IP address for device: %s\n", nf2->device_name);
 299                                         fprintf(stderr, "Either run this program as root or ask an administrator\n");
 300                                         fprintf(stderr, "to assign an IP address to the device\n");
 301                                         return -1;
 302                                 }
 303 
 304                                 /* Set the addres and attempt to bind to the socket */
 305                                 address.sin_family = AF_INET;
 306                                 address.sin_addr.s_addr = sin->sin_addr.s_addr;
 307                                 address.sin_port = htons(0);
 308                                 if (bind(nf2->fd,(struct sockaddr *)&address,sizeof(address)) == -1) {
 309                                         perror("bind: binding");
 310                                         return -1;
 311                                 }
 312                         }
 313                 }
 314         }
 315         else
 316         {
 317                 strcpy(filename, "/dev/");
 318                 strcat(filename, nf2->device_name);
 319                 nf2->fd = fileno(fopen(filename, "w+"));
 320                 if (nf2->fd == -1)
 321                 {
 322                         perror("fileno: creating descriptor");
 323                         return -1;
 324                 }
 325         }
 326 
 327         return 0;
 328 }

 347 /*
 348  *  Read the version info from the Virtex and CPCI
 349  */
 350 void nf2_read_info(struct nf2device *nf2)
 351 {
 352         int i;
 353         int md5_good = 1;
 354         unsigned md5[MD5_LEN];
 355         unsigned cpci_id;
 356         unsigned nf2_cpci_id;
 357 
 358         // Read the CPCI version/revision
 359         readReg(nf2, CPCI_REG_ID, &cpci_id);
 360         cpci_version = cpci_id & 0xffffff;
 361         cpci_revision = cpci_id >> 24;
 362 
 363         // Verify the MD5 checksum of the device ID block
 364         for  (i = 0; i < MD5_LEN; i++) {
 365                 readReg(nf2, DEV_ID_MD5_0_REG + i * 4, &md5[i]);
 366         }
 367         md5_good &= md5[0] == DEV_ID_MD5_VALUE_0;
 368         md5_good &= md5[1] == DEV_ID_MD5_VALUE_1;
 369         md5_good &= md5[2] == DEV_ID_MD5_VALUE_2;
 370         md5_good &= md5[3] == DEV_ID_MD5_VALUE_3;
 371 
 372         // Process only if the MD5 sum is good
 373         if (md5_good) {
 374                 // Read the version and revision
 375                 readReg(nf2, DEV_ID_DEVICE_ID_REG, &nf2_device_id);
 376                 readReg(nf2, DEV_ID_REVISION_REG, &nf2_revision);
 377                 readReg(nf2, DEV_ID_CPCI_ID_REG, &nf2_cpci_id);
 378                 nf2_cpci_version = nf2_cpci_id & 0xffffff;
 379                 nf2_cpci_revision = nf2_cpci_id >> 24;
 380 
 381                 // Read the version string
 382                 for (i = 0; i < (DEVICE_STR_LEN / 4) - 2; i++)
 383                 {
 384                         readReg(nf2, DEV_ID_DEV_STR_0_REG + i * 4, (unsigned *)(nf2_device_str + i * 4));
 385 
 386                         // Perform byte swapping if necessary
 387                         *(unsigned *)(nf2_device_str + i * 4) = ntohl(*(unsigned *)(nf2_device_str + i * 4));
 388                 }
 389                 nf2_device_str[DEVICE_STR_LEN - 1] = '\0';
 390         }
 391 }
