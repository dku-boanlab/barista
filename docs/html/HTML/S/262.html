  45 recv_of_set_dp_desc(struct datapath *dp, const struct sender * sender,
  46     const struct openflow_extension_header  *ofexth)
  47 {
  48     struct openflow_ext_set_dp_desc * set_dp_desc = (struct openflow_ext_set_dp_desc * )
  49                 ofexth;
  50     memcpy(dp->dp_desc, set_dp_desc->dp_desc, DESC_STR_LEN);
  51     dp->dp_desc[DESC_STR_LEN-1] = 0;        // force null for safety
  52     return 0;

  56 int
  57 openflow_ext_recv_msg(struct sw_chain *chain, const struct sender *sender,
  58                  const void *ofph)
  59 {
  60         int error = 0;
  61     const struct openflow_queue_command_header  *ofexth = ofph;
  62 
  63     switch (ntohl(ofexth->header.subtype)) {
  64         /**** added here as a place holder
  65          * case OFP_EXT_QUEUE_MODIFY: {
  66          *                              recv_of_exp_queue_modify(dp,sender,oh);
  67          *                             return 0;
  68          *                         }
  69          * case OFP_EXT_QUEUE_DELETE: {
  70          *                             recv_of_exp_queue_delete(dp,sender,oh);
  71          *                             return 0;
  72          *                         }
  73          */
  74         case OFP_EXT_SET_DESC:
  75             return recv_of_set_dp_desc(chain->dp,sender,ofexth);
  76         default:
  77            VLOG_ERR("Received unknown command of type %d",
  78                    ntohl(ofexth->header.subtype));
  79            return -EINVAL;
  80     }
  81 
  82         return error;
  83 }
