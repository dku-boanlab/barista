  71 struct vconn_class {
  72     /* Prefix for connection names, e.g. "nl", "tcp". */
  73     const char *name;
  74 
  75     /* Attempts to connect to an OpenFlow device.  'name' is the full
  76      * connection name provided by the user, e.g. "nl:0", "tcp:1.2.3.4".  This
  77      * name is useful for error messages but must not be modified.
  78      *
  79      * 'suffix' is a copy of 'name' following the colon and may be modified.
  80      *
  81      * Returns 0 if successful, otherwise a positive errno value.  If
  82      * successful, stores a pointer to the new connection in '*vconnp'.
  83      *
  84      * The open function must not block waiting for a connection to complete.
  85      * If the connection cannot be completed immediately, it should return
  86      * EAGAIN (not EINPROGRESS, as returned by the connect system call) and
  87      * continue the connection in the background. */
  88     int (*open)(const char *name, char *suffix, struct vconn **vconnp);
  89 
  90     /* Closes 'vconn' and frees associated memory. */
  91     void (*close)(struct vconn *vconn);
  92 
  93     /* Tries to complete the connection on 'vconn'.  If 'vconn''s connection is
  94      * complete, returns 0 if the connection was successful or a positive errno
  95      * value if it failed.  If the connection is still in progress, returns
  96      * EAGAIN.
  97      *
  98      * The connect function must not block waiting for the connection to
  99      * complete; instead, it should return EAGAIN immediately. */
 100     int (*connect)(struct vconn *vconn);
 101 
 102     /* Tries to receive an OpenFlow message from 'vconn'.  If successful,
 103      * stores the received message into '*msgp' and returns 0.  The caller is
 104      * responsible for destroying the message with ofpbuf_delete().  On
 105      * failure, returns a positive errno value and stores a null pointer into
 106      * '*msgp'.
 107      *
 108      * If the connection has been closed in the normal fashion, returns EOF.
 109      *
 110      * The recv function must not block waiting for a packet to arrive.  If no
 111      * packets have been received, it should return EAGAIN. */
 112     int (*recv)(struct vconn *vconn, struct ofpbuf **msgp);
 113 
 114     /* Tries to queue 'msg' for transmission on 'vconn'.  If successful,
 115      * returns 0, in which case ownership of 'msg' is transferred to the vconn.
 116      * Success does not guarantee that 'msg' has been or ever will be delivered
 117      * to the peer, only that it has been queued for transmission.
 118      *
 119      * Returns a positive errno value on failure, in which case the caller
 120      * retains ownership of 'msg'.
 121      *
 122      * The send function must not block.  If 'msg' cannot be immediately
 123      * accepted for transmission, it should return EAGAIN. */
 124     int (*send)(struct vconn *vconn, struct ofpbuf *msg);
 125 
 126     /* Arranges for the poll loop to wake up when 'vconn' is ready to take an
 127      * action of the given 'type'. */
 128     void (*wait)(struct vconn *vconn, enum vconn_wait_type type);
 129 };

 146 struct pvconn_class {
 147     /* Prefix for connection names, e.g. "ptcp", "pssl". */
 148     const char *name;
 149 
 150     /* Attempts to start listening for OpenFlow connections.  'name' is the
 151      * full connection name provided by the user, e.g. "nl:0", "tcp:1.2.3.4".
 152      * This name is useful for error messages but must not be modified.
 153      *
 154      * 'suffix' is a copy of 'name' following the colon and may be modified.
 155      *
 156      * Returns 0 if successful, otherwise a positive errno value.  If
 157      * successful, stores a pointer to the new connection in '*pvconnp'.
 158      *
 159      * The listen function must not block.  If the connection cannot be
 160      * completed immediately, it should return EAGAIN (not EINPROGRESS, as
 161      * returned by the connect system call) and continue the connection in the
 162      * background. */
 163     int (*listen)(const char *name, char *suffix, struct pvconn **pvconnp);
 164 
 165     /* Closes 'pvconn' and frees associated memory. */
 166     void (*close)(struct pvconn *pvconn);
 167 
 168     /* Tries to accept a new connection on 'pvconn'.  If successful, stores the
 169      * new connection in '*new_vconnp' and returns 0.  Otherwise, returns a
 170      * positive errno value.
 171      *
 172      * The accept function must not block waiting for a connection.  If no
 173      * connection is ready to be accepted, it should return EAGAIN. */
 174     int (*accept)(struct pvconn *pvconn, struct vconn **new_vconnp);
 175 
 176     /* Arranges for the poll loop to wake up when a connection is ready to be
 177      * accepted on 'pvconn'. */
 178     void (*wait)(struct pvconn *pvconn);
 179 };
