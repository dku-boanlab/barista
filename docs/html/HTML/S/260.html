 140 static int
 141 do_uninstall(struct datapath *dpinst, struct sw_table *flowtab,
 142              struct sw_flow *flow, enum ofp_flow_removed_reason reason)
 143 {
 144         /* XXX: Remove the entry from hardware.  If you need to do any other
 145          * clean-up associated with the entry, do it here.
 146          */
 147         dp_send_flow_end(dpinst, flow, reason);
 148         list_del_rcu(&flow->node);
 149         list_del_rcu(&flow->iter_node);
 150         flow_deferred_free(flow);
 151         return 1;

 154 static int
 155 tmpl_uninstall_flow(struct datapath *dpinst, struct sw_table *flowtab,
 156                     const struct sw_flow_key *key, uint16_t out_port,
 157                     uint16_t priority, int strict)
 158 {
 159         struct tmpl_flowtable *myflowtab = (struct tmpl_flowtable *)flowtab;
 160         struct sw_flow *flow;
 161         unsigned int count = 0;
 162 
 163         list_for_each_entry(flow, &myflowtab->flows, node) {
 164                 if (flow_matches_desc(&flow->key, key, strict)
 165                     && (!strict || (flow->priority == priority)))
 166                         count += do_uninstall(dpinst, flowtab,
 167                                               flow, OFPRR_DELETE);
 168         }
 169 
 170         if (count != 0)
 171                 atomic_sub(count, &myflowtab->num_flows);
 172         return count;
 173 }

 175 static int
 176 tmpl_flow_timeout(struct datapath *dpinst, struct sw_table *flowtab)
 177 {
 178         struct tmpl_flowtable *myflowtab = (struct tmpl_flowtable *)flowtab;
 179         struct sw_flow *flow;
 180         int num_uninst_flows = 0;
 181         uint64_t num_forw_packets = 0;
 182         uint64_t num_forw_bytes = 0;
 183         int reason;
 184 
 185         mutex_lock(&dp_mutex);
 186         list_for_each_entry(flow, &myflowtab->flows, node) {
 187                 /* XXX: Retrieve the packet and byte counts associated with this
 188                  * entry and store them in "packet_count" and "byte_count".
 189                  */
 190 #if 0
 191                 num_forw_pakcets = flow->packet_count + get_hwmib(...);
 192                 num_forw_bytes = flow->byte_count + get_hwmib(...);
 193 #endif
 194                 if (num_forw_packets > flow->packet_count
 195                     && flow->idle_timeout != OFP_FLOW_PERMANENT) {
 196                         flow->packet_count = num_forw_packets;
 197                         flow->byte_count = num_forw_bytes;
 198                         flow->used = get_jiffies_64();
 199                 }
 200                 reason = flow_timeout(flow);
 201                 if (reason >= 0) {
 202                         num_uninst_flows += do_uninstall(dpinst, flowtab,
 203                                                          flow, reason);
 204                 }
 205         }
 206         mutex_unlock(&dp_mutex);
 207 
 208         if (num_uninst_flows != 0)
 209                 atomic_sub(num_uninst_flows, &myflowtab->num_flows);
 210         return num_uninst_flows;
 211 }

 213 static void
 214 tmpl_destroy_flowtable(struct sw_table *flowtab)
 215 {
 216         struct tmpl_flowtable *myflowtab = (struct tmpl_flowtable *)flowtab;
 217 
 218         if (myflowtab == NULL) {
 219                 return;
 220         }
 221 
 222         /* XXX: This table is being destroyed, so free any data that you
 223          * don't want to leak.
 224          */
 225         while (!list_empty(&myflowtab->flows)) {
 226                 struct sw_flow *flow = list_entry(myflowtab->flows.next,
 227                                                   struct sw_flow, node);
 228                 list_del(&flow->node);
 229                 flow_free(flow);
 230         }
 231 
 232         kfree(myflowtab);
 233 }

 235 static int
 236 tmpl_iterate_flowtable(struct sw_table *flowtab, const struct sw_flow_key *key,
 237                        uint16_t out_port, struct sw_table_position *position,
 238                        int (*callback) (struct sw_flow *, void *),
 239                        void *private)
 240 {
 241         struct tmpl_flowtable *myflowtab = (struct tmpl_flowtable *)flowtab;
 242         struct sw_flow *flow;
 243         unsigned long start;
 244         int error = 0;
 245 
 246         start = ~position->private[0];
 247         list_for_each_entry(flow, &myflowtab->iter_flows, iter_node) {
 248                 if (flow->serial <= start
 249                     && flow_matches_2wild(key, &flow->key)) {
 250                         error = callback(flow, private);
 251                         if (error) {
 252                                 position->private[0] = ~flow->serial;
 253                                 return error;
 254                         }
 255                 }
 256         }
 257 
 258         return error;
 259 }

 273 static struct sw_table *
 274 tmpl_create_flowtable(void)
 275 {
 276         struct tmpl_flowtable *myflowtab;
 277         struct sw_table *flowtab;
 278 
 279         myflowtab = kzalloc(sizeof(*myflowtab), GFP_KERNEL);
 280         if (myflowtab == NULL)
 281                 return NULL;
 282 
 283         flowtab = &myflowtab->flowtab;
 284         flowtab->lookup = tmpl_flowtable_lookup;
 285         flowtab->insert = tmpl_install_flow;
 286         flowtab->modify = tmpl_modify_flow;
 287         flowtab->delete = tmpl_uninstall_flow;
 288         flowtab->timeout = tmpl_flow_timeout;
 289         flowtab->destroy = tmpl_destroy_flowtable;
 290         flowtab->iterate = tmpl_iterate_flowtable;
 291         flowtab->stats = tmpl_get_flowstats;
 292 
 293         myflowtab->max_flows = TMPL_MAX_FLOWS;
 294         atomic_set(&myflowtab->num_flows, 0);
 295         INIT_LIST_HEAD(&myflowtab->flows);
 296         INIT_LIST_HEAD(&myflowtab->iter_flows);
 297         myflowtab->next_serial = 0;
 298 
 299         return flowtab;
 300 }
