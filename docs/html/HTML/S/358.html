  35 #define LENGTH_OID 16
  36 struct tcaphash_context_t {
  37   struct tcaphash_context_key_t * key;
  38   guint32 session_id;
  39   guint32 first_frame;
  40   guint32 last_frame;
  41   nstime_t begin_time;  /* time of arrival of TC_BEGIN */
  42   nstime_t end_time;    /* time of closing message */
  43   gboolean responded;   /* true, if request has been responded */
  44   gboolean closed;
  45   gboolean upper_dissector;
  46   gboolean oid_present;
  47   gchar oid[LENGTH_OID+1];
  48   gboolean subdissector_present;
  49   dissector_handle_t subdissector_handle;
  50   void (* callback) (tvbuff_t *,packet_info *, proto_tree *, struct tcaphash_context_t *);
  51   struct tcaphash_begincall_t * begincall; 
  52   struct tcaphash_contcall_t * contcall; 
  53   struct tcaphash_endcall_t * endcall;
  54   struct tcaphash_ansicall_t * ansicall; 
  55 };
  56 
  57 struct tcaphash_begincall_t {
  58   struct tcaphash_begin_info_key_t * beginkey;
  59   struct tcaphash_context_t * context;
  60   gboolean father;
  61   struct tcaphash_begincall_t * next_begincall; 
  62   struct tcaphash_begincall_t * previous_begincall; 
  63 };
  64 
  65 struct tcaphash_contcall_t { 
  66   struct tcaphash_cont_info_key_t * contkey;
  67   struct tcaphash_context_t * context;
  68   gboolean father;
  69   struct tcaphash_contcall_t * next_contcall; 
  70   struct tcaphash_contcall_t * previous_contcall; 
  71 };
  72 
  73 struct tcaphash_endcall_t {
  74   struct tcaphash_end_info_key_t * endkey;
  75   struct tcaphash_context_t * context;
  76   gboolean father;
  77   struct tcaphash_endcall_t * next_endcall; 
  78   struct tcaphash_endcall_t * previous_endcall; 
  79 };
  80 
  81 struct tcaphash_ansicall_t {
  82   struct tcaphash_ansi_info_key_t * ansikey;
  83   struct tcaphash_context_t * context;
  84   gboolean father;
  85   struct tcaphash_ansicall_t * next_ansicall; 
  86   struct tcaphash_ansicall_t * previous_ansicall; 
  87 };
