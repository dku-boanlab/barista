  42 /*
  43  * Tagging support.
  44  *
  45  * A 'tag' represents an arbitrary category.  Currently, tags are used to
  46  * represent categories of flows and in particular the dependencies for a flow
  47  * switching decision.  For example, if a flow's output port is based on
  48  * knowledge that source MAC 00:02:e3:0f:80:a4 is on eth0, then a tag that
  49  * represents that dependency is attached to that flow in the flowtracking hash
  50  * table.
  51  *
  52  * As this example shows, the universe of possible categories is very large,
  53  * and even the number of categories that are in use at a given time can be
  54  * very large.  This means that keeping track of category membership via
  55  * conventional means (lists, bitmaps, etc.) is likely to be expensive.
  56  *
  57  * Tags are actually implemented via a "superimposed coding", as discussed in
  58  * Knuth TAOCP v.3 section 6.5 "Retrieval on Secondary Keys".  A tag is an
  59  * unsigned integer in which exactly 2 bits are set to 1 and the rest set to 0.
  60  * For 32-bit integers (as currently used) there are 32 * 31 / 2 = 496 unique
  61  * tags; for 64-bit integers there are 64 * 63 / 2 = 2,016.
  62  *
  63  * Because there is a small finite number of unique tags, tags must collide
  64  * after some number of them have been created.  In practice we generally
  65  * create tags by choosing bits randomly.
  66  *
  67  * The key property of tags is that we can combine them without increasing the
  68  * amount of data required using bitwise-OR, since the result has the 1-bits
  69  * from both tags set.  The necessary tradeoff is that the result is even more
  70  * ambiguous: if combining two tags yields a value with 4 bits set to 1, then
  71  * the result value will test as having 4 * 3 / 2 = 6 unique tags, not just the
  72  * two tags that we combined.
  73  *
  74  * The upshot is this: a value that is the bitwise-OR combination of a number
  75  * of tags will always include the tags that were combined, but it may contain
  76  * any number of additional tags as well.  This is acceptable for flowtracking,
  77  * since we want to be sure that we catch every flow that needs to be
  78  * revalidated, but it is OK if we revalidate a few extra flows as well.
  79  *
  80  * If we combine too many tags, then the result will have every bit set, so
  81  * that it will test as including every tag.  Fortunately, this is not a big
  82  * problem for us: although there are many flows overall, each individual flow
  83  * belongs only to a small number of categories.
  84  */
  85

  94 /* Returns true if 'a' and 'b' have at least one tag in common,
  95  * false if their set of tags is disjoint. . */
  96 static inline bool
  97 tag_intersects(tag_type a, tag_type b)
  98 {
  99     tag_type x = a & b;
 100     return (x & (x - 1)) != 0;
 101 }
 102 
 103 /* Returns true if 'tag' is a valid tag, that is, if exactly two bits are set
 104  * to 1 and the rest to 0.   Otherwise, returns false. */
 105 static inline bool
 106 tag_is_valid(tag_type tag)
 107 {
 108     tag_type x = tag & (tag - 1);
 109     tag_type y = x & (x - 1);
 110     return x && !y;
 111 }

 113 /*
 114  * A tag set accumulates tags with reduced ambiguity compared to a single tag.
 115  * The flow tracking uses tag sets to keep track of tags that need to
 116  * revalidated after a number of packets have been processed.
 117  */
 118 #define TAG_SET_SIZE 4
 119 struct tag_set {
 120     tag_type total;
 121     tag_type tags[TAG_SET_SIZE];
 122     unsigned int n;
 123 };
 124 
 125 void tag_set_init(struct tag_set *);
 126 void tag_set_add(struct tag_set *, tag_type);
 127 static inline bool tag_set_is_empty(const struct tag_set *);
 128 static inline bool tag_set_intersects(const struct tag_set *, tag_type);
 129 
 130 /* Returns true if 'set' will match no tags at all,
 131  * false if it will match at least one tag. */
 132 static inline bool
 133 tag_set_is_empty(const struct tag_set *set)
 134 {
 135     return !set->n;
 136 }
 137 
 138 /* Returns true if any of the tags in 'tags' are also in 'set',
 139  * false if the intersection is empty. */
 140 static inline bool
 141 tag_set_intersects(const struct tag_set *set, tag_type tags)
 142 {
 143     BUILD_ASSERT_DECL(TAG_SET_SIZE == 4);
 144     return (tag_intersects(set->total, tags)
 145             && (tag_intersects(set->tags[0], tags)
 146                 || tag_intersects(set->tags[1], tags)
 147                 || tag_intersects(set->tags[2], tags)
 148                 || tag_intersects(set->tags[3], tags)));
 149 }
