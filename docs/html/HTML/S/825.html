  60 /* Returns a string that represents the contents of the Ethernet frame in the
  61  * 'len' bytes starting at 'data' to 'stream' as output by tcpdump.
  62  * 'total_len' specifies the full length of the Ethernet frame (of which 'len'
  63  * bytes were captured).
  64  *
  65  * The caller must free the returned string.
  66  *
  67  * This starts and kills a tcpdump subprocess so it's quite expensive. */
  68 char *
  69 ofp_packet_to_string(const void *data, size_t len, size_t total_len UNUSED)
  70 {
  71     struct ds ds = DS_EMPTY_INITIALIZER;
  72     struct ofpbuf buf;
  73 
  74     char command[128];
  75     FILE *pcap;
  76     FILE *tcpdump;
  77     int status;
  78     int c;
  79 
  80     buf.data = (void *) data;
  81     buf.size = len;
  82 
  83     pcap = tmpfile();
  84     if (!pcap) {
  85         ofp_error(errno, "tmpfile");
  86         return xstrdup("<error>");
  87     }
  88     pcap_write_header(pcap);
  89     pcap_write(pcap, &buf);
  90     fflush(pcap);
  91     if (ferror(pcap)) {
  92         ofp_error(errno, "error writing temporary file");
  93     }
  94     rewind(pcap);
  95 
  96     snprintf(command, sizeof command, "/usr/sbin/tcpdump -e -n -r /dev/fd/%d 2>/dev/null",
  97              fileno(pcap));
  98     tcpdump = popen(command, "r");
  99     fclose(pcap);
 100     if (!tcpdump) {
 101         ofp_error(errno, "exec(\"%s\")", command);
 102         return xstrdup("<error>");
 103     }
 104 
 105     while ((c = getc(tcpdump)) != EOF) {
 106         ds_put_char(&ds, c);
 107     }
 108 
 109     status = pclose(tcpdump);
 110     if (WIFEXITED(status)) {
 111         if (WEXITSTATUS(status))
 112             ofp_error(0, "tcpdump exited with status %d", WEXITSTATUS(status));
 113     } else if (WIFSIGNALED(status)) {
 114         ofp_error(0, "tcpdump exited with signal %d", WTERMSIG(status)); 
 115     }
 116     return ds_cstr(&ds);
 117 }

 119 /* Pretty-print the OFPT_PACKET_IN packet of 'len' bytes at 'oh' to 'stream'
 120  * at the given 'verbosity' level. */
 121 static void
 122 ofp_packet_in(struct ds *string, const void *oh, size_t len, int verbosity)
 123 {
 124     const struct ofp_packet_in *op = oh;
 125     size_t data_len;
 126 
 127     ds_put_format(string, " total_len=%"PRIu16" in_port=",
 128                   ntohs(op->total_len));
 129     ofp_print_port_name(string, ntohs(op->in_port));
 130 
 131     if (op->reason == OFPR_ACTION)
 132         ds_put_cstr(string, " (via action)");
 133     else if (op->reason != OFPR_NO_MATCH)
 134         ds_put_format(string, " (***reason %"PRIu8"***)", op->reason);
 135 
 136     data_len = len - offsetof(struct ofp_packet_in, data);
 137     ds_put_format(string, " data_len=%zu", data_len);
 138     if (htonl(op->buffer_id) == UINT32_MAX) {
 139         ds_put_format(string, " (unbuffered)");
 140         if (ntohs(op->total_len) != data_len)
 141             ds_put_format(string, " (***total_len != data_len***)");
 142     } else {
 143         ds_put_format(string, " buffer=0x%08"PRIx32, ntohl(op->buffer_id));
 144         if (ntohs(op->total_len) < data_len)
 145             ds_put_format(string, " (***total_len < data_len***)");
 146     }
 147     ds_put_char(string, '\n');
 148 
 149     if (verbosity > 0) {
 150         struct flow flow;
 151         struct ofpbuf packet;
 152         struct ofp_match match;
 153         packet.data = (void *) op->data;
 154         packet.size = data_len;
 155         flow_extract(&packet, ntohs(op->in_port), &flow);
 156         match.wildcards = 0;
 157         match.in_port = flow.in_port;
 158         memcpy(match.dl_src, flow.dl_src, ETH_ADDR_LEN);
 159         memcpy(match.dl_dst, flow.dl_dst, ETH_ADDR_LEN);
 160         match.dl_vlan = flow.dl_vlan;
 161         match.dl_type = flow.dl_type;
 162         match.nw_proto = flow.nw_proto;
 163         match.dl_vlan_pcp = flow.dl_vlan_pcp;
 164         match.nw_tos = flow.nw_tos;
 165         match.nw_src = flow.nw_src;
 166         match.nw_dst = flow.nw_dst;
 167         match.tp_src = flow.tp_src;
 168         match.tp_dst = flow.tp_dst;
 169         ofp_print_match(string, &match, verbosity);
 170         ds_put_char(string, '\n');
 171     }
 172     if (verbosity > 1) {
 173         char *packet = ofp_packet_to_string(op->data, data_len,
 174                                             ntohs(op->total_len)); 
 175         ds_put_cstr(string, packet);
 176         free(packet);
 177     }
 178 }

 279     if (actions_len < sizeof *ah) {

 283 
 284     type = ntohs(ah->type);
 285     len = ntohs(ah->len);
 286     if (actions_len < len) {
 287         ds_put_format(string, "***truncated action %"PRIu16"***\n", type);
 288         return -1;
 289     }
 290

 292         ds_put_format(string,

 301             ds_put_format(string, 
 302                           "***action %"PRIu16" wrong length: %zu %zu %zu***\n", type, len, act->min_size, act->max_size);

 304         }
 305     }

 316                 if (oa->max_len) {
 317                     ds_put_format(string, ":%"PRIu16, ntohs(oa->max_len));
 318                 } else {
 319                     ds_put_cstr(string, ":all");
 320                 }
 321             }
 322         }
 323         break;
 324     }
 325 
 326     case OFPAT_ENQUEUE: {
 327         struct ofp_action_enqueue *ea = (struct ofp_action_enqueue *)ah;
 328         uint16_t port = ntohs(ea->port);
 329         uint32_t queue = ntohl(ea->queue_id);

 333     }
 334 
 335     case OFPAT_SET_VLAN_VID: {

 339     }

 342         struct ofp_action_vlan_pcp *va = (struct ofp_action_vlan_pcp *)ah;

 372         struct ofp_action_nw_addr *na = (struct ofp_action_nw_addr *)ah;
 373         ds_put_format(string, "mod_nw_dst:"IP_FMT, IP_ARGS(&na->nw_addr));
 374         break;
 375     }
 376 
 377     case OFPAT_SET_NW_TOS: {
 378        struct ofp_action_nw_tos *nt = (struct ofp_action_nw_tos *)ah;
 379        ds_put_format(string, "mod_nw_tos:0x%"PRIx8, nt->nw_tos);
 380         break;
 381     }
 382 
 383     case OFPAT_SET_TP_SRC: {
 384         struct ofp_action_tp_port *ta = (struct ofp_action_tp_port *)ah;
 385         ds_put_format(string, "mod_tp_src:%d", ntohs(ta->tp_port));
 386         break;
 387     }
 388 
 389     case OFPAT_SET_TP_DST: {
 390         struct ofp_action_tp_port *ta = (struct ofp_action_tp_port *)ah;
 391         ds_put_format(string, "mod_tp_dst:%d", ntohs(ta->tp_port));
 392         break;
 393     }
 394 
 395     case OFPAT_VENDOR: {
 396         struct ofp_action_vendor_header *avh 
 397                 = (struct ofp_action_vendor_header *)ah;
 398         if (len < sizeof *avh) {
 399             ds_put_format(string, "***ofpat_vendor truncated***\n");

 403         if (1 == 0) {
 404             /* Code to output vendor action */
 405         } else {
 406             ds_put_format(string, "vendor action:0x%x", ntohl(avh->vendor));
 407         }
 408         break;
 409     }
 410 
 411     default:
 412         ds_put_format(string, "(decoder %"PRIu16" not implemented)", type);
 413         break;
 414     }
 415 
 416     return len;
 417 }

 419 static void 
 420 ofp_print_actions(struct ds *string, const struct ofp_action_header *action,
 421                   size_t actions_len) 
 422 {
 423     uint8_t *p = (uint8_t *)action;

 438     }

 441 /* Pretty-print the OFPT_PACKET_OUT packet of 'len' bytes at 'oh' to 'string'
 442  * at the given 'verbosity' level. */
 443 static void ofp_packet_out(struct ds *string, const void *oh, size_t len,
 444                            int verbosity) 
 445 {
 446     const struct ofp_packet_out *opo = oh;
 447     size_t actions_len = ntohs(opo->actions_len);
 448 
 449     ds_put_cstr(string, " in_port=");
 450     ofp_print_port_name(string, ntohs(opo->in_port));
 451 
 452     ds_put_format(string, " actions_len=%zu ", actions_len);
 453     if (actions_len > (ntohs(opo->header.length) - sizeof *opo)) {
 454         ds_put_format(string, "***packet too short for action length***\n");
 455         return;
 456     }
 457     ofp_print_actions(string, opo->actions, actions_len);
 458 
 459     if (ntohl(opo->buffer_id) == UINT32_MAX) {
 460         int data_len = len - sizeof *opo - actions_len;
 461         ds_put_format(string, " data_len=%d", data_len);
 462         if (verbosity > 0 && len > sizeof *opo) {
 463             char *packet = ofp_packet_to_string(
 464                     (uint8_t *)opo->actions + actions_len, data_len, data_len);
 465             ds_put_char(string, '\n');
 466             ds_put_cstr(string, packet);
 467             free(packet);
 468         }
 469     } else {
 470         ds_put_format(string, " buffer=0x%08"PRIx32, ntohl(opo->buffer_id));
 471     }
 472     ds_put_char(string, '\n');
 473 }

 667 char *
 668 ofp_match_to_string(const struct ofp_match *om, int verbosity)
 669 {
 670     struct ds f = DS_EMPTY_INITIALIZER;
 671     uint32_t w = ntohl(om->wildcards);
 672     bool skip_type = false;
 673     bool skip_proto = false;
 674 
 675     if (!(w & OFPFW_DL_TYPE)) {
 676         skip_type = true;
 677         if (om->dl_type == htons(ETH_TYPE_IP)) {
 678             if (!(w & OFPFW_NW_PROTO)) {
 679                 skip_proto = true;
 680                 if (om->nw_proto == IP_TYPE_ICMP) {
 681                     ds_put_cstr(&f, "icmp,");
 682                 } else if (om->nw_proto == IP_TYPE_TCP) {
 683                     ds_put_cstr(&f, "tcp,");
 684                 } else if (om->nw_proto == IP_TYPE_UDP) {
 685                     ds_put_cstr(&f, "udp,");
 686                 } else {
 687                     ds_put_cstr(&f, "ip,");
 688                     skip_proto = false;
 689                 }
 690             } else {
 691                 ds_put_cstr(&f, "ip,");
 692             }
 693         } else if (om->dl_type == htons(ETH_TYPE_ARP)) {
 694             ds_put_cstr(&f, "arp,");
 695         } else {
 696             skip_type = false;
 697         }
 698     }
 699     print_wild(&f, "in_port=", w & OFPFW_IN_PORT, verbosity,
 700                "%d", ntohs(om->in_port));
 701     print_wild(&f, "dl_vlan=", w & OFPFW_DL_VLAN, verbosity,
 702                "0x%04x", ntohs(om->dl_vlan));
 703     print_wild(&f, "dl_vlan_pcp=", w & OFPFW_DL_VLAN_PCP, verbosity,
 704                "0x%02x", om->dl_vlan_pcp);
 705     print_wild(&f, "dl_src=", w & OFPFW_DL_SRC, verbosity,
 706                ETH_ADDR_FMT, ETH_ADDR_ARGS(om->dl_src));
 707     print_wild(&f, "dl_dst=", w & OFPFW_DL_DST, verbosity,
 708                ETH_ADDR_FMT, ETH_ADDR_ARGS(om->dl_dst));
 709     if (!skip_type) {
 710         print_wild(&f, "dl_type=", w & OFPFW_DL_TYPE, verbosity,
 711                    "0x%04x", ntohs(om->dl_type));
 712     }
 713     print_ip_netmask(&f, "nw_src=", om->nw_src,
 714                      (w & OFPFW_NW_SRC_MASK) >> OFPFW_NW_SRC_SHIFT, verbosity);
 715     print_ip_netmask(&f, "nw_dst=", om->nw_dst,
 716                      (w & OFPFW_NW_DST_MASK) >> OFPFW_NW_DST_SHIFT, verbosity);
 717     print_wild(&f, "nw_tos=", w & OFPFW_NW_TOS, verbosity,
 718               "0x%02x", om->nw_tos);
 719     if (!skip_proto) {
 720         print_wild(&f, "nw_proto=", w & OFPFW_NW_PROTO, verbosity,
 721                    "%u", om->nw_proto);
 722     }
 723     if (om->nw_proto == IP_TYPE_ICMP) {
 724         print_wild(&f, "icmp_type=", w & OFPFW_ICMP_TYPE, verbosity,
 725                    "%d", ntohs(om->icmp_type));
 726         print_wild(&f, "icmp_code=", w & OFPFW_ICMP_CODE, verbosity,
 727                    "%d", ntohs(om->icmp_code));
 728     } else {
 729         print_wild(&f, "tp_src=", w & OFPFW_TP_SRC, verbosity,
 730                    "%d", ntohs(om->tp_src));
 731         print_wild(&f, "tp_dst=", w & OFPFW_TP_DST, verbosity,
 732                    "%d", ntohs(om->tp_dst));
 733     }
 734     return ds_cstr(&f);
 735 }

 972                      int verbosity)
 973 {
 974     const char *body = body_;
 975     const char *pos = body;
 976     for (;;) {
 977         const struct ofp_flow_stats *fs;
 978         ptrdiff_t bytes_left = body + len - pos;
 979         size_t length;
 980 
 981         if (bytes_left < sizeof *fs) {
 982             if (bytes_left != 0) {
 983                 ds_put_format(string, " ***%td leftover bytes at end***",
 984                               bytes_left);
 985             }
 986             break;
 987         }
 988 
 989         fs = (const void *) pos;
 990         length = ntohs(fs->length);
 991         if (length < sizeof *fs) {

 997                           " ***length=%zu but only %td bytes left***",
 998                           length, bytes_left);
 999             break;
1000         } else if ((length - sizeof *fs) % sizeof fs->actions[0]) {
1001             ds_put_format(string,
1002                           " ***length=%zu has %zu bytes leftover in "
1003                           "final action***",
1004                           length,
1005                           (length - sizeof *fs) % sizeof fs->actions[0]);
1006             break;
1007         }
1008 
1009         ds_put_format(string, "  cookie=%"PRIu64", ", ntohll(fs->cookie));
1010         ds_put_format(string, "duration_sec=%"PRIu32"s, ",
1011                     ntohl(fs->duration_sec));

1016                     fs->match.wildcards ? ntohs(fs->priority) : (uint16_t)-1);

1525 /* Composes and returns a string representing the OpenFlow packet of 'len'
1526  * bytes at 'oh' at the given 'verbosity' level.  0 is a minimal amount of
1527  * verbosity and higher numbers increase verbosity.  The caller is responsible
1528  * for freeing the string. */
1529 char *
1530 ofp_to_string(const void *oh_, size_t len, int verbosity)
1531 {
1532     struct ds string = DS_EMPTY_INITIALIZER;
1533     const struct ofp_header *oh = oh_;
1534     const struct openflow_packet *pkt;
1535 
1536     if (len < sizeof(struct ofp_header)) {
1537         ds_put_cstr(&string, "OpenFlow packet too short:\n");
1538         ds_put_hex_dump(&string, oh, len, 0, true);
1539         return ds_cstr(&string);
1540     } else if (oh->version != OFP_VERSION) {
1541         ds_put_format(&string, "Bad OpenFlow version %"PRIu8":\n", oh->version);
1542         ds_put_hex_dump(&string, oh, len, 0, true);
1543         return ds_cstr(&string);
1544     }
1545 
1546     for (pkt = packets; ; pkt++) {
1547         if (pkt >= &packets[ARRAY_SIZE(packets)]) {
1548             ds_put_format(&string, "Unknown OpenFlow packet type %"PRIu8":\n",
1549                           oh->type);
1550             ds_put_hex_dump(&string, oh, len, 0, true);
1551             return ds_cstr(&string);
1552         } else if (oh->type == pkt->type) {
1553             break;
1554         }
1555     }
1556 
1557     ds_put_format(&string, "%s (xid=0x%"PRIx32"):", pkt->name, ntohl(oh->xid));
1558 
1559     if (ntohs(oh->length) > len)
1560         ds_put_format(&string, " (***truncated to %zu bytes from %"PRIu16"***)",
1561                 len, ntohs(oh->length));
1562     else if (ntohs(oh->length) < len) {
1563         ds_put_format(&string, " (***only uses %"PRIu16" bytes out of %zu***)\n",
1564                 ntohs(oh->length), len);
1565         len = ntohs(oh->length);
1566     }
1567 
1568     if (len < pkt->min_size) {
1569         ds_put_format(&string, " (***length=%zu < min_size=%zu***)\n",
1570                 len, pkt->min_size);
1571     } else if (!pkt->printer) {
1572         if (len > sizeof *oh) {
1573             ds_put_format(&string, " length=%"PRIu16" (decoder not implemented)\n",
1574                           ntohs(oh->length)); 
1575         }
1576     } else {
1577         pkt->printer(&string, oh, len, verbosity);
1578     }
1579     if (verbosity >= 3) {
1580         ds_put_hex_dump(&string, oh, len, 0, true);
1581     }
1582     if (string.string[string.length - 1] != '\n') {
1583         ds_put_char(&string, '\n');
1584     }
1585     return ds_cstr(&string);
1586 }

1588 /* Returns the name for the specified OpenFlow message type as a string,
1589  * e.g. "OFPT_FEATURES_REPLY".  If no name is known, the string returned is a
1590  * hex number, e.g. "0x55".
1591  *
1592  * The caller must free the returned string when it is no longer needed. */
1593 char *
1594 ofp_message_type_to_string(uint8_t type)
1595 {
1596     struct ds s = DS_EMPTY_INITIALIZER;
1597     const struct openflow_packet *pkt;
1598     for (pkt = packets; ; pkt++) {
1599         if (pkt >= &packets[ARRAY_SIZE(packets)]) {
1600             ds_put_format(&s, "0x%02"PRIx8, type);
1601             break;
1602         } else if (type == pkt->type) {
1603             const char *p;
1604 
1605             ds_put_cstr(&s, "OFPT_");
1606             for (p = pkt->name; *p; p++) {
1607                 ds_put_char(&s, toupper((unsigned char) *p));
1608             }
1609             break;
1610         }
1611     }
1612     return ds_cstr(&s);
1613 }
