  29 /* Verifies that 'hmap' contains exactly the 'n' values in 'values'. */
  30 static void
  31 check_hmap(struct hmap *hmap, const int values[], size_t n,
  32            hash_func *hash)
  33 {
  34     int *sort_values, *hmap_values;
  35     struct element *e;
  36     size_t i;
  37 
  38     /* Check that all the values are there in iteration. */
  39     sort_values = xmalloc(sizeof *sort_values * n);
  40     hmap_values = xmalloc(sizeof *sort_values * n);
  41 
  42     i = 0;
  43     HMAP_FOR_EACH (e, struct element, node, hmap) {
  44         assert(i < n);
  45         hmap_values[i++] = e->value;
  46     }
  47     assert(i == n);
  48 
  49     memcpy(sort_values, values, sizeof *sort_values * n);
  50     qsort(sort_values, n, sizeof *sort_values, compare_ints);
  51     qsort(hmap_values, n, sizeof *hmap_values, compare_ints);
  52 
  53     for (i = 0; i < n; i++) {
  54         assert(sort_values[i] == hmap_values[i]);
  55     }
  56 
  57     free(hmap_values);
  58     free(sort_values);
  59 
  60     /* Check that all the values are there in lookup. */
  61     for (i = 0; i < n; i++) {
  62         size_t count = 0;
  63 
  64         HMAP_FOR_EACH_WITH_HASH (e, struct element, node,
  65                                  hash(values[i]), hmap) {
  66             count += e->value == values[i];
  67         }
  68         assert(count == 1);
  69     }
  70 
  71     /* Check counters. */
  72     assert(hmap_is_empty(hmap) == !n);
  73     assert(hmap_count(hmap) == n);
  74 }

  76 /* Puts the 'n' values in 'values' into 'elements', and then puts those
  77  * elements into 'hmap'. */
  78 static void
  79 make_hmap(struct hmap *hmap, struct element elements[],
  80           int values[], size_t n, hash_func *hash)
  81 {
  82     size_t i;
  83 
  84     hmap_init(hmap);
  85     for (i = 0; i < n; i++) {
  86         elements[i].value = i;

  92 static void
  93 shuffle(int *p, size_t n)
  94 {
  95     for (; n > 1; n--, p++) {
  96         int *q = &p[rand() % n];
  97         int tmp = *p;
  98         *p = *q;
  99         *q = tmp;
 100     }

 150 /* Tests basic hmap insertion and deletion. */
 151 static void
 152 test_hmap_insert_delete(hash_func *hash)
 153 {
 154     enum { N_ELEMS = 100 };
 155 
 156     struct element elements[N_ELEMS];
 157     int values[N_ELEMS];
 158     struct hmap hmap;
 159     size_t i;
 160 
 161     hmap_init(&hmap);
 162     for (i = 0; i < N_ELEMS; i++) {
 163         elements[i].value = i;
 164         hmap_insert(&hmap, &elements[i].node, hash(i));
 165         values[i] = i;
 166         check_hmap(&hmap, values, i + 1, hash);
 167     }
 168     shuffle(values, N_ELEMS);
 169     for (i = 0; i < N_ELEMS; i++) {
 170         hmap_remove(&hmap, &elements[values[i]].node);
 171         check_hmap(&hmap, values + (i + 1), N_ELEMS - (i + 1), hash);
 172     }
 173     hmap_destroy(&hmap);
 174 }

 208 /* Tests that HMAP_FOR_EACH_SAFE properly allows for deletion of the current
 209  * element of a hmap.  */
 210 static void
 211 test_hmap_for_each_safe(hash_func *hash)
 212 {
 213     enum { MAX_ELEMS = 10 };
 214     size_t n;
 215     unsigned long int pattern;
 216 
 217     for (n = 0; n <= MAX_ELEMS; n++) {
 218         for (pattern = 0; pattern < 1ul << n; pattern++) {
 219             struct element elements[MAX_ELEMS];
 220             int values[MAX_ELEMS];
 221             struct hmap hmap;
 222             struct element *e, *next;
 223             size_t n_remaining;
 224             int i;
 225 
 226             make_hmap(&hmap, elements, values, n, hash);
 227 
 228             i = 0;
 229             n_remaining = n;
 230             HMAP_FOR_EACH_SAFE (e, next, struct element, node, &hmap) {
 231                 assert(i < n);
 232                 if (pattern & (1ul << e->value)) {
 233                     size_t j;
 234                     hmap_remove(&hmap, &e->node);
 235                     for (j = 0; ; j++) {
 236                         assert(j < n_remaining);
 237                         if (values[j] == e->value) {
 238                             values[j] = values[--n_remaining];
 239                             break;
 240                         }
 241                     }
 242                 }
 243                 check_hmap(&hmap, values, n_remaining, hash);
 244                 i++;
 245             }
 246             assert(i == n);
 247 
 248             for (i = 0; i < n; i++) {
 249                 if (pattern & (1ul << i)) {
 250                     n_remaining++;
 251                 }
 252             }
 253             assert(n == n_remaining);
 254 
 255             hmap_destroy(&hmap);
 256         }
 257     }
 258 }
