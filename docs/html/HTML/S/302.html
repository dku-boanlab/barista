  53 static u32 __random32(struct rnd_state *state)
  54 {
  55 #define TAUSWORTHE(s,a,b,c,d) ((s&c)<<d) ^ (((s <<a) ^ s)>>b)
  56 
  57         state->s1 = TAUSWORTHE(state->s1, 13, 19, 4294967294UL, 12);
  58         state->s2 = TAUSWORTHE(state->s2, 2, 25, 4294967288UL, 4);
  59         state->s3 = TAUSWORTHE(state->s3, 3, 11, 4294967280UL, 17);
  60 
  61         return (state->s1 ^ state->s2 ^ state->s3);
  62 }

  64 static void __set_random32(struct rnd_state *state, unsigned long s)
  65 {
  66         if (s == 0)
  67                 s = 1;      /* default seed is 1 */
  68 
  69 #define LCG(n) (69069 * n)
  70         state->s1 = LCG(s);
  71         state->s2 = LCG(state->s1);
  72         state->s3 = LCG(state->s2);
  73 
  74         /* "warm it up" */

  76         __random32(state);
  77         __random32(state);
  78         __random32(state);
  79         __random32(state);
  80         __random32(state);

  96 /**
  97  *      srandom32 - add entropy to pseudo random number generator
  98  *      @seed: seed value
  99  *
 100  *      Add some additional seeding to the random32() pool.
 101  *      Note: this pool is per cpu so it only affects current CPU.
 102  */
 103 void srandom32(u32 entropy)
 104 {
 105         struct rnd_state *state = &net_rand_state[smp_processor_id()];
 106         __set_random32(state, state->s1 ^ entropy);

 112 /*
 113  *      Generate some initially weak seeding values to allow
 114  *      to start the random32() engine.
 115  */
 116 int __init random32_init(void)
 117 {
 118         int i;
 119 
 120         for (i = 0; i < NR_CPUS; i++) {
 121                 struct rnd_state *state = &net_rand_state[i];
 122                 __set_random32(state, i + jiffies);
 123         }
 124         random32_reseed();
 125         return 0;
 126 }
