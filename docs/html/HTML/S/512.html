  25 #ifndef MPA_H
  26 #define MPA_H 1
  27 
  28 struct mpa {
  29         unsigned emphasis   :2;
  30         unsigned original   :1;
  31         unsigned copyright  :1;
  32         unsigned modeext    :2;
  33         unsigned mode       :2;
  34         unsigned private    :1;
  35         unsigned padding    :1;
  36         unsigned frequency  :2;
  37         unsigned bitrate    :4;
  38         unsigned protection :1;
  39         unsigned layer      :2;
  40         unsigned version    :2;
  41         unsigned sync       :11;
  42 };
  43 
  44 #define MPA_UNMARSHAL_SYNC(n)       ((n) >> 21 & 0x7ff)
  45 #define MPA_UNMARSHAL_VERSION(n)    ((n) >> 19 & 0x3)
  46 #define MPA_UNMARSHAL_LAYER(n)      ((n) >> 17 & 0x3)
  47 #define MPA_UNMARSHAL_PROTECTION(n) ((n) >> 16 & 0x1)
  48 #define MPA_UNMARSHAL_BITRATE(n)    ((n) >> 12 & 0xf)
  49 #define MPA_UNMARSHAL_FREQUENCY(n)  ((n) >> 10 & 0x3)
  50 #define MPA_UNMARSHAL_PADDING(n)    ((n) >>  9 & 0x1)
  51 #define MPA_UNMARSHAL_PRIVATE(n)    ((n) >>  8 & 0x1)
  52 #define MPA_UNMARSHAL_MODE(n)       ((n) >>  6 & 0x3)
  53 #define MPA_UNMARSHAL_MODEEXT(n)    ((n) >>  4 & 0x3)
  54 #define MPA_UNMARSHAL_COPYRIGHT(n)  ((n) >>  3 & 0x1)
  55 #define MPA_UNMARSHAL_ORIGINAL(n)   ((n) >>  2 & 0x1)
  56 #define MPA_UNMARSHAL_EMPHASIS(n)   ((n) >>  0 & 0x3)
  57 
  58 #define MPA_UNMARSHAL(mpa, n) do { \
  59         (mpa)->sync       = MPA_UNMARSHAL_SYNC(n);       \
  60         (mpa)->version    = MPA_UNMARSHAL_VERSION(n);    \
  61         (mpa)->layer      = MPA_UNMARSHAL_LAYER(n);      \
  62         (mpa)->protection = MPA_UNMARSHAL_PROTECTION(n); \
  63         (mpa)->bitrate    = MPA_UNMARSHAL_BITRATE(n);    \
  64         (mpa)->frequency  = MPA_UNMARSHAL_FREQUENCY(n);  \
  65         (mpa)->padding    = MPA_UNMARSHAL_PADDING(n);    \
  66         (mpa)->private    = MPA_UNMARSHAL_PRIVATE(n);    \
  67         (mpa)->mode       = MPA_UNMARSHAL_MODE(n);       \
  68         (mpa)->modeext    = MPA_UNMARSHAL_MODEEXT(n);    \
  69         (mpa)->copyright  = MPA_UNMARSHAL_COPYRIGHT(n);  \
  70         (mpa)->original   = MPA_UNMARSHAL_ORIGINAL(n);   \
  71         (mpa)->emphasis   = MPA_UNMARSHAL_EMPHASIS(n);   \
  72         } while (0)
  73 
  74 int mpa_version(const struct mpa *);
  75 int mpa_layer(const struct mpa *);
  76 unsigned mpa_samples(const struct mpa *);
  77 unsigned mpa_bitrate(const struct mpa *);
  78 unsigned mpa_frequency(const struct mpa *);
  79 unsigned mpa_padding(const struct mpa *);
  80 
  81 #define MPA_DATA_BYTES(mpa) (mpa_bitrate(mpa) * mpa_samples(mpa) \
  82                 / mpa_frequency(mpa) / 8)
  83 #define MPA_BYTES(mpa) (MPA_DATA_BYTES(mpa) + mpa_padding(mpa))
  84 #define MPA_DURATION_NS(mpa) \
  85         (1000000000 / mpa_frequency(mpa) * mpa_samples(mpa))
  86 
  87 enum { MPA_SYNC = 0x7ff };
  88 
  89 #define MPA_SYNC_VALID(mpa)      ((mpa)->sync == MPA_SYNC)
  90 #define MPA_VERSION_VALID(mpa)   (mpa_version(mpa) >= 0)
  91 #define MPA_LAYER_VALID(mpa)     (mpa_layer(mpa) >= 0)
  92 #define MPA_BITRATE_VALID(mpa)   (mpa_bitrate(mpa) > 0)
  93 #define MPA_FREQUENCY_VALID(mpa) (mpa_frequency(mpa) > 0)
  94 #define MPA_VALID(mpa) (MPA_SYNC_VALID(mpa) \
  95                 && MPA_VERSION_VALID(mpa) && MPA_LAYER_VALID(mpa) \
  96                 && MPA_BITRATE_VALID(mpa) && MPA_FREQUENCY_VALID(mpa))
  97 
  98 #endif
