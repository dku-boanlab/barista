  75 int init(struct oflops_context *ctx, char * config_str) {
  76   char *pos = NULL;
  77   char *param = config_str;
  78   char *value = NULL;
  79   //init counters
  80   sendno = 0;
  81   TAILQ_INIT(&my_tailq_head);
  82   finished = 0;
  83   //open file for storing measurement
  84   measure_output = fopen("measure.log", "w");
  85 
  86   while(*config_str == ' ') {
  87     config_str++;
  88   }
  89   param = config_str;
  90   while(1) {
  91     pos = index(param, ' ');
  92 
  93     if((pos == NULL)) {
  94       if (*param != '\0') {
  95         pos = param + strlen(param) + 1;
  96       } else
  97         break;
  98     }
  99     *pos='\0';
 100     pos++;
 101     value = index(param,'=');
 102     *value = '\0';
 103     value++;
 104     //fprintf(stderr, "param = %s, value = %s\n", param, value);
 105     if(value != NULL) {
 106       if(strcmp(param, "flows") == 0) {
 107         flows = atoi(value);
 108         if(flows <= 0)
 109           perror_and_exit("Invalid flow number",1);
 110       } else if(strcmp(param, "network") == 0) {
 111         network = (char *)xmalloc(strlen(value) + 1);
 112         strcpy(network, value);
 113       } else if(strcmp(param, "pkt_size") == 0) {
 114         //parse int to get pkt size
 115         pkt_size = strtol(value, NULL, 0);
 116         if((pkt_size < MIN_PKT_SIZE) && (pkt_size > MAX_PKT_SIZE))  {
 117           perror_and_exit("Invalid packet size value", 1);
 118         }
 119       } else if(strcmp(param, "data_rate") == 0) {
 120         //parse int to get pkt size
 121         datarate = strtol(value, NULL, 0);
 122         if((datarate <= 0) || (datarate > 1010))  {
 123           perror_and_exit("Invalid data rate param(Values between 1 and 1010)", 1);
 124         }
 125       }  else if(strcmp(param, "probe_rate") == 0) {
 126         //parse int to get pkt size
 127         proberate = strtol(value, NULL, 0);
 128         if((proberate <= 0) || (proberate >= 1010)) {
 129           perror_and_exit("Invalid probe rate param(Value between 1 and 1010)", 1);
 130         }
 131       } else {
 132         fprintf(stderr, "Invalid parameter:%s\n", param);
 133       }
 134       param = pos;
 135     }
 136   } 
 137 
 138   //calculate sendind interval
 139   data_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (datarate * mbits_to_bits);
 140   fprintf(stderr, "Sending data interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 141       (uint32_t)data_snd_interval, (uint32_t)pkt_size, (uint32_t)datarate);
 142   probe_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (proberate * mbits_to_bits);
 143   fprintf(stderr, "Sending probe interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 144       (uint32_t)probe_snd_interval, (uint32_t)pkt_size, (uint32_t)proberate);
 145 
 146   return 0;
 147 }
