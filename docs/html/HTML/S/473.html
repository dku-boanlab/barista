  26 #ifndef __NLPID_H__
  27 #define __NLPID_H__
  28 
  29 /* X.263 / ISO/IEC TR 9577 NLPID values. */
  30 
  31 #define NLPID_NULL              0x00
  32 #define NLPID_IPI_T_70          0x01    /* T.70, when an IPI */
  33 #define NLPID_SPI_X_29          0x01    /* X.29, when an SPI */
  34 #define NLPID_X_633             0x03    /* X.633 */
  35 #define NLPID_Q_931             0x08    /* Q.931, Q.932, X.36, ISO 11572, ISO 11582 */
  36 #define NLPID_Q_933             0x08    /* Q.933, on Frame Relay */
  37 #define NLPID_Q_2931            0x09    /* Q.2931 */
  38 #define NLPID_Q_2119            0x0c    /* Q.2119 */
  39 #define NLPID_SNAP              0x80
  40 #define NLPID_ISO8473_CLNP      0x81    /* X.233 */
  41 #define NLPID_ISO9542_ESIS      0x82
  42 #define NLPID_ISO10589_ISIS     0x83
  43 #define NLPID_ISO10747_IDRP     0x85
  44 #define NLPID_ISO9542X25_ESIS   0x8a
  45 #define NLPID_ISO10030          0x8c
  46 #define NLPID_ISO11577          0x8d    /* X.273 */
  47 #define NLPID_IP6               0x8e
  48 #define NLPID_COMPRESSED        0xb0    /* "Data compression protocol" */
  49 #define NLPID_SNDCF             0xc1    /* "SubNetwork Dependent Convergence Function */
  50 #define NLPID_IP                0xcc
  51 #define NLPID_PPP               0xcf
  52 
  53 extern const value_string nlpid_vals[];
  54 
  55 /*
  56  * 0x09 is, in Frame Relay, LMI, Q.2931.
  57  */
  58 #define NLPID_LMI               0x09    /* LMI */
  59 
  60 #endif
