 148 static int cli_load(struct cli_def *cli, UNUSED(const char *command), char *argv[], int argc)
 149 {
 150     // load components
 151     if (component_load(cli, cli_ctx)) {
 152         cli_print(cli, "Failed to load component configurations");
 153         return CLI_ERROR;
 154     }
 155 
 156     // load applications
 157     if (application_load(cli, cli_ctx)) {
 158         cli_print(cli, "Failed to load application configurations");
 159         return CLI_ERROR;
 160     }
 161 
 162     // autostart
 163     if (component_activate(cli, "log")) {
 164         cli_print(cli, "Failed to activate the log component");
 165         return CLI_ERROR;
 166     }
 167 
 168     // automatically load operator-defined policies
 169     FILE *fp = fopen(DEFAULT_ODP_FILE, "r");
 170     if (fp != NULL) {
 171         char buf[__CONF_STR_LEN] = {0};
 172 
 173         cli_print(cli, "Operator-defined policy file: %s", DEFAULT_ODP_FILE);
 174 
 175         while (fgets(buf, __CONF_STR_LEN-1, fp) != NULL) {
 176             if (buf[0] == '#') continue;
 177 
 178             int cnt = 0;
 179             char *args[__CONF_ARGC + 1] = {0};
 180 
 181             str2args(buf, &cnt, args, __CONF_ARGC);
 182 
 183             if (cnt != 3) continue;
 184 
 185             if (strcmp(args[0], "component") == 0) {
 186                 component_add_policy(cli, args[1], args[2]);
 187             } else if (strcmp(args[0], "application") == 0) {
 188                 application_add_policy(cli, args[1], args[2]);
 189             }
 190         }
 191 
 192         fclose(fp);
 193     } else {
 194         cli_print(cli, "No operator-defined policy");
 195     }
 196 
 197     return CLI_OK;
 198 }

 705         if (component_cli(cli, &argv[0]) < 0)
 706             return CLI_ERROR;
 707         else
 708             return CLI_OK;
 709     }
 710 
 711     return CLI_ERROR;
 712 }

 714 static int cli_application(struct cli_def *cli, UNUSED(const char *command), char *argv[], int argc)
 715 {
 716     if (argc > 0) {
 717         if (application_cli(cli, &argv[0]) < 0)
 718             return CLI_ERROR;
 719         else
 720             return CLI_OK;
 721     }
 722 
 723     return CLI_ERROR;
 724 }

 726 static int cli_exit(struct cli_def *cli, UNUSED(const char *command), char *argv[], int argc)
 727 {
 728     component_stop(cli);
 729 
 730     waitsec(0, 1000 * 1000);
 731 
 732     application_stop(cli);
 733 
 734     waitsec(0, 1000 * 1000);
 735 
 736     cli_print(cli, " ");
 737 
 738     component_deactivate(cli, "log");
 739 
 740     waitsec(0, 1000);
 741 
 742     int pid;
 743     char *argb[] = {"pkill", "-9", "barista_sb", NULL};
 744     posix_spawn(&pid, "/usr/bin/pkill", NULL, NULL, argb, NULL);
 745 
 746     char *argr[] = {"pkill", "-9", "barista_rsm", NULL};
 747     posix_spawn(&pid, "/usr/bin/pkill", NULL, NULL, argr, NULL);
 748 
 749     cli_print(cli, "\nReady to be terminated");
 750 
 751     return CLI_OK;
 752 }

 754 static int check_auth(const char *username, const char *password)
 755 {
 756     if (strcasecmp(username, "admin") != 0)
 757         return CLI_ERROR;
 758     if (strcasecmp(password, "secret") != 0)
 759         return CLI_ERROR;
 760     return CLI_OK;
 761 }

 764 {
 765     return CLI_OK;

 768 static int check_enable(const char *password)

 770     return !strcasecmp(password, "topsecret");

 774 {
 775     return CLI_QUIT;

 778 struct arg_struct {
 779     struct cli_def *cli;
 780     int acc_sock;
 781 };

 783 static void *connection_handler(void *arguments)
 784 {
 785     struct arg_struct *args = (struct arg_struct *)arguments;
 786     struct cli_def *cli = args->cli;
 787     int acc_sock = args->acc_sock;
 788 
 789     cli_loop(cli, acc_sock);
 790 
 791     return NULL;
 792 }

 798

 800     int cli_sock, acc_sock;
 801     struct sockaddr_in addr;
 802     int on = 1;
 803 
 804     signal(SIGCHLD, SIG_IGN);
 805 
 806     cli = cli_init();
 807     cli_set_banner(cli, "Barista Console");
 808     cli_set_hostname(cli, "Barista");

 834     cli_register_command(cli, cc, "application", cli_set_deactivate_application, PRIVILEGE_PRIVILEGED, MODE_EXEC, "[Application Name], Deactivate applications");
 835 
 836     c = cli_register_command(cli, NULL, "odp", NULL, PRIVILEGE_UNPRIVILEGED, MODE_EXEC, NULL);
 837 
 838     cc = cli_register_command(cli, c, "add", NULL, PRIVILEGE_PRIVILEGED, MODE_EXEC, NULL);
 839     cli_register_command(cli, cc, "component", cli_policy_add_component, PRIVILEGE_PRIVILEGED, MODE_EXEC, "[Component Name] [Policy...], Add a policy to a component");
 840     cli_register_command(cli, cc, "application", cli_policy_add_application, PRIVILEGE_PRIVILEGED, MODE_EXEC, "[Application Name] [Policy...], Add a policy to an application");

 879     cli_register_command(cli, c, "traffic", cli_stat_traffic, PRIVILEGE_UNPRIVILEGED, MODE_EXEC, "[Time(sec)], Show the traffic statistics for the last n seconds");

 881     cli_register_command(cli, NULL, "component", cli_component, PRIVILEGE_UNPRIVILEGED, MODE_EXEC, "[Component Name] [Arguments...], Run a command in a component");
 882     cli_register_command(cli, NULL, "application", cli_application, PRIVILEGE_UNPRIVILEGED, MODE_EXEC, "[Application Name] [Arguments...], Run a command in an application");
 883 
 884     cli_register_command(cli, NULL, "exit", cli_exit, PRIVILEGE_PRIVILEGED, MODE_EXEC, "Terminate the Barista NOS");
 885 
 886     cli_set_auth_callback(cli, check_auth);
 887     cli_set_enable_callback(cli, check_enable);
 888 
 889     if ((cli_sock = socket(AF_INET, SOCK_STREAM, 0)) < 0) {
 890         perror("socket");
 891         return -1;
 892     }
 893 
 894     setsockopt(cli_sock, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on));
 895 
 896     memset(&addr, 0, sizeof(addr));
 897     addr.sin_family = AF_INET;
 898     addr.sin_addr.s_addr = htonl(INADDR_ANY);
 899     addr.sin_port = htons(__CLI_PORT);
 900 
 901     if (bind(cli_sock, (struct sockaddr *)&addr, sizeof(addr)) < 0) {
 902         perror("bind");
 903         return -1;
 904     }
 905 
 906     if (listen(cli_sock, 10) < 0) {
 907         perror("listen");
 908         return -1;
 909     }
 910 
 911     if (ctx->autostart) {
 912         component_load(NULL, cli_ctx);
 913         application_load(NULL, cli_ctx);
 914 
 915         waitsec(0, 1000 * 1000);
 916 
 917         component_activate(NULL, "log");
 918 
 919         waitsec(0, 1000 * 1000);
 920 
 921         component_start(NULL);
 922 
 923         waitsec(0, 1000 * 1000);
 924 
 925         application_start(NULL);
 926 
 927         // automatically load operator-defined policies
 928         FILE *fp = fopen(DEFAULT_ODP_FILE, "r");
 929         if (fp != NULL) {
 930             char buf[__CONF_STR_LEN] = {0};
 931 
 932             PRINTF("Operator-defined policy file: %s\n", DEFAULT_ODP_FILE);
 933 
 934             while (fgets(buf, __CONF_STR_LEN-1, fp) != NULL) {
 935                 if (buf[0] == '#') continue;
 936 
 937                 int cnt = 0;
 938                 char *args[__CONF_ARGC + 1] = {0};
 939 
 940                 str2args(buf, &cnt, args, __CONF_ARGC);
 941 
 942                 if (cnt != 3) continue;
 943 
 944                 if (strcmp(args[0], "component") == 0) {
 945                     component_add_policy(NULL, args[1], args[2]);
 946                 } else if (strcmp(args[0], "application") == 0) {
 947                     application_add_policy(NULL, args[1], args[2]);
 948                 }
 949             }
 950 
 951             fclose(fp);
 952         } else {
 953             PRINTF("No operator-defined policy\n");
 954         }
 955 
 956         if (system("touch /tmp/barista.lock") != 0) {
 957             cli_print(cli, "Failed to create /tmp/barista.lock\n");
 958         }
 959     }
 960 
 961     sigint_func = signal(SIGINT, sigint_handler);
 962     sigkill_func = signal(SIGKILL, sigkill_handler);
 963     sigterm_func = signal(SIGTERM, sigterm_handler);
 964 
 965     PRINTF("Listening on port %d for the Barista CLI\n", __CLI_PORT);
 966 
 967     while ((acc_sock = accept(cli_sock, NULL, 0))) {
 968         socklen_t len = sizeof(addr);
 969         if (getpeername(acc_sock, (struct sockaddr *)&addr, &len) >= 0) {
 970             PRINTF("Accepted connection from %s\n", inet_ntoa(addr.sin_addr));
 971         }
 972 
 973         struct arg_struct args;
 974         args.cli = cli;
 975         args.acc_sock = acc_sock;
 976 
 977         pthread_t thread;
 978         if (pthread_create(&thread, NULL, connection_handler, (void *)&args) < 0) {
 979             PERROR("pthread_create");
 980             return -1;
 981         }
 982     }
 983 
 984     if (acc_sock < 0) {
 985         PERROR("accept");
 986         return -1;
 987     }
 988 
 989     cli_done(cli);
 990 
 991     return 0;
 992 }
