 250 /** Handle timer event
 251  * @param ctx pointer to opaque context
 252  * @param te pointer to timer event
 253  */
 254 int handle_timer_event(struct oflops_context * ctx, struct timer_event *te) {  
 255   char *str = te->arg; 
 256   struct timeval now;
 257   struct tailq_entry *item;
 258   int len;
 259   void *b;
 260   int i;
 261 
 262   //terminate process 
 263   if (strcmp(str, BYESTR) == 0) {
 264     printf("terminating test....\n");
 265     oflops_end_test(ctx,1);
 266     finished = 0;
 267 
 268     for (i = 0; i < 2;i++) {
 269       
 270       float mean = ((count[i] == 0)?FLT_MAX:sum[i]/count[i]);
 271       float var = ((count[i] == 0)?FLT_MAX:(sum_square[i]/count[i]) - mean*mean);
 272       char msg[400];
 273       gettimeofday(&now, NULL);
 274       snprintf(msg, 400, "case %d : mean = %f, var=%f", i, mean, var); //, sqrt(var));
 275       oflops_log(now, GENERIC_MSG, msg);
 276     }
 277     return 0;    
 278   } else if (strcmp(str, SND_PKT) == 0) {
 279     sendno++;
 280     *((uint32_t *)(buf + MEASUREMENT_PACKET_HEADER)) = sendno;
 281     
 282     gettimeofday(&now, NULL);
 283     
 284     item = xmalloc(sizeof(struct tailq_entry));
 285     set_timeval(&item->value.scheduled, &now);
 286     item->value.id = sendno;
 287     pthread_mutex_lock(&mutex1);
 288     TAILQ_INSERT_TAIL(&my_tailq_head, item, entries);
 289     pthread_mutex_unlock(&mutex1);
 290     oflops_send_raw_mesg(ctx, probe_send/*OFLOPS_DATA1*/, buf, pkt_size);
 291     
 292     if(sendno % 10000 == 0) 
 293       printf("send id : %d\n", sendno);
 294     
 295     //schedule next packet send
 296     if(!finished) {
 297       add_time(&te->sched_time, 0, probe_snd_interval);
 298       oflops_schedule_timer_event(ctx,&te->sched_time, SND_PKT);
 299     }
 300   } else if (strcmp(str, SND_ACT) == 0) {
 301     printf("sending flow modification\n");
 302     len = make_ofp_flow_modify(&b, fl_probe, command, command_len, 
 303                                1, 1200);
 304     oflops_send_of_mesg(ctx, b);
 305     free(b);
 306   }
 307   return 0;
 308 }

 328 /** Handle pcap event.
 329  * @param ctx pointer to opaque context
 330  * @param pe pcap event
 331  * @param ch enumeration of channel that pcap event is triggered
 332  */
 333 int 
 334 handle_pcap_event(struct oflops_context *ctx, struct pcap_event * pe, oflops_channel_name ch) {
 335   struct pktgen_hdr *pktgen;
 336   int dir, len, res;
 337   void *b;
 338   struct ofp_header *ofp;
 339   struct pcap_event *ofp_msg;
 340   struct ofp_error_msg *err_p = NULL;
 341 
 342   if (ch == OFLOPS_CONTROL) {
 343     dir = append_data_to_flow(pe->data,pe->pcaphdr);
 344     while(contains_next_msg(dir) > 0) {
 345       len = get_next_msg(dir, &ofp_msg);
 346       ofp = (struct ofp_header *)ofp_msg->data;
 347       switch(ofp->type) {
 348       case OFPT_FLOW_MOD:
 349         oflops_log(pe->pcaphdr.ts,OFPT_FLOW_MOD_ADD, "stats request send");
 350       case OFPT_ECHO_REQUEST:
 351         make_ofp_hello(&b);
 352         ((struct ofp_header *)b)->type = OFPT_ECHO_REPLY;
 353         res = oflops_send_of_mesgs(ctx, b, sizeof(struct ofp_hello));
 354         free(b);
 355         break;  
 356       case OFPT_ERROR:
 357         err_p = (struct ofp_error_msg *)ofp;
 358         char *msg = xmalloc(sizeof("OFPT_ERROR(type: XXXXXXXXXX, code: XXXXXXXXXX)"));
 359         sprintf(msg, "OFPT_ERROR(type: %d, code: %d)", ntohs(err_p->type), ntohs(err_p->code));
 360         oflops_log(pe->pcaphdr.ts, OFPT_ERROR_MSG, msg);
 361         fprintf(stderr, "%s\n", msg);
 362         break;   
 363       }
 364     }
 365   } else if ((ch == probe_receive_1) || (ch == probe_receive_2)) {
 366     //if(dummy_parse_packet(pe->data, pe->pcaphdr.caplen) == 0) {
 367     struct flow fl;
 368     int c = 0;
 369     pktgen = extract_pktgen_pkt(pe->data, pe->pcaphdr.caplen, &fl);
 370     if(((vlan_id) && (fl.dl_vlan == 16)) ||
 371        ((vlan_strip) && (fl.dl_vlan == 0))) {
 372       c = 1;
 373     }
 374     //printf("%d %d %x %d\n", vlan_id, vlan_strip, fl.dl_vlan, c);
 375     uint64_t diff = 1000000*(pe->pcaphdr.ts.tv_sec - htonl(pktgen->time.tv_sec)) +
 376       (pe->pcaphdr.ts.tv_usec - htonl(pktgen->time.tv_usec));
 377     sum[c] += diff;
 378     sum_square[c] += diff*diff;
 379     count[c]++;
 380     if(htonl(pktgen->seq_num) % 100000 == 0)
 381       printf("data packet received %llu\n", (long long unsigned int)ntohl(pktgen->seq_num));
 382     if(fprintf(measure_output, "%lu;%lu.%06lu;%lu.%06lu;%d\n", 
 383                (long unsigned int)htonl(pktgen->seq_num),  
 384                (long unsigned int)htonl(pktgen->time.tv_sec), 
 385                (long unsigned int)htonl(pktgen->time.tv_usec), 
 386                (long unsigned int)pe->pcaphdr.ts.tv_sec, 
 387                (long unsigned int)pe->pcaphdr.ts.tv_usec, c) < 0)  
 388       perror_and_exit("fprintf fail", 1); 
 389     fflush(measure_output);
 390   }
 391   return 0;
 392 }

 467 /*
 468  * read the first 2 hex characters and return the byte.
 469  */
 470 uint8_t
 471 read_hex(const char *data) {
 472   uint8_t ret = 0;
 473   int i;
 474 
 475   for(i = 0 ; i < 2 ; i++) {
 476     ret = ret << 4;
 477     if((*data >= 'A') && (*data <= 'F')) {
 478       ret += 10 + (*data - 'A');
 479     } else if((*data >= 'a') && (*data <= 'f')) {
 480       ret += 10 + ((*data) - 'a');
 481     } else if((*data >= '0') && (*data <= '9')) {
 482       ret += ((*data) - '0');
 483     }
 484     data++;
 485   }
 486   return ret;
 487 }

 493 int init(struct oflops_context *ctx, char * config_str) {
 494   char *pos = NULL;
 495   char *param = config_str;
 496   int len = strlen(config_str);
 497   char *value = NULL;
 498   char *action;
 499 
 500   struct ofp_action_output *act_out;
 501 
 502   //init counters
 503   sendno = 0;
 504   TAILQ_INIT(&my_tailq_head);
 505   finished = 0;
 506 
 507   while(*config_str == ' ') {
 508     config_str++;
 509   }
 510   param = config_str;
 511   while(1) {
 512     pos = index(param, ' ');
 513 
 514     if((pos == NULL)) {
 515       if (*param != '\0') {
 516         pos = param + strlen(param) + 1;
 517       } else
 518         break;
 519     }
 520     *pos='\0';
 521     pos++;
 522     value = index(param,'=');
 523     if(value != NULL) {
 524       *value = '\0';
 525       value++;
 526     }
 527     fprintf(stderr, "param = %s, value = %s\n", param, value);
 528     if(strcmp(param, "pkt_size") == 0) {
 529       //parse int to get pkt size
 530       pkt_size = strtol(value, NULL, 0);
 531       if((pkt_size < MIN_PKT_SIZE) && (pkt_size > MAX_PKT_SIZE))
 532         perror_and_exit("Invalid packet size value", 1);
 533     } else if(strcmp(param, "data_rate") == 0) {
 534       //parse int to get rate of background data
 535       datarate = strtol(value, NULL, 0);
 536       if((datarate <= 0) || (datarate > 1010))
 537         perror_and_exit("Invalid data rate param(Values between 1 and 1010)", 1);
 538     }  else if(strcmp(param, "probe_rate") == 0) {
 539       //parse int to get measurement probe rate
 540       proberate = strtol(value, NULL, 0);
 541       if((proberate <= 0) || (proberate >= 1010)) 
 542         perror_and_exit("Invalid probe rate param(Value between 1 and 1010)", 1);
 543     }  else if(strcmp(param, "action") == 0) {
 544       char *p = value;
 545       while((*p != ' ') && (*p != '\0') && (config_str + len > p)) {
 546         action = p;
 547         //find where value ends and set it to null to extract the string.
 548         p = index(p, ',');
 549         if(p == NULL) {
 550             p = config_str + len + 1;
 551             *p='\0';
 552         } else {
 553           *p = '\0'; 
 554           p++;
 555         }
 556         
 557         //set null char to split action param and action value
 558         param = index(action, '/');
 559           if(param != NULL) {
 560             *param = '\0';
 561             param++;
 562           }
 563           
 564           printf("%s : %s\n", action, param);
 565           
 566           //check if action value is correct
 567           if(*action < '0' || *action > '9') { printf("invalid action: %1s", action); continue; }
 568           //append the action to the action list.
 569           append_action((*action) - '0', param);
 570         }
 571       } else if(strcmp(param, "table") == 0) {
 572         //parse int to get pkt size
 573         table = strtol(value, NULL, 0);
 574         if((table < 0) && (table > 2))  
 575           perror_and_exit("Invalid table number", 1);
 576       } else if(strcmp(param, "flows") == 0) {
 577         //parse int to get pkt size
 578         flows = strtol(value, NULL, 0);
 579         if(flows <= 0)  
 580           perror_and_exit("Invalid flow number", 1);
 581       } else if(strcmp(param, "vlan_id") == 0) {
 582         //change vlan_id
 583       //      char test = ('0' + new_vlan_id);
 584         append_action(1, new_vlan_id);
 585         vlan_id = 1;
 586       } else if(strcmp(param, "vlan_pcp") == 0) {
 587         //change vlan_id
 588         append_action(2, new_vlan_pcp);
 589         vlan_pcp = 1;
 590       } else if(strcmp(param, "vlan_strip") == 0) {
 591         //change vlan_id
 592         append_action(3, NULL);
 593         vlan_strip = 1;
 594       } else 
 595         fprintf(stderr, "Invalid parameter:%s\n", param);
 596       param = pos;
 597     }
 598 
 599   //calculate sendind interval
 600   data_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (datarate * mbits_to_bits);
 601   fprintf(stderr, "Sending data interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 602       (uint32_t)data_snd_interval, (uint32_t)pkt_size, (uint32_t)datarate);
 603   probe_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (proberate * mbits_to_bits);
 604   fprintf(stderr, "Sending probe interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 605       (uint32_t)probe_snd_interval, (uint32_t)pkt_size, (uint32_t)proberate);
 606 
 607   //by default the new rule should be redirected to port 2 to make the measurement easier
 608   fprintf(stderr, "by default output packet to port 1\n");
 609   command_len += sizeof(struct ofp_action_output);
 610   command = realloc(command, command_len);
 611   act_out = (struct ofp_action_output *)
 612     (command + (command_len - sizeof(struct ofp_action_output)));
 613   act_out->type = htons(0);
 614   act_out->len = htons(8);
 615   act_out->max_len = htons(0);
 616   act_out->port = htons(probe_receive_2 - 1);
 617

 620   sum_square[0] = 0;
 621   sum_square[1] = 0;
 622   sum[0] = 0;
 623   sum[1] = 0;
 624   return 0;

 748 /*
 749  * check if the char array contains hex like characters only.
 750  */
 751 int
 752 is_hex(const char *data, int len) {
 753   int i;
 754   for(i = 0 ; i < len; i++) {
 755     if(!( ((*data >= 'A') && (*data <= 'F')) ||
 756           ((*data >= 'a') && (*data <= 'f')) || 
 757           ((*data >= '0') && (*data <= '9')) )) 
 758       return 0;
 759     data++;
 760   }
 761   return 1;
 762 }
