  53 /**
  54  * \brief Function to generate a flow from a user input
  55  * \param cli The CLI pointer
  56  * \param cmd Command
  57  * \param dpid Datapath ID
  58  * \param options Match and action data
  59  */
  60 static int flow_push(cli_t *cli, char *cmd, char *dpid, char *options)
  61 {
  62     flow_t flow = {0};
  63 
  64     // dpid
  65     flow.dpid = strtoull(dpid, NULL, 0);
  66 
  67     // default setup
  68     flow.buffer_id = -1;
  69 
  70     // action check
  71     int action_check = 0;
  72 
  73     // fields
  74     char *token = strtok(options, " ;");
  75 
  76     do {
  77         char field[__CONF_WORD_LEN] = {0};
  78         char value[__CONF_WORD_LEN] = {0};
  79 
  80         if (sscanf(token, "%[^'=']=%[^'=']", field, value) != 2) {
  81             cli_print(cli, "wrong format");
  82             return -1;
  83         }
  84 
  85         if (strcmp(field, "priority") == 0) {
  86             flow.priority = atoi(value);
  87         } else if (strcmp(field, "idle_timeout") == 0) {
  88             flow.idle_timeout = atoi(value);
  89         } else if (strcmp(field, "hard_timeout") == 0) {
  90             flow.hard_timeout = atoi(value);
  91         } else if (strcmp(field, "proto") == 0) {
  92             if (strcmp(value, "ipv4") == 0) {
  93                 flow.proto |= PROTO_IPV4;
  94             } else if (strcmp(value, "tcp") == 0) {
  95                 flow.proto |= PROTO_TCP;
  96             } else if (strcmp(value, "udp") == 0) {
  97                 flow.proto |= PROTO_UDP;
  98             } else if (strcmp(value, "icmp") == 0) {
  99                 flow.proto |= PROTO_ICMP;
 100             }
 101         } else if (strcmp(field, "vlan_id") == 0) {
 102             flow.proto |= PROTO_VLAN;
 103             flow.vlan_id = atoi(value);
 104         } else if (strcmp(field, "vlan_pcp") == 0) {
 105             if (flow.proto & PROTO_VLAN) {
 106                 flow.vlan_pcp = atoi(value);
 107             }
 108         } else if (strcmp(field, "src_mac") == 0) {
 109             int val[ETH_ALEN];
 110             if (sscanf(value, "%02x:%02x:%02x:%02x:%02x:%02x", &val[0], &val[1], &val[2], &val[3], &val[4], &val[5]) != 6) {
 111                 cli_print(cli, "wrong value");
 112                 return -1;
 113             }
 114             int i;
 115             for (i=0; i<ETH_ALEN; i++) {
 116                 flow.src_mac[i] = (uint8_t)val[i];
 117             }
 118         } else if (strcmp(field, "dst_mac") == 0) {
 119             int val[ETH_ALEN];
 120             if (sscanf(value, "%02x:%02x:%02x:%02x:%02x:%02x", &val[0], &val[1], &val[2], &val[3], &val[4], &val[5]) != 6) {
 121                 cli_print(cli, "wrong value");
 122                 return -1;
 123             }
 124             int i;
 125             for (i=0; i<ETH_ALEN; i++) {
 126                 flow.dst_mac[i] = (uint8_t)val[i];
 127             }
 128         } else if ((flow.proto & PROTO_IPV4) && strcmp(field, "src_ip") == 0) {
 129             flow.src_ip = inet_addr(value);
 130         } else if ((flow.proto & PROTO_IPV4) && strcmp(field, "dst_ip") == 0) {
 131             flow.dst_ip = inet_addr(value);
 132         } else if ((flow.proto & PROTO_TCP || flow.proto & PROTO_UDP) && strcmp(field, "src_port") == 0) {
 133             flow.src_port = atoi(value);
 134         } else if ((flow.proto & PROTO_TCP || flow.proto & PROTO_UDP) && strcmp(field, "dst_port") == 0) {
 135             flow.dst_port = atoi(value);
 136         } else if (flow.proto & PROTO_ICMP && strcmp(field, "icmp_type") == 0) {
 137             flow.type = atoi(value);
 138         } else if (flow.proto & PROTO_ICMP && strcmp(field, "icmp_code") == 0) {
 139             flow.code = atoi(value);
 140         } else if (strcmp(field, "in_port") == 0) {
 141             flow.port = atoi(value);
 142         } else if (strcmp(field, "action") == 0) {
 143             if (flow.num_actions >= __MAX_NUM_ACTIONS) {
 144                 cli_print(cli, "action overflow");
 145                 return -1;
 146             }
 147 
 148             if (strncmp(value, "output", 6) == 0) {
 149                 char forward[__CONF_WORD_LEN] = {0};
 150                 char out_port[__CONF_WORD_LEN] = {0};
 151 
 152                 if (sscanf(value, "%[^','],%[^',']", forward, out_port) == 2) {
 153                     action_check = TRUE;
 154                     flow.action[flow.num_actions].type = ACTION_OUTPUT;
 155                     flow.action[flow.num_actions].port = atoi(out_port);
 156                     flow.num_actions++;
 157                 } else {
 158                     cli_print(cli, "wrong value");
 159                     return -1;
 160                 }
 161             } else if (strcmp(value, "broadcast") == 0) {
 162                 action_check = TRUE;
 163                 flow.action[flow.num_actions].type = ACTION_OUTPUT;
 164                 flow.action[flow.num_actions].port = PORT_FLOOD;
 165                 flow.num_actions++;
 166             } else {
 167                 cli_print(cli, "wrong value");
 168                 return -1; 
 169             }
 170         } else {
 171             cli_print(cli, "wrong field");
 172             return -1;
 173         }
 174     } while ((token = strtok(NULL, " ;")));
 175 
 176     if (action_check == FALSE) {
 177         cli_print(cli, "no action");
 178         return -1;
 179     }
 180 
 181     if (!(flow.proto & PROTO_VLAN)) {
 182         flow.vlan_id = VLAN_NONE;
 183     }
 184 
 185     if (strcmp(cmd, "add") == 0) {
 186         ev_dp_insert_flow(FLOW_PUSH_ID, &flow);
 187     } else if (strcmp(cmd, "del") == 0) {
 188         ev_dp_delete_flow(FLOW_PUSH_ID, &flow);
 189     }
 190 
 191     return 0;
 192 }

 195  * \brief The CLI function

 199 int flow_push_cli(cli_t *cli, char **args)
 200 {
 201     if (args[0] != NULL && args[1] != NULL && args[2] != NULL && args[3] == NULL) {
 202         if (flow_push(cli, args[0], args[1], args[2]) == 0) 
 203             return 0;
 204         else
 205             return -1;
 206     }
 207 
 208     cli_print(cli, "<Command Usage>");
 209     cli_print(cli, "  flow_push [command] [datapath ID] [field=value];[field=value];...;[action=value]}");
 210     cli_print(cli, "command -> { add | del }");
 211     cli_print(cli, "dpid    -> datapath id");
 212     cli_print(cli, "fields  -> priority=[priority] (default: 32768)");
 213     cli_print(cli, "           idle_timeout=[time(sec)] (default: 0)");
 214     cli_print(cli, "           hard_timeout=[time(sec)] (default: 0)");
 215     cli_print(cli, "           proto={ ipv4 | tcp | udp | icmp }");
 216     cli_print(cli, "           vlan_id=[vlan id]");
 217     cli_print(cli, "           vlan_pcp=[vlan priority]");
 218     cli_print(cli, "           src_mac=[xx:xx:xx:xx:xx:xx]");
 219     cli_print(cli, "           dst_mac=[xx.xx.xx.xx.xx.xx]");
 220     cli_print(cli, "           src_ip=[x.x.x.x]");
 221     cli_print(cli, "           dst_ip=[x.x.x.x]");
 222     cli_print(cli, "           src_port=[port number] for TCP/UDP");
 223     cli_print(cli, "           dst_port=[port number] for TCP/UDP");
 224     cli_print(cli, "           icmp_type=[type] for ICMP");
 225     cli_print(cli, "           icmp_code=[code] for ICMP");
 226     cli_print(cli, "           in_port=[port number]");
 227     cli_print(cli, "actions -> action={ output,[port_number] | broadcast }");
 228 
 229     return 0;
 230 }
