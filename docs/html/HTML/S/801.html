 102 int
 103 main(int argc, char *argv[])
 104 {
 105     struct settings s;
 106 
 107     struct list relays = LIST_INITIALIZER(&relays);
 108 
 109     struct secchan secchan;
 110 
 111     struct pvconn *monitor;
 112 
 113     struct pvconn *listeners[MAX_MGMT];
 114     size_t n_listeners;
 115 
 116     char *local_rconn_name;
 117     struct rconn *async_rconn, *local_rconn, *remote_rconn;
 118     struct relay *controller_relay;
 119     struct discovery *discovery;
 120     struct switch_status *switch_status;
 121     struct port_watcher *pw;
 122     int i;
 123     int retval;
 124 
 125     set_program_name(argv[0]);
 126     register_fault_handlers();
 127     time_init();
 128     vlog_init();
 129     parse_options(argc, argv, &s);
 130     signal(SIGPIPE, SIG_IGN);
 131 
 132     secchan.hooks = NULL;
 133     secchan.n_hooks = 0;
 134     secchan.allocated_hooks = 0;
 135 
 136     /* Start listening for management and monitoring connections. */
 137     n_listeners = 0;
 138     for (i = 0; i < s.n_listeners; i++) {
 139         listeners[n_listeners++] = open_passive_vconn(s.listener_names[i]);
 140     }
 141     monitor = s.monitor_name ? open_passive_vconn(s.monitor_name) : NULL;
 142 
 143     /* Initialize switch status hook. */
 144     switch_status_start(&secchan, &s, &switch_status);
 145 
 146     die_if_already_running();
 147     daemonize();
 148 
 149     /* Start listening for vlogconf requests. */
 150     retval = vlog_server_listen(NULL, NULL);
 151     if (retval) {
 152         ofp_fatal(retval, "Could not listen for vlog connections");
 153     }
 154 
 155     VLOG_INFO("OpenFlow reference implementation version %s", VERSION BUILDNR);
 156     VLOG_INFO("OpenFlow protocol version 0x%02x", OFP_VERSION);
 157 
 158     /* Check datapath name, to try to catch command-line invocation errors. */
 159     if (strncmp(s.dp_name, "nl:", 3) && strncmp(s.dp_name, "unix:", 5)
 160         && !s.controller_names[0]) {
 161         VLOG_WARN("Controller not specified and datapath is not nl: or "
 162                   "unix:.  (Did you forget to specify the datapath?)");
 163     }
 164 
 165     if (!strncmp(s.dp_name, "nl:", 3)) {
 166         /* Connect to datapath with a subscription for asynchronous events.  By
 167          * separating the connection for asynchronous events from that for
 168          * request and replies we prevent the socket receive buffer from being
 169          * filled up by received packet data, which in turn would prevent
 170          * getting replies to any Netlink messages we send to the kernel. */
 171         async_rconn = rconn_create(0, s.max_backoff);
 172         rconn_connect(async_rconn, s.dp_name);
 173         switch_status_register_category(switch_status, "async",
 174                                         rconn_status_cb, async_rconn);
 175     } else {
 176         /* No need for a separate asynchronous connection: we must be connected
 177          * to the user datapath, which is smart enough to discard packet events
 178          * instead of message replies.  In fact, having a second connection
 179          * would work against us since we'd get double copies of asynchronous
 180          * event messages (the user datapath provides no way to turn off
 181          * asynchronous events). */
 182         async_rconn = NULL;
 183     }
 184 
 185     /* Connect to datapath without a subscription, for requests and replies. */
 186     local_rconn_name = vconn_name_without_subscription(s.dp_name);
 187     local_rconn = rconn_create(0, s.max_backoff);
 188     rconn_connect(local_rconn, local_rconn_name);
 189     free(local_rconn_name);
 190     switch_status_register_category(switch_status, "local",
 191                                     rconn_status_cb, local_rconn);
 192 
 193     /* Connect to controller. */
 194     remote_rconn = rconn_create(s.probe_interval, s.max_backoff);
 195     if (s.controller_names[0]) {
 196         retval = rconn_connect(remote_rconn, s.controller_names[0]);
 197         if (retval == EAFNOSUPPORT) {
 198             ofp_fatal(0, "No support for %s vconn", s.controller_names[0]);
 199         }
 200     }
 201     switch_status_register_category(switch_status, "remote",
 202                                     rconn_status_cb, remote_rconn);
 203 
 204     /* Start relaying. */
 205     controller_relay = relay_create(async_rconn, local_rconn, remote_rconn,
 206                                     false);
 207     list_push_back(&relays, &controller_relay->node);
 208 
 209     /* Set up hooks. */
 210     port_watcher_start(&secchan, local_rconn, remote_rconn, &pw);
 211     discovery = s.discovery ? discovery_init(&s, pw, switch_status) : NULL;
 212     if (s.enable_stp) {
 213         stp_start(&secchan, pw, local_rconn, remote_rconn);
 214     }
 215     if (s.in_band) {
 216         in_band_start(&secchan, &s, switch_status, pw, remote_rconn);
 217     }
 218     if (s.fail_mode == FAIL_OPEN) {
 219         fail_open_start(&secchan, &s, switch_status,
 220                         local_rconn, remote_rconn);
 221     }
 222     if (s.num_controllers > 1) {
 223         failover_start(&secchan, &s, switch_status, remote_rconn);
 224     }
 225     if (s.n_listeners > 0) {
 226         protocol_stat_start(&secchan, &s, local_rconn, remote_rconn);
 227     }
 228     if (s.rate_limit) {
 229         rate_limit_start(&secchan, &s, switch_status, remote_rconn);
 230     }
 231     if (s.emerg_flow) {
 232         emerg_flow_start(&secchan, &s, switch_status, local_rconn, remote_rconn);
 233     }
 234 
 235     while (s.discovery || rconn_is_alive(remote_rconn)) {
 236         struct relay *r, *n;
 237         size_t i;
 238 
 239         /* Do work. */
 240         LIST_FOR_EACH_SAFE (r, n, struct relay, node, &relays) {
 241             relay_run(r, &secchan);
 242         }
 243         for (i = 0; i < n_listeners; i++) {
 244             for (;;) {
 245                 struct relay *r = relay_accept(&s, listeners[i]);
 246                 if (!r) {
 247                     break;
 248                 }
 249                 list_push_back(&relays, &r->node);
 250             }
 251         }
 252         if (monitor) {
 253             struct vconn *new = accept_vconn(monitor);
 254             if (new) {
 255                 /* XXX should monitor async_rconn too but rconn_add_monitor()
 256                  * takes ownership of the vconn passed in. */
 257                 rconn_add_monitor(local_rconn, new);
 258             }
 259         }
 260         for (i = 0; i < secchan.n_hooks; i++) {
 261             if (secchan.hooks[i].class->periodic_cb) {
 262                 secchan.hooks[i].class->periodic_cb(secchan.hooks[i].aux);
 263             }
 264         }
 265         if (s.discovery) {
 266             char *controller_name;
 267             if (rconn_is_connectivity_questionable(remote_rconn)) {
 268                 discovery_question_connectivity(discovery);
 269             }
 270             if (discovery_run(discovery, &controller_name)) {
 271                 if (controller_name) {
 272                     rconn_connect(remote_rconn, controller_name);
 273                 } else {
 274                     rconn_disconnect(remote_rconn);
 275                 }
 276             }
 277         }
 278 
 279         /* Wait for something to happen. */
 280         LIST_FOR_EACH (r, struct relay, node, &relays) {
 281             relay_wait(r);
 282         }
 283         for (i = 0; i < n_listeners; i++) {
 284             pvconn_wait(listeners[i]);
 285         }
 286         if (monitor) {
 287             pvconn_wait(monitor);
 288         }
 289         for (i = 0; i < secchan.n_hooks; i++) {
 290             if (secchan.hooks[i].class->wait_cb) {
 291                 secchan.hooks[i].class->wait_cb(secchan.hooks[i].aux);
 292             }
 293         }
 294         if (discovery) {
 295             discovery_wait(discovery);
 296         }
 297         poll_block();
 298     }
 299 
 300     return 0;
 301 }

 303 static struct pvconn *
 304 open_passive_vconn(const char *name)
 305 {
 306     struct pvconn *pvconn;
 307     int retval;
 308 
 309     retval = pvconn_open(name, &pvconn);
 310     if (retval && retval != EAGAIN) {
 311         ofp_fatal(retval, "opening %s", name);
 312     }
 313     return pvconn;
 314 }

 316 static struct vconn *
 317 accept_vconn(struct pvconn *pvconn)
 318 {
 319     struct vconn *new;
 320     int retval;
 321 
 322     retval = pvconn_accept(pvconn, OFP_VERSION, &new);
 323     if (retval && retval != EAGAIN) {
 324         VLOG_WARN_RL(&rl, "accept failed (%s)", strerror(retval));
 325     }
 326     return new;

 329 void
 330 add_hook(struct secchan *secchan, const struct hook_class *class, void *aux)
 331 {
 332     struct hook *hook;
 333 
 334     if (secchan->n_hooks >= secchan->allocated_hooks) {
 335         secchan->hooks = x2nrealloc(secchan->hooks, &secchan->allocated_hooks,
 336                                     sizeof *secchan->hooks);
 337     }
 338     hook = &secchan->hooks[secchan->n_hooks++];
 339     hook->class = class;
 340     hook->aux = aux;
 341 }

 343 struct ofp_packet_in *
 344 get_ofp_packet_in(struct relay *r)
 345 {
 346     struct ofpbuf *msg = r->halves[HALF_LOCAL].rxbuf;
 347     struct ofp_header *oh = msg->data;
 348     if (oh->type == OFPT_PACKET_IN) {
 349         if (msg->size >= offsetof (struct ofp_packet_in, data)) {
 350             return msg->data;
 351         } else {
 352             VLOG_WARN("packet too short (%zu bytes) for packet_in",
 353                       msg->size);
 354         }
 355     }
 356     return NULL;
 357 }

 359 bool
 360 get_ofp_packet_eth_header(struct relay *r, struct ofp_packet_in **opip,
 361                           struct eth_header **ethp)
 362 {
 363     const int min_len = offsetof(struct ofp_packet_in, data) + ETH_HEADER_LEN;
 364     struct ofp_packet_in *opi = get_ofp_packet_in(r);
 365     if (opi && ntohs(opi->header.length) >= min_len) {
 366         *opip = opi;
 367         *ethp = (void *) opi->data;
 368         return true;
 369     }
 370     return false;
 371 }

 375 /* Returns a malloc'd string containing a copy of 'vconn_name' modified not to
 376  * subscribe to asynchronous messages such as 'ofp_packet_in' events (if
 377  * possible). */
 378 static char *
 379 vconn_name_without_subscription(const char *vconn_name)
 380 {
 381     int nl_index;
 382     if (sscanf(vconn_name, "nl:%d", &nl_index) == 1) {
 383         /* nl:123 or nl:123:1 opens a netlink connection to local datapath 123.
 384          * nl:123:0 opens a netlink connection to local datapath 123 without
 385          * obtaining a subscription for ofp_packet_in or ofp_flow_removed
 386          * messages. */
 387         return xasprintf("nl:%d:0", nl_index);
 388     } else {
 389         /* We don't have a way to specify not to subscribe to those messages
 390          * for other transports.  (That's a defect: really this should be in
 391          * the OpenFlow protocol, not the Netlink transport). */
 392         VLOG_WARN_RL(&rl, "new management connection will receive "
 393                      "asynchronous messages");
 394         return xstrdup(vconn_name);
 395     }
 396 }

 398 static struct relay *
 399 relay_accept(const struct settings *s, struct pvconn *pvconn)
 400 {
 401     struct vconn *new_remote, *new_local;
 402     struct rconn *r1, *r2;
 403     char *vconn_name;
 404     int retval;
 405 
 406     new_remote = accept_vconn(pvconn);
 407     if (!new_remote) {
 408         return NULL;
 409     }
 410 
 411     vconn_name = vconn_name_without_subscription(s->dp_name);
 412     retval = vconn_open(vconn_name, OFP_VERSION, &new_local);
 413     if (retval) {
 414         VLOG_ERR_RL(&rl, "could not connect to %s (%s)",
 415                     vconn_name, strerror(retval));
 416         vconn_close(new_remote);
 417         free(vconn_name);
 418         return NULL;
 419     }
 420 
 421     /* Create and return relay. */
 422     r1 = rconn_create(0, 0);
 423     rconn_connect_unreliably(r1, vconn_name, new_local);
 424     free(vconn_name);
 425 
 426     r2 = rconn_create(0, 0);
 427     rconn_connect_unreliably(r2, "passive", new_remote);
 428 
 429     return relay_create(NULL, r1, r2, true);
 430 }

 432 static struct relay *
 433 relay_create(struct rconn *async, struct rconn *local, struct rconn *remote,
 434              bool is_mgmt_conn)
 435 {
 436     struct relay *r = xcalloc(1, sizeof *r);
 437     r->halves[HALF_LOCAL].rconn = local;
 438     r->halves[HALF_REMOTE].rconn = remote;
 439     r->is_mgmt_conn = is_mgmt_conn;
 440     r->async_rconn = async;
 441     return r;
 442 }

 444 static bool
 445 call_local_packet_cbs(struct secchan *secchan, struct relay *r)
 446 {
 447     const struct hook *h;
 448     for (h = secchan->hooks; h < &secchan->hooks[secchan->n_hooks]; h++) {
 449         bool (*cb)(struct relay *, void *aux) = h->class->local_packet_cb;
 450         if (cb && (cb)(r, h->aux)) {
 451             return true;
 452         }
 453     }
 454     return false;
 455 }

 457 static bool
 458 call_remote_packet_cbs(struct secchan *secchan, struct relay *r)
 459 {
 460     const struct hook *h;
 461     for (h = secchan->hooks; h < &secchan->hooks[secchan->n_hooks]; h++) {
 462         bool (*cb)(struct relay *, void *aux) = h->class->remote_packet_cb;
 463         if (cb && (cb)(r, h->aux)) {
 464             return true;
 465         }
 466     }
 467     return false;
 468 }

 470 static void
 471 relay_run(struct relay *r, struct secchan *secchan)
 472 {
 473     int iteration;
 474     int i;
 475 
 476     if (r->async_rconn) {
 477         rconn_run(r->async_rconn);
 478     }
 479     for (i = 0; i < 2; i++) {
 480         rconn_run(r->halves[i].rconn);
 481     }
 482 
 483     /* Limit the number of iterations to prevent other tasks from starving. */
 484     for (iteration = 0; iteration < 50; iteration++) {
 485         bool progress = false;
 486         for (i = 0; i < 2; i++) {
 487             struct half *this = &r->halves[i];
 488             struct half *peer = &r->halves[!i];
 489 
 490             if (!this->rxbuf) {
 491                 this->rxbuf = rconn_recv(this->rconn);
 492                 if (!this->rxbuf && i == HALF_LOCAL && r->async_rconn) {
 493                     this->rxbuf = rconn_recv(r->async_rconn);
 494                 }
 495                 if (this->rxbuf && (i == HALF_REMOTE || !r->is_mgmt_conn)) {
 496                     if (i == HALF_LOCAL
 497                         ? call_local_packet_cbs(secchan, r)
 498                         : call_remote_packet_cbs(secchan, r))
 499                     {
 500                         ofpbuf_delete(this->rxbuf);
 501                         this->rxbuf = NULL;
 502                         progress = true;
 503                         break;
 504                     }
 505                 }
 506             }
 507 
 508             if (this->rxbuf && !this->n_txq) {
 509                 int retval = rconn_send(peer->rconn, this->rxbuf,
 510                                         &this->n_txq);
 511                 if (retval != EAGAIN) {
 512                     if (!retval) {
 513                         progress = true;
 514                     } else {
 515                         ofpbuf_delete(this->rxbuf);
 516                     }
 517                     this->rxbuf = NULL;
 518                 }
 519             }
 520         }
 521         if (!progress) {
 522             break;
 523         }
 524     }
 525 
 526     if (r->is_mgmt_conn) {
 527         for (i = 0; i < 2; i++) {
 528             struct half *this = &r->halves[i];
 529             if (!rconn_is_alive(this->rconn)) {
 530                 relay_destroy(r);
 531                 return;
 532             }
 533         }
 534     }
 535 }

 537 static void
 538 relay_wait(struct relay *r)
 539 {
 540     int i;
 541 
 542     if (r->async_rconn) {
 543         rconn_run_wait(r->async_rconn);
 544     }
 545     for (i = 0; i < 2; i++) {
 546         struct half *this = &r->halves[i];
 547 
 548         rconn_run_wait(this->rconn);
 549         if (!this->rxbuf) {
 550             rconn_recv_wait(this->rconn);
 551             if (i == HALF_LOCAL && r->async_rconn) {
 552                 rconn_recv_wait(r->async_rconn);
 553             }
 554         }
 555     }
 556 }
 557 
 558 static void
 559 relay_destroy(struct relay *r)
 560 {
 561     int i;
 562 
 563     list_remove(&r->node);
 564     rconn_destroy(r->async_rconn);
 565     for (i = 0; i < 2; i++) {
 566         struct half *this = &r->halves[i];
 567         rconn_destroy(this->rconn);
 568         ofpbuf_delete(this->rxbuf);
 569     }
 570     free(r);
 571 }

 575 static void
 576 parse_options(int argc, char *argv[], struct settings *s)
 577 {
 578     enum {
 579         OPT_ACCEPT_VCONN = UCHAR_MAX + 1,
 580         OPT_NO_RESOLV_CONF,
 581         OPT_INACTIVITY_PROBE,
 582         OPT_MAX_IDLE,
 583         OPT_MAX_BACKOFF,
 584         OPT_RATE_LIMIT,
 585         OPT_BURST_LIMIT,
 586         OPT_BOOTSTRAP_CA_CERT,
 587         OPT_STP,
 588         OPT_NO_STP,
 589         OPT_OUT_OF_BAND,
 590         OPT_IN_BAND,
 591         OPT_EMERG_FLOW,
 592         VLOG_OPTION_ENUMS,
 593         LEAK_CHECKER_OPTION_ENUMS
 594     };
 595     static struct option long_options[] = {
 596         {"accept-vconn", required_argument, 0, OPT_ACCEPT_VCONN},
 597         {"no-resolv-conf", no_argument, 0, OPT_NO_RESOLV_CONF},
 598         {"fail",        required_argument, 0, 'F'},
 599         {"inactivity-probe", required_argument, 0, OPT_INACTIVITY_PROBE},
 600         {"max-idle",    required_argument, 0, OPT_MAX_IDLE},
 601         {"max-backoff", required_argument, 0, OPT_MAX_BACKOFF},
 602         {"listen",      required_argument, 0, 'l'},
 603         {"monitor",     required_argument, 0, 'm'},
 604         {"rate-limit",  optional_argument, 0, OPT_RATE_LIMIT},
 605         {"burst-limit", required_argument, 0, OPT_BURST_LIMIT},
 606         {"stp",         no_argument, 0, OPT_STP},
 607         {"no-stp",      no_argument, 0, OPT_NO_STP},
 608         {"out-of-band", no_argument, 0, OPT_OUT_OF_BAND},
 609         {"in-band",     no_argument, 0, OPT_IN_BAND},
 610         {"emerg-flow",  no_argument, 0, OPT_EMERG_FLOW},
 611         {"verbose",     optional_argument, 0, 'v'},
 612         {"help",        no_argument, 0, 'h'},
 613         {"version",     no_argument, 0, 'V'},
 614         DAEMON_LONG_OPTIONS,
 615         VLOG_LONG_OPTIONS,
 616         LEAK_CHECKER_LONG_OPTIONS,
 617 #ifdef HAVE_OPENSSL
 618         VCONN_SSL_LONG_OPTIONS
 619         {"bootstrap-ca-cert", required_argument, 0, OPT_BOOTSTRAP_CA_CERT},
 620 #endif
 621         {0, 0, 0, 0},
 622     };
 623     char *short_options = long_options_to_short_options(long_options);
 624     char *accept_re = NULL;
 625     int retval;
 626 
 627     /* Set defaults that we can figure out before parsing options. */
 628     s->n_listeners = 0;
 629     s->monitor_name = NULL;
 630     s->fail_mode = FAIL_OPEN;
 631     s->max_idle = 15;
 632     s->probe_interval = 15;
 633     s->max_backoff = 15;
 634     s->update_resolv_conf = true;
 635     s->rate_limit = 0;
 636     s->burst_limit = 0;
 637     s->enable_stp = false;
 638     s->in_band = true;
 639     s->emerg_flow = false;
 640     for (;;) {
 641         int c;
 642 
 643         c = getopt_long(argc, argv, short_options, long_options, NULL);
 644         if (c == -1) {
 645             break;
 646         }
 647 
 648         switch (c) {
 649         case OPT_ACCEPT_VCONN:
 650             accept_re = optarg[0] == '^' ? optarg : xasprintf("^%s", optarg);
 651             break;
 652 
 653         case OPT_NO_RESOLV_CONF:
 654             s->update_resolv_conf = false;
 655             break;
 656 
 657         case 'F':
 658             if (!strcmp(optarg, "open")) {
 659                 s->fail_mode = FAIL_OPEN;
 660             } else if (!strcmp(optarg, "closed")) {
 661                 s->fail_mode = FAIL_CLOSED;
 662             } else {
 663                 ofp_fatal(0, "-f or --fail argument must be \"open\" "
 664                           "or \"closed\"");
 665             }
 666             break;
 667 
 668         case OPT_INACTIVITY_PROBE:
 669             s->probe_interval = atoi(optarg);
 670             if (s->probe_interval < 1) {
 671                 ofp_fatal(0, "--inactivity-probe argument must be at least 1");
 672             }
 673             break;
 674 
 675         case OPT_MAX_IDLE:
 676             if (!strcmp(optarg, "permanent")) {
 677                 s->max_idle = OFP_FLOW_PERMANENT;
 678             } else {
 679                 s->max_idle = atoi(optarg);
 680                 if (s->max_idle < 1 || s->max_idle > 65535) {
 681                     ofp_fatal(0, "--max-idle argument must be between 1 and "
 682                               "65535 or the word 'permanent'");
 683                 }
 684             }
 685             break;
 686 
 687         case OPT_MAX_BACKOFF:
 688             s->max_backoff = atoi(optarg);
 689             if (s->max_backoff < 1) {
 690                 ofp_fatal(0, "--max-backoff argument must be at least 1");
 691             } else if (s->max_backoff > 3600) {
 692                 s->max_backoff = 3600;
 693             }
 694             break;
 695 
 696         case OPT_RATE_LIMIT:
 697             if (optarg) {
 698                 s->rate_limit = atoi(optarg);
 699                 if (s->rate_limit < 1) {
 700                     ofp_fatal(0, "--rate-limit argument must be at least 1");
 701                 }
 702             } else {
 703                 s->rate_limit = 1000;
 704             }
 705             break;
 706 
 707         case OPT_BURST_LIMIT:
 708             s->burst_limit = atoi(optarg);
 709             if (s->burst_limit < 1) {
 710                 ofp_fatal(0, "--burst-limit argument must be at least 1");
 711             }
 712             break;
 713 
 714         case OPT_STP:
 715             s->enable_stp = true;
 716             break;
 717 
 718         case OPT_NO_STP:
 719             s->enable_stp = false;
 720             break;
 721 
 722         case OPT_OUT_OF_BAND:
 723             s->in_band = false;
 724             break;
 725 
 726         case OPT_IN_BAND:
 727             s->in_band = true;
 728             break;
 729 
 730         case OPT_EMERG_FLOW:
 731             s->emerg_flow = true;
 732             break;
 733 
 734         case 'l':
 735             if (s->n_listeners >= MAX_MGMT) {
 736                 ofp_fatal(0,
 737                           "-l or --listen may be specified at most %d times",
 738                           MAX_MGMT);
 739             }
 740             s->listener_names[s->n_listeners++] = optarg;
 741             break;
 742 
 743         case 'm':
 744             if (s->monitor_name) {
 745                 ofp_fatal(0, "-m or --monitor may only be specified once");
 746             }
 747             s->monitor_name = optarg;
 748             break;
 749 
 750         case 'h':
 751             usage();
 752 
 753         case 'V':
 754             printf("%s %s compiled "__DATE__" "__TIME__"\n",
 755                    program_name, VERSION BUILDNR);
 756             exit(EXIT_SUCCESS);
 757 
 758         DAEMON_OPTION_HANDLERS
 759 
 760         VLOG_OPTION_HANDLERS
 761 
 762         LEAK_CHECKER_OPTION_HANDLERS
 763 
 764 #ifdef HAVE_OPENSSL
 765         VCONN_SSL_OPTION_HANDLERS
 766 
 767         case OPT_BOOTSTRAP_CA_CERT:
 768             vconn_ssl_set_ca_cert_file(optarg, true);
 769             break;
 770 #endif
 771 
 772         case '?':
 773             exit(EXIT_FAILURE);
 774 
 775         default:
 776             abort();
 777         }
 778     }
 779     free(short_options);
 780 
 781     argc -= optind;
 782     argv += optind;
 783     if (argc < 1 || argc > 2) {
 784         ofp_fatal(0, "need one or two non-option arguments; "
 785                   "use --help for usage");
 786     }
 787 
 788     /* Local and remote vconns. */
 789     s->dp_name = argv[0];
 790     {
 791         char *curr;
 792         char *save;
 793         int i;
 794 
 795         s->num_controllers = 0;
 796         for (i = 0; i < MAX_CONTROLLERS; ++i)
 797             s->controller_names[i] = NULL;
 798         if (argc > 1) {
 799             for (curr = strtok_r(argv[1], ",,", &save), i = 0;
 800                  curr && i < MAX_CONTROLLERS;
 801                  curr = strtok_r(NULL, ",,", &save), ++i) {
 802                 s->controller_names[i] = xstrdup(curr);
 803                 ++s->num_controllers;
 804             }
 805         }
 806     }
 807 
 808     /* Set accept_controller_regex. */
 809     if (!accept_re) {
 810         accept_re = vconn_ssl_is_configured() ? "^ssl:.*" : ".*";
 811     }
 812     retval = regcomp(&s->accept_controller_regex, accept_re,
 813                      REG_NOSUB | REG_EXTENDED);
 814     if (retval) {
 815         size_t length = regerror(retval, &s->accept_controller_regex, NULL, 0);
 816         char *buffer = xmalloc(length);
 817         regerror(retval, &s->accept_controller_regex, buffer, length);
 818         ofp_fatal(0, "%s: %s", accept_re, buffer);
 819     }
 820     s->accept_controller_re = accept_re;
 821 
 822     /* Mode of operation. */
 823     s->discovery = s->controller_names[0] == NULL;
 824     if (s->discovery && !s->in_band) {
 825         ofp_fatal(0, "Cannot perform discovery with out-of-band control");
 826     }
 827 
 828     /* Rate limiting. */
 829     if (s->rate_limit) {
 830         if (s->rate_limit < 100) {
 831             VLOG_WARN("Rate limit set to unusually low value %d",
 832                       s->rate_limit);
 833         }
 834         if (!s->burst_limit) {
 835             s->burst_limit = s->rate_limit / 4;
 836         }
 837         s->burst_limit = MAX(s->burst_limit, 1);
 838         s->burst_limit = MIN(s->burst_limit, INT_MAX / 1000);
 839     }
 840 }

 848            "CONTROLLER is an active OpenFlow connection method; if it is\n"
