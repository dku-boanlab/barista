  41 static inline uint16_t
  42 port_array_extract_bits__(uint16_t data, int start, int count)
  43 {
  44     return (data >> start) & ((1u << count) - 1);
  45 }

  47 /* Level 1: most-significant bits. */
  48 #define PORT_ARRAY_L1_BITS 5
  49 #define PORT_ARRAY_L1_SHIFT (PORT_ARRAY_L3_BITS + PORT_ARRAY_L2_BITS)
  50 #define PORT_ARRAY_L1_SIZE (1u << PORT_ARRAY_L1_BITS)
  51 #define PORT_ARRAY_L1(IDX) \
  52     port_array_extract_bits__(IDX, PORT_ARRAY_L1_SHIFT, PORT_ARRAY_L1_BITS)

  61 /* Level 3: least-significant bits. */

  74 /* A "sparse array" of up to 65536 elements (numbered 0...65535), implemented
  75  * as a 3-level trie.  Most efficient when the elements are clustered
  76  * together. */
  77 struct port_array {
  78     struct port_array_l2 *l1[1u << PORT_ARRAY_L1_BITS];
  79 };
  80 
  81 struct port_array_l2 {
  82     struct port_array_l3 *l2[1u << PORT_ARRAY_L2_BITS];
  83 };
  84 
  85 struct port_array_l3 {
  86     void *l3[1u << PORT_ARRAY_L3_BITS];
  87 };
  88 
  89 /* Returns the value of the element numbered 'idx' in 'pa', or a null pointer
  90  * if no element numbered 'idx' has been set. */
  91 static inline void *
  92 port_array_get(const struct port_array *pa, uint16_t idx)
  93 {
  94     unsigned int l1_idx = PORT_ARRAY_L1(idx);
  95     unsigned int l2_idx = PORT_ARRAY_L2(idx);
  96     unsigned int l3_idx = PORT_ARRAY_L3(idx);
  97     return pa->l1[l1_idx]->l2[l2_idx]->l3[l3_idx];
  98 }
