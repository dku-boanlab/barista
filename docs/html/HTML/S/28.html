  48 static inline uint64_t htonll(uint64_t n)
  49 {
  50     return htonl(1) == 1 ? n : ((uint64_t) htonl(n) << 32) | htonl(n >> 32);

 159 int fakeswitch_get_count(struct fakeswitch *fs)
 160 {
 161     int ret = fs->count;
 162     int count;
 163     int msglen;
 164     struct ofp_header * ofph;
 165     fs->count = 0;
 166     fs->probe_state = 0;        // reset packet state
 167     // keep reading until there is nothing to clear out the queue
 168     while( (count = msgbuf_read(fs->inbuf,fs->sock)) > 0) {
 169         while(count > 0) {
 170             // need to read msg by msg to ensure framing isn't broken
 171             ofph = msgbuf_peek(fs->inbuf);
 172             msglen = ntohs(ofph->length);
 173             if(count < msglen)
 174                 break;     // msg not all there yet; 
 175             msgbuf_pull(fs->inbuf, NULL, ntohs(ofph->length));
 176             count -= msglen;
 177         }
 178     }
 179     return ret;
 180 }

 284 static int packet_out_is_lldp(struct ofp_packet_out * po){
 285         char * ptr = (char *) po;
 286         ptr += sizeof(struct ofp_packet_out) + ntohs(po->actions_len);
 287         struct ether_header * ethernet = (struct ether_header *) ptr;
 288         unsigned short ethertype = ntohs(ethernet->ether_type);
 289         if (ethertype == ETHERTYPE_VLAN) {
 290                 ethernet = (struct ether_header *) ((char *) ethernet) +4;
 291                 ethertype = ntohs(ethernet->ether_type);
 292         }
 293         
 294         return ethertype == ETHERTYPE_LLDP;
 295 }

 336 void fakeswitch_change_status(struct fakeswitch *fs, int new_status) {
 337     if( fs->delay == 0) {
 338         fakeswitch_change_status_now(fs, new_status);
 339         debug_msg(fs, " switched to next status %d", new_status);
 340     } else {
 341         fs->switch_status = WAITING;

 345         fs->delay_start.tv_usec += (fs->delay % 1000 ) * 1000;
 346         debug_msg(fs, " delaying next status %d by %d ms", new_status, fs->delay);
 347     }

 353 void fakeswitch_handle_read(struct fakeswitch *fs)
 354 {
 355     int count;
 356     struct ofp_header * ofph;
 357     struct ofp_header echo;
 358     struct ofp_header barrier;
 359     char buf[BUFLEN];
 360     count = msgbuf_read(fs->inbuf, fs->sock);   // read any queued data
 361     if (count <= 0)
 362     {
 363         fprintf(stderr, "controller msgbuf_read() = %d:  ", count);
 364         if(count < 0)
 365             perror("msgbuf_read");
 366         else
 367             fprintf(stderr, " closed connection ");
 368         fprintf(stderr, "... exiting\n");
 369         exit(1);
 370     }
 371     while((count= msgbuf_count_buffered(fs->inbuf)) >= sizeof(struct ofp_header ))
 372     {
 373         ofph = msgbuf_peek(fs->inbuf);
 374         if(count < ntohs(ofph->length))
 375             return;     // msg not all there yet
 376         msgbuf_pull(fs->inbuf, NULL, ntohs(ofph->length));
 377         switch(ofph->type)
 378         {
 379             struct ofp_flow_mod * fm;
 380             struct ofp_packet_out *po;
 381             struct ofp_stats_request * stats_req;
 382             case OFPT_PACKET_OUT:
 383                 po = (struct ofp_packet_out *) ofph;
 384                 if ( fs->switch_status == READY_TO_SEND && ! packet_out_is_lldp(po)) { 
 385                     // assume this is in response to what we sent
 386                     fs->count++;        // got response to what we went
 387                     fs->probe_state--;
 388                 }
 389                 break;
 390             case OFPT_FLOW_MOD:
 391                 fm = (struct ofp_flow_mod *) ofph;
 392                 if(fs->switch_status == READY_TO_SEND && (fm->command == htons(OFPFC_ADD) || 
 393                         fm->command == htons(OFPFC_MODIFY_STRICT)))
 394                 {
 395                     fs->count++;        // got response to what we went
 396                     fs->probe_state--;
 397                 }
 398                 break;
 399             case OFPT_FEATURES_REQUEST:
 400                 // pull msgs out of buffer
 401                 debug_msg(fs, "got feature_req");
 402                 // Send features reply
 403                 count = make_features_reply(fs->id, ofph->xid, buf, BUFLEN);
 404                 msgbuf_push(fs->outbuf, buf, count);
 405                 debug_msg(fs, "sent feature_rsp");
 406                 fakeswitch_change_status(fs, fs->learn_dstmac ? LEARN_DSTMAC : READY_TO_SEND);
 407                 break;
 408             case OFPT_SET_CONFIG:
 409                 // pull msgs out of buffer
 410                 debug_msg(fs, "parsing set_config");
 411                 parse_set_config(ofph);
 412                 break;
 413             case OFPT_GET_CONFIG_REQUEST:
 414                 // pull msgs out of buffer
 415                 debug_msg(fs, "got get_config_request");
 416                 count = make_config_reply(ofph->xid, buf, BUFLEN);
 417                 msgbuf_push(fs->outbuf, buf, count);
 418                 if ((fs->mode == MODE_LATENCY)  && ( fs->probe_state == 1 )) {     
 419                     fs->probe_state = 0;       // restart probe state b/c some 
 420                                                // controllers block on config
 421                         debug_msg(fs, "reset probe state b/c of get_config_reply");
 422                 }
 423                 debug_msg(fs, "sent get_config_reply");
 424                 break;
 425             case OFPT_VENDOR:
 426                 // pull msgs out of buffer
 427                 debug_msg(fs, "got vendor");
 428                 count = make_vendor_reply(ofph->xid, buf, BUFLEN);
 429                 msgbuf_push(fs->outbuf, buf, count);
 430                 debug_msg(fs, "sent vendor");
 431                 // apply nox hack; nox ignores packet_in until this msg is sent
 432                 fs->probe_state=0;
 433                 break;
 434             case OFPT_HELLO:
 435                 debug_msg(fs, "got hello");
 436                 // we already sent our own HELLO; don't respond
 437                 break;
 438             case OFPT_ECHO_REQUEST:
 439                 debug_msg(fs, "got echo, sent echo_resp");
 440                 echo.version= OFP_VERSION;
 441                 echo.length = htons(sizeof(echo));
 442                 echo.type   = OFPT_ECHO_REPLY;
 443                 echo.xid = ofph->xid;
 444                 msgbuf_push(fs->outbuf,(char *) &echo, sizeof(echo));
 445                 break;
 446             case OFPT_BARRIER_REQUEST:
 447                 debug_msg(fs, "got barrier, sent barrier_resp");
 448                 barrier.version= OFP_VERSION;
 449                 barrier.length = htons(sizeof(barrier));
 450                 barrier.type   = OFPT_BARRIER_REPLY;
 451                 barrier.xid = ofph->xid;
 452                 msgbuf_push(fs->outbuf,(char *) &barrier, sizeof(barrier));
 453                 break;
 454             case OFPT_STATS_REQUEST:
 455                 stats_req  = (struct ofp_stats_request *) ofph;
 456                 if ( ntohs(stats_req->type) == OFPST_DESC ) {
 457                     count = make_stats_desc_reply(stats_req, buf, BUFLEN);
 458                     msgbuf_push(fs->outbuf, buf, count);
 459                     debug_msg(fs, "sent description stats_reply");
 460                     if ((fs->mode == MODE_LATENCY)  && ( fs->probe_state == 1 )) {     
 461                         fs->probe_state = 0;       // restart probe state b/c some 
 462                                        // controllers block on config
 463                                 debug_msg(fs, "reset probe state b/c of desc_stats_request");
 464                     }
 465                 } else {
 466                     debug_msg(fs, "Silently ignoring non-desc stats_request msg\n");
 467                 }
 468                 break;
 469             default: 
 470     //            if(fs->debug)
 471                     fprintf(stderr, "Ignoring OpenFlow message type %d\n", ofph->type);
 472         };
 473         if(fs->probe_state < 0)
 474         {
 475                 debug_msg(fs, "WARN: Got more responses than probes!!: : %d",
 476                             fs->probe_state);
 477                 fs->probe_state =0;
 478         }
 479     }
 480 }

 482 static void fakeswitch_handle_write(struct fakeswitch *fs)
 483 {
 484     char buf[BUFLEN];
 485     int count ;
 486     int send_count = 0 ;
 487     int throughput_buffer = BUFLEN;
 488     int i;
 489     if( fs->switch_status == READY_TO_SEND) 
 490     {
 491         if ((fs->mode == MODE_LATENCY)  && ( fs->probe_state == 0 ))      
 492             send_count = 1;                 // just send one packet
 493         else if ((fs->mode == MODE_THROUGHPUT) && 
 494                 (msgbuf_count_buffered(fs->outbuf) < throughput_buffer))  // keep buffer full
 495             send_count = (throughput_buffer - msgbuf_count_buffered(fs->outbuf)) / fs->probe_size;
 496         for (i = 0; i < send_count; i++)
 497         {
 498             // queue up packet
 499             
 500             fs->probe_state++;
 501             // TODO come back and remove this copy
 502             count = make_packet_in(fs->id, fs->xid++, fs->current_buffer_id, buf, BUFLEN, fs->current_mac_address);
 503             fs->current_mac_address = ( fs->current_mac_address + 1 ) % fs->total_mac_addresses;
 504             fs->current_buffer_id =  ( fs->current_buffer_id + 1 ) % NUM_BUFFER_IDS;
 505             msgbuf_push(fs->outbuf, buf, count);
 506             debug_msg(fs, "send message %d", i);
 507         }
 508     } else if( fs->switch_status == WAITING) 
 509     {
 510         struct timeval now;
 511         gettimeofday(&now, NULL);
 512         if (timercmp(&now, &fs->delay_start, > ))
 513         {
 514             fakeswitch_change_status_now(fs, fs->next_status);
 515             debug_msg(fs, " delay is over: switching to state %d", fs->next_status);
 516         }
 517     } else if (  fs->switch_status == LEARN_DSTMAC) 
 518     {
 519         // we should learn the dst mac addresses
 520         fakeswitch_learn_dstmac(fs);
 521         fakeswitch_change_status(fs, READY_TO_SEND);
 522     }
 523     // send any data if it's queued
 524     if( msgbuf_count_buffered(fs->outbuf) > 0)
 525         msgbuf_write(fs->outbuf, fs->sock, 0);
 526 }
 527 /***********************************************************************/
 528 void fakeswitch_handle_io(struct fakeswitch *fs, void *pfd_events)
 529 {
 530     #ifdef USE_EPOLL
 531     int events = *((int*) pfd_events);
 532     if(events & EPOLLIN) {
 533         fakeswitch_handle_read(fs);
 534     } else if(events & EPOLLOUT) {
 535         fakeswitch_handle_write(fs);
 536     }
 537     #else
 538     struct pollfd *pfd = (struct pollfd*)pfd_events;
 539     if(pfd->revents & POLLIN)
 540         fakeswitch_handle_read(fs);
 541     if(pfd->revents & POLLOUT)
 542         fakeswitch_handle_write(fs);
 543     #endif
 544 }
 545 /************************************************************************/
 546 static int debug_msg(struct fakeswitch * fs, char * msg, ...)
 547 {
 548     va_list aq;
 549     if(fs->debug == 0 )
 550         return 0;
 551     fprintf(stderr,"\n-------Switch %d: ", fs->id);
 552     va_start(aq,msg);
 553     vfprintf(stderr,msg,aq);
 554     if(msg[strlen(msg)-1] != '\n')
 555         fprintf(stderr, "\n");
 556     return 1;
 557 }
