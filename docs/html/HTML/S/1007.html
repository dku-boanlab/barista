  44 /**
  45  * \brief Function to push a log message into a log queue
  46  * \param msg Log message
  47  */
  48 static int push_msg_into_queue(char *msg)
  49 {
  50     pthread_spin_lock(&log_lock);
  51 
  52     if (write_msg >= BATCH_LOGS-1) {
  53         FILE *fp = fopen(log_file, "a");
  54         if (fp != NULL) {
  55             int i;
  56             for (i=0; i<write_msg; i++) {
  57                 fputs(msgs[i], fp);
  58                 memset(msgs[i], 0, __CONF_STR_LEN);
  59             }
  60 
  61             write_msg = 0;
  62 
  63             fclose(fp);
  64         } else {
  65             pthread_spin_unlock(&log_lock);
  66 
  67             return -1;
  68         }
  69     }
  70 
  71     snprintf(msgs[write_msg++], __CONF_STR_LEN-1, "%s", msg);
  72 
  73     pthread_spin_unlock(&log_lock);
  74 
  75     return 0;
  76 }

 112 /**
 113  * \brief The main function
 114  * \param activated The activation flag of this component
 115  * \param argc The number of arguments
 116  * \param argv Arguments
 117  */
 118 int log_main(int *activated, int argc, char **argv)
 119 {
 120     msgs = (char **)MALLOC(sizeof(char *) * BATCH_LOGS);
 121     if (msgs == NULL) {
 122         PERROR("malloc");
 123         return -1;
 124     }
 125 
 126     int k;
 127     for (k=0; k<BATCH_LOGS; k++) {
 128         char *m = (char *)MALLOC(sizeof(char) * __CONF_STR_LEN);
 129         if (m == NULL) {
 130             PERROR("malloc");
 131             return -1;
 132         }
 133 
 134         msgs[k] = m;
 135 
 136         memset(msgs[k], 0, __CONF_STR_LEN);
 137     }
 138 
 139     pthread_spin_init(&log_lock, PTHREAD_PROCESS_PRIVATE);
 140 
 141     activate();
 142 
 143     LOG_INFO(LOG_ID, "Init - Logging mechanism");
 144 
 145     while (*activated) {
 146         pthread_spin_lock(&log_lock);
 147 
 148         if (write_msg) {
 149             FILE *fp = fopen(log_file, "a");
 150             if (fp != NULL) {
 151                 int i;
 152                 for (i=0; i<write_msg; i++) {
 153                     fputs(msgs[i], fp);
 154                     memset(msgs[i], 0, __CONF_STR_LEN);
 155                 }
 156                 write_msg = 0;
 157                 fclose(fp);
 158             } else {
 159                 pthread_spin_unlock(&log_lock);
 160                 continue;
 161             }
 162         }
 163 
 164         pthread_spin_unlock(&log_lock);
 165 
 166         waitsec(LOG_UPDATE_TIME, 0);
 167     }
 168 
 169     return 0;
 170 }

 183 
 184     if (write_msg) {
 185         FILE *fp = fopen(log_file, "a");
 186         if (fp != NULL) {
 187             int i;

 189                 fputs(msgs[i], fp);
 190             }
 191             fclose(fp);
 192         }
 193     }
 194 
 195     pthread_spin_unlock(&log_lock);
 196     pthread_spin_destroy(&log_lock);
 197 
 198     int k;
 199     for (k=0; k<BATCH_LOGS; k++) {
 200         FREE(msgs[k]);
 201     }
 202 
 203     FREE(msgs);
 204

 218 /**
 219  * \brief The handler function
 220  * \param ev Read-only event
 221  * \param ev_out Read-write event (if this component has the write permission)
 222  */
 223 int log_handler(const event_t *ev, event_out_t *ev_out)
 224 {
 225     time_t timer;
 226     struct tm *tm_info;
 227     char now[__CONF_WORD_LEN] = {0}, out[__CONF_STR_LEN] = {0};
 228 
 229     time(&timer);
 230     tm_info = localtime(&timer);
 231     strftime(now, __CONF_WORD_LEN, "%Y:%m:%d %H:%M:%S", tm_info);
 232 
 233     switch (ev->type) {
 234     case EV_LOG_DEBUG:
 235         PRINT_EV("EV_LOG_DEBUG\n");
 236         {
 237             snprintf(out, __CONF_STR_LEN-1, "%s <DEBUG> (%010u) %s\n", now, ev->id, ev->log);
 238 #ifdef __ENABLE_DEBUG
 239             PRINTF("%s <DEBUG> (%010u) %s\n", now, ev->id, ev->log);
 240 #endif /* __ENABLE_DEBUG */
 241             push_msg_into_queue(out);
 242             ev_log_update_msgs(LOG_ID, out);
 243         }
 244         break;
 245     case EV_LOG_INFO:
 246         PRINT_EV("EV_LOG_INFO\n");
 247         {
 248             snprintf(out, __CONF_STR_LEN-1, "%s <INFO> (%010u) %s\n", now, ev->id, ev->log);
 249             PRINTF("%s <INFO> (%010u) %s\n", now, ev->id, ev->log);
 250             push_msg_into_queue(out);
 251             ev_log_update_msgs(LOG_ID, out);
 252         }
 253         break;
 254     case EV_LOG_WARN:
 255         PRINT_EV("EV_LOG_WARN\n");
 256         {
 257             snprintf(out, __CONF_STR_LEN-1, "%s <WARN> (%010u) %s\n", now, ev->id, ev->log);
 258             PRINTF(ANSI_COLOR_MAGENTA "%s <WARN> (%010u) %s" ANSI_COLOR_RESET "\n", now, ev->id, ev->log);
 259             push_msgs_into_file(out);
 260             ev_log_update_msgs(LOG_ID, out);
 261         }
 262         break;
 263     case EV_LOG_ERROR:
 264         PRINT_EV("EV_LOG_ERROR\n");
 265         {
 266             snprintf(out, __CONF_STR_LEN-1, "%s <ERROR> (%010u) %s\n", now, ev->id, ev->log);
 267             PRINTF(ANSI_COLOR_MAGENTA "%s <ERROR> (%010u) %s" ANSI_COLOR_RESET "\n", now, ev->id, ev->log);
 268             push_msgs_into_file(out);
 269             ev_log_update_msgs(LOG_ID, out);
 270         }
 271         break;
 272     case EV_LOG_FATAL:
 273         PRINT_EV("EV_LOG_FATAL\n");
 274         {
 275             snprintf(out, __CONF_STR_LEN-1, "%s <FATAL> (%010u) %s\n", now, ev->id, ev->log);
 276             PRINTF(ANSI_COLOR_RED "%s <FATAL> (%010u) %s" ANSI_COLOR_RESET "\n", now, ev->id, ev->log);
 277             push_msgs_into_file(out);
 278             ev_log_update_msgs(LOG_ID, out);
 279         }
 280         break;
 281     default:
 282         break;
 283     }
 284 
 285     return 0;
 286 }
