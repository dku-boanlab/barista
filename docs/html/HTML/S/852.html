  34 #ifndef TYPE_PROPS_H
  35 #define TYPE_PROPS_H 1
  36 
  37 #include <limits.h>
  38 
  39 #define TYPE_IS_INTEGER(TYPE) ((TYPE) 1.5 == (TYPE) 1)
  40 #define TYPE_IS_SIGNED(TYPE) ((TYPE) 0 > (TYPE) -1)
  41 #define TYPE_VALUE_BITS(TYPE) (sizeof(TYPE) * CHAR_BIT - TYPE_IS_SIGNED(TYPE))
  42 #define TYPE_MINIMUM(TYPE) (TYPE_IS_SIGNED(TYPE) \
  43                             ? ~(TYPE)0 << TYPE_VALUE_BITS(TYPE) \
  44                             : 0)
  45 #define TYPE_MAXIMUM(TYPE) (TYPE_IS_SIGNED(TYPE) \
  46                             ? ~(~(TYPE)0 << TYPE_VALUE_BITS(TYPE)) \
  47                             : (TYPE)-1)
  48 
  49 #endif /* type-props.h */
