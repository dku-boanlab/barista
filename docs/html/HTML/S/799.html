  64 static void
  65 failover_status_cb(struct status_reply *status_reply, void *context_)
  66 {
  67         struct failover_context *context = context_;
  68         int i;
  69 
  70         status_reply_put(status_reply, "num-controllers=%d",
  71                          context->settings->num_controllers);
  72 
  73         for (i = 0; i < MAX_CONTROLLERS; ++i) {
  74                 if (context->settings->controller_names[i] == NULL)
  75                         continue;
  76                 status_reply_put(status_reply, "controller#%d=%s",
  77                                  i, context->settings->controller_names[i]);
  78         }
  79 }

  81 static bool
  82 is_timed_out(const struct failover_peer *peer, int max_backoff)
  83 {
  84         unsigned int sat_value = sat_add(peer->epoch, max_backoff);
  85         return time_now() >= sat_value;
  86 }
  87 
  88 static void
  89 failover_periodic_cb(void *context_)
  90 {
  91         struct failover_context *context = context_;
  92         char *curr_peer = NULL;
  93         char *prev_peer = NULL;
  94 
  95         if (rconn_is_connected(context->remote_rconn))
  96                 return;
  97 
  98         if (!is_timed_out(context->peers[context->index],
  99                           context->settings->max_backoff)) {
 100                 return;
 101         }
 102 
 103         rconn_disconnect(context->remote_rconn);
 104         prev_peer = (char *)context->settings->controller_names[context->index];
 105         context->index = (context->index + 1)
 106                 % context->settings->num_controllers;
 107         curr_peer = (char *)context->settings->controller_names[context->index];
 108         rconn_connect(context->remote_rconn,
 109                       context->settings->controller_names[context->index]);
 110         context->peers[context->index]->epoch = time_now();
 111         VLOG_INFO("Switching over to %s, from %s", curr_peer, prev_peer);
 112 }

 114 void
 115 failover_start(struct secchan *secchan, const struct settings *settings,
 116                struct switch_status *switch_status, struct rconn *remote_rconn)
 117 {
 118         struct failover_context *context = NULL;
 119         int i;
 120         static struct hook_class failover_hook_class = {
 121                 NULL,           /* local_packet_cb */
 122                 NULL,           /* remote_packet_cb */
 123                 failover_periodic_cb,   /* periodic_cb */
 124                 NULL,           /* wait_cb */
 125                 NULL,           /* closing_cb */
 126         };
 127 
 128         context = xmalloc(sizeof(*context));
 129         context->settings = settings;
 130         context->secchan = secchan;
 131         context->remote_rconn = remote_rconn;
 132         context->index = 0;
 133         for (i = 0; i < MAX_CONTROLLERS; ++i) {
 134                 context->peers[i] = NULL;
 135                 if (settings->controller_names[i] == NULL)
 136                         continue;
 137                 context->peers[i] = xmalloc(sizeof(struct failover_peer));
 138                 context->peers[i]->epoch = time_now();
 139         }
 140 
 141         switch_status_register_category(switch_status, "failover",
 142                                         failover_status_cb, context);
 143         add_hook(secchan, &failover_hook_class, context);
 144 }
