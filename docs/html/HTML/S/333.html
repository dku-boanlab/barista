 104 int main(int argc, char *argv[])
 105 {
 106     struct settings s;
 107     struct command *p;
 108 
 109     set_program_name(argv[0]);
 110     time_init();
 111     vlog_init();
 112     parse_options(argc, argv, &s);
 113     signal(SIGPIPE, SIG_IGN);
 114 
 115     argc -= optind;
 116     argv += optind;
 117     if (argc < 1)
 118         ofp_fatal(0, "missing command name; use --help for help");
 119 
 120     for (p = all_commands; p->name != NULL; p++) {
 121         if (!strcmp(p->name, argv[0])) {
 122             int n_arg = argc - 1;
 123             if (n_arg < p->min_args)
 124                 ofp_fatal(0, "'%s' command requires at least %d arguments",
 125                           p->name, p->min_args);
 126             else if (n_arg > p->max_args)
 127                 ofp_fatal(0, "'%s' command takes at most %d arguments",
 128                           p->name, p->max_args);
 129             else {
 130                 p->handler(&s, argc, argv);
 131                 if (ferror(stdout)) {
 132                     ofp_fatal(0, "write to stdout failed");
 133                 }
 134                 if (ferror(stderr)) {
 135                     ofp_fatal(0, "write to stderr failed");
 136                 }
 137                 exit(0);
 138             }
 139         }
 140     }
 141     ofp_fatal(0, "unknown command '%s'; use --help for help", argv[0]);
 142 
 143     return 0;
 144 }

 146 static void
 147 parse_options(int argc, char *argv[], struct settings *s)
 148 {
 149     enum {
 150         OPT_STRICT = UCHAR_MAX + 1
 151     };
 152     static struct option long_options[] = {
 153         {"timeout", required_argument, 0, 't'},
 154         {"verbose", optional_argument, 0, 'v'},
 155         {"strict", no_argument, 0, OPT_STRICT},
 156         {"help", no_argument, 0, 'h'},
 157         {"version", no_argument, 0, 'V'},
 158         VCONN_SSL_LONG_OPTIONS
 159         {0, 0, 0, 0},
 160     };
 161     char *short_options = long_options_to_short_options(long_options);
 162 
 163     /* Set defaults that we can figure out before parsing options. */
 164     s->strict = false;
 165 
 166     for (;;) {
 167         unsigned long int timeout;
 168         int c;
 169 
 170         c = getopt_long(argc, argv, short_options, long_options, NULL);
 171         if (c == -1) {
 172             break;
 173         }
 174 
 175         switch (c) {
 176         case 't':
 177             timeout = strtoul(optarg, NULL, 10);
 178             if (timeout <= 0) {
 179                 ofp_fatal(0, "value %s on -t or --timeout is not at least 1",
 180                           optarg);
 181             } else {
 182                 time_alarm(timeout);
 183             }
 184             break;
 185 
 186         case 'h':
 187             usage();
 188 
 189         case 'V':
 190             printf("%s %s compiled "__DATE__" "__TIME__"\n",
 191                    program_name, VERSION BUILDNR);
 192             exit(EXIT_SUCCESS);
 193 
 194         case 'v':
 195             vlog_set_verbosity(optarg);
 196             break;
 197 
 198         case OPT_STRICT:
 199             s->strict = true;
 200             break;
 201 
 202         VCONN_SSL_OPTION_HANDLERS
 203 
 204         case '?':
 205             exit(EXIT_FAILURE);
 206 
 207         default:
 208             abort();
 209         }
 210     }
 211     free(short_options);
 212 }

 270 static void run(int retval, const char *message, ...)
 271 {
 272     if (retval) {
 273         va_list args;
 274 
 275         fprintf(stderr, "%s: ", program_name);
 276         va_start(args, message);
 277         vfprintf(stderr, message, args);
 278         va_end(args);
 279         if (retval == EOF) {
 280             fputs(": unexpected end of file\n", stderr);
 281         } else {
 282             fprintf(stderr, ": %s\n", strerror(retval));
 283         }
 284 
 285         exit(EXIT_FAILURE);

 442 static void
 443 dump_stats_transaction(const char *vconn_name, struct ofpbuf *request)
 444 {
 445     uint32_t send_xid = ((struct ofp_header *) request->data)->xid;
 446     struct vconn *vconn;
 447     bool done = false;
 448     
 449     open_vconn(vconn_name, &vconn);
 450     send_openflow_buffer(vconn, request);
 451     while (!done) {
 452         uint32_t recv_xid;
 453         struct ofpbuf *reply;
 454 
 455         run(vconn_recv_block(vconn, &reply), "OpenFlow packet receive failed");
 456 
 457         recv_xid = ((struct ofp_header *) reply->data)->xid;
 458         if (send_xid == recv_xid) {
 459             struct ofp_stats_reply *osr;
 460           
 461             ofp_print(stdout, reply->data, reply->size, 1);
 462 
 463             osr = ofpbuf_at(reply, 0, sizeof *osr);
 464             done = !osr || !(ntohs(osr->flags) & OFPSF_REPLY_MORE);
 465         } else {
 466             VLOG_DBG("received reply with xid %08"PRIx32" "
 467                      "!= expected %08"PRIx32, recv_xid, send_xid);
 468         }
 469         ofpbuf_delete(reply);
 470     }
 471     vconn_close(vconn);
 472 }

 511 static void
 512 do_status(const struct settings *s UNUSED, int argc, char *argv[])
 513 {
 514     struct nicira_header *request, *reply;
 515     struct vconn *vconn;
 516     struct ofpbuf *b;
 517 
 518     request = make_openflow(sizeof *request, OFPT_VENDOR, &b);
 519     request->vendor = htonl(NX_VENDOR_ID);
 520     request->subtype = htonl(NXT_STATUS_REQUEST);
 521     if (argc > 2) {
 522         ofpbuf_put(b, argv[2], strlen(argv[2]));
 523     }
 524     open_vconn(argv[1], &vconn);
 525     run(vconn_transact(vconn, b, &b), "talking to %s", argv[1]);
 526     vconn_close(vconn);
 527 
 528     if (b->size < sizeof *reply) {
 529         ofp_fatal(0, "short reply (%zu bytes)", b->size);
 530     }
 531     reply = b->data;
 532     if (reply->header.type != OFPT_VENDOR
 533         || reply->vendor != ntohl(NX_VENDOR_ID)
 534         || reply->subtype != ntohl(NXT_STATUS_REPLY)) {
 535         ofp_print(stderr, b->data, b->size, 2);
 536         ofp_fatal(0, "bad reply");
 537     }
 538 
 539     fwrite(reply + 1, b->size, 1, stdout);
 540 }

 758 static uint32_t
 759 str_to_ip(const char *str_, uint32_t *ip)
 760 {
 761     char *str = xstrdup(str_);
 762     char *save_ptr = NULL;
 763     const char *name, *netmask;
 764     struct in_addr in_addr;
 765     int n_wild, retval;
 766 
 767     name = strtok_r(str, "//", &save_ptr);
 768     retval = name ? lookup_ip(name, &in_addr) : EINVAL;
 769     if (retval) {
 770         ofp_fatal(0, "%s: could not convert to IP address", str);
 771     }
 772     *ip = in_addr.s_addr;
 773 
 774     netmask = strtok_r(NULL, "//", &save_ptr);
 775     if (netmask) {
 776         uint8_t o[4];
 777         if (sscanf(netmask, "%"SCNu8".%"SCNu8".%"SCNu8".%"SCNu8,
 778                    &o[0], &o[1], &o[2], &o[3]) == 4) {
 779             uint32_t nm = (o[0] << 24) | (o[1] << 16) | (o[2] << 8) | o[3];
 780             int i;
 781 
 782             /* Find first 1-bit. */
 783             for (i = 0; i < 32; i++) {
 784                 if (nm & (1u << i)) {
 785                     break;
 786                 }
 787             }
 788             n_wild = i;
 789 
 790             /* Verify that the rest of the bits are 1-bits. */
 791             for (; i < 32; i++) {
 792                 if (!(nm & (1u << i))) {
 793                     ofp_fatal(0, "%s: %s is not a valid netmask",
 794                               str, netmask);
 795                 }
 796             }
 797         } else {
 798             int prefix = atoi(netmask);
 799             if (prefix <= 0 || prefix > 32) {
 800                 ofp_fatal(0, "%s: network prefix bits not between 1 and 32",
 801                           str);
 802             }
 803             n_wild = 32 - prefix;
 804         }
 805     } else {
 806         n_wild = 0;
 807     }
 808 
 809     free(str);
 810     return n_wild;
 811 }

 976 
 977     for (f = fields; f < &fields[ARRAY_SIZE(fields)]; f++) {
 978         if (!strcmp(f->name, name)) {
 979             *f_out = f;
 980             return true;
 981         }
 982     }
 983     *f_out = NULL;
 984     return false;

 987 static void
 988 str_to_flow(char *string, struct ofp_match *match, struct ofpbuf *actions,
 989             uint8_t *table_idx, uint16_t *out_port, uint16_t *priority,
 990             uint16_t *idle_timeout, uint16_t *hard_timeout,
 991             uint64_t *cookie)
 992 {
 993     char *save_ptr = NULL;
 994     char *name;
 995     uint32_t wildcards;
 996 
 997     if (table_idx) {
 998         *table_idx = 0xff;
 999     }
1000     if (out_port) {
1001         *out_port = OFPP_NONE;
1002     }
1003     if (priority) {
1004         *priority = OFP_DEFAULT_PRIORITY;
1005     }
1006     if (idle_timeout) {
1007         *idle_timeout = DEFAULT_IDLE_TIMEOUT;
1008     }
1009     if (hard_timeout) {
1010         *hard_timeout = OFP_FLOW_PERMANENT;
1011     }
1012     if (cookie) {
1013         *cookie = 0;
1014     }
1015     if (actions) {
1016         char *act_str = strstr(string, "actions");
1017         if (!act_str) {
1018             ofp_fatal(0, "must specify an action");
1019         }
1020         *(act_str-1) = '\0';
1021 
1022         act_str = strchr(act_str, '=');
1023         if (!act_str) {
1024             ofp_fatal(0, "must specify an action");
1025         }
1026 
1027         act_str++;
1028 
1029         str_to_action(act_str, actions);
1030     }
1031     memset(match, 0, sizeof *match);
1032     wildcards = OFPFW_ALL;
1033     for (name = strtok_r(string, "=, \t\r\n", &save_ptr); name;
1034          name = strtok_r(NULL, "=, \t\r\n", &save_ptr)) {
1035         const struct protocol *p;
1036 
1037         if (parse_protocol(name, &p)) {
1038             wildcards &= ~OFPFW_DL_TYPE;
1039             match->dl_type = htons(p->dl_type);
1040             if (p->nw_proto) {
1041                 wildcards &= ~OFPFW_NW_PROTO;
1042                 match->nw_proto = p->nw_proto;
1043             }
1044         } else {
1045             const struct field *f;
1046             char *value;
1047 
1048             value = strtok_r(NULL, ", \t\r\n", &save_ptr);
1049             if (!value) {
1050                 ofp_fatal(0, "field %s missing value", name);
1051             }
1052 
1053             if (table_idx && !strcmp(name, "table")) {
1054                 *table_idx = atoi(value);
1055             } else if (out_port && !strcmp(name, "out_port")) {
1056                 *out_port = atoi(value);
1057             } else if (priority && !strcmp(name, "priority")) {
1058                 *priority = atoi(value);
1059             } else if (idle_timeout && !strcmp(name, "idle_timeout")) {
1060                 *idle_timeout = atoi(value);
1061             } else if (hard_timeout && !strcmp(name, "hard_timeout")) {
1062                 *hard_timeout = atoi(value);
1063             } else if (cookie && !strcmp(name, "cookie")) {
1064                 *cookie = atoi(value);
1065             } else if (parse_field(name, &f)) {
1066                 void *data = (char *) match + f->offset;
1067                 if (!strcmp(value, "*") || !strcmp(value, "ANY")) {
1068                     wildcards |= f->wildcard;
1069                 } else {
1070                     wildcards &= ~f->wildcard;
1071                     if (f->type == F_U8) {
1072                         *(uint8_t *) data = str_to_u32(value);
1073                     } else if (f->type == F_U16) {
1074                         *(uint16_t *) data = htons(str_to_u32(value));
1075                     } else if (f->type == F_MAC) {
1076                         str_to_mac(value, data);
1077                     } else if (f->type == F_IP) {
1078                         wildcards |= str_to_ip(value, data) << f->shift;
1079                     } else {
1080                         NOT_REACHED();
1081                     }
1082                 }
1083             } else {
1084                 ofp_fatal(0, "unknown keyword %s", name);
1085             }
1086         }
1087     }
1088     match->wildcards = htonl(wildcards);
1089 }

1407             }

1422     opm->port_no = osf->ports[port_idx].port_no;

1424     opm->config = htonl(0);
1425     opm->mask = htonl(0);
1426     opm->advertise = htonl(0);
1427

1429

1434         opm->mask |= htonl(OFPPC_PORT_DOWN);

1575 /* Length of queue request; works with 16-bit property values like min_rate */
1576 #define Q_REQ_LEN(prop_count) \
1577     (sizeof(struct ofp_packet_queue) + Q_PROP_LEN(prop_count))
1578 
1579 #define Q_PROP_LEN(prop_count) \
1580     ((prop_count) * sizeof(struct ofp_queue_prop_min_rate))
1581 
1582 /*
1583  * Execute a queue add/mod/del operation
1584  *
1585  * All commands must specify a port and queue id.
1586  * Add may specify a bandwidth value
1587  * Modify must specify a bandwidth value
1588  *
1589  * To simplify things, always allocate space for all three parameters
1590  * (port, queue, min-bw):
1591  *     openflow_queue_header (w/ ofp header, port)
1592  *     ofp_queue (with q_id and offset to properties list)
1593  *     ofp_queue_prop_min_rate (w/ prop header and rate info)
1594  */
1595 static struct openflow_queue_command_header *
1596 queue_req_create(int cmd, struct ofpbuf **b, uint16_t port,
1597                  uint32_t q_id, uint16_t min_rate)
1598 {
1599     struct openflow_queue_command_header *request;
1600     struct ofp_packet_queue *queue;
1601     struct ofp_queue_prop_min_rate *min_rate_prop;
1602     int req_bytes;
1603 
1604     req_bytes = sizeof(*request) + sizeof(*queue) + sizeof(*min_rate_prop);
1605     request = make_openflow(req_bytes, OFPT_VENDOR, b);
1606     if (request == NULL) {
1607         return NULL;
1608     }
1609     request->header.vendor = htonl(OPENFLOW_VENDOR_ID);
1610     request->header.subtype = htonl(cmd);
1611     request->port = htons(port);
1612 
1613     /* Will get complicated when queue properties w/ different struct sizes */
1614     queue = S_PTR(request, struct openflow_queue_command_header, body);
1615     queue->queue_id = htonl(q_id);
1616     queue->len = htons(Q_REQ_LEN(1));
1617 
1618     min_rate_prop = S_PTR(queue, struct ofp_packet_queue, properties);
1619     min_rate_prop->prop_header.property = htons(OFPQT_MIN_RATE);
1620     min_rate_prop->prop_header.len = htons(Q_PROP_LEN(1));
1621     min_rate_prop->rate = htons(min_rate);
1622 
1623     return request;
1624 }
