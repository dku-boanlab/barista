  34 /** 
  35  * \brief Function to parse a string and then make arguments 
  36  * \param str Raw string
  37  * \param argc The number of arguments
  38  * \param argv The pointer array for parsed arguments
  39  * \param max_argc The maximum number of arguments
  40  */
  41 int str2args(char *str, int *argc, char **argv, int max_argc)
  42 {
  43     uint8_t single_quotes;
  44     uint8_t double_quotes;
  45     uint8_t delim;
  46 
  47     single_quotes = 0;
  48     double_quotes = 0;
  49     delim = 1;
  50 
  51     *argc = 0;
  52 
  53     int i;
  54     int len = strlen(str);
  55     for (i = 0; i < len; i++) {
  56         if (str[i] == '\'') {
  57             if (single_quotes)
  58                 str[i] = '\0';
  59             else
  60                 i++;
  61             single_quotes = !single_quotes;
  62             goto __non_space;
  63         } else if (str[i] == '\"') {
  64             if (double_quotes)
  65                 str[i] = '\0';
  66             else
  67                 i++;
  68             double_quotes = !double_quotes;
  69             goto __non_space;
  70         }
  71 
  72         if (single_quotes || double_quotes)
  73             continue;
  74 
  75         if (isspace(str[i])) {
  76             delim = 1;
  77             str[i] = '\0';
  78             continue;
  79         }
  80 
  81 __non_space:
  82         if (delim == 1) {
  83             delim = 0;
  84             argv[(*argc)++] = &str[i];
  85             if (*argc > max_argc)
  86                 break;
  87         }
  88     }
  89 
  90     argv[*argc] = NULL;
  91 
  92     return 0;
  93 }

  95 /**
  96  * \brief Function to read the raw string in a file
  97  * \param fname File name
  98  * \return Raw string retrieved from the file
  99  */
 100 char *str_read(char *fname)
 101 {
 102     size_t hav_read = 0;
 103     FILE *fp = fopen(fname, "r");
 104     if (fp == NULL)
 105         return NULL;
 106 
 107     // get the size of a file
 108     fseek(fp, 0L, SEEK_END);
 109     int size = ftell(fp);
 110     fseek(fp, 0L, SEEK_SET);
 111 
 112     char *file = calloc(1, size + 1);
 113     if (file == NULL)
 114         return NULL;
 115 
 116     file[size] = '\0';
 117 
 118     while ((hav_read += fread(file, 1, size, fp)) < size);
 119 
 120     fclose(fp);
 121 
 122     return file;
 123 }

 125 /**
 126  * \brief Function to remove all comments and new lines
 127  * \param raw Raw string
 128  * \return Pre-processed string
 129  */
 130 char *str_preproc(char *raw)
 131 {
 132     char *line;
 133     int llen;
 134 
 135     int len = strlen(raw);
 136 
 137     LINE_FOREACH(line, llen, raw, len) {
 138         int i, iscomment = 0;
 139         for (i = 0; i < llen; i++) {
 140             if (!iscomment && line[i] == '#')
 141                 iscomment = 1;
 142             if (iscomment)
 143                 line[i] = ' ';
 144 
 145             else if (line[i] == '\\' && line[i + 1] == '\n')
 146                 line[i + 1] = line[i] = ' ';
 147         }
 148     }
 149 
 150     return raw;
 151 }
