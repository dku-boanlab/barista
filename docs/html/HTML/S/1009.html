  37 /** \brief The pointer of the first entry in a domain socket queue */

  56 static void init_queue(void)

  63 /**
  64  * \brief Function to push a socket into a domain socket queue
  65  * \param v Domain socket
  66  */
  67 static void push_back(int v)
  68 {
  69     if (sb_size == 0) {
  70         head = tail = FIRST;
  71     } else {
  72         tail = (LAST - tail > 0) ? tail + 1 : FIRST;
  73     }
  74 
  75     *tail = v;
  76 
  77     sb_size++;
  78 }
  79 
  80 /**
  81  * \brief Function to pop a socket from a domain socket queue
  82  * \return Domain socket
  83  */
  84 static int pop_front(void)
  85 {
  86     int ret = -1;
  87 
  88     if (sb_size > 0) {
  89         ret = *head;
  90         head = (LAST - head > 0) ? head + 1 : FIRST;
  91         sb_size--;
  92     }
  93 
  94     return ret;
  95 }

 146 /**
 147  * \brief Function to handle incoming messages from domain sockets
 148  * \param sock Domain socket
 149  * \param rx_buf Input buffer
 150  * \param bytes The size of the input buffer
 151  */
 152 static int msg_proc(int sock, uint8_t *rx_buf, int bytes)
 153 {
 154     context_t *c = &context[sock];
 155 
 156     uint8_t *temp = c->temp;
 157     int need = c->need;
 158     int done = c->done;
 159 
 160     int buf_ptr = 0;
 161     while (bytes > 0) {
 162         if (0 < done && done < 4) {
 163             if (done + bytes < 4) {
 164                 memmove(temp + done, rx_buf + buf_ptr, bytes);
 165 
 166                 done += bytes;
 167                 bytes = 0;
 168 
 169                 break;
 170             } else {
 171                 memmove(temp + done, rx_buf + buf_ptr, (4 - done));
 172 
 173                 bytes -= (4 - done);
 174                 buf_ptr += (4 - done);
 175 
 176                 need = ntohs(((struct ofp_header *)temp)->length) - 4;
 177                 done = 4;
 178             }
 179         }
 180 
 181         if (need == 0) {
 182             struct ofp_header *ofph = (struct ofp_header *)(rx_buf + buf_ptr);
 183 
 184             if (bytes < 4) {
 185                 memmove(temp, rx_buf + buf_ptr, bytes);
 186 
 187                 need = 0;
 188                 done = bytes;
 189                 bytes = 0;
 190             } else {
 191                 uint16_t len = ntohs(ofph->length);
 192 
 193                 if (bytes < len) {
 194                     memmove(temp, rx_buf + buf_ptr, bytes);
 195 
 196                     need = len - bytes;
 197                     done = bytes;
 198                     bytes = 0;
 199                 } else {
 200                     raw_msg_t msg = {0};
 201 
 202                     msg.fd = sock;
 203                     msg.length = ntohs(ofph->length);
 204 
 205                     msg.data = rx_buf + buf_ptr;
 206 
 207                     ev_ofp_msg_in(CONN_SB_ID, &msg);
 208 
 209                     bytes -= len;
 210                     buf_ptr += len;
 211                     need = 0;
 212                     done = 0;
 213                 }
 214             }
 215         } else {
 216             struct ofp_header *ofph = (struct ofp_header *)temp;
 217 
 218             if (need > bytes) {
 219                 memmove(temp + done, rx_buf + buf_ptr, bytes);
 220 
 221                 need -= bytes;
 222                 done += bytes;
 223                 bytes = 0;
 224             } else {
 225                 raw_msg_t msg = {0};
 226 
 227                 msg.fd = sock;
 228                 msg.length = ntohs(ofph->length);
 229 
 230                 memmove(temp + done, rx_buf + buf_ptr, need);
 231                 msg.data = temp;
 232 
 233                 ev_ofp_msg_in(CONN_SB_ID, &msg);
 234 
 235                 bytes -= need;
 236                 buf_ptr += need;
 237                 need = 0;
 238                 done = 0;
 239             }
 240         }
 241     }
 242 
 243     c->need = need;
 244     c->done = done;
 245 
 246     return bytes;
 247 }

 338 /**
 339  * \brief Function to receive raw messages from domain sockets
 340  * \return NULL
 341  */
 342 static void *do_tasks(void *null)
 343 {
 344     int wsock, bytes, done = 0;
 345     uint8_t rx_buf[BUFFER_SIZE] = {0};
 346 
 347     pthread_mutex_lock(&queue_mutex);
 348 
 349     while (conn_sb_on) {
 350         pthread_cond_wait(&queue_cond, &queue_mutex);
 351 
 352 FETCH_ONE_FD:
 353         if (conn_sb_on == FALSE) { // clean up
 354             pthread_mutex_unlock(&queue_mutex);
 355             break;
 356         }
 357 
 358         wsock = pop_front();
 359         if (wsock < 0) continue;
 360         pthread_mutex_unlock(&queue_mutex);
 361 
 362         done = 0;
 363         bytes = 1;
 364         while (bytes > 0) {
 365             bytes = read(wsock, rx_buf, BUFFER_SIZE);
 366             if (bytes < 0) {
 367                 if (errno != EAGAIN) {
 368                     done = 1;
 369                 }
 370                 break;
 371             } else if (bytes == 0) {
 372                 done = 1;
 373                 break;
 374             }
 375 
 376             bytes = sb_recv_cb(wsock, rx_buf, bytes);
 377             if (bytes == -1) {
 378                 done = 1;
 379                 break;
 380             }
 381         }
 382 
 383         if (done) {
 384             // closed connection
 385             switch_t sw = {0};
 386             sw.fd = wsock;
 387             ev_sw_expired_conn(CONN_SB_ID, &sw);
 388 
 389             clean_context(wsock);
 390             close(wsock);
 391         } else {
 392             if (relink_epoll(epoll, wsock, EPOLLIN | EPOLLET | EPOLLONESHOT) < 0) {
 393                 // closed connection
 394                 switch_t sw = {0};
 395                 sw.fd = wsock;
 396                 ev_sw_expired_conn(CONN_SB_ID, &sw);
 397 
 398                 clean_context(wsock);
 399                 close(wsock);
 400             }
 401         }
 402 
 403         pthread_mutex_lock(&queue_mutex);
 404 
 405         if (get_qsize() > 0) {
 406             goto FETCH_ONE_FD;
 407         }
 408     }
 409 
 410     return NULL;
 411 }

 441 /**
 442  * \brief Function to set the non-blocking mode to a socket
 443  * \param fd Socket
 444  */
 445 static int nonblocking_mode(const int fd)
 446 {
 447     int flags;
 448 
 449     if ((flags = fcntl(fd, F_GETFL, 0)) < 0) {
 450         if (errno != EAGAIN) {
 451             PERROR("fcntl");
 452             return -1;
 453         }
 454     }
 455 
 456     flags |= O_NONBLOCK;
 457     if (fcntl(fd, F_SETFL, flags) < 0) {
 458         if (errno != EAGAIN) {
 459             PERROR("fcntl");
 460             return -1;
 461         }
 462     }
 463 
 464     return 0;
 465 }

 467 /**
 468  * \brief Function to initialize a domain socket
 469  * \return None
 470  */
 471 static int init_socket(void)
 472 {
 473     if ((sc = socket(AF_UNIX, SOCK_STREAM, 0)) < 0) {
 474         PERROR("socket");
 475         return -1;
 476     }
 477 
 478     if (nonblocking_mode(sc) < 0) {
 479         PERROR("nonblocking_mode");
 480         close(sc);
 481         return -1;
 482     }
 483 
 484     struct sockaddr_un local_addr;
 485 
 486     bzero(&local_addr, sizeof(struct sockaddr_un));
 487     local_addr.sun_family = AF_UNIX;
 488     strcpy(local_addr.sun_path, DS_PATH);
 489     unlink(local_addr.sun_path);
 490 
 491     int len = strlen(local_addr.sun_path) + sizeof(local_addr.sun_family);
 492 
 493     if (bind(sc, (struct sockaddr *)&local_addr, len) == -1) {
 494         PERROR("bind");
 495         close(sc);
 496         return -1;
 497     }
 498 
 499     if (listen(sc, SOMAXCONN) == -1) {
 500         PERROR("listen");
 501         close(sc);
 502         return -1;
 503     }
 504 
 505     if (link_epoll(epoll, sc, EPOLLIN | EPOLLET) < 0) {
 506         PERROR("link_epoll");
 507         close(sc);
 508         return -1;
 509     }
 510 
 511     return sc;
 512 }

 514 /**
 515  * \brief Function to receive connections from domain sockets
 516  * \return NULL
 517  */
 518 static void *socket_listen(void *arg)
 519 {
 520     int csock;
 521     int nums = 0;
 522     struct sockaddr_in client;
 523     socklen_t len = sizeof(struct sockaddr);
 524 
 525     while (conn_sb_on) {
 526         nums = epoll_wait(epoll, events, MAXEVENTS, 100);
 527 
 528         if (conn_sb_on == FALSE) break;
 529 
 530         int i;
 531         for (i=0; i<nums; i++) {
 532             if (events[i].data.fd == sc) {
 533                 while (1) {
 534                     if ((csock = accept(sc, (struct sockaddr *)&client, &len)) < 0) {
 535                         if (errno != EAGAIN && errno != EWOULDBLOCK) {
 536                             PERROR("accept");
 537                         }
 538                         break;
 539                     }
 540 
 541                     // clean up the previous segments
 542                     clean_context(csock);
 543 
 544                     // new connection
 545                     switch_t sw = {0};
 546                     sw.fd = csock;
 547                     ev_sw_new_conn(CONN_SB_ID, &sw);
 548 
 549                     if (nonblocking_mode(csock) < 0) {
 550                         // closed connection
 551                         switch_t sw = {0};
 552                         sw.fd = csock;
 553                         ev_sw_expired_conn(CONN_SB_ID, &sw);
 554 
 555                         close(csock);
 556                         break;
 557                     }
 558 
 559                     if (link_epoll(epoll, csock, EPOLLIN | EPOLLET | EPOLLONESHOT) < 0) {
 560                         // closed connection
 561                         switch_t sw = {0};
 562                         sw.fd = csock;
 563                         ev_sw_expired_conn(CONN_SB_ID, &sw);
 564 
 565                         close(csock);
 566                         break;
 567                     }
 568                 }
 569             } else {
 570                 pthread_mutex_lock(&queue_mutex);
 571                 push_back(events[i].data.fd);
 572                 pthread_mutex_unlock(&queue_mutex);
 573                 pthread_cond_signal(&queue_cond);
 574             }
 575         }
 576 
 577         if (nums < 0 && errno != EINTR)
 578             break;
 579     }
 580 
 581     if (nums < 0)
 582         PERROR("epoll_wait");
 583 
 584     close(epoll);
 585     close(sc);
 586 
 587     LOG_INFO(CONN_SB_ID, "Closed a socket");
 588 
 589     return NULL;
 590 }

 670 /**
 671  * \brief The handler function
 672  * \param ev Read-only event
 673  * \param ev_out Read-write event (if this component has the write permission)
 674  */
 675 int conn_sb_handler(const event_t *ev, event_out_t *ev_out)
 676 {
 677     switch (ev->type) {
 678     case EV_OFP_MSG_OUT:
 679         PRINT_EV("EV_OFP_MSG_OUT\n");
 680         {
 681             const raw_msg_t *msg = ev->msg;
 682             int fd = msg->fd;
 683 
 684             int remain = msg->length;
 685             int bytes = 0;
 686             int done = 0;
 687 
 688             while (remain > 0) {
 689                 bytes = write(fd, msg->data + done, remain);
 690                 if (bytes < 0 && errno != EAGAIN) {
 691                     break;
 692                 } else if (bytes < 0) {
 693                     continue;
 694                 }
 695 
 696                 remain -= bytes;
 697                 done += bytes;
 698             }
 699         }
 700         break;
 701     default:
 702         break;
 703     }
 704 
 705     return 0;
 706 }
