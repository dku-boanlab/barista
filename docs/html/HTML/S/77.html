 269 int 
 270 destroy(struct oflops_context *ctx) {
 271   char msg[1024];
 272   struct timeval now;
 273   FILE *out = fopen(logfile, "w");
 274   struct entry *np;
 275   long long t[3], t_sq[3], mean, std;
 276   int count[3], min_id[3], max_id[3];
 277   int ch;
 278   float loss;
 279   int xid;
 280 
 281   gettimeofday(&now, NULL);
 282   printf("destroying code\n");
 283   snprintf(msg, 1024, "OFPT_ERROR_DELAY:%lld", delay_false_modificaton);
 284   oflops_log(now, GENERIC_MSG, msg);
 285   snprintf(msg, 1024, "OFPT_INSERT_DELAY:%lld", delay_modificaton);
 286   oflops_log(now, GENERIC_MSG, msg);
 287 
 288   for(ch = 0; ch < ctx->n_channels-1; ch++) {
 289     count[ch] = 0;
 290     min_id[ch] =  INT_MAX;
 291     max_id[ch] =  INT_MIN;
 292     t[ch] = 0;
 293     t_sq[ch] = 0;    
 294   }
 295 
 296   for (np = head.tqh_first; np != NULL; np = np->entries.tqe_next) {
 297     if(fprintf(out, "%lu;%lu.%06lu;%lu.%06lu;%d\n", 
 298                (long unsigned int)np->id,  
 299                (long unsigned int)np->snd.tv_sec, 
 300                (long unsigned int)np->snd.tv_usec,
 301                (long unsigned int)np->rcv.tv_sec, 
 302                (long unsigned int)np->rcv.tv_usec,  np->ch) < 0)  
 303       perror_and_exit("fprintf fail", 1); 
 304     ch = np->ch - 1;
 305     count[ch]++; 
 306     min_id[ch] = (np->id < min_id[ch])?np->id:min_id[ch];
 307     max_id[ch] = (np->id > max_id[ch])?np->id:max_id[ch];
 308     t[ch] += time_diff(&np->snd, &np->rcv);
 309     t_sq[ch] += time_diff(&np->snd, &np->rcv)*time_diff(&np->snd, &np->rcv);
 310     free(np);
 311   }
 312 
 313   for(ch = 0; ch < ctx->n_channels-1; ch++) {
 314     if(count[ch] == 0) continue;
 315     mean = t[ch]/count[ch];
 316     std = (t_sq[ch]/count[ch]) - mean*mean;
 317     if(std >= 0) std = sqrt(std); else std = LONG_MAX;
 318     loss = (float)count[ch]/(float)(max_id[ch] - min_id[ch]);
 319     snprintf(msg, 1024, "statistics:port:%d:%lld:%lld:%.4f:%d", 
 320              ctx->channels[ch + 1].of_port, mean, std, loss, count[ch]);
 321     oflops_log(now, GENERIC_MSG, msg);
 322   }
 323 
 324   t[0] = 0;
 325   t_sq[0] = 0;
 326   for (xid = 1 ; xid < trans_id; xid++) {
 327     t[0] +=delay[xid];
 328     t_sq[0] += delay[xid]*delay[xid];
 329   }
 330   std = (t_sq[0]/trans_id) - mean*mean;
 331   std = (std >= 0)?sqrt(std):LONG_MAX;
 332   snprintf(msg, 1024, "ofp_echo_statistics:%lld:%lld", 
 333            t[0]/trans_id, std);
 334   oflops_log(now, GENERIC_MSG, msg);
 335   return 0;
 336 }

 338 /** Handle timer event
 339  * @param ctx pointer to opaque context
 340  * @param te pointer to timer event
 341  */
 342 int handle_timer_event(struct oflops_context * ctx, struct timer_event *te) {  
 343   char *str = te->arg; 
 344   int len;
 345   void *b;
 346   struct timeval now;
 347   struct ofp_flow_mod *ofp;
 348   struct ofp_header *ofph;
 349 
 350   //terminate process 
 351   if (strcmp(str, BYESTR) == 0) {
 352     printf("terminating test....\n");
 353     oflops_end_test(ctx,1);
 354     finished = 0;
 355     return 0;    
 356   } else if(strcmp(str, OFP_PING) == 0) {
 357     len = make_ofp_hello(&b);
 358     ofph = (struct ofp_header *)b;
 359     ofph->type = OFPT_ECHO_REQUEST;
 360     ofph->xid = htonl(trans_id++);
 361     oflops_send_of_mesgs(ctx, b, len);
 362     free(b);
 363     gettimeofday(&now, NULL);
 364     add_time(&now, 1, 0);
 365     oflops_schedule_timer_event(ctx,&now, OFP_PING);
 366 
 367 
 368   } else if  (strcmp(str, SND_FALSE_ACT) == 0) {
 369     struct ofp_action_output *cmd;
 370     int cmd_len;
 371     uint32_t tmp = fl_probe->nw_src;
 372     fl_probe->nw_src = inet_addr("10.5.5.5");
 373     cmd_len = sizeof(struct ofp_action_output);
 374     cmd = (struct ofp_action_output *)xmalloc(cmd_len);
 375     cmd->type = htons(0);
 376     cmd->len = htons(8);
 377     cmd->max_len = htons(0);
 378     cmd->port = htons(50);
 379     len = make_ofp_flow_modify(&b, fl_probe, (char *)cmd, cmd_len, 
 380                                1, 1200);
 381     ofp = (struct ofp_flow_mod *) b;
 382     ofp->buffer_id = htonl(1);
 383     send_false_mod= 1;
 384     printf("sending false modification to measure delay\n");
 385     oflops_send_of_mesg(ctx, b);
 386     free(b);
 387     free(cmd);
 388     fl_probe->nw_src = tmp;
 389   } else if (strcmp(str, SND_ACT) == 0) {
 390     len = make_ofp_flow_modify(&b, fl_probe, command, command_len, 
 391                                1, 1200);
 392     send_mod= 1;
 393     oflops_send_of_mesg(ctx, b);
 394     free(b);
 395   } else if(strcmp(str, SNMPGET) == 0) {
 396     oflops_snmp_get(ctx, ctx->cpuOID, ctx->cpuOID_len);
 397     int i;
 398     for(i=0;i<ctx->n_channels;i++) {
 399       oflops_snmp_get(ctx, ctx->channels[i].inOID, ctx->channels[i].inOID_len);
 400       oflops_snmp_get(ctx, ctx->channels[i].outOID, ctx->channels[i].outOID_len);
 401     }    
 402     gettimeofday(&now, NULL);
 403     add_time(&now, 10, 0);
 404     oflops_schedule_timer_event(ctx,&now, SNMPGET);
 405   }
 406   return 0;
 407 }

 426 /** Handle pcap event.
 427  * @param ctx pointer to opaque context
 428  * @param pe pcap event
 429  * @param ch enumeration of channel that pcap event is triggered
 430  */
 431 int 
 432 handle_pcap_event(struct oflops_context *ctx, struct pcap_event * pe, oflops_channel_name ch) {
 433   struct pktgen_hdr *pktgen;
 434   int dir, len;
 435   struct ofp_header *ofp;
 436   struct pcap_event *ofp_msg;
 437   struct ofp_error_msg *err_p = NULL;
 438   char msg[1024];
 439 
 440   if (ch == OFLOPS_CONTROL) {
 441     dir = append_data_to_flow(pe->data,pe->pcaphdr);
 442     while(contains_next_msg(dir) > 0) {
 443       len = get_next_msg(dir, &ofp_msg);
 444       ofp = (struct ofp_header *)ofp_msg->data;
 445       switch(ofp->type) {
 446       case OFPT_FLOW_MOD:
 447         if (send_false_mod) {
 448           if(send_mod) {
 449             memcpy(&true_modification, &pe->pcaphdr.ts, sizeof(struct timeval));
 450           } else {
 451             memcpy(&false_modification, &pe->pcaphdr.ts, sizeof(struct timeval));
 452           }
 453         }
 454         if(send_mod) {
 455           oflops_log(pe->pcaphdr.ts,OFPT_FLOW_MOD_ADD, "flow modification send");
 456         }
 457         break;
 458       case OFPT_ECHO_REQUEST:
 459         if(ntohl(ofp->xid) < 100) 
 460           memcpy(&ofp_ping_timestamp[ntohl(ofp->xid)], &pe->pcaphdr.ts, sizeof(struct timeval));
 461         break;
 462       case OFPT_ECHO_REPLY:
 463         if(ntohl(ofp->xid) < 100 && ntohl(ofp->xid) > 0) {
 464           delay[ntohl(ofp->xid)] = time_diff(&ofp_ping_timestamp[ntohl(ofp->xid)], &pe->pcaphdr.ts);
 465           snprintf(msg, 1024, "%d", time_diff(&ofp_ping_timestamp[ntohl(ofp->xid)], &pe->pcaphdr.ts));
 466           oflops_log(pe->pcaphdr.ts, OFPT_ECHO_REPLY_MSG, msg);
 467         }
 468         break;    
 469       case OFPT_ERROR:
 470         err_p = (struct ofp_error_msg *)ofp;
 471         if(send_false_mod)  {
 472           delay_false_modificaton = time_diff(&false_modification, &pe->pcaphdr.ts);
 473         } else {
 474           snprintf(msg, 1024, "%d:%d", ntohs(err_p->type), ntohs(err_p->code));
 475           oflops_log(pe->pcaphdr.ts, OFPT_ERROR_MSG, msg);
 476           fprintf(stderr, "OFPT_ERROR_MSG:%s\n", msg);
 477           break;   
 478         }
 479       }
 480     }
 481   } else if ((ch == OFLOPS_DATA1) || (ch == OFLOPS_DATA2) || (ch == OFLOPS_DATA3)) {
 482     struct flow fl;
 483     pktgen = extract_pktgen_pkt(pe->data, pe->pcaphdr.caplen, &fl);
 484     if((ch == OFLOPS_DATA3) && (!first_pkt)) {
 485       delay_modificaton = time_diff(&true_modification, &pe->pcaphdr.ts);
 486       oflops_log(pe->pcaphdr.ts, GENERIC_MSG, "first packet on new port");
 487       first_pkt = 1;
 488     }
 489     if(htonl(pktgen->seq_num) % 100000 == 0)
 490       printf("data packet received %d\n", htonl(pktgen->seq_num));
 491     
 492     struct entry *n1 = malloc(sizeof(struct entry));
 493     n1->snd.tv_sec = htonl(pktgen->tv_sec);
 494     n1->snd.tv_usec = htonl(pktgen->tv_usec);
 495     memcpy(&n1->rcv, &pe->pcaphdr.ts, sizeof(struct timeval));
 496     n1->id = htonl(pktgen->seq_num);
 497     n1->ch = ch;
 498     TAILQ_INSERT_TAIL(&head, n1, entries);
 499   }
 500   return 0;
 501 }

 527 int 
 528 handle_snmp_event(struct oflops_context * ctx, struct snmp_event * se) {
 529   netsnmp_variable_list *vars;
 530   int len = 1024;
 531   char msg[1024], log[1024];
 532   struct timeval now;
 533   int i;
 534   gettimeofday(&now, NULL);
 535 
 536   for(vars = se->pdu->variables; vars; vars = vars->next_variable)  {
 537     snprint_value(msg, len, vars->name, vars->name_length, vars);
 538     if((vars->name_length == ctx->cpuOID_len) &&
 539        (memcmp(vars->name, ctx->cpuOID,  ctx->cpuOID_len * sizeof(oid)) == 0) ) {
 540       snprintf(log, len, "cpu : %s %%", msg);
 541       oflops_log(now, SNMP_MSG, log);
 542     } else {
 543       for(i=0;i<ctx->n_channels;i++) {
 544         if((vars->name_length == ctx->channels[i].inOID_len) &&
 545            (memcmp(vars->name, ctx->channels[i].inOID,  
 546                    ctx->channels[i].inOID_len * sizeof(oid)) == 0) ) {
 547           snprintf(log, len, "port %d : rx %s pkts",  
 548                    (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 549           oflops_log(now, SNMP_MSG, log);
 550           break;
 551         }
 552         
 553         if((vars->name_length == ctx->channels[i].outOID_len) &&
 554            (memcmp(vars->name, ctx->channels[i].outOID,  
 555                    ctx->channels[i].outOID_len * sizeof(oid))==0) ) {
 556           snprintf(log, len, "port %d : tx %s pkts",  
 557                    (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 558           oflops_log(now, SNMP_MSG, log);
 559           break;
 560         }
 561       } //for
 562     }// if cpu
 563   }// variable iterator
 564   return 0;
 565 }

 613 /*
 614  * read the first 2 hex characters and return the byte.
 615  */
 616 uint8_t
 617 read_hex(const char *data) {
 618   uint8_t ret = 0;
 619   int i;
 620 
 621   for(i = 0 ; i < 2 ; i++) {
 622     ret = ret << 4;
 623     if((*data >= 'A') && (*data <= 'F')) {
 624       ret += 10 + (*data - 'A');
 625     } else if((*data >= 'a') && (*data <= 'f')) {
 626       ret += 10 + ((*data) - 'a');
 627     } else if((*data >= '0') && (*data <= '9')) {
 628       ret += ((*data) - '0');
 629     }
 630     data++;
 631   }
 632   return ret;
 633 }

 635 /**
 636  * Initialization code with parameters
 637  * @param ctx 
 638  */
 639 int init(struct oflops_context *ctx, char * config_str) {
 640   char *pos = NULL;
 641   char *param = config_str;
 642   int len = strlen(config_str);
 643   char *value = NULL;
 644   char *action;
 645 
 646   struct ofp_action_output *act_out;
 647 
 648   //init counters
 649   finished = 0;
 650 
 651   struct timeval now;
 652   gettimeofday(&now, NULL);
 653 
 654   cli_param = strdup(config_str);
 655 
 656 
 657   while(*config_str == ' ') {
 658     config_str++;
 659   }
 660   param = config_str;
 661   while(1) {
 662     pos = index(param, ' ');
 663 
 664     if((pos == NULL)) {
 665       if (*param != '\0') {
 666         pos = param + strlen(param) + 1;
 667       } else
 668         break;
 669     }
 670     *pos='\0';
 671     pos++;
 672     value = index(param,'=');
 673     *value = '\0';
 674     value++;
 675     //fprintf(stderr, "param = %s, value = %s\n", param, value);
 676     if(value != NULL) {
 677       if(strcmp(param, "pkt_size") == 0) {
 678         //parse int to get pkt size
 679         pkt_size = strtol(value, NULL, 0);
 680         if((pkt_size < MIN_PKT_SIZE) && (pkt_size > MAX_PKT_SIZE))
 681           perror_and_exit("Invalid packet size value", 1);
 682       } else if(strcmp(param, "data_rate") == 0) {
 683         //parse int to get rate of background data
 684         datarate = strtol(value, NULL, 0);
 685         if((datarate < 0) || (datarate > 1010))
 686           perror_and_exit("Invalid data rate param(Values between 1 and 1010)", 1);
 687       }  else if(strcmp(param, "probe_rate") == 0) {
 688         //parse int to get measurement probe rate
 689         proberate = strtol(value, NULL, 0);
 690         if((proberate <= 0) || (proberate >= 1010)) 
 691           perror_and_exit("Invalid probe rate param(Value between 1 and 1010)", 1);
 692       }  else if(strcmp(param, "action") == 0) {
 693         char *p = value;
 694         while((*p != ' ') && (*p != '\0') && (config_str + len > p)) {
 695           action = p;
 696           //find where value ends and set it to null to extract the string.
 697           p = index(p, ',');
 698           if(p == NULL) {
 699             p = config_str + len + 1;
 700             *p='\0';
 701           } else {
 702             *p = '\0'; 
 703             p++;
 704           }
 705           
 706           //set null char to split action param and action value
 707           param = index(action, '/');
 708           if(param != NULL) {
 709             *param = '\0';
 710             param++;
 711           }
 712 
 713           //check if action value is correct and append it at the end of the action list
 714           if(*action >= '0' && *action <= '9') {
 715               append_action((*action) - '0', param);
 716             } else if (*action == 'a') {
 717               append_action(10, param);
 718             } else { 
 719               printf("invalid action: %1s", action); 
 720               continue;
 721             }     
 722         }
 723       } else if(strcmp(param, "table") == 0) {
 724         //parse int to get pkt size
 725         table = strtol(value, NULL, 0);
 726         if((table < 0) && (table > 2))  
 727           perror_and_exit("Invalid table number", 1);
 728       } else if(strcmp(param, "flows") == 0) { 
 729         //parse int to get pkt size
 730         flows = strtol(value, NULL, 0);
 731         if(flows <= 0)  
 732           perror_and_exit("Invalid flow number", 1);
 733       } else 
 734         fprintf(stderr, "Invalid parameter:%s\n", param);
 735       param = pos;
 736     }
 737   } 
 738 
 739   //calculate sendind interval
 740   if(datarate > 0) {
 741     data_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (datarate * mbits_to_bits);
 742     fprintf(stderr, "Sending data interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 743             (uint32_t)data_snd_interval, (uint32_t)pkt_size, (uint32_t)datarate);
 744   } else {
 745     fprintf(stderr, "background data probe is disabled\n");
 746     data_snd_interval = 0;
 747   }
 748   probe_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (proberate * mbits_to_bits);
 749   fprintf(stderr, "Sending probe interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 750       (uint32_t)probe_snd_interval, (uint32_t)pkt_size, (uint32_t)proberate);
 751 
 752   //by default the new rule should be redirected to port 2 to make the measurement easier
 753   /* fprintf(stderr, "by default output packet to port 1\n"); */
 754   /* command_len += sizeof(struct ofp_action_output); */
 755   /* command = realloc(command, command_len); */
 756   /* act_out = (struct ofp_action_output *) */
 757   /*   (command + (command_len - sizeof(struct ofp_action_output))); */
 758   /* act_out->type = htons(0); */
 759   /* act_out->len = htons(8); */
 760   /* act_out->max_len = htons(0); */
 761   /* act_out->port = htons(ctx->channels[OFLOPS_DATA3].of_port); */
 762   return 0;
 763 }
