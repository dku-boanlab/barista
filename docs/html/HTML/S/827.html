  86     struct ofpbuf *txbuf;
  87     struct poll_waiter *tx_waiter;
  88 
  89     /* rx_want and tx_want record the result of the last call to SSL_read()
  90      * and SSL_write(), respectively:
  91      *
  92      *    - If the call reported that data needed to be read from the file
  93      *      descriptor, the corresponding member is set to SSL_READING.
  94      *
  95      *    - If the call reported that data needed to be written to the file
  96      *      descriptor, the corresponding member is set to SSL_WRITING.
  97      *
  98      *    - Otherwise, the member is set to SSL_NOTHING, indicating that the
  99      *      call completed successfully (or with an error) and that there is no
 100      *      need to block.
 101      *
 102      * These are needed because there is no way to ask OpenSSL what a data read
 103      * or write would require without giving it a buffer to receive into or
 104      * data to send, respectively.  (Note that the SSL_want() status is
 105      * overwritten by each SSL_read() or SSL_write() call, so we can't rely on
 106      * its value.)
 107      *
 108      * A single call to SSL_read() or SSL_write() can perform both reading
 109      * and writing and thus invalidate not one of these values but actually
 110      * both.  Consider this situation, for example:
 111      *
 112      *    - SSL_write() blocks on a read, so tx_want gets SSL_READING.
 113      *
 114      *    - SSL_read() laters succeeds reading from 'fd' and clears out the
 115      *      whole receive buffer, so rx_want gets SSL_READING.
 116      *
 117      *    - Client calls vconn_wait(WAIT_RECV) and vconn_wait(WAIT_SEND) and
 118      *      blocks.
 119      *
 120      *    - Now we're stuck blocking until the peer sends us data, even though
 121      *      SSL_write() could now succeed, which could easily be a deadlock
 122      *      condition.
 123      *
 124      * On the other hand, we can't reset both tx_want and rx_want on every call
 125      * to SSL_read() or SSL_write(), because that would produce livelock,
 126      * e.g. in this situation:
 127      *
 128      *    - SSL_write() blocks, so tx_want gets SSL_READING or SSL_WRITING.
 129      *
 130      *    - SSL_read() blocks, so rx_want gets SSL_READING or SSL_WRITING,
 131      *      but tx_want gets reset to SSL_NOTHING.
 132      *
 133      *    - Client calls vconn_wait(WAIT_RECV) and vconn_wait(WAIT_SEND) and
 134      *      blocks.
 135      *
 136      *    - Client wakes up immediately since SSL_NOTHING in tx_want indicates
 137      *      that no blocking is necessary.
 138      *
 139      * The solution we adopt here is to set tx_want to SSL_NOTHING after
 140      * calling SSL_read() only if the SSL state of the connection changed,
 141      * which indicates that an SSL-level renegotiation made some progress, and
 142      * similarly for rx_want and SSL_write().  This prevents both the
 143      * deadlock and livelock situations above.
 144      */
 145     int rx_want, tx_want;
 146 };

 199 static int
 200 new_ssl_vconn(const char *name, int fd, enum session_type type,
 201               enum ssl_state state, const struct sockaddr_in *sin,
 202               struct vconn **vconnp)
 203 {
 204     struct ssl_vconn *sslv;
 205     SSL *ssl = NULL;
 206     int on = 1;
 207     int retval;
 208 
 209     /* Check for all the needful configuration. */
 210     if (!has_private_key) {
 211         VLOG_ERR("Private key must be configured to use SSL");
 212         goto error;
 213     }
 214     if (!has_certificate) {
 215         VLOG_ERR("Certificate must be configured to use SSL");
 216         goto error;
 217     }
 218     if (!has_ca_cert && !bootstrap_ca_cert) {
 219         VLOG_ERR("CA certificate must be configured to use SSL");
 220         goto error;
 221     }
 222     if (!SSL_CTX_check_private_key(ctx)) {
 223         VLOG_ERR("Private key does not match certificate public key: %s",
 224                  ERR_error_string(ERR_get_error(), NULL));
 225         goto error;
 226     }
 227 
 228     /* Disable Nagle. */
 229     retval = setsockopt(fd, IPPROTO_TCP, TCP_NODELAY, &on, sizeof on);
 230     if (retval) {
 231         VLOG_ERR("%s: setsockopt(TCP_NODELAY): %s", name, strerror(errno));
 232         close(fd);
 233         return errno;
 234     }
 235 
 236     /* Create and configure OpenSSL stream. */
 237     ssl = SSL_new(ctx);
 238     if (ssl == NULL) {
 239         VLOG_ERR("SSL_new: %s", ERR_error_string(ERR_get_error(), NULL));
 240         close(fd);
 241         return ENOPROTOOPT;
 242     }
 243     if (SSL_set_fd(ssl, fd) == 0) {
 244         VLOG_ERR("SSL_set_fd: %s", ERR_error_string(ERR_get_error(), NULL));
 245         goto error;
 246     }
 247     if (bootstrap_ca_cert && type == CLIENT) {
 248         SSL_set_verify(ssl, SSL_VERIFY_NONE, NULL);
 249     }
 250 
 251     /* Create and return the ssl_vconn. */
 252     sslv = xmalloc(sizeof *sslv);
 253     vconn_init(&sslv->vconn, &ssl_vconn_class, EAGAIN, sin->sin_addr.s_addr,
 254                name, true);
 255     sslv->state = state;
 256     sslv->type = type;
 257     sslv->fd = fd;
 258     sslv->ssl = ssl;
 259     sslv->rxbuf = NULL;
 260     sslv->txbuf = NULL;
 261     sslv->tx_waiter = NULL;
 262     sslv->rx_want = sslv->tx_want = SSL_NOTHING;
 263     *vconnp = &sslv->vconn;
 264     return 0;
 265 
 266 error:
 267     if (ssl) {
 268         SSL_free(ssl);
 269     }
 270     close(fd);
 271     return ENOPROTOOPT;
 272 }

 281 static int
 282 ssl_open(const char *name, char *suffix, struct vconn **vconnp)
 283 {
 284     char *save_ptr, *host_name, *port_string;
 285     struct sockaddr_in sin;
 286     int retval;
 287     int fd;
 288 
 289     retval = ssl_init();
 290     if (retval) {
 291         return retval;
 292     }
 293 
 294     /* Glibc 2.7 has a bug in strtok_r when compiling with optimization that
 295      * can cause segfaults here:
 296      * http://sources.redhat.com/bugzilla/show_bug.cgi?id=5614.
 297      * Using "::" instead of the obvious ":" works around it. */
 298     host_name = strtok_r(suffix, "::", &save_ptr);
 299     port_string = strtok_r(NULL, "::", &save_ptr);
 300     if (!host_name) {
 301         ofp_error(0, "%s: bad peer name format", name);
 302         return EAFNOSUPPORT;
 303     }
 304 
 305     memset(&sin, 0, sizeof sin);
 306     sin.sin_family = AF_INET;
 307     if (lookup_ip(host_name, &sin.sin_addr)) {
 308         return ENOENT;
 309     }
 310     sin.sin_port = htons(port_string && *port_string ? atoi(port_string)
 311                          : OFP_SSL_PORT);
 312 
 313     /* Create socket. */
 314     fd = socket(AF_INET, SOCK_STREAM, 0);
 315     if (fd < 0) {
 316         VLOG_ERR("%s: socket: %s", name, strerror(errno));
 317         return errno;
 318     }
 319     retval = set_nonblocking(fd);
 320     if (retval) {
 321         close(fd);
 322         return retval;
 323     }
 324 
 325     /* Connect socket. */
 326     retval = connect(fd, (struct sockaddr *) &sin, sizeof sin);
 327     if (retval < 0) {
 328         if (errno == EINPROGRESS) {
 329             return new_ssl_vconn(name, fd, CLIENT, STATE_TCP_CONNECTING,
 330                                  &sin, vconnp);
 331         } else {
 332             int error = errno;
 333             VLOG_ERR("%s: connect: %s", name, strerror(error));
 334             close(fd);
 335             return error;
 336         }
 337     } else {
 338         return new_ssl_vconn(name, fd, CLIENT, STATE_SSL_CONNECTING,
 339                              &sin, vconnp);
 340     }
 341 }

 343 static int
 344 do_ca_cert_bootstrap(struct vconn *vconn)
 345 {
 346     struct ssl_vconn *sslv = ssl_vconn_cast(vconn);
 347     STACK_OF(X509) *chain;
 348     X509 *ca_cert;
 349     FILE *file;
 350     int error;
 351     int fd;
 352 
 353     chain = SSL_get_peer_cert_chain(sslv->ssl);
 354     if (!chain || !sk_X509_num(chain)) {
 355         VLOG_ERR("could not bootstrap CA cert: no certificate presented by "
 356                  "peer");
 357         return EPROTO;
 358     }
 359     ca_cert = sk_X509_value(chain, sk_X509_num(chain) - 1);
 360 
 361     /* Check that 'ca_cert' is self-signed.  Otherwise it is not a CA
 362      * certificate and we should not attempt to use it as one. */
 363     error = X509_check_issued(ca_cert, ca_cert);
 364     if (error) {
 365         VLOG_ERR("could not bootstrap CA cert: obtained certificate is "
 366                  "not self-signed (%s)",
 367                  X509_verify_cert_error_string(error));
 368         if (sk_X509_num(chain) < 2) {
 369             VLOG_ERR("only one certificate was received, so probably the peer "
 370                      "is not configured to send its CA certificate");
 371         }
 372         return EPROTO;
 373     }
 374 
 375     fd = open(ca_cert_file, O_CREAT | O_EXCL | O_WRONLY, 0444);
 376     if (fd < 0) {
 377         VLOG_ERR("could not bootstrap CA cert: creating %s failed: %s",
 378                  ca_cert_file, strerror(errno));
 379         return errno;
 380     }
 381 
 382     file = fdopen(fd, "w");
 383     if (!file) {
 384         int error = errno;
 385         VLOG_ERR("could not bootstrap CA cert: fdopen failed: %s",
 386                  strerror(error));
 387         unlink(ca_cert_file);
 388         return error;
 389     }
 390 
 391     if (!PEM_write_X509(file, ca_cert)) {
 392         VLOG_ERR("could not bootstrap CA cert: PEM_write_X509 to %s failed: "
 393                  "%s", ca_cert_file, ERR_error_string(ERR_get_error(), NULL));
 394         fclose(file);
 395         unlink(ca_cert_file);
 396         return EIO;
 397     }
 398 
 399     if (fclose(file)) {
 400         int error = errno;
 401         VLOG_ERR("could not bootstrap CA cert: writing %s failed: %s",
 402                  ca_cert_file, strerror(error));
 403         unlink(ca_cert_file);
 404         return error;
 405     }
 406 
 407     VLOG_INFO("successfully bootstrapped CA cert to %s", ca_cert_file);
 408     log_ca_cert(ca_cert_file, ca_cert);
 409     bootstrap_ca_cert = false;
 410     has_ca_cert = true;
 411 
 412     /* SSL_CTX_add_client_CA makes a copy of ca_cert's relevant data. */
 413     SSL_CTX_add_client_CA(ctx, ca_cert);
 414 
 415     /* SSL_CTX_use_certificate() takes ownership of the certificate passed in.
 416      * 'ca_cert' is owned by sslv->ssl, so we need to duplicate it. */
 417     ca_cert = X509_dup(ca_cert);
 418     if (!ca_cert) {
 419         out_of_memory();
 420     }
 421     if (SSL_CTX_load_verify_locations(ctx, ca_cert_file, NULL) != 1) {
 422         VLOG_ERR("SSL_CTX_load_verify_locations: %s",
 423                  ERR_error_string(ERR_get_error(), NULL));
 424         return EPROTO;
 425     }
 426     VLOG_INFO("killing successful connection to retry using CA cert");
 427     return EPROTO;
 428 }

 430 static int
 431 ssl_connect(struct vconn *vconn)
 432 {
 433     struct ssl_vconn *sslv = ssl_vconn_cast(vconn);
 434     int retval;
 435 
 436     switch (sslv->state) {
 437     case STATE_TCP_CONNECTING:
 438         retval = check_connection_completion(sslv->fd);
 439         if (retval) {
 440             return retval;
 441         }
 442         sslv->state = STATE_SSL_CONNECTING;
 443         /* Fall through. */
 444 
 445     case STATE_SSL_CONNECTING:
 446         retval = (sslv->type == CLIENT
 447                    ? SSL_connect(sslv->ssl) : SSL_accept(sslv->ssl));
 448         if (retval != 1) {
 449             int error = SSL_get_error(sslv->ssl, retval);
 450             if (retval < 0 && ssl_wants_io(error)) {
 451                 return EAGAIN;
 452             } else {
 453                 int unused;
 454                 interpret_ssl_error((sslv->type == CLIENT ? "SSL_connect"
 455                                      : "SSL_accept"), retval, error, &unused);
 456                 shutdown(sslv->fd, SHUT_RDWR);
 457                 return EPROTO;
 458             }
 459         } else if (bootstrap_ca_cert) {
 460             return do_ca_cert_bootstrap(vconn);
 461         } else if ((SSL_get_verify_mode(sslv->ssl)
 462                     & (SSL_VERIFY_NONE | SSL_VERIFY_PEER))
 463                    != SSL_VERIFY_PEER) {
 464             /* Two or more SSL connections completed at the same time while we
 465              * were in bootstrap mode.  Only one of these can finish the
 466              * bootstrap successfully.  The other one(s) must be rejected
 467              * because they were not verified against the bootstrapped CA
 468              * certificate.  (Alternatively we could verify them against the CA
 469              * certificate, but that's more trouble than it's worth.  These
 470              * connections will succeed the next time they retry, assuming that
 471              * they have a certificate against the correct CA.) */
 472             VLOG_ERR("rejecting SSL connection during bootstrap race window");
 473             return EPROTO;
 474         } else {
 475             return 0;
 476         }
 477     }
 478 
 479     NOT_REACHED();
 480 }

 494 static int
 495 interpret_ssl_error(const char *function, int ret, int error,
 496                     int *want)
 497 {
 498     *want = SSL_NOTHING;
 499 
 500     switch (error) {
 501     case SSL_ERROR_NONE:
 502         VLOG_ERR_RL(&rl, "%s: unexpected SSL_ERROR_NONE", function);
 503         break;
 504 
 505     case SSL_ERROR_ZERO_RETURN:
 506         VLOG_ERR_RL(&rl, "%s: unexpected SSL_ERROR_ZERO_RETURN", function);
 507         break;
 508 
 509     case SSL_ERROR_WANT_READ:
 510         *want = SSL_READING;
 511         return EAGAIN;
 512 
 513     case SSL_ERROR_WANT_WRITE:
 514         *want = SSL_WRITING;
 515         return EAGAIN;
 516 
 517     case SSL_ERROR_WANT_CONNECT:
 518         VLOG_ERR_RL(&rl, "%s: unexpected SSL_ERROR_WANT_CONNECT", function);
 519         break;
 520 
 521     case SSL_ERROR_WANT_ACCEPT:
 522         VLOG_ERR_RL(&rl, "%s: unexpected SSL_ERROR_WANT_ACCEPT", function);
 523         break;
 524 
 525     case SSL_ERROR_WANT_X509_LOOKUP:
 526         VLOG_ERR_RL(&rl, "%s: unexpected SSL_ERROR_WANT_X509_LOOKUP",
 527                     function);
 528         break;
 529 
 530     case SSL_ERROR_SYSCALL: {
 531         int queued_error = ERR_get_error();
 532         if (queued_error == 0) {
 533             if (ret < 0) {
 534                 int status = errno;
 535                 VLOG_WARN_RL(&rl, "%s: system error (%s)",
 536                              function, strerror(status));
 537                 return status;
 538             } else {
 539                 VLOG_WARN_RL(&rl, "%s: unexpected SSL connection close",
 540                              function);
 541                 return EPROTO;
 542             }
 543         } else {
 544             VLOG_WARN_RL(&rl, "%s: %s",
 545                          function, ERR_error_string(queued_error, NULL));
 546             break;
 547         }
 548     }
 549 
 550     case SSL_ERROR_SSL: {
 551         int queued_error = ERR_get_error();
 552         if (queued_error != 0) {
 553             VLOG_WARN_RL(&rl, "%s: %s",
 554                          function, ERR_error_string(queued_error, NULL));
 555         } else {
 556             VLOG_ERR_RL(&rl, "%s: SSL_ERROR_SSL without queued error",
 557                         function);
 558         }
 559         break;
 560     }
 561 
 562     default:
 563         VLOG_ERR_RL(&rl, "%s: bad SSL error code %d", function, error);
 564         break;
 565     }
 566     return EIO;
 567 }

 608     if (old_state != SSL_get_state(sslv->ssl)) {
 609         sslv->tx_want = SSL_NOTHING;

 613         }

 618         rx->size += ret;
 619         if (ret == want_bytes) {
 620             if (rx->size > sizeof(struct ofp_header)) {

 622                 sslv->rxbuf = NULL;
 623                 return 0;
 624             } else {
 625                 goto again;
 626             }
 627         }
 628         return EAGAIN;
 629     } else {
 630         int error = SSL_get_error(sslv->ssl, ret);
 631         if (error == SSL_ERROR_ZERO_RETURN) {
 632             /* Connection closed (EOF). */
 633             if (rx->size) {
 634                 VLOG_WARN_RL(&rl, "SSL_read: unexpected connection close");
 635                 return EPROTO;
 636             } else {
 637                 return EOF;
 638             }
 639         } else {
 640             return interpret_ssl_error("SSL_read", ret, error, &sslv->rx_want);
 641         }
 642     }
 643 }

 658                                        want_to_poll_events(sslv->tx_want),

 662 static int
 663 ssl_do_tx(struct vconn *vconn)
 664 {
 665     struct ssl_vconn *sslv = ssl_vconn_cast(vconn);
 666 
 667     for (;;) {
 668         int old_state = SSL_get_state(sslv->ssl);
 669         int ret = SSL_write(sslv->ssl, sslv->txbuf->data, sslv->txbuf->size);
 670         if (old_state != SSL_get_state(sslv->ssl)) {
 671             sslv->rx_want = SSL_NOTHING;
 672         }
 673         sslv->tx_want = SSL_NOTHING;
 674         if (ret > 0) {
 675             ofpbuf_pull(sslv->txbuf, ret);
 676             if (sslv->txbuf->size == 0) {
 677                 return 0;
 678             }
 679         } else {
 680             int ssl_error = SSL_get_error(sslv->ssl, ret);
 681             if (ssl_error == SSL_ERROR_ZERO_RETURN) {
 682                 VLOG_WARN_RL(&rl, "SSL_write: connection closed");
 683                 return EPIPE;
 684             } else {
 685                 return interpret_ssl_error("SSL_write", ret, ssl_error,
 686                                            &sslv->tx_want);
 687             }
 688         }
 689     }
 690 }

 694 {
 695     struct vconn *vconn = vconn_;
 696     struct ssl_vconn *sslv = ssl_vconn_cast(vconn);
 697     int error = ssl_do_tx(vconn);
 698     if (error != EAGAIN) {
 699         ssl_clear_txbuf(sslv);
 700     } else {
 701         ssl_register_tx_waiter(vconn);

 705 static int
 706 ssl_send(struct vconn *vconn, struct ofpbuf *buffer)
 707 {
 708     struct ssl_vconn *sslv = ssl_vconn_cast(vconn);
 709 
 710     if (sslv->txbuf) {
 711         return EAGAIN;
 712     } else {
 713         int error;
 714 
 715         sslv->txbuf = buffer;
 716         error = ssl_do_tx(vconn);
 717         switch (error) {
 718         case 0:
 719             ssl_clear_txbuf(sslv);
 720             return 0;
 721         case EAGAIN:
 722             leak_checker_claim(buffer);
 723             ssl_register_tx_waiter(vconn);
 724             return 0;
 725         default:
 726             sslv->txbuf = NULL;
 727             return error;
 728         }
 729     }
 730 }

 732 static void
 733 ssl_wait(struct vconn *vconn, enum vconn_wait_type wait)
 734 {
 735     struct ssl_vconn *sslv = ssl_vconn_cast(vconn);
 736 
 737     switch (wait) {
 738     case WAIT_CONNECT:
 739         if (vconn_connect(vconn) != EAGAIN) {
 740             poll_immediate_wake();
 741         } else {
 742             switch (sslv->state) {
 743             case STATE_TCP_CONNECTING:
 744                 poll_fd_wait(sslv->fd, POLLOUT);
 745                 break;
 746 
 747             case STATE_SSL_CONNECTING:
 748                 /* ssl_connect() called SSL_accept() or SSL_connect(), which
 749                  * set up the status that we test here. */
 750                 poll_fd_wait(sslv->fd,
 751                              want_to_poll_events(SSL_want(sslv->ssl)));
 752                 break;
 753 
 754             default:
 755                 NOT_REACHED();
 756             }
 757         }
 758         break;
 759 
 760     case WAIT_RECV:
 761         if (sslv->rx_want != SSL_NOTHING) {
 762             poll_fd_wait(sslv->fd, want_to_poll_events(sslv->rx_want));
 763         } else {
 764             poll_immediate_wake();
 765         }
 766         break;
 767 
 768     case WAIT_SEND:
 769         if (!sslv->txbuf) {
 770             /* We have room in our tx queue. */
 771             poll_immediate_wake();
 772         } else {
 773             /* The call to ssl_tx_poll_callback() will wake us up. */
 774         }
 775         break;
 776 
 777     default:
 778         NOT_REACHED();
 779     }
 780 }

 809 static int
 810 pssl_open(const char *name, char *suffix, struct pvconn **pvconnp)
 811 {
 812     struct sockaddr_in sin;
 813     struct pssl_pvconn *pssl;
 814     int retval;
 815     int fd;
 816     unsigned int yes = 1;
 817 
 818     retval = ssl_init();
 819     if (retval) {
 820         return retval;
 821     }
 822 
 823     /* Create socket. */
 824     fd = socket(AF_INET, SOCK_STREAM, 0);
 825     if (fd < 0) {
 826         int error = errno;
 827         VLOG_ERR("%s: socket: %s", name, strerror(error));
 828         return error;
 829     }
 830 
 831     if (setsockopt(fd, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof yes) < 0) {
 832         int error = errno;
 833         VLOG_ERR("%s: setsockopt(SO_REUSEADDR): %s", name, strerror(errno));
 834         return error;
 835     }
 836 
 837     memset(&sin, 0, sizeof sin);
 838     sin.sin_family = AF_INET;
 839     sin.sin_addr.s_addr = htonl(INADDR_ANY);
 840     sin.sin_port = htons(atoi(suffix) ? atoi(suffix) : OFP_SSL_PORT);
 841     retval = bind(fd, (struct sockaddr *) &sin, sizeof sin);
 842     if (retval < 0) {
 843         int error = errno;
 844         VLOG_ERR("%s: bind: %s", name, strerror(error));
 845         close(fd);
 846         return error;
 847     }
 848 
 849     retval = listen(fd, 10);
 850     if (retval < 0) {
 851         int error = errno;
 852         VLOG_ERR("%s: listen: %s", name, strerror(error));
 853         close(fd);
 854         return error;
 855     }
 856 
 857     retval = set_nonblocking(fd);
 858     if (retval) {
 859         close(fd);
 860         return retval;
 861     }
 862 
 863     pssl = xmalloc(sizeof *pssl);
 864     pvconn_init(&pssl->pvconn, &pssl_pvconn_class, name);
 865     pssl->fd = fd;
 866     *pvconnp = &pssl->pvconn;
 867     return 0;
 868 }

 978 static DH *
 979 tmp_dh_callback(SSL *ssl UNUSED, int is_export UNUSED, int keylength)
 980 {
 981     struct dh {
 982         int keylength;
 983         DH *dh;
 984         DH *(*constructor)(void);
 985     };
 986 
 987     static struct dh dh_table[] = {
 988         {1024, NULL, get_dh1024},
 989         {2048, NULL, get_dh2048},
 990         {4096, NULL, get_dh4096},
 991     };
 992 
 993     struct dh *dh;
 994 
 995     for (dh = dh_table; dh < &dh_table[ARRAY_SIZE(dh_table)]; dh++) {
 996         if (dh->keylength == keylength) {
 997             if (!dh->dh) {
 998                 dh->dh = dh->constructor();
 999                 if (!dh->dh) {
1000                     ofp_fatal(ENOMEM, "out of memory constructing "
1001                               "Diffie-Hellman parameters");
1002                 }
1003             }
1004             return dh->dh;
1005         }
1006     }
1007     VLOG_ERR_RL(&rl, "no Diffie-Hellman parameters for key length %d",
1008                 keylength);
1009     return NULL;
1010 }

1047 /* Reads the X509 certificate or certificates in file 'file_name'.  On success,
1048  * stores the address of the first element in an array of pointers to
1049  * certificates in '*certs' and the number of certificates in the array in
1050  * '*n_certs', and returns 0.  On failure, stores a null pointer in '*certs', 0
1051  * in '*n_certs', and returns a positive errno value.
1052  *
1053  * The caller is responsible for freeing '*certs'. */
1054 static int
1055 read_cert_file(const char *file_name, X509 ***certs, size_t *n_certs)
1056 {
1057     FILE *file;
1058     size_t allocated_certs = 0;
1059 
1060     *certs = NULL;
1061     *n_certs = 0;
1062 
1063     file = fopen(file_name, "r");
1064     if (!file) {
1065         VLOG_ERR("failed to open %s for reading: %s",
1066                  file_name, strerror(errno));
1067         return errno;
1068     }
1069 
1070     for (;;) {
1071         X509 *certificate;
1072         int c;
1073 
1074         /* Read certificate from file. */
1075         certificate = PEM_read_X509(file, NULL, NULL, NULL);
1076         if (!certificate) {
1077             size_t i;
1078 
1079             VLOG_ERR("PEM_read_X509 failed reading %s: %s",
1080                      file_name, ERR_error_string(ERR_get_error(), NULL));
1081             for (i = 0; i < *n_certs; i++) {
1082                 X509_free((*certs)[i]);
1083             }
1084             free(*certs);
1085             *certs = NULL;
1086             *n_certs = 0;
1087             return EIO;
1088         }
1089 
1090         /* Add certificate to array. */
1091         if (*n_certs >= allocated_certs) {
1092             *certs = x2nrealloc(*certs, &allocated_certs, sizeof **certs);
1093         }
1094         (*certs)[(*n_certs)++] = certificate;
1095 
1096         /* Are there additional certificates in the file? */
1097         do {
1098             c = getc(file);
1099         } while (isspace(c));
1100         if (c == EOF) {
1101             break;
1102         }
1103         ungetc(c, file);
1104     }
1105     fclose(file);
1106     return 0;
1107 }

1136 /* Logs fingerprint of CA certificate 'cert' obtained from 'file_name'. */
1137 static void
1138 log_ca_cert(const char *file_name, X509 *cert)
1139 {
1140     unsigned char digest[EVP_MAX_MD_SIZE];
1141     unsigned int n_bytes;
1142     struct ds fp;
1143     char *subject;
1144 
1145     ds_init(&fp);
1146     if (!X509_digest(cert, EVP_sha1(), digest, &n_bytes)) {
1147         ds_put_cstr(&fp, "<out of memory>");
1148     } else {
1149         unsigned int i;
1150         for (i = 0; i < n_bytes; i++) {
1151             if (i) {
1152                 ds_put_char(&fp, ':');
1153             }
1154             ds_put_format(&fp, "%02hhx", digest[i]);
1155         }
1156     }
1157     subject = X509_NAME_oneline(X509_get_subject_name(cert), NULL, 0);
1158     VLOG_INFO("Trusting CA cert from %s (%s) (fingerprint %s)", file_name,
1159               subject ? subject : "<out of memory>", ds_cstr(&fp));
1160     free(subject);
1161     ds_destroy(&fp);
1162 }

1164 /* Sets 'file_name' as the name of the file from which to read the CA
1165  * certificate used to verify the peer within SSL connections.  If 'bootstrap'
1166  * is false, the file must exist.  If 'bootstrap' is false, then the file is
1167  * read if it is exists; if it does not, then it will be created from the CA
1168  * certificate received from the peer on the first SSL connection. */
1169 void
1170 vconn_ssl_set_ca_cert_file(const char *file_name, bool bootstrap)
1171 {
1172     X509 **certs;
1173     size_t n_certs;
1174     struct stat s;
1175 
1176     if (ssl_init()) {
1177         return;
1178     }
1179 
1180     if (bootstrap && stat(file_name, &s) && errno == ENOENT) {
1181         bootstrap_ca_cert = true;
1182         ca_cert_file = xstrdup(file_name);
1183     } else if (!read_cert_file(file_name, &certs, &n_certs)) {
1184         size_t i;
1185 
1186         /* Set up list of CAs that the server will accept from the client. */
1187         for (i = 0; i < n_certs; i++) {
1188             /* SSL_CTX_add_client_CA makes a copy of the relevant data. */
1189             if (SSL_CTX_add_client_CA(ctx, certs[i]) != 1) {
1190                 VLOG_ERR("failed to add client certificate %d from %s: %s",
1191                          i, file_name,
1192                          ERR_error_string(ERR_get_error(), NULL));
1193             } else {
1194                 log_ca_cert(file_name, certs[i]);
1195             }
1196             X509_free(certs[i]);
1197         }
1198 
1199         /* Set up CAs for OpenSSL to trust in verifying the peer's
1200          * certificate. */
1201         if (SSL_CTX_load_verify_locations(ctx, file_name, NULL) != 1) {
1202             VLOG_ERR("SSL_CTX_load_verify_locations: %s",
1203                      ERR_error_string(ERR_get_error(), NULL));
1204             return;
1205         }
1206 
1207         has_ca_cert = true;
1208     }
