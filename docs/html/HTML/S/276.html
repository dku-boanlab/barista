  90 static struct sw_flow *
  91 nf2_lookup_flowtable(struct sw_table *flowtab, const struct sw_flow_key *key)
  92 {
  93         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
  94         struct sw_flow *flow;
  95 
  96         list_for_each_entry(flow, &nf2flowtab->flows, node) {
  97                 if (flow_matches_1wild(key, &flow->key)) {
  98                         return flow;
  99                 }
 100         }
 101 
 102         return NULL;
 103 }

 105 static int
 106 nf2_install_flow(struct sw_table *flowtab, struct sw_flow *flow)
 107 {
 108         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 109 
 110         /* Delete flows that match exactly. */
 111         nf2_uninstall_flow(NULL, flowtab, &flow->key, OFPP_NONE,
 112                            flow->priority, true);
 113 
 114         if (nf2_are_actions_supported(flow)) {
 115                 if (nf2_build_and_write_flow(flow)) {
 116                         return 0;
 117                 }
 118         } else {
 119                 /* Unsupported actions or no netdevice. */
 120                 return 0;
 121         }
 122 
 123         atomic_inc(&nf2flowtab->num_flows);
 124         list_add_rcu(&flow->node, &nf2flowtab->flows);
 125         list_add_rcu(&flow->iter_node, &nf2flowtab->iter_flows);
 126         return 1;
 127 }

 129 static int
 130 nf2_modify_flow(struct sw_table *flowtab, const struct sw_flow_key *key,
 131                 uint16_t priority, int strict,
 132                 const struct ofp_action_header *actions, size_t actions_len)
 133 {
 134         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 135         struct sw_flow *flow;
 136         unsigned int count = 0;
 137 
 138         list_for_each_entry(flow, &nf2flowtab->flows, node) {
 139                 if (flow_matches_desc(&flow->key, key, strict)
 140                     && (!strict || flow->priority == priority)) {
 141                         flow_replace_acts(flow, actions, actions_len);
 142                         if (nf2_are_actions_supported(flow)) {
 143                                 count += nf2_modify_acts(flowtab, flow);
 144                         }
 145                 } else {
 146                         return 0;
 147                 }
 148         }
 149 
 150         return count;
 151 }

 153 static int
 154 nf2_has_conflict(struct sw_table *flowtab, const struct sw_flow_key *key,
 155                  uint16_t priority, int strict)
 156 {
 157         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 158         struct sw_flow *flow;
 159 
 160         list_for_each_entry(flow, &nf2flowtab->flows, node) {
 161                 if (flow_matches_desc(&flow->key, key, strict)
 162                     && (flow->priority == priority)) {
 163                         return true;
 164                 }
 165         }
 166         return false;
 167 }

 183 static int
 184 do_uninstall(struct datapath *dpinst, struct sw_table *flowtab,
 185              struct sw_flow *flow, enum ofp_flow_removed_reason reason)
 186 {
 187         if (flow != NULL && flow->private != NULL) {
 188                 if (dpinst != NULL)
 189                         dp_send_flow_end(dpinst, flow, reason);
 190                 list_del_rcu(&flow->node);
 191                 list_del_rcu(&flow->iter_node);
 192                 nf2_delete_private(flow->private);
 193                 do_deferred_uninstall(flow);
 194                 return 1;
 195         }
 196 
 197         return 0;
 198 }

 200 static int
 201 nf2_uninstall_flow(struct datapath *dpinst, struct sw_table *flowtab,
 202                    const struct sw_flow_key *key, uint16_t out_port,
 203                    uint16_t priority, int strict)
 204 {
 205         struct net_device *netdev;
 206         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 207         struct sw_flow *flow;
 208         struct nf2_flow *nf2flow;
 209         unsigned int count = 0;
 210 
 211         netdev = nf2_get_net_device();
 212         if (netdev == NULL)
 213                 return 0;
 214 
 215         list_for_each_entry(flow, &nf2flowtab->flows, node) {
 216                 if (flow_matches_desc(&flow->key, key, strict)
 217                     && (!strict || flow->priority == priority)
 218                     && flow_has_out_port(flow, out_port)) {
 219                         nf2flow = flow->private;
 220                         if (nf2flow != NULL) {
 221                                 flow->packet_count
 222                                         += nf2_get_packet_count(netdev,
 223                                                                 nf2flow);
 224                                 flow->byte_count += nf2_get_byte_count(netdev,
 225                                                                        nf2flow);
 226                         }
 227                         count += do_uninstall(dpinst, flowtab,
 228                                               flow, OFPRR_DELETE);
 229                 }
 230         }
 231         if (count != 0)
 232                 atomic_sub(count, &nf2flowtab->num_flows);
 233 
 234         nf2_free_net_device(netdev);
 235 
 236         return count;
 237 }

 239 static int
 240 nf2_flow_timeout(struct datapath *dpinst, struct sw_table *flowtab)
 241 {
 242         struct net_device *netdev;
 243         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 244         struct sw_flow *flow;
 245         struct nf2_flow *nf2flow;
 246         int num_uninst_flows = 0;
 247         uint64_t num_forw_packets = 0;
 248         int reason;
 249 
 250         netdev = nf2_get_net_device();
 251         if (netdev == NULL)
 252                 return num_uninst_flows;
 253 
 254         mutex_lock(&dp_mutex);
 255         list_for_each_entry(flow, &nf2flowtab->flows, node) {
 256                 nf2flow = flow->private;
 257                 if (nf2flow != NULL) {
 258                         num_forw_packets = flow->packet_count
 259                                 + nf2_get_packet_count(netdev, nf2flow);
 260                         flow->byte_count += nf2_get_byte_count(netdev, nf2flow);
 261                 }
 262                 if (num_forw_packets > flow->packet_count
 263                     && flow->idle_timeout != OFP_FLOW_PERMANENT) {
 264                         flow->packet_count = num_forw_packets;
 265                         flow->used = get_jiffies_64();
 266                 }
 267                 reason = flow_timeout(flow);
 268                 if (reason >= 0) {
 269                         num_uninst_flows += do_uninstall(dpinst, flowtab,
 270                                                          flow, reason);
 271                 }
 272         }
 273         mutex_unlock(&dp_mutex);
 274 
 275         nf2_clear_watchdog(netdev);
 276 
 277         nf2_free_net_device(netdev);
 278 
 279         if (num_uninst_flows != 0)
 280                 atomic_sub(num_uninst_flows, &nf2flowtab->num_flows);
 281         return num_uninst_flows;
 282 }

 284 static void
 285 nf2_destroy_flowtable(struct sw_table *flowtab)
 286 {
 287         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 288         struct nf2_flow *nf2flow = NULL;
 289 
 290         if (nf2flowtab == NULL)
 291                 return;
 292 
 293         while (!list_empty(&nf2flowtab->flows)) {
 294                 struct sw_flow *flow = list_entry(nf2flowtab->flows.next,
 295                                                   struct sw_flow, node);
 296 
 297                 list_del(&flow->node);
 298                 if (flow->private) {
 299                         nf2flow = (struct nf2_flow *)flow->private;
 300 
 301                         if (nf2flow->type == NF2_TABLE_EXACT) {
 302                                 nf2_add_free_exact(nf2flow);
 303                         } else if (nf2flow->type == NF2_TABLE_WILDCARD) {
 304                                 nf2_add_free_wildcard(nf2flow);
 305                         }
 306                         flow->private = NULL;
 307                 }
 308                 flow_free(flow);
 309         }
 310         kfree(nf2flowtab);
 311 
 312         nf2_destroy_exact_freelist();
 313         nf2_destroy_wildcard_freelist();
 314 }

 316 static int
 317 nf2_iterate_flowtable(struct sw_table *flowtab, const struct sw_flow_key *key,
 318                       uint16_t out_port, struct sw_table_position *position,
 319                       int (*callback) (struct sw_flow *, void *), void *private)
 320 {
 321         unsigned long start;
 322         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 323         struct sw_flow *flow;
 324         int error = 0;
 325 
 326         start = ~position->private[0];
 327         list_for_each_entry(flow, &nf2flowtab->iter_flows, iter_node) {
 328                 if (flow->serial <= start && flow_matches_2wild(key, &flow->key)
 329                     && flow_has_out_port(flow, out_port)) {
 330                         error = callback(flow, private);
 331                         if (error != 0) {
 332                                 position->private[0] = ~flow->serial;
 333                                 return error;
 334                         }
 335                 }
 336         }
 337 
 338         return error;
 339 }

 341 unsigned long int
 342 get_lookup_stats(void)
 343 {
 344         struct net_device *netdev;
 345         unsigned long int num_searched = 0;
 346 
 347         netdev = nf2_get_net_device();
 348         if (netdev == NULL)
 349                 return num_searched;
 350 
 351         num_searched = nf2_get_missed_count(netdev);
 352         nf2_free_net_device(netdev);
 353         num_searched += get_lookup_matched_stats();
 354         return num_searched;
 355 }

 372 static void
 373 nf2_get_flowstats(struct sw_table *flowtab, struct sw_table_stats *stats)
 374 {
 375         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 376 
 377         stats->name = "nf2";
 378         stats->wildcards = OPENFLOW_WILDCARD_TABLE_SIZE - 8;
 379         stats->n_flows = atomic_read(&nf2flowtab->num_flows);

 385 static struct sw_table *
 386 nf2_create_flowtable(void)
 387 {
 388         struct net_device *netdev;
 389         struct nf2_flowtable *nf2flowtab;
 390         struct sw_table *flowtab;
 391 
 392         netdev = nf2_get_net_device();
 393         if (netdev == NULL)
 394                 return NULL;
 395 
 396         nf2_reset_card(netdev);
 397         nf2_free_net_device(netdev);
 398 
 399         nf2flowtab = kzalloc(sizeof(*nf2flowtab), GFP_KERNEL);
 400         if (nf2flowtab == NULL) {
 401                 nf2_free_net_device(netdev);
 402                 return NULL;
 403         }
 404 
 405         flowtab = &nf2flowtab->flowtab;
 406         flowtab->n_lookup = (unsigned long long)get_lookup_stats();
 407         flowtab->n_matched = (unsigned long long)get_lookup_matched_stats();
 408         flowtab->lookup = nf2_lookup_flowtable;
 409         flowtab->insert = nf2_install_flow;
 410         flowtab->modify = nf2_modify_flow;
 411         flowtab->has_conflict = nf2_has_conflict;
 412         flowtab->delete = nf2_uninstall_flow;
 413         flowtab->timeout = nf2_flow_timeout;
 414         flowtab->destroy = nf2_destroy_flowtable;
 415         flowtab->iterate = nf2_iterate_flowtable;
 416         flowtab->stats = nf2_get_flowstats;
 417 #define RESERVED_FOR_CPU2NETFPGA        8
 418         nf2flowtab->max_flows = OPENFLOW_NF2_EXACT_TABLE_SIZE
 419                 + OPENFLOW_WILDCARD_TABLE_SIZE - RESERVED_FOR_CPU2NETFPGA;
 420         atomic_set(&nf2flowtab->num_flows, 0);
 421         INIT_LIST_HEAD(&nf2flowtab->flows);
 422         INIT_LIST_HEAD(&nf2flowtab->iter_flows);
 423         nf2flowtab->next_serial = 0;
 424 
 425         nf2_init_wildcard_freelist();
 426         nf2_write_static_wildcard();
 427         nf2_init_exact_freelist();
 428 
 429         return flowtab;
 430 }
