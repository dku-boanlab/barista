  14 struct tlv_header {
  15     union {
  16         struct {
  17             uint16_t length:9;
  18             uint16_t type:7; /* One of the LLDP_TLVT_ */
  19         };
  20         uint16_t raw;
  21     };
  22     uint8_t data[0];         /* 0 ~ 511 bytes */
  23 };
