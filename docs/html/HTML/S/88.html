 135 int handle_timer_event(struct oflops_context * ctx, struct timer_event *te)
 136 {
 137   struct timeval now;
 138   char * str;
 139 
 140   gettimeofday(&now,NULL);
 141   str = (char *) te->arg;
 142  
 143   if(!strcmp(str,SNMPGET)) {
 144     oflops_snmp_get(ctx, ctx->cpuOID, ctx->cpuOID_len);
 145     int i;
 146     for(i=0;i<ctx->n_channels;i++) {
 147       oflops_snmp_get(ctx, ctx->channels[i].inOID, ctx->channels[i].inOID_len);
 148       oflops_snmp_get(ctx, ctx->channels[i].outOID, ctx->channels[i].outOID_len);
 149     }  
 150     gettimeofday(&now, NULL);
 151     add_time(&now, 1, 0);
 152     oflops_schedule_timer_event(ctx,&now, SNMPGET);
 153   } else if(!strcmp(str,BYESTR)) {
 154     oflops_end_test(ctx,1);
 155   } else
 156     fprintf(stderr, "Unknown timer event: %s", str);
 157   return 0;
 158 }

 160 int 
 161 destroy(oflops_context *ctx) {
 162   struct entry *np;
 163   long long t = 0, t_sq = 0, mean, std;
 164   int count = 0, min_id =  INT_MAX, max_id =  INT_MIN, delay;
 165   float loss;
 166   char msg[1024];
 167   struct timeval now;
 168   gettimeofday(&now, NULL);
 169   
 170   for (np = head.tqh_first; np != NULL; np = np->entries.tqe_next) {
 171     count++; 
 172     min_id = (np->id < min_id)?np->id:min_id;
 173     max_id = (np->id > max_id)?np->id:max_id;
 174     delay = time_diff(&np->snd, &np->rcv);
 175     t += delay;
 176     t_sq += delay*delay;
 177     free(np);
 178   }
 179   if(count > 0) {
 180     mean = t/count;
 181     std = (t_sq/count) - mean*mean;
 182     printf("std:%f\n", std);
 183     std=(std >= 0)?(long)sqrt((double)std):LONG_MAX;
 184     loss = (float)count/(float)(max_id - min_id);
 185     snprintf(msg, 1024, "statistics:port:%lld:%lld:%.4f:%d", mean, std, loss, count);
 186     oflops_log(now, GENERIC_MSG, msg);
 187   }
 188   return 0;
 189 }

 205 /** Handle pcap event.
 206  * @param ctx pointer to opaque context
 207  * @param pe pcap event
 208  * @param ch enumeration of channel that pcap event is triggered
 209  */
 210 int handle_pcap_event(struct oflops_context *ctx, struct pcap_event * pe, oflops_channel_name ch) {
 211   struct pktgen_hdr *pktgen;
 212   int dir, len;
 213   struct ofp_header *ofp;
 214   struct pcap_event *ofp_msg;
 215   struct ofp_packet_in *pkt_in = NULL;
 216   char msg[1024];
 217   struct flow fl;
 218   struct in_addr addr;
 219   if (ch == OFLOPS_CONTROL) {
 220     dir = append_data_to_flow(pe->data,pe->pcaphdr);
 221     while(contains_next_msg(dir) > 0) {
 222       len = get_next_msg(dir, &ofp_msg);
 223       ofp = (struct ofp_header *)ofp_msg->data;
 224       if(ofp->type ==  OFPT_PACKET_IN) {
 225         pkt_in = (struct ofp_error_msg *)ofp;
 226         pktgen = extract_pktgen_pkt(pkt_in->data, pe->pcaphdr.caplen - sizeof(struct ofp_packet_in), &fl);
 227         if(pktgen == NULL) 
 228           return 0;
 229         addr.s_addr = fl.nw_dst;
 230 
 231         struct entry *n1 = malloc(sizeof(struct entry));
 232         n1->snd.tv_sec = htonl(pktgen->tv_sec);
 233         n1->snd.tv_usec = htonl(pktgen->tv_usec);
 234         memcpy(&n1->rcv, &pe->pcaphdr.ts, sizeof(struct timeval));
 235         n1->id = htonl(pktgen->seq_num);
 236         n1->ch = ch;
 237         TAILQ_INSERT_TAIL(&head, n1, entries);
 238 
 239         snprintf(msg, 1024, 
 240                  "%llu.%llu:%d:%s:%d:%d", 
 241                  (unsigned long long)ntohl(pktgen->tv_sec),
 242                  (unsigned long long)ntohl(pktgen->tv_usec),
 243                  ntohl(pkt_in->buffer_id),
 244                  inet_ntoa(addr),
 245                  ntohl(pktgen->seq_num),
 246                  ntohs(ofp->length));
 247         //printf("%s\n", msg);
 248         oflops_log(pe->pcaphdr.ts, OFPT_PACKET_IN_MSG, msg);
 249         //fprintf(stderr, "%s\n", msg);
 250       }
 251     }
 252   }
 253   else
 254     fprintf(stderr, "wtf! why channel %u?", ch);
 255 
 256   return 0;
 257 }

 259 int 
 260 handle_snmp_event(struct oflops_context * ctx, struct snmp_event * se) {
 261   netsnmp_variable_list *vars;
 262   int len = 1024, i;
 263   char msg[1024], log[1024];
 264   struct timeval now;
 265 
 266   for(vars = se->pdu->variables; vars; vars = vars->next_variable)  {
 267     snprint_value(msg, len, vars->name, vars->name_length, vars);
 268     if((vars->name_length == ctx->cpuOID_len) &&
 269        (memcmp(vars->name, ctx->cpuOID,  ctx->cpuOID_len * sizeof(oid)) == 0) ) {
 270       snprintf(log, len, "cpu : %s %%", msg);
 271       oflops_log(now, SNMP_MSG, log);
 272     } else {
 273       for(i=0;i<ctx->n_channels;i++) {
 274         if((vars->name_length == ctx->channels[i].inOID_len) &&
 275            (memcmp(vars->name, ctx->channels[i].inOID,  
 276                    ctx->channels[i].inOID_len * sizeof(oid)) == 0) ) {
 277           snprintf(log, len, "port %d : rx %s pkts",  
 278                    (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 279           oflops_log(now, SNMP_MSG, log);
 280           break;
 281         }
 282         
 283         if((vars->name_length == ctx->channels[i].outOID_len) &&
 284            (memcmp(vars->name, ctx->channels[i].outOID,  
 285                    ctx->channels[i].outOID_len * sizeof(oid))==0) ) {
 286           snprintf(log, len, "port %d : tx %s pkts",  
 287                    (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 288           oflops_log(now, SNMP_MSG, log);
 289           break;
 290         }
 291       } //for
 292     }// if cpu
 293   }
 294   return 0;
 295 }

 325 /**
 326  * Initialization code with parameters
 327  * @param ctx 
 328  */
 329 int init(struct oflops_context *ctx, char * config_str) {
 330   char *pos = NULL;
 331   char *param = config_str;
 332   char *value = NULL;
 333   struct timeval now;
 334 
 335   //init counters
 336   finished = 0;
 337   gettimeofday(&now, NULL);
 338   cli_param = strdup(config_str);
 339 
 340   while(*config_str == ' ') {
 341     config_str++;
 342   }
 343   param = config_str;
 344   while(1) {
 345     pos = index(param, ' ');
 346 
 347     if((pos == NULL)) {
 348       if (*param != '\0') {
 349         pos = param + strlen(param) + 1;
 350       } else
 351         break;
 352     }
 353     *pos='\0';
 354     pos++;
 355     value = index(param,'=');
 356     *value = '\0';
 357     value++;
 358     //fprintf(stderr, "param = %s, value = %s\n", param, value);
 359     if(value != NULL) {
 360       if(strcmp(param, "pkt_size") == 0) {
 361         //parse int to get pkt size
 362         pkt_size = strtol(value, NULL, 0);
 363         if((pkt_size < MIN_PKT_SIZE) && (pkt_size > MAX_PKT_SIZE))
 364           perror_and_exit("Invalid packet size value", 1);
 365       }  else if(strcmp(param, "probe_rate") == 0) {
 366         //parse int to get measurement probe rate
 367         proberate = strtol(value, NULL, 0);
 368         if((proberate <= 0) || (proberate >= 1010)) 
 369           perror_and_exit("Invalid probe rate param(Value between 1 and 1010)", 1);
 370       } else if(strcmp(param, "flows") == 0) {
 371         //parse int to get pkt size
 372         flows = strtol(value, NULL, 0);
 373         if(flows <= 0)  
 374           perror_and_exit("Invalid flow number", 1);
 375       } else 
 376         fprintf(stderr, "Invalid parameter:%s\n", param);
 377       param = pos;
 378     }
 379   } 
 380 
 381   //calculate sendind interval
 382   probe_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (proberate * mbits_to_bits);
 383   fprintf(stderr, "Sending probe interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 384           (uint32_t)probe_snd_interval, (uint32_t)pkt_size, (uint32_t)proberate);
 385   return 0;
 386 }
