  25 #ifndef _TAP_VOIP_H_
  26 #define _TAP_VOIP_H_
  27 
  28 /* defines voip call state */
  29 typedef enum _voip_call_state {
  30                 VOIP_NO_STATE,
  31         VOIP_CALL_SETUP,
  32         VOIP_RINGING,
  33         VOIP_IN_CALL,
  34         VOIP_CANCELLED,
  35         VOIP_COMPLETED,
  36         VOIP_REJECTED,
  37         VOIP_UNKNOWN
  38 } voip_call_state;
  39 
  40 typedef enum _voip_call_active_state {
  41         VOIP_ACTIVE,
  42         VOIP_INACTIVE
  43 } voip_call_active_state;
  44 
  45 /* structure for common/proprietary VoIP calls TAP */
  46 typedef struct _voip_packet_info_t
  47 {
  48         gchar *protocol_name;
  49         gchar *call_id;
  50         voip_call_state call_state;
  51         voip_call_active_state call_active_state;
  52         gchar *from_identity;
  53         gchar *to_identity;
  54         gchar *call_comment;
  55         gchar *frame_label;
  56         gchar *frame_comment;
  57 } voip_packet_info_t;
  58 
  59 #endif  /* _TAP_VOIP_H_ */
