 127 /**
 128  * \brief Function to summarize the traffic history for the last n seconds
 129  * \param cli The CLI pointer
 130  * \param seconds Time period to query
 131  */
 132 static int traffic_stat_summary(cli_t *cli, char *seconds)
 133 {
 134     int sec = atoi(seconds);
 135     if (sec > TRAFFIC_MGMT_HISTORY) sec = TRAFFIC_MGMT_HISTORY;
 136     else if (tr_history_ptr < TRAFFIC_MGMT_HISTORY && tr_history_ptr < sec) sec = tr_history_ptr;
 137 
 138     int ptr = tr_history_ptr - sec;
 139     if (ptr < 0) ptr = TRAFFIC_MGMT_HISTORY + ptr;
 140 
 141     traffic_t tr = {0};
 142 
 143     int i;
 144     for (i=0; i<sec; i++) {
 145         tr.in_pkt_cnt += tr_history[ptr % TRAFFIC_MGMT_HISTORY].in_pkt_cnt;
 146         tr.in_byte_cnt += tr_history[ptr % TRAFFIC_MGMT_HISTORY].in_byte_cnt;
 147 
 148         tr.out_pkt_cnt += tr_history[ptr % TRAFFIC_MGMT_HISTORY].out_pkt_cnt;
 149         tr.out_byte_cnt += tr_history[ptr % TRAFFIC_MGMT_HISTORY].out_byte_cnt;
 150 
 151         ptr++;
 152     }
 153 
 154     double in_pkt_cnt = tr.in_pkt_cnt * 1.0 / sec;
 155     double in_byte_cnt = tr.in_byte_cnt * 1.0 / sec;
 156 
 157     double out_pkt_cnt = tr.out_pkt_cnt * 1.0 / sec;
 158     double out_byte_cnt = tr.out_byte_cnt * 1.0 / sec;
 159 
 160     cli_print(cli, "<Traffic Statistics for %d seconds>", sec);
 161     cli_print(cli, "  - Cumulative values");
 162     cli_print(cli, "    Inbound packet count : %lu packets", tr.in_pkt_cnt);
 163     cli_print(cli, "    Inbound byte count   : %lu bytes", tr.in_byte_cnt);
 164     cli_print(cli, "    Outbound packet count: %lu packets", tr.out_pkt_cnt);
 165     cli_print(cli, "    Outbound byte count  : %lu bytes", tr.out_byte_cnt);
 166     cli_print(cli, "  - Average values");
 167     cli_print(cli, "    Inbound packet count : %.2f packets", in_pkt_cnt);
 168     cli_print(cli, "    Inbound byte count   : %.2f bytes", in_byte_cnt);
 169     cli_print(cli, "    Outbound packet count: %.2f packets", out_pkt_cnt);
 170     cli_print(cli, "    Outbound byte count  : %.2f bytes", out_byte_cnt);
 171 
 172     return 0;
 173 }

 193 /**
 194  * \brief The handler function
 195  * \param ev Read-only event
 196  * \param ev_out Read-write event (if this component has the write permission)
 197  */
 198 int traffic_mgmt_handler(const event_t *ev, event_out_t *ev_out)
 199 {
 200     switch (ev->type) {
 201     case EV_OFP_MSG_IN:
 202         PRINT_EV("EV_OFP_MSG_IN\n");
 203         {
 204             pthread_spin_lock(&lock);
 205 
 206             traffic.in_pkt_cnt++;
 207             traffic.in_byte_cnt += ev->msg->length;
 208 
 209             pthread_spin_unlock(&lock);
 210         }
 211         break;
 212     case EV_OFP_MSG_OUT:
 213         PRINT_EV("EV_OFP_MSG_OUT\n");
 214         {
 215             pthread_spin_lock(&lock);
 216 
 217             traffic.out_pkt_cnt++;
 218             traffic.out_byte_cnt += ev->msg->length;
 219 
 220             pthread_spin_unlock(&lock);
 221         }
 222         break;
 223     default:
 224         break;
 225     }
 226 
 227     return 0;
 228 }
