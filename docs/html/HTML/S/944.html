  25 /**
  26  * \brief Function to find configuration files
  27  * \param temp The paths of the files that will be discovered
  28  * \param cnt The number of the files that will be discovered
  29  * \param curr The current directory
  30  */
  31 int get_external_configurations(char (*temp)[__CONF_WORD_LEN], int *cnt, char *curr)
  32 {
  33     DIR *dir = NULL;
  34     struct dirent *entry = NULL;
  35 
  36     if (!(dir = opendir(curr))) {
  37         return -1;
  38     } else if (!(entry = readdir(dir))) {
  39         closedir(dir);
  40         return -1;
  41     }
  42 
  43     do {
  44         char path[__CONF_WORD_LEN] = {0};
  45         int len = snprintf(path, sizeof(path)-1, "%s/%s", curr, entry->d_name);
  46         path[len] = '\0';
  47 
  48         if (entry->d_type == DT_DIR) {
  49             if(strncmp(entry->d_name, ".", 1) == 0 || strncmp(entry->d_name, "..", 2) == 0)
  50                 continue;
  51             get_external_configurations(temp, cnt, path);
  52         } else {
  53             if (strstr(entry->d_name, ".conf") != NULL) {
  54                 strcpy(temp[(*cnt)], path);
  55                 (*cnt)++;
  56             }
  57         }
  58     } while ((entry = readdir(dir)) != NULL);
  59 
  60     closedir(dir);
  61 
  62     return 0;
  63 }
