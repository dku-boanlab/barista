  30 static struct sw_flow **find_bucket(struct sw_table *swt,
  31                                                                         const struct sw_flow_key *key)
  32 {
  33         struct sw_table_hash *th = (struct sw_table_hash *) swt;
  34         unsigned int crc = crc32_calculate(&th->crc32, key, 
  35                                 offsetof(struct sw_flow_key, wildcards));
  36         return &th->buckets[crc & th->bucket_mask];
  37 }

  39 static struct sw_flow *table_hash_lookup(struct sw_table *swt,
  40                                                                                  const struct sw_flow_key *key)
  41 {
  42         struct sw_flow *flow = *find_bucket(swt, key);
  43         return flow && flow_keys_equal(&flow->key, key) ? flow : NULL;

  46 static int table_hash_insert(struct sw_table *swt, struct sw_flow *flow)
  47 {
  48         struct sw_table_hash *th = (struct sw_table_hash *) swt;
  49         struct sw_flow **bucket;
  50         int retval;
  51 
  52         if (flow->key.wildcards != 0)
  53                 return 0;
  54 
  55         bucket = find_bucket(swt, &flow->key);
  56         if (*bucket == NULL) {
  57                 th->n_flows++;
  58                 rcu_assign_pointer(*bucket, flow);
  59                 retval = 1;
  60         } else {
  61                 struct sw_flow *old_flow = *bucket;
  62                 if (flow_keys_equal(&old_flow->key, &flow->key)) {
  63                         rcu_assign_pointer(*bucket, flow);
  64                         flow_deferred_free(old_flow);
  65                         retval = 1;
  66                 } else {
  67                         retval = 0;
  68                 }
  69         }
  70         return retval;
  71 }

  73 static int table_hash_modify(struct sw_table *swt, 
  74                 const struct sw_flow_key *key, uint16_t priority, int strict,
  75                 const struct ofp_action_header *actions, size_t actions_len) 
  76 {
  77         struct sw_table_hash *th = (struct sw_table_hash *) swt;
  78         unsigned int count = 0;
  79 
  80         if (key->wildcards == 0) {
  81                 struct sw_flow **bucket = find_bucket(swt, key);
  82                 struct sw_flow *flow = *bucket;
  83                 if (flow && flow_matches_desc(&flow->key, key, strict)
  84                                 && (!strict || (flow->priority == priority))) {
  85                         flow_replace_acts(flow, actions, actions_len);
  86                         count = 1;
  87                 }
  88         } else {
  89                 unsigned int i;
  90 
  91                 for (i = 0; i <= th->bucket_mask; i++) {
  92                         struct sw_flow **bucket = &th->buckets[i];
  93                         struct sw_flow *flow = *bucket;
  94                         if (flow && flow_matches_desc(&flow->key, key, strict)
  95                                         && (!strict || (flow->priority == priority))) {
  96                                 flow_replace_acts(flow, actions, actions_len);
  97                                 count++;
  98                         }
  99                 }
 100         }
 101         return count;
 102 }

 133 static int do_delete(struct datapath *dp, struct sw_flow **bucket, 
 134                         struct sw_flow *flow, enum ofp_flow_removed_reason reason)
 135 {
 136         dp_send_flow_end(dp, flow, reason);

 142 /* Returns number of deleted flows.  We can ignore the priority
 143  * argument, since all exact-match entries are the same (highest)
 144  * priority. */
 145 static int table_hash_delete(struct datapath *dp, struct sw_table *swt,
 146                                         const struct sw_flow_key *key,  uint16_t out_port, 
 147                                         uint16_t priority, int strict)
 148 {
 149         struct sw_table_hash *th = (struct sw_table_hash *) swt;
 150         unsigned int count = 0;
 151 
 152         if (key->wildcards == 0) {
 153                 struct sw_flow **bucket = find_bucket(swt, key);
 154                 struct sw_flow *flow = *bucket;
 155                 if (flow && flow_keys_equal(&flow->key, key)
 156                                 && flow_has_out_port(flow, out_port)) 
 157                         count = do_delete(dp, bucket, flow, OFPRR_DELETE);
 158         } else {
 159                 unsigned int i;
 160 
 161                 for (i = 0; i <= th->bucket_mask; i++) {
 162                         struct sw_flow **bucket = &th->buckets[i];
 163                         struct sw_flow *flow = *bucket;
 164                         if (flow && flow_matches_desc(&flow->key, key, strict)
 165                                         && flow_has_out_port(flow, out_port))
 166                                 count += do_delete(dp, bucket, flow, OFPRR_DELETE);
 167                 }
 168         }
 169         th->n_flows -= count;
 170         return count;
 171 }

 173 static int table_hash_timeout(struct datapath *dp, struct sw_table *swt)
 174 {
 175         struct sw_table_hash *th = (struct sw_table_hash *) swt;
 176         unsigned int i;
 177         int count = 0;
 178 
 179         if (mutex_lock_interruptible(&dp_mutex))
 180                 return 0;
 181         for (i = 0; i <= th->bucket_mask; i++) {
 182                 struct sw_flow **bucket = &th->buckets[i];
 183                 struct sw_flow *flow = *bucket;
 184                 if (flow) {
 185                         int reason = flow_timeout(flow);
 186                         if (reason >= 0) {
 187                                 count += do_delete(dp, bucket, flow, reason); 
 188                         }
 189                 }
 190         }
 191         th->n_flows -= count;
 192         mutex_unlock(&dp_mutex);
 193 
 194         return count;
 195 }

 197 static void table_hash_destroy(struct sw_table *swt)
 198 {
 199         struct sw_table_hash *th = (struct sw_table_hash *) swt;
 200         unsigned int i;
 201         for (i = 0; i <= th->bucket_mask; i++)
 202         if (th->buckets[i])
 203                 flow_free(th->buckets[i]);
 204         kmem_free(th->buckets, (th->bucket_mask + 1) * sizeof *th->buckets);
 205         kfree(th);

 208 static int table_hash_iterate(struct sw_table *swt,
 209                               const struct sw_flow_key *key, uint16_t out_port,
 210                               struct sw_table_position *position,
 211                               int (*callback)(struct sw_flow *, void *private),
 212                               void *private) 
 213 {
 214         struct sw_table_hash *th = (struct sw_table_hash *) swt;
 215 
 216         if (position->private[0] > th->bucket_mask)
 217                 return 0;
 218 
 219         if (key->wildcards == 0) {
 220                 struct sw_flow *flow;
 221                 int error;
 222 
 223                 flow = table_hash_lookup(swt, key);
 224                 if (!flow || !flow_has_out_port(flow, out_port))
 225                         return 0;
 226 
 227                 error = callback(flow, private);
 228                 if (!error)
 229                         position->private[0] = -1;
 230                 return error;
 231         } else {
 232                 int i;
 233 
 234                 for (i = position->private[0]; i <= th->bucket_mask; i++) {
 235                         struct sw_flow *flow = th->buckets[i];
 236                         if (flow && flow_matches_1wild(&flow->key, key)
 237                                         && flow_has_out_port(flow, out_port)) {
 238                                 int error = callback(flow, private);
 239                                 if (error) {
 240                                         position->private[0] = i;
 241                                         return error;
 242                                 }
 243                         }
 244                 }
 245                 return 0;
 246         }
 247 }

 260 struct sw_table *table_hash_create(unsigned int polynomial,
 261                         unsigned int n_buckets)
 262 {
 263         struct sw_table_hash *th;
 264         struct sw_table *swt;
 265 
 266         th = kzalloc(sizeof *th, GFP_KERNEL);
 267         if (th == NULL)
 268                 return NULL;
 269 
 270         BUG_ON(n_buckets & (n_buckets - 1));
 271         th->buckets = kmem_zalloc(n_buckets * sizeof *th->buckets);
 272         if (th->buckets == NULL) {
 273                 printk(KERN_EMERG "failed to allocate %u buckets\n",
 274                        n_buckets);
 275                 kfree(th);
 276                 return NULL;
 277         }
 278         th->bucket_mask = n_buckets - 1;
 279 
 280         swt = &th->swt;
 281         swt->lookup = table_hash_lookup;
 282         swt->insert = table_hash_insert;
 283         swt->has_conflict = table_hash_has_conflict;
 284         swt->delete = table_hash_delete;
 285         swt->timeout = table_hash_timeout;
 286         swt->destroy = table_hash_destroy;
 287         swt->iterate = table_hash_iterate;
 288         swt->stats = table_hash_stats;
 289 
 290         crc32_init(&th->crc32, polynomial);
 291         th->n_flows = 0;
 292 
 293         return swt;
 294 }

 373 static int table_hash2_iterate(struct sw_table *swt,
 374                                const struct sw_flow_key *key, uint16_t out_port,
 375                                struct sw_table_position *position,
 376                                int (*callback)(struct sw_flow *, void *),
 377                                void *private)

 452 static void *
 453 kmem_alloc(size_t size)
 454 {
 455         void *ptr;
 456 
 457 #ifdef KMALLOC_MAX_SIZE
 458         if (size > KMALLOC_MAX_SIZE)
 459                 return NULL;
 460 #endif
 461         ptr = kmalloc(size, GFP_KERNEL);
 462         if (!ptr) {
 463                 ptr = vmalloc(size);
 464                 if (ptr)
 465                         printk(KERN_NOTICE "openflow: used vmalloc for %lu "
 466                                "bytes\n", (unsigned long)size);
 467         }
 468         return ptr;
 469 }
