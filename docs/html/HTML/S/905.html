  43 /* Initializes 'q' as an empty packet queue. */
  44 void
  45 queue_init(struct ofp_queue *q)
  46 {
  47     q->n = 0;
  48     q->head = NULL;
  49     q->tail = NULL;
  50 }
  51 
  52 /* Destroys 'q' and all of the packets that it contains. */
  53 void
  54 queue_destroy(struct ofp_queue *q)
  55 {
  56     struct ofpbuf *cur, *next;
  57     for (cur = q->head; cur != NULL; cur = next) {
  58         next = cur->next;
  59         ofpbuf_delete(cur);
  60     }
  61 }
  62 
  63 /* Removes and destroys all of the packets in 'q', rendering it empty. */
  64 void
  65 queue_clear(struct ofp_queue *q)
  66 {
  67     queue_destroy(q);
  68     queue_init(q);
  69 }

  71 /* Advances the first packet in 'q' from 'q->head' to 'next', which should be
  72  * the second packet in the queue.
  73  *
  74  * The odd, unsafe interface here allows the first packet in the queue to be
  75  * passed to a function for possible consumption (and destruction) and only
  76  * dropped from the queue if that function actually accepts it. */
  77 void
  78 queue_advance_head(struct ofp_queue *q, struct ofpbuf *next)
  79 {
  80     assert(q->n);
  81     assert(q->head);
  82     q->head = next;
  83     if (q->head == NULL) {
  84         q->tail = NULL;
  85     }
  86     q->n--;
  87 }
  88 
  89 /* Appends 'b' to the tail of 'q'. */
  90 void
  91 queue_push_tail(struct ofp_queue *q, struct ofpbuf *b)
  92 {
  93     check_queue(q);
  94     leak_checker_claim(b);
  95 
  96     b->next = NULL;
  97     if (q->n++) {
  98         q->tail->next = b;
  99     } else {
 100         q->head = b;
 101     }
 102     q->tail = b;
 103 
 104     check_queue(q);
 105 }

 107 /* Removes the first buffer from 'q', which must not be empty, and returns
 108  * it.  The caller must free the buffer (with ofpbuf_delete()) when it is no
 109  * longer needed. */
 110 struct ofpbuf *
 111 queue_pop_head(struct ofp_queue *q)
 112 {
 113     struct ofpbuf *head = q->head;
 114     queue_advance_head(q, head->next);
 115     return head;
 116 }
