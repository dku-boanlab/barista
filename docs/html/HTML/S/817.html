  58  * provided to set_pidfile().  The caller must free the returned string. */

  60 make_pidfile_name(const char *name) 
  61 {
  62     return (!name ? xasprintf("%s/%s.pid", ofp_rundir, program_name)
  63             : *name == '/' ? xstrdup(name)
  64             : xasprintf("%s/%s", ofp_rundir, name));

  67 /* Sets up a following call to daemonize() to create a pidfile named 'name'.
  68  * If 'name' begins with '/', then it is treated as an absolute path.
  69  * Otherwise, it is taken relative to RUNDIR, which is $(prefix)/var/run by
  70  * default.

 106 /* If a pidfile has been configured and that pidfile already exists and is
 107  * locked by a running process, returns the pid of the running process.
 108  * Otherwise, returns 0. */
 109 static pid_t
 110 already_running(void)
 111 {
 112     pid_t pid = 0;
 113     if (pidfile) {
 114         int fd = open(pidfile, O_RDWR);
 115         if (fd >= 0) {
 116             struct flock lck;
 117             lck.l_type = F_WRLCK;
 118             lck.l_whence = SEEK_SET;
 119             lck.l_start = 0;
 120             lck.l_len = 0;
 121             if (fcntl(fd, F_GETLK, &lck) != -1 && lck.l_type != F_UNLCK) {
 122                 pid = lck.l_pid;
 123             }
 124             close(fd);
 125         }
 126     }
 127     return pid;
 128 }

 130 /* If a locked pidfile exists, issue a warning message and, unless
 131  * ignore_existing_pidfile() has been called, terminate the program. */
 132 void
 133 die_if_already_running(void)
 134 {
 135     pid_t pid = already_running();
 136     if (pid) {
 137         if (!force) {
 138             ofp_fatal(0, "%s: already running as pid %ld",
 139                       get_pidfile(), (long int) pid);
 140         } else {
 141             VLOG_WARN("%s: %s already running as pid %ld",
 142                       get_pidfile(), program_name, (long int) pid);
 143         }
 144     }
 145 }

 147 /* If a pidfile has been configured, creates it and stores the running process'
 148  * pid init.  Ensures that the pidfile will be deleted when the process
 149  * exits. */
 150 static void
 151 make_pidfile(void)
 152 {
 153     if (pidfile) {
 154         /* Create pidfile via temporary file, so that observers never see an
 155          * empty pidfile or an unlocked pidfile. */
 156         long int pid = getpid();
 157         char *tmpfile;
 158         int fd;
 159 
 160         tmpfile = xasprintf("%s.tmp%ld", pidfile, pid);
 161         fatal_signal_add_file_to_unlink(tmpfile);
 162         fd = open(tmpfile, O_CREAT | O_WRONLY | O_TRUNC, 0666);
 163         if (fd >= 0) {
 164             struct flock lck;
 165             lck.l_type = F_WRLCK;
 166             lck.l_whence = SEEK_SET;
 167             lck.l_start = 0;
 168             lck.l_len = 0;
 169             if (fcntl(fd, F_SETLK, &lck) != -1) {
 170                 char *text = xasprintf("%ld\n", pid);
 171                 if (write(fd, text, strlen(text)) == strlen(text)) {
 172                     fatal_signal_add_file_to_unlink(pidfile);
 173                     if (rename(tmpfile, pidfile) < 0) {
 174                         VLOG_ERR("failed to rename \"%s\" to \"%s\": %s",
 175                                  tmpfile, pidfile, strerror(errno));
 176                         fatal_signal_remove_file_to_unlink(pidfile);
 177                         close(fd);
 178                     } else {
 179                         /* Keep 'fd' open to retain the lock. */
 180                     }
 181                     free(text);
 182                 } else {
 183                     VLOG_ERR("%s: write failed: %s", tmpfile, strerror(errno));
 184                     close(fd);
 185                 }
 186             } else {
 187                 VLOG_ERR("%s: fcntl failed: %s", tmpfile, strerror(errno));
 188                 close(fd);
 189             }
 190         } else {
 191             VLOG_ERR("%s: create failed: %s", tmpfile, strerror(errno));
 192         }
 193         fatal_signal_remove_file_to_unlink(tmpfile);
 194         free(tmpfile);
 195     }
 196     free(pidfile);
 197     pidfile = NULL;
 198 }

 200 /* If configured with set_pidfile() or set_detach(), creates the pid file and
 201  * detaches from the foreground session.  */
 202 void
 203 daemonize(void)
 204 {
 205     if (detach) {
 206         char c = 0;
 207         int fds[2];
 208         if (pipe(fds) < 0) {
 209             ofp_fatal(errno, "pipe failed");
 210         }
 211 
 212         switch (fork()) {
 213         default:
 214             /* Parent process: wait for child to create pidfile, then exit. */
 215             close(fds[1]);
 216             fatal_signal_fork();
 217             if (read(fds[0], &c, 1) != 1) {
 218                 ofp_fatal(errno, "daemon child failed to signal startup");
 219             }
 220             exit(0);
 221 
 222         case 0:
 223             /* Child process. */
 224             close(fds[0]);
 225             make_pidfile();
 226             write(fds[1], &c, 1);
 227             close(fds[1]);
 228             setsid();
 229             chdir("/");
 230             break;
 231 
 232         case -1:
 233             /* Error. */
 234             ofp_fatal(errno, "could not fork");
 235             break;
 236         }
 237     } else {
 238         make_pidfile();
 239     }
 240 }

 253 /* Opens and reads a PID from 'pidfile'.  Returns the nonnegative PID if
 254  * successful, otherwise a negative errno value. */
 255 pid_t
 256 read_pidfile(const char *pidfile)
 257 {
 258     char line[128];
 259     struct flock lck;
 260     FILE *file;
 261     int error;
 262 
 263     file = fopen(pidfile, "r");
 264     if (!file) {
 265         error = errno;
 266         VLOG_WARN("%s: open: %s", pidfile, strerror(error));
 267         goto error;
 268     }
 269 
 270     lck.l_type = F_WRLCK;
 271     lck.l_whence = SEEK_SET;
 272     lck.l_start = 0;
 273     lck.l_len = 0;
 274     if (fcntl(fileno(file), F_GETLK, &lck)) {
 275         error = errno;
 276         VLOG_WARN("%s: fcntl: %s", pidfile, strerror(error));
 277         goto error;
 278     }
 279     if (lck.l_type == F_UNLCK) {
 280         error = ESRCH;
 281         VLOG_WARN("%s: pid file is not locked", pidfile);
 282         goto error;
 283     }
 284 
 285     if (!fgets(line, sizeof line, file)) {
 286         if (ferror(file)) {
 287             error = errno;
 288             VLOG_WARN("%s: read: %s", pidfile, strerror(error));
 289         } else {
 290             error = ESRCH;
 291             VLOG_WARN("%s: read: unexpected end of file", pidfile);
 292         }
 293         goto error;
 294     }
 295 
 296     if (lck.l_pid != strtoul(line, NULL, 10)) {
 297         error = ESRCH;
 298         VLOG_WARN("l_pid (%ld) != %s pid (%s)",
 299                    (long int) lck.l_pid, pidfile, line);
 300         goto error;
 301     }
 302 
 303     fclose(file);
 304     return lck.l_pid;
 305 
 306 error:
 307     if (file) {
 308         fclose(file);
 309     }
 310     return -error;
 311 }
