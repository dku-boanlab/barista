  59 enum port_state {
  60     P_DISABLED = 1 << 0,
  61     P_LISTENING = 1 << 1,
  62     P_LEARNING = 1 << 2,
  63     P_FORWARDING = 1 << 3,
  64     P_BLOCKING = 1 << 4
  65 };

  67 struct lswitch {
  68     /* If nonnegative, the switch sets up flows that expire after the given
  69      * number of seconds (or never expire, if the value is OFP_FLOW_PERMANENT).
  70      * Otherwise, the switch processes every packet. */
  71     int max_idle;
  72 
  73     unsigned long long int datapath_id;
  74     uint32_t capabilities;
  75     time_t last_features_request;
  76     struct mac_learning *ml;    /* NULL to act as hub instead of switch. */
  77 
  78     /* Number of outgoing queued packets on the rconn. */
  79     int n_queued;
  80 
  81     /* Spanning tree protocol implementation.
  82      *
  83      * We implement STP states by, whenever a port's STP state changes,
  84      * querying all the flows on the switch and then deleting any of them that
  85      * are inappropriate for a port's STP state. */
  86     long long int next_query;   /* Next time at which to query all flows. */
  87     long long int last_query;   /* Last time we sent a query. */
  88     long long int last_reply;   /* Last time we received a query reply. */
  89     unsigned int port_states[STP_MAX_PORTS];
  90     uint32_t query_xid;         /* XID used for query. */
  91     int n_flows, n_no_recv, n_no_send;
  92 };

 114 /* Creates and returns a new learning switch.
 115  *
 116  * If 'learn_macs' is true, the new switch will learn the ports on which MAC
 117  * addresses appear.  Otherwise, the new switch will flood all packets.
 118  *
 119  * If 'max_idle' is nonnegative, the new switch will set up flows that expire
 120  * after the given number of seconds (or never expire, if 'max_idle' is
 121  * OFP_FLOW_PERMANENT).  Otherwise, the new switch will process every packet.
 122  *
 123  * 'rconn' is used to send out an OpenFlow features request. */
 124 struct lswitch *
 125 lswitch_create(struct rconn *rconn, bool learn_macs, int max_idle)
 126 {
 127     struct lswitch *sw;
 128     size_t i;
 129 
 130     sw = xcalloc(1, sizeof *sw);
 131     sw->max_idle = max_idle;
 132     sw->datapath_id = 0;
 133     sw->last_features_request = time_now() - 1;
 134     sw->ml = learn_macs ? mac_learning_create() : NULL;
 135     sw->next_query = LLONG_MIN;
 136     sw->last_query = LLONG_MIN;
 137     sw->last_reply = LLONG_MIN;
 138     for (i = 0; i < STP_MAX_PORTS; i++) {
 139         sw->port_states[i] = P_DISABLED;
 140     }
 141     send_features_request(sw, rconn);
 142     return sw;
 143 }

 145 /* Destroys 'sw'. */
 146 void
 147 lswitch_destroy(struct lswitch *sw)
 148 {
 149     if (sw) {
 150         mac_learning_destroy(sw->ml);
 151         free(sw);
 152     }

 155 /* Takes care of necessary 'sw' activity, except for receiving packets (which
 156  * the caller must do). */
 157 void
 158 lswitch_run(struct lswitch *sw, struct rconn *rconn)
 159 {
 160     long long int now = time_msec();
 161 
 162     if (sw->ml) {
 163         mac_learning_run(sw->ml, NULL);
 164     }
 165 
 166     /* If we're waiting for more replies, keeping waiting for up to 10 s. */
 167     if (sw->last_reply != LLONG_MIN) {
 168         if (now - sw->last_reply > 10000) {
 169             VLOG_ERR_RL(&rl, "%012llx: No more flow stat replies last 10 s",
 170                         sw->datapath_id);
 171             sw->last_reply = LLONG_MIN;
 172             sw->last_query = LLONG_MIN;
 173             schedule_query(sw, 0);
 174         } else {
 175             return;
 176         }
 177     }
 178 
 179     /* If we're waiting for any reply at all, keep waiting for up to 10 s. */
 180     if (sw->last_query != LLONG_MIN) {
 181         if (now - sw->last_query > 10000) {
 182             VLOG_ERR_RL(&rl, "%012llx: No flow stat replies in last 10 s",
 183                         sw->datapath_id);
 184             sw->last_query = LLONG_MIN;
 185             schedule_query(sw, 0);
 186         } else {
 187             return;
 188         }
 189     }
 190 
 191     /* If it's time to send another query, do so. */
 192     if (sw->next_query != LLONG_MIN && now >= sw->next_query) {
 193         sw->next_query = LLONG_MIN;
 194         if (!rconn_is_connected(rconn)) {
 195             schedule_query(sw, 1000);
 196         } else {
 197             struct ofp_stats_request *osr;
 198             struct ofp_flow_stats_request *ofsr;
 199             struct ofpbuf *b;
 200             int error;
 201 
 202             VLOG_DBG("%012llx: Sending flow stats request to implement STP",
 203                      sw->datapath_id);
 204 
 205             sw->last_query = now;
 206             sw->query_xid = random_uint32();
 207             sw->n_flows = 0;
 208             sw->n_no_recv = 0;
 209             sw->n_no_send = 0;
 210             osr = make_openflow_xid(sizeof *osr + sizeof *ofsr,
 211                                     OFPT_STATS_REQUEST, sw->query_xid, &b);
 212             osr->type = htons(OFPST_FLOW);
 213             osr->flags = htons(0);
 214             ofsr = (struct ofp_flow_stats_request *) osr->body;
 215             ofsr->match.wildcards = htonl(OFPFW_ALL);
 216             ofsr->table_id = 0xff;
 217             ofsr->out_port = htons(OFPP_NONE);
 218 
 219             error = rconn_send(rconn, b, NULL);
 220             if (error) {
 221                 VLOG_WARN_RL(&rl, "%012llx: sending flow stats request "
 222                              "failed: %s", sw->datapath_id, strerror(error));
 223                 ofpbuf_delete(b);
 224                 schedule_query(sw, 1000);
 225             }
 226         }
 227     }
 228 }

 230 static void
 231 wait_timeout(long long int started)
 232 {
 233     long long int now = time_msec();
 234     long long int timeout = 10000 - (now - started);
 235     if (timeout <= 0) {
 236         poll_immediate_wake();
 237     } else {
 238         poll_timer_wait(timeout);
 239     }
 240 }

 242 void
 243 lswitch_wait(struct lswitch *sw)
 244 {
 245     if (sw->ml) {
 246         mac_learning_wait(sw->ml);
 247     }
 248 
 249     if (sw->last_reply != LLONG_MIN) {
 250         wait_timeout(sw->last_reply);
 251     } else if (sw->last_query != LLONG_MIN) {
 252         wait_timeout(sw->last_query);
 253     }

 256 /* Processes 'msg', which should be an OpenFlow received on 'rconn', according
 257  * to the learning switch state in 'sw'.  The most likely result of processing
 258  * is that flow-setup and packet-out OpenFlow messages will be sent out on
 259  * 'rconn'.  */
 260 void
 261 lswitch_process_packet(struct lswitch *sw, struct rconn *rconn,
 262                        const struct ofpbuf *msg)
 263 {
 264     struct processor {
 265         uint8_t type;
 266         size_t min_size;
 267         packet_handler_func *handler;
 268     };
 269     static const struct processor processors[] = {
 270         {
 271             OFPT_ECHO_REQUEST,
 272             sizeof(struct ofp_header),
 273             process_echo_request
 274         },
 275         {
 276             OFPT_FEATURES_REPLY,
 277             sizeof(struct ofp_switch_features),
 278             process_switch_features
 279         },
 280         {
 281             OFPT_PACKET_IN,
 282             offsetof(struct ofp_packet_in, data),
 283             process_packet_in
 284         },
 285         {
 286             OFPT_PORT_STATUS,
 287             sizeof(struct ofp_port_status),
 288             process_port_status
 289         },
 290         {
 291             OFPT_STATS_REPLY,
 292             offsetof(struct ofp_stats_reply, body),
 293             process_stats_reply
 294         },
 295         {
 296             OFPT_FLOW_REMOVED,
 297             sizeof(struct ofp_flow_removed),
 298             NULL
 299         },
 300     };
 301     const size_t n_processors = ARRAY_SIZE(processors);
 302     const struct processor *p;
 303     struct ofp_header *oh;
 304 
 305     oh = msg->data;
 306     if (sw->datapath_id == 0
 307         && oh->type != OFPT_ECHO_REQUEST
 308         && oh->type != OFPT_FEATURES_REPLY) {
 309         send_features_request(sw, rconn);
 310         return;
 311     }
 312 
 313     for (p = processors; p < &processors[n_processors]; p++) {
 314         if (oh->type == p->type) {
 315             if (msg->size < p->min_size) {
 316                 VLOG_WARN_RL(&rl, "%012llx: %s: too short (%zu bytes) for "
 317                              "type %"PRIu8" (min %zu)", sw->datapath_id,
 318                              rconn_get_name(rconn), msg->size, oh->type,
 319                              p->min_size);
 320                 return;
 321             }
 322             if (p->handler) {
 323                 (p->handler)(sw, rconn, msg->data);
 324             }
 325             return;
 326         }
 327     }
 328     if (VLOG_IS_DBG_ENABLED()) {
 329         char *p = ofp_to_string(msg->data, msg->size, 2);
 330         VLOG_DBG_RL(&rl, "%012llx: OpenFlow packet ignored: %s",
 331                     sw->datapath_id, p);
 332         free(p);
 333     }
 334 }

 336 static void
 337 send_features_request(struct lswitch *sw, struct rconn *rconn)
 338 {
 339     time_t now = time_now();
 340     if (now >= sw->last_features_request + 1) {
 341         struct ofpbuf *b;
 342         struct ofp_switch_config *osc;
 343 
 344         /* Send OFPT_FEATURES_REQUEST. */
 345         make_openflow(sizeof(struct ofp_header), OFPT_FEATURES_REQUEST, &b);
 346         queue_tx(sw, rconn, b);
 347 
 348         /* Send OFPT_SET_CONFIG. */
 349         osc = make_openflow(sizeof *osc, OFPT_SET_CONFIG, &b);
 350         osc->flags = 0;
 351         osc->miss_send_len = htons(OFP_DEFAULT_MISS_SEND_LEN);
 352         queue_tx(sw, rconn, b);
 353 
 354         sw->last_features_request = now;
 355     }
 356 }

 358 static void
 359 queue_tx(struct lswitch *sw, struct rconn *rconn, struct ofpbuf *b)
 360 {
 361     int retval = rconn_send_with_limit(rconn, b, &sw->n_queued, 10);
 362     if (retval && retval != ENOTCONN) {
 363         if (retval == EAGAIN) {
 364             VLOG_INFO_RL(&rl, "%012llx: %s: tx queue overflow",
 365                          sw->datapath_id, rconn_get_name(rconn));
 366         } else {
 367             VLOG_WARN_RL(&rl, "%012llx: %s: send: %s",
 368                          sw->datapath_id, rconn_get_name(rconn),
 369                          strerror(retval));
 370         }
 371     }
 372 }

 374 static void
 375 schedule_query(struct lswitch *sw, long long int delay)
 376 {
 377     long long int now = time_msec();
 378     if (sw->next_query == LLONG_MIN || sw->next_query > now + delay) {
 379         sw->next_query = now + delay;
 380     }

 383 static void
 384 process_switch_features(struct lswitch *sw, struct rconn *rconn, void *osf_)
 385 {
 386     struct ofp_switch_features *osf = osf_;
 387     size_t n_ports = ((ntohs(osf->header.length)
 388                        - offsetof(struct ofp_switch_features, ports))
 389                       / sizeof *osf->ports);
 390     size_t i;
 391 
 392     sw->datapath_id = ntohll(osf->datapath_id);
 393     sw->capabilities = ntohl(osf->capabilities);
 394     for (i = 0; i < n_ports; i++) {
 395         process_phy_port(sw, rconn, &osf->ports[i]);
 396     }
 397     if (sw->capabilities & OFPC_STP) {
 398         schedule_query(sw, 1000);
 399     }
 400 }

 402 static void
 403 process_packet_in(struct lswitch *sw, struct rconn *rconn, void *opi_)
 404 {
 405     struct ofp_packet_in *opi = opi_;
 406     uint16_t in_port = ntohs(opi->in_port);
 407     uint16_t out_port = OFPP_FLOOD;
 408 
 409     size_t pkt_ofs, pkt_len;
 410     struct ofpbuf pkt;
 411     struct flow flow;
 412 
 413     /* Extract flow data from 'opi' into 'flow'. */
 414     pkt_ofs = offsetof(struct ofp_packet_in, data);
 415     pkt_len = ntohs(opi->header.length) - pkt_ofs;
 416     pkt.data = opi->data;
 417     pkt.size = pkt_len;
 418     flow_extract(&pkt, in_port, &flow);
 419 
 420     if (may_learn(sw, in_port) && sw->ml) {
 421         if (mac_learning_learn(sw->ml, flow.dl_src, 0, in_port)) {
 422             VLOG_DBG_RL(&rl, "%012llx: learned that "ETH_ADDR_FMT" is on "
 423                         "port %"PRIu16, sw->datapath_id,
 424                         ETH_ADDR_ARGS(flow.dl_src), in_port);
 425         }
 426     }
 427 
 428     if (eth_addr_is_reserved(flow.dl_src)) {
 429         goto drop_it;
 430     }
 431 
 432     if (!may_recv(sw, in_port, false)) {
 433         /* STP prevents receiving anything on this port. */
 434         goto drop_it;
 435     }
 436 
 437     if (sw->ml) {
 438         uint16_t learned_port = mac_learning_lookup(sw->ml, flow.dl_dst, 0);
 439         if (may_send(sw, learned_port)) {
 440             out_port = learned_port;
 441         }
 442     }
 443 
 444     if (in_port == out_port) {
 445         /* Don't send out packets on their input ports. */
 446         goto drop_it;
 447     } else if (sw->max_idle >= 0 && (!sw->ml || out_port != OFPP_FLOOD)) {
 448         /* The output port is known, or we always flood everything, so add a
 449          * new flow. */
 450         queue_tx(sw, rconn, make_add_simple_flow(&flow, ntohl(opi->buffer_id),
 451                                                  out_port, sw->max_idle));
 452 
 453         /* If the switch didn't buffer the packet, we need to send a copy. */
 454         if (ntohl(opi->buffer_id) == UINT32_MAX) {
 455             queue_tx(sw, rconn,
 456                      make_unbuffered_packet_out(&pkt, in_port, out_port));
 457         }
 458     } else {
 459         /* We don't know that MAC, or we don't set up flows.  Send along the
 460          * packet without setting up a flow. */
 461         struct ofpbuf *b;
 462         if (ntohl(opi->buffer_id) == UINT32_MAX) {
 463             b = make_unbuffered_packet_out(&pkt, in_port, out_port);
 464         } else {
 465             b = make_buffered_packet_out(ntohl(opi->buffer_id),
 466                                          in_port, out_port);
 467         }
 468         queue_tx(sw, rconn, b);
 469     }
 470     return;
 471 
 472 drop_it:
 473     if (sw->max_idle >= 0) {
 474         /* Set up a flow to drop packets. */
 475         queue_tx(sw, rconn, make_add_flow(&flow, ntohl(opi->buffer_id),
 476                                           sw->max_idle, 0));
 477     } else {
 478         /* Just drop the packet, since we don't set up flows at all.
 479          * XXX we should send a packet_out with no actions if buffer_id !=
 480          * UINT32_MAX, to avoid clogging the kernel buffers. */
 481     }
 482     return;
 483 }

 499 static void
 500 process_phy_port(struct lswitch *sw, struct rconn *rconn UNUSED, void *opp_)
 501 {
 502     const struct ofp_phy_port *opp = opp_;
 503     uint16_t port_no = ntohs(opp->port_no);
 504     if (sw->capabilities & OFPC_STP && port_no < STP_MAX_PORTS) {
 505         uint32_t config = ntohl(opp->config);
 506         uint32_t state = ntohl(opp->state);
 507         unsigned int *port_state = &sw->port_states[port_no];
 508         unsigned int new_port_state;
 509 
 510         if (!(config & (OFPPC_NO_STP | OFPPC_PORT_DOWN))
 511             && !(state & OFPPS_LINK_DOWN))
 512         {
 513             switch (state & OFPPS_STP_MASK) {
 514             case OFPPS_STP_LISTEN:
 515                 new_port_state = P_LISTENING;
 516                 break;
 517             case OFPPS_STP_LEARN:
 518                 new_port_state = P_LEARNING;
 519                 break;
 520             case OFPPS_STP_FORWARD:
 521                 new_port_state = P_FORWARDING;
 522                 break;
 523             case OFPPS_STP_BLOCK:
 524                 new_port_state = P_BLOCKING;
 525                 break;
 526             default:
 527                 new_port_state = P_DISABLED;
 528                 break;
 529             }
 530         } else {
 531             new_port_state = P_FORWARDING;
 532         }
 533         if (*port_state != new_port_state) {
 534             *port_state = new_port_state;
 535             schedule_query(sw, 1000);
 536         }
 537     }
 538 }

 540 static unsigned int
 541 get_port_state(const struct lswitch *sw, uint16_t port_no)
 542 {
 543     return (port_no >= STP_MAX_PORTS || !(sw->capabilities & OFPC_STP)
 544             ? P_FORWARDING
 545             : sw->port_states[port_no]);
 546 }

 548 static bool
 549 may_learn(const struct lswitch *sw, uint16_t port_no)
 550 {
 551     return get_port_state(sw, port_no) & (P_LEARNING | P_FORWARDING);
 552 }
 553 
 554 static bool
 555 may_recv(const struct lswitch *sw, uint16_t port_no, bool any_actions)
 556 {
 557     unsigned int state = get_port_state(sw, port_no);
 558     return !(any_actions
 559              ? state & (P_DISABLED | P_LISTENING | P_BLOCKING)
 560              : state & (P_DISABLED | P_LISTENING | P_BLOCKING | P_LEARNING));
 561 }

 563 static bool
 564 may_send(const struct lswitch *sw, uint16_t port_no)
 565 {
 566     return get_port_state(sw, port_no) & P_FORWARDING;
 567 }

 569 static void
 570 process_flow_stats(struct lswitch *sw, struct rconn *rconn,
 571                    const struct ofp_flow_stats *ofs)
 572 {
 573     const char *end = (char *) ofs + ntohs(ofs->length);
 574     bool delete = false;
 575 
 576     /* Decide to delete the flow if it matches on an STP-disabled physical
 577      * port.  But don't delete it if the flow just drops all received packets,
 578      * because that's a perfectly reasonable thing to do for disabled physical
 579      * ports. */
 580     if (!(ofs->match.wildcards & htonl(OFPFW_IN_PORT))) {
 581         if (!may_recv(sw, ntohs(ofs->match.in_port),
 582                       end > (char *) ofs->actions)) {
 583             delete = true;
 584             sw->n_no_recv++;
 585         }
 586     }
 587 
 588     /* Decide to delete the flow if it forwards to an STP-disabled physical
 589      * port. */
 590     if (!delete) {
 591         const struct ofp_action_header *a;
 592         size_t len;
 593 
 594         for (a = ofs->actions; (char *) a < end; a += len / 8) {
 595             len = ntohs(a->len);
 596             if (len > end - (char *) a) {
 597                 VLOG_DBG_RL(&rl, "%012llx: action exceeds available space "
 598                             "(%zu > %td)",
 599                             sw->datapath_id, len, end - (char *) a);
 600                 break;
 601             } else if (len % 8) {
 602                 VLOG_DBG_RL(&rl, "%012llx: action length (%zu) not multiple "
 603                             "of 8 bytes", sw->datapath_id, len);
 604                 break;
 605             }
 606 
 607             if (a->type == htons(OFPAT_OUTPUT)) {
 608                 struct ofp_action_output *oao = (struct ofp_action_output *) a;
 609                 if (!may_send(sw, ntohs(oao->port))) {
 610                     delete = true;
 611                     sw->n_no_send++;
 612                     break;
 613                 }
 614             }
 615         }
 616     }
 617 
 618     /* Delete the flow. */
 619     if (delete) {
 620         struct ofp_flow_mod *ofm;
 621         struct ofpbuf *b;
 622 
 623         ofm = make_openflow(offsetof(struct ofp_flow_mod, actions),
 624                             OFPT_FLOW_MOD, &b);
 625         ofm->match = ofs->match;
 626         ofm->command = OFPFC_DELETE_STRICT;
 627         rconn_send(rconn, b, NULL);
 628     }
 629 }

 631 static void
 632 process_stats_reply(struct lswitch *sw, struct rconn *rconn, void *osr_)
 633 {
 634     struct ofp_stats_reply *osr = osr_;
 635     struct flow_stats_iterator i;
 636     const struct ofp_flow_stats *fs;
 637 
 638     if (sw->last_query == LLONG_MIN
 639         || osr->type != htons(OFPST_FLOW)
 640         || osr->header.xid != sw->query_xid) {
 641         return;
 642     }
 643     for (fs = flow_stats_first(&i, osr); fs; fs = flow_stats_next(&i)) {
 644         sw->n_flows++;
 645         process_flow_stats(sw, rconn, fs);
 646     }
 647     if (!(osr->flags & htons(OFPSF_REPLY_MORE))) {
 648         VLOG_DBG("%012llx: Deleted %d of %d received flows to "
 649                  "implement STP, %d because of no-recv, %d because of "
 650                  "no-send", sw->datapath_id,
 651                  sw->n_no_recv + sw->n_no_send, sw->n_flows,
 652                  sw->n_no_recv, sw->n_no_send);
 653         sw->last_query = LLONG_MIN;
 654         sw->last_reply = LLONG_MIN;
 655     } else {
 656         sw->last_reply = time_msec();
 657     }
 658 }
