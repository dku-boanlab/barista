 220 /*
 221  * toXxx(L,idx) gets a Xxx from an index (Lua Error if fails)
 222  * checkXxx(L,idx) gets a Xxx from an index after calling check_code (No Lua Error if it fails)
 223  * pushXxx(L,xxx) pushes an Xxx into the stack
 224  * isXxx(L,idx) tests whether we have an Xxx at idx
 225  *
 226  * LUA_CLASS_DEFINE must be used without trailing ';'
 227  */
 228 #define WSLUA_CLASS_DEFINE(C,check_code,push_code) \
 229 C to##C(lua_State* L, int index) { \
 230     C* v = (C*)lua_touserdata (L, index); \
 231     if (!v) luaL_typerror(L,index,#C); \
 232     return *v; \
 233 } \
 234 C check##C(lua_State* L, int index) { \
 235     C* p; \
 236     luaL_checktype(L,index,LUA_TUSERDATA); \
 237     p = (C*)luaL_checkudata(L, index, #C); \
 238     check_code; \
 239     return p ? *p : NULL; \
 240 } \
 241 C* push##C(lua_State* L, C v) { \
 242     C* p = lua_newuserdata(L,sizeof(C)); *p = v; \
 243     luaL_getmetatable(L, #C); lua_setmetatable(L, -2); \
 244         push_code; \
 245     return p; \
 246 }\
 247 gboolean is##C(lua_State* L,int i) { \
 248         void *p; \
 249         if(!lua_isuserdata(L,i)) return FALSE; \
 250         p = lua_touserdata(L, i); \
 251         lua_getfield(L, LUA_REGISTRYINDEX, #C); \
 252         if (p == NULL || !lua_getmetatable(L, i) || !lua_rawequal(L, -1, -2)) p=NULL; \
 253         lua_pop(L, 2); \
 254         return p ? TRUE : FALSE; \
 255 } \
 256 C shift##C(lua_State* L,int i) { \
 257     C* p; \
 258         if(!lua_isuserdata(L,i)) return NULL; \
 259         p = lua_touserdata(L, i); \
 260         lua_getfield(L, LUA_REGISTRYINDEX, #C); \
 261         if (p == NULL || !lua_getmetatable(L, i) || !lua_rawequal(L, -1, -2)) p=NULL; \
 262         lua_pop(L, 2); \
 263         if (p) { lua_remove(L,i); return *p; }\
 264         else return NULL;\
 265 } \
 266 int dummy##C

 270 #define WSLUA_REGISTER_CLASS(C) { \
 271         luaL_register (L, #C, C ## _methods); \
 272         luaL_newmetatable (L, #C); \
 273         luaL_register (L, NULL, C ## _meta); \
 274         lua_pushliteral(L, "__index"); \
 275         lua_pushvalue(L, -3); \
 276         lua_rawset(L, -3); \
 277         lua_pushliteral(L, "__metatable"); \
 278         lua_pushvalue(L, -3); \
 279         lua_rawset(L, -3); \
 280         lua_pop(L, 1); \
 281 }
