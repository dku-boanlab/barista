  50 /**
  51  * \brief Function to add a flow
  52  * \param list Flow table mapped to a datapath ID
  53  * \param flow NEW flow
  54  */
  55 static int add_flow(flow_table_t *list, const flow_t *flow)
  56 {
  57     int ret = TRUE;
  58 
  59     pthread_mutex_lock(&list->lock);
  60 
  61     flow_t *curr = list->head;
  62     while (curr != NULL) {
  63         if (FLOW_COMPARE(curr, flow)) {
  64             ret = FALSE;
  65             break;
  66         }
  67 
  68         curr = curr->next;
  69     }
  70 
  71     pthread_mutex_unlock(&list->lock);
  72 
  73     if (ret) {
  74         flow_t *new = flow_dequeue();
  75         if (new == NULL) {
  76             PERROR("flow_q_dequeue");
  77             return FALSE;
  78         }
  79 
  80         new->dpid = flow->dpid;
  81         new->port = flow->port;
  82 
  83         new->insert_time = time(NULL);
  84 
  85         new->idle_timeout = flow->idle_timeout;
  86         new->hard_timeout = flow->hard_timeout;
  87 
  88         new->wildcards = flow->wildcards;
  89 
  90         new->vlan_id = flow->vlan_id;
  91         new->vlan_pcp = flow->vlan_pcp;
  92 
  93         new->proto = flow->proto;
  94 
  95         new->remote = flow->remote;
  96 
  97         memmove(new->src_mac, flow->src_mac, ETH_ALEN);
  98         memmove(new->dst_mac, flow->dst_mac, ETH_ALEN);
  99 
 100         new->src_ip = flow->src_ip;
 101         new->dst_ip = flow->dst_ip;
 102 
 103         new->src_port = flow->src_port;
 104         new->dst_port = flow->dst_port;
 105 
 106         new->num_actions = flow->num_actions;
 107         memmove(new->action, flow->action, sizeof(action_t) * new->num_actions);
 108 
 109         new->prev = new->next = new->r_next = NULL;
 110 
 111         flow_t out = {0};
 112         memmove(&out, new, sizeof(flow_t));
 113 
 114         pthread_mutex_lock(&list->lock);
 115 
 116         if (list->head == NULL) {
 117             list->head = new;
 118             list->tail = new;
 119         } else {
 120             new->prev = list->tail;
 121             list->tail->next = new;
 122             list->tail = new;
 123         }
 124 
 125         num_flows++;
 126 
 127         pthread_mutex_unlock(&list->lock);
 128 
 129         if (new->remote == FALSE)
 130             ev_flow_added(FLOW_MGMT_ID, &out);
 131     }
 132 
 133     return ret;
 134 }

 148 /**
 149  * \brief Function to delete a flow
 150  * \param list Flow table mapped to a datapath ID
 151  * \param flow Existing flow to delete
 152  */
 153 static int delete_flow(flow_table_t *list, const flow_t *flow)
 154 {
 155     flow_table_t tmp_list = {0};
 156 
 157     pthread_mutex_lock(&list->lock);
 158 
 159     flow_t *curr = list->head;
 160     while (curr != NULL) {
 161         if (FLOW_COMPARE(curr, flow)) {
 162             if (tmp_list.head == NULL) {
 163                 tmp_list.head = curr;
 164                 tmp_list.tail = curr;
 165                 curr->r_next = NULL;
 166             } else {
 167                 tmp_list.tail->r_next = curr;
 168                 tmp_list.tail = curr;
 169                 curr->r_next = NULL;
 170             }
 171         }
 172 
 173         curr = curr->next;
 174     }
 175 
 176     curr = tmp_list.head;
 177     while (curr != NULL) {
 178         flow_t *tmp = curr;
 179 
 180         curr = curr->r_next;
 181 
 182         if (tmp->prev != NULL && tmp->next != NULL) {
 183             tmp->prev->next = tmp->next;
 184             tmp->next->prev = tmp->prev;
 185         } else if (tmp->prev == NULL && tmp->next != NULL) {
 186             list->head = tmp->next;
 187             tmp->next->prev = NULL;
 188         } else if (tmp->prev != NULL && tmp->next == NULL) {
 189             list->tail = tmp->prev;
 190             tmp->prev->next = NULL;
 191         } else if (tmp->prev == NULL && tmp->next == NULL) {
 192             list->head = NULL;
 193             list->tail = NULL;
 194         }
 195 
 196         num_flows--;
 197 
 198         tmp->prev = tmp->next = tmp->r_next = NULL;
 199 
 200         if (tmp->remote == FALSE)
 201             ev_flow_deleted(FLOW_MGMT_ID, tmp);
 202 
 203         flow_enqueue(tmp);
 204     }
 205 
 206     pthread_mutex_unlock(&list->lock);
 207 
 208     return 0;
 209 }

 306 void *timeout_thread(void *arg)
 307 {
 308     time_t previous_time = time(NULL);
 309     time_t current_time = time(NULL);
 310 
 311     while (flow_mgmt_on) {
 312         previous_time = current_time;
 313         current_time = time(NULL);
 314 
 315         if (previous_time != current_time) {
 316             int i;
 317             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 318                 flow_table_t tmp_list = {0};
 319 
 320                 pthread_mutex_lock(&flow_table[i].lock);
 321 
 322                 flow_t *curr = flow_table[i].head;
 323                 while (curr != NULL) {
 324                     if ((current_time - curr->insert_time) > curr->hard_timeout) {
 325                         if (tmp_list.head == NULL) {
 326                             tmp_list.head = curr;
 327                             tmp_list.tail = curr;
 328                             curr->r_next = NULL;
 329                         } else {
 330                             tmp_list.tail->r_next = curr;
 331                             tmp_list.tail = curr;
 332                             curr->r_next = NULL;
 333                         }
 334 
 335                         num_flows--;
 336                     }
 337 
 338                     curr = curr->next;
 339                 }
 340 
 341                 curr = tmp_list.head;
 342                 while (curr != NULL) {
 343                     flow_t *tmp = curr;
 344 
 345                     curr = curr->r_next;
 346 
 347                     if (tmp->prev != NULL && tmp->next != NULL) {
 348                         tmp->prev->next = tmp->next;
 349                         tmp->next->prev = tmp->prev;
 350                     } else if (tmp->prev == NULL && tmp->next != NULL) {
 351                         flow_table[i].head = tmp->next;
 352                         tmp->next->prev = NULL;
 353                     } else if (tmp->prev != NULL && tmp->next == NULL) {
 354                         flow_table[i].tail = tmp->prev;
 355                         tmp->prev->next = NULL;
 356                     } else if (tmp->prev == NULL && tmp->next == NULL) {
 357                         flow_table[i].head = NULL;
 358                         flow_table[i].tail = NULL;
 359                     }
 360 
 361                     tmp->prev = tmp->next = tmp->r_next = NULL;
 362 
 363                     if (tmp->remote == FALSE)
 364                         ev_flow_deleted(FLOW_MGMT_ID, tmp);
 365 
 366                     flow_enqueue(tmp);
 367                 }
 368 
 369                 pthread_mutex_unlock(&flow_table[i].lock);
 370             }
 371         }
 372 
 373         waitsec(1, 0);
 374     }
 375 
 376     return NULL;
 377 }

 381 /**
 382  * \brief The main function
 383  * \param activated The activation flag of this component
 384  * \param argc The number of arguments
 385  * \param argv Arguments
 386  */
 387 int flow_mgmt_main(int *activated, int argc, char **argv)
 388 {
 389     LOG_INFO(FLOW_MGMT_ID, "Init - Flow management");
 390 
 391     flow_mgmt_on = TRUE;
 392 
 393     num_flows = 0;
 394 
 395     flow_table = (flow_table_t *)MALLOC(sizeof(flow_table_t) * __DEFAULT_TABLE_SIZE);
 396     if (flow_table == NULL) {
 397         PERROR("malloc");
 398         return -1;
 399     }
 400 
 401     memset(flow_table, 0, sizeof(flow_table_t) * __DEFAULT_TABLE_SIZE);
 402 
 403     int i;
 404     for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 405         pthread_mutex_init(&flow_table[i].lock, NULL);
 406     }
 407 
 408     flow_q_init();
 409 
 410     pthread_t thread;
 411     if (pthread_create(&thread, NULL, timeout_thread, NULL) < 0) {
 412         PERROR("pthread_create");
 413     }
 414 
 415     activate();
 416 
 417     return 0;
 418 }

 420 /**
 421  * \brief The cleanup function
 422  * \param activated The activation flag of this component
 423  */
 424 int flow_mgmt_cleanup(int *activated)
 425 {
 426     LOG_INFO(FLOW_MGMT_ID, "Clean up - Flow management");
 427 
 428     deactivate();
 429 
 430     flow_mgmt_on = FALSE;
 431 
 432     int i;
 433     for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 434         pthread_mutex_lock(&flow_table[i].lock);
 435 
 436         flow_t *curr = flow_table[i].head;
 437         while (curr != NULL) {
 438             flow_t *tmp = curr;
 439             curr = curr->next;
 440             FREE(tmp);
 441         }
 442 
 443         pthread_mutex_unlock(&flow_table[i].lock);
 444         pthread_mutex_destroy(&flow_table[i].lock);
 445     }
 446 
 447     flow_q_destroy();
 448 
 449     FREE(flow_table);
 450 
 451     return 0;
 452 }

 623                           curr->dpid, curr->idle_timeout, curr->hard_timeout, curr->port,
 624                           curr->src_mac[0], curr->src_mac[1], curr->src_mac[2], curr->src_mac[3], curr->src_mac[4], curr->src_mac[5],
 625                           curr->dst_mac[0], curr->dst_mac[1], curr->dst_mac[2], curr->dst_mac[3], curr->dst_mac[4], curr->dst_mac[5],
 626                           curr->vlan_id, curr->vlan_pcp, proto, srcip, dstip, curr->src_port, curr->dst_port,
 627                           curr->num_actions, curr->pkt_count, curr->byte_count);
 628             } else if (curr->proto & PROTO_IPV4) {
 629                 cli_print(cli, "  Flow #%d\n    %s, DPID: %lu, idle_timeout: %u, hard_timeout: %u, in_port: %u,\n"
 630                                "     src_mac: %02x:%02x:%02x:%02x:%02x:%02x, dst_mac: %02x:%02x:%02x:%02x:%02x:%02x,\n"
 631                                "     vlan: %u, vlan_pcp: %u, %s, src_ip: %s, dst_ip: %s\n"
 632                                "     # of actions: %d, pkt_count: %lu, byte_count: %lu",
 633                           ++cnt, (curr->remote == TRUE) ? "Remote" : "Local", 
 634                           curr->dpid, curr->idle_timeout, curr->hard_timeout, curr->port,
 635                           curr->src_mac[0], curr->src_mac[1], curr->src_mac[2], curr->src_mac[3], curr->src_mac[4], curr->src_mac[5],

 637                           curr->vlan_id, curr->vlan_pcp, proto, srcip, dstip,
 638                           curr->num_actions, curr->pkt_count, curr->byte_count);
 639             } else {
 640                 cli_print(cli, "  Flow #%d\n    %s, DPID: %lu, idle_timeout: %u, hard_timeout: %u, in_port: %u,\n"
 641                                "     src_mac: %02x:%02x:%02x:%02x:%02x:%02x, dst_mac: %02x:%02x:%02x:%02x:%02x:%02x,\n"
 642                                "     vlan: %u, vlan_pcp: %u, %s\n"
 643                                "     # of actions: %d, pkt_count: %lu, byte_count: %lu",
 644                           ++cnt, (curr->remote == TRUE) ? "Remote" : "Local",
 645                           curr->dpid, curr->idle_timeout, curr->hard_timeout, curr->port,
 646                           curr->src_mac[0], curr->src_mac[1], curr->src_mac[2], curr->src_mac[3], curr->src_mac[4], curr->src_mac[5],
 647                           curr->dst_mac[0], curr->dst_mac[1], curr->dst_mac[2], curr->dst_mac[3], curr->dst_mac[4], curr->dst_mac[5],
 648                           curr->vlan_id, curr->vlan_pcp, proto,
 649                           curr->num_actions, curr->pkt_count, curr->byte_count);
 650             }
 651 
 652             curr = curr->next;
 653         }
 654 
 655         pthread_mutex_unlock(&flow_table[i].lock);
 656     }
 657 
 658     if (!cnt)
 659         cli_print(cli, "  No flow");
 660 
 661     return 0;
 662 }

 690 /**
 691  * \brief The handler function
 692  * \param ev Read-only event
 693  * \param ev_out Read-write event (if this component has the write permission)
 694  */
 695 int flow_mgmt_handler(const event_t *ev, event_out_t *ev_out)
 696 {
 697     switch(ev->type) {
 698     case EV_DP_INSERT_FLOW:
 699         PRINT_EV("EV_DP_INSERT_FLOW\n");
 700         {
 701             const flow_t *flow = ev->flow;
 702             flow_table_t *flow_tbl = &flow_table[flow->dpid % __DEFAULT_TABLE_SIZE];
 703             add_flow(flow_tbl, ev->flow);
 704         }
 705         break;
 706     case EV_DP_MODIFY_FLOW:
 707         PRINT_EV("EV_DP_MODIFY_FLOW\n");
 708         {
 709             const flow_t *flow = ev->flow;
 710             flow_table_t *flow_tbl = &flow_table[flow->dpid % __DEFAULT_TABLE_SIZE];
 711             modify_flow(flow_tbl, ev->flow);
 712         }
 713         break;
 714     case EV_DP_DELETE_FLOW:
 715         PRINT_EV("EV_DP_DELETE_FLOW\n");
 716         {
 717             const flow_t *flow = ev->flow;
 718             flow_table_t *flow_tbl = &flow_table[flow->dpid % __DEFAULT_TABLE_SIZE];
 719             delete_flow(flow_tbl, ev->flow);
 720         }
 721         break;
 722     case EV_DP_FLOW_EXPIRED:
 723         PRINT_EV("EV_DP_FLOW_EXPIRED\n");
 724         {
 725             const flow_t *flow = ev->flow;
 726             flow_table_t *flow_tbl = &flow_table[flow->dpid % __DEFAULT_TABLE_SIZE];
 727             delete_flow(flow_tbl, ev->flow);
 728         }
 729         break;
 730     case EV_DP_FLOW_DELETED:
 731         PRINT_EV("EV_DP_FLOW_DELETED\n");
 732         {
 733             const flow_t *flow = ev->flow;
 734             flow_table_t *flow_tbl = &flow_table[flow->dpid % __DEFAULT_TABLE_SIZE];
 735             delete_flow(flow_tbl, ev->flow);
 736         }
 737         break;
 738     case EV_DP_FLOW_STATS:
 739         PRINT_EV("EV_DP_FLOW_STATS\n");
 740         {
 741             const flow_t *flow = ev->flow;
 742             flow_table_t *flow_tbl = &flow_table[flow->dpid % __DEFAULT_TABLE_SIZE];
 743             update_flow(flow_tbl, ev->flow);
 744         }
 745         break;
 746     case EV_SW_DISCONNECTED:
 747         PRINT_EV("EV_SW_DISCONNECTED\n");
 748         {
 749             const switch_t *sw = ev->sw;
 750             flow_table_t *flow_tbl = &flow_table[sw->dpid % __DEFAULT_TABLE_SIZE];
 751             delete_all_flows(flow_tbl, ev->sw->dpid);
 752         }
 753         break;
 754     case EV_FLOW_ADDED:
 755         PRINT_EV("EV_FLOW_ADDED\n");
 756         {
 757             const flow_t *flow = ev->flow;
 758 
 759             if (flow->remote == FALSE)
 760                 break;
 761 
 762             flow_table_t *flow_tbl = &flow_table[flow->dpid % __DEFAULT_TABLE_SIZE];
 763             add_flow(flow_tbl, ev->flow);
 764         }
 765         break;
 766     case EV_FLOW_DELETED:
 767         PRINT_EV("EV_FLOW_DELETED\n");
 768         {
 769             const flow_t *flow = ev->flow;
 770 
 771             if (flow->remote == FALSE)
 772                 break;
 773 
 774             flow_table_t *flow_tbl = &flow_table[flow->dpid % __DEFAULT_TABLE_SIZE];
 775             delete_flow(flow_tbl, ev->flow);
 776         }
 777         break;
 778     case EV_FLOW_GET_INFO:
 779         PRINT_EV("EV_FLOW_GET_INFO\n");
 780         {
 781             flow_t *flow = ev_out->flow_data;
 782 
 783             uint64_t dpid = flow->dpid;
 784 
 785             flow = (flow_t *)MALLOC(sizeof(flow_t) * num_flows);
 786             if (flow == NULL) {
 787                 break;
 788             }
 789 
 790             int i, idx = 0;
 791             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 792                 pthread_mutex_lock(&flow_table[i].lock);
 793 
 794                 flow_t *curr = flow_table[i].head;
 795                 while (curr != NULL) {
 796                     if (curr->dpid == dpid) {
 797                         memmove(&flow[idx], curr, sizeof(flow_t));
 798                         if (idx == 0) {
 799                             flow[idx].prev = NULL;
 800 
 801                             if (num_flows > 1) {
 802                                 flow[idx].next = &flow[idx + 1];
 803                             } else {
 804                                 flow[idx].next = NULL;
 805                             }
 806                         } else {
 807                             flow[idx].prev = &flow[idx - 1];
 808 
 809                             if (idx < num_flows - 1) {
 810                                 flow[idx].next = &flow[idx + 1];
 811                             } else {
 812                                 flow[idx].next = NULL;
 813                             }
 814                         }
 815                         flow[idx].r_next = NULL;
 816 
 817                         idx++;
 818                     }
 819 
 820                     curr = curr->next;
 821                 }
 822 
 823                 pthread_mutex_unlock(&flow_table[i].lock);
 824             }
 825         }
 826         break;
 827     case EV_FLOW_GET_ALL_INFO:
 828         PRINT_EV("EV_FLOW_GET_ALL_INFO\n");
 829         {
 830             flow_t *flow = ev_out->flow_data;
 831 
 832             flow = (flow_t *)MALLOC(sizeof(flow_t) * num_flows);
 833             if (flow == NULL) {
 834                 break;
 835             }
 836 
 837             int i, idx = 0;
 838             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 839                 pthread_mutex_lock(&flow_table[i].lock);
 840 
 841                 flow_t *curr = flow_table[i].head;
 842                 while (curr != NULL) {
 843                     memmove(&flow[idx], curr, sizeof(flow_t));
 844                     if (idx == 0) {
 845                         flow[idx].prev = NULL;
 846 
 847                         if (num_flows > 1) {
 848                             flow[idx].next = &flow[idx + 1];
 849                         } else {
 850                             flow[idx].next = NULL;
 851                         }
 852                     } else {
 853                         flow[idx].prev = &flow[idx - 1];
 854 
 855                         if (idx < num_flows - 1) {
 856                             flow[idx].next = &flow[idx + 1];
 857                         } else {
 858                             flow[idx].next = NULL;
 859                         }
 860                     }
 861                     flow[idx].r_next = NULL;
 862 
 863                     idx++;
 864 
 865                     curr = curr->next;
 866                 }
 867 
 868                 pthread_mutex_unlock(&flow_table[i].lock);
 869             }
 870         }
 871         break;
 872     default:
 873         break;
 874     }
 875 
 876     return 0;
 877 }
