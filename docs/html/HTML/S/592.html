  29 #if GLIB_MAJOR_VERSION >= 2
  30 /*
  31  * XXX - "isprint()" can return "true" for non-ASCII characters, but
  32  * those don't work with GTK+ 1.3 or later, as they take UTF-8 strings
  33  * as input.  Until we fix up Wireshark to properly handle non-ASCII
  34  * characters in all output (both GUI displays and text printouts)
  35  * in those versions of GTK+, we work around the problem by escaping
  36  * all characters that aren't printable ASCII.
  37  *
  38  * We don't know what version of GTK+ we're using, as dissectors don't
  39  * use any GTK+ stuff; we use GLib as a proxy for that, with GLib 2.x
  40  * implying GTK+ 1.3 or later (we don't support GLib 1.3[.x]).
  41  */
  42 #undef isprint
  43 #define isprint(c) (c >= 0x20 && c < 0x7f)
