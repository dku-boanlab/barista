 128 /*
 129  * returns the time until the next packet send.
 130  * @param generator the generator mumber for which we
 131  * have to send next packet
 132  * @return number of milliseconds until next packet.
 133  */
 134 int
 135 get_next_pkt(int num_generator) {
 136   int i, tm, min_tm = 1000, min_generator = -1;
 137   struct timeval now;
 138   gettimeofday(&now, NULL);
 139   for (i = 0; i < num_generator; i++) {
 140     if(( (tm = time_diff(&now, &generator_state[i]->timestamp)) <=0) && (tm < min_tm)) {
 141       min_tm = tm;
 142       min_generator = i;
 143     }
 144   }
 145   return min_generator;
 146 }

 148 int
 149 send_pkt(struct oflops_context *ctx, int ix) {
 150   struct timeval now;
 151   struct pkt_details *state = generator_state[ix];
 152 
 153   gettimeofday(&now, NULL);
 154   state->pktgen->seq_num = htonl(state->seq_num++);
 155   state->pktgen->time.tv_sec = htonl(now.tv_sec);
 156   state->pktgen->time.tv_sec = htonl(now.tv_usec);
 157     oflops_send_raw_mesg(ctx, state->traffic_gen, state->data, state->data_len);
 158     //printf("%d: %ld.%06ld\n", state->traffic_gen, now.tv_sec, now.tv_usec);
 159   add_time(&state->timestamp, 0, ctx->channels[state->traffic_gen].det->delay);
 160   return 1;
 161 }

 164 read_mac_addr(uint8_t *addr, char *str) {
 165   char *p = str, *tmp;
 166   int i = 0;
 167   do {    
 168     tmp = index(p, ':');
 169     if(tmp != NULL) {
 170       *tmp = '\0';
 171       tmp++;
 172     }
 173     addr[i] = (uint8_t)strtol(p, NULL, 16);
 174     i++;
 175     p = tmp;
 176   } while (p!= NULL);
 177 
 178   //fprintf(stderr, "mac %x:%x:%x:%x:%x:%x\n", addr[0], addr[1], addr[2], addr[3], addr[4], addr[5]);
 179   return 0;

 182 int
 183 innitialize_generator_packet(struct pkt_details *state, struct traf_gen_det *det) {  
 184   state->data = (void *)xmalloc(det->pkt_size); 
 185   state->data_len = det->pkt_size;
 186 
 187   bzero((void *)state->data, state->data_len);
 188   if(state->data_len < sizeof(struct ether_vlan_header) + sizeof(struct iphdr) + sizeof(struct tcphdr)) {
 189     printf("packet size is too small\n");
 190     return 0;
 191   }
 192   //ethernet header with default values
 193   state->eth_vlan = (struct ether_vlan_header *) state->data;
 194   state->eth = (struct ether_header *) state->data;
 195   read_mac_addr(state->eth_vlan->ether_dhost, det->mac_dst);
 196   read_mac_addr(state->eth_vlan->ether_shost, det->mac_src);
 197   if(det->vlan != 0 && det->vlan != 0xffff) {
 198     state->eth_vlan->tpid = htons(0x8100);
 199     state->eth_vlan->vid = htons(det->vlan) >>4;
 200     state->eth_vlan->ether_type = htons(ETHERTYPE_IP);
 201     state->ip = (struct iphdr *)(state->data + sizeof(struct ether_vlan_header));
 202     state->ip->tot_len=htons(state->data_len - sizeof(struct ether_vlan_header)); 
 203   state->udp = (struct udphdr *)
 204     (state->data + sizeof(struct ether_vlan_header) + sizeof(struct iphdr));
 205   state->udp->len = htons(state->data_len - sizeof(struct ether_vlan_header) - sizeof(struct iphdr));
 206   state->pktgen = (struct pktgen_hdr *)
 207     (state->data + sizeof(struct ether_vlan_header) + sizeof(struct iphdr) + sizeof(struct udphdr));
 208   } else {
 209     state->eth->ether_type = htons(ETHERTYPE_IP);
 210     state->ip = (struct iphdr *)(state->data + sizeof(struct ether_header));
 211     state->ip->tot_len=htons(state->data_len - sizeof(struct ether_header)); 
 212     state->udp = (struct udphdr *)
 213       (state->data + sizeof(struct ether_header) + sizeof(struct iphdr));
 214     state->udp->len = htons(state->data_len - sizeof(struct ether_header) - sizeof(struct iphdr));
 215     state->pktgen = (struct pktgen_hdr *)
 216       (state->data + sizeof(struct ether_header) + sizeof(struct iphdr) + sizeof(struct udphdr));
 217   }
 218   //ip header with default values
 219   state->ip->protocol=1;
 220   state->ip->ihl=5;
 221   state->ip->version=4;
 222   //state->ip->check = htons(0x9a97);
 223   //total packet size without ethernet header
 224   state->ip->ttl = 100;
 225   state->ip->protocol = IPPROTO_UDP; //udp protocol
 226   state->ip->saddr = inet_addr(det->src_ip); 
 227   state->ip->daddr = inet_addr(det->dst_ip_min); //test.nw_dst;
 228   
 229   state->ip->check=ip_sum_calc(20, state->ip);
 230 
 231   state->udp->source = htons(det->udp_src_port);
 232   state->udp->dest = htons(det->udp_dst_port);
 233   
 234   state->pktgen->magic = 0xbe9be955;
 235 
 236   return 1;
 237 }

 239 int 
 240 init_traffic_gen(oflops_context *ctx) {
 241   int num_generator = 0;
 242   int ix;
 243   
 244   srand(getpid());
 245 
 246   for(ix = 0; ix < ctx->n_channels; ix++) {
 247     if(ctx->channels[ix].det != NULL) {
 248       num_generator++;
 249       generator_state = (struct pkt_details **)realloc(generator_state, num_generator);
 250       generator_state[num_generator - 1] = (struct pkt_details *)xmalloc(sizeof(struct pkt_details));
 251       generator_state[num_generator - 1]->traffic_gen = ix;
 252       gettimeofday(&(generator_state[num_generator - 1]->timestamp), NULL);
 253       add_time(&(generator_state[num_generator - 1]->timestamp), 0,  (rand()%1000000));
 254       innitialize_generator_packet(generator_state[num_generator - 1], ctx->channels[ix].det);
 255     }
 256   }
 257   return num_generator;
 258 }

 261 start_user_traffic_generator(oflops_context *ctx) {
 262   int num_generator = init_traffic_gen(ctx), generator;  
 263   while(ctx->should_end == 0) {
 264     generator = get_next_pkt(num_generator);
 265     if(generator >= 0 ) {

 267     }

 269   return 1;

 373 struct pktgen_hdr *
 374 extract_pktgen_pkt(unsigned char *b, int len, struct flow *fl) {
 375   struct ether_header *ether = (struct ether_header *)b;
 376   struct ether_vlan_header *ether_vlan = (struct ether_vlan_header *)b;
 377   
 378   if( (ntohs(ether->ether_type) == 0x8100) && (ntohs(ether_vlan->ether_type) == 0x0800)) {
 379     b = b + sizeof(struct ether_vlan_header);
 380     len -= sizeof(struct ether_vlan_header);
 381   } else if(ntohs(ether->ether_type) == 0x0800) {
 382     b = b + sizeof(struct ether_header);
 383     len -= sizeof(struct ether_header);
 384   } else {
 385     printf("Invalid ether type found: %x\n", ntohs(ether->ether_type));
 386     return NULL;
 387   }
 388 
 389   struct iphdr *ip_p = (struct iphdr *) b;
 390   if (len < 4*ip_p->ihl) {
 391     printf("capture too small for ip: %d\n", len);
 392     return 0;
 393   }
 394   b = b + 4*ip_p->ihl;
 395   len -=  4*ip_p->ihl;
 396 
 397   if(fl!= NULL) {
 398     //ethenet fields
 399     memcpy(fl->dl_src, ether->ether_shost, 6);
 400     memcpy(fl->dl_dst, ether->ether_dhost, 6);
 401     if(ntohs(ether->ether_type) == 0x8100) {
 402       fl->dl_type = ntohs(ether_vlan->ether_type);
 403       fl->dl_vlan = (0x0FFF&ntohs(ether_vlan->vid<<4));
 404     } else {
 405       fl->dl_type = ntohs(ether->ether_type);
 406       fl->dl_vlan = 0;
 407     }
 408 
 409     //ip fields
 410     fl->nw_src = ip_p->saddr;
 411     fl->nw_dst = ip_p->daddr;
 412     
 413     //tcp/udp fields
 414     struct udphdr *udp_p = (struct udphdr *)b;
 415     fl->tp_src = ntohs(udp_p->source);
 416     fl->tp_dst = ntohs(udp_p->dest);
 417     
 418   }
 419 
 420   b += sizeof(struct udphdr);
 421 
 422   return (struct pktgen_hdr *)b;
 423 }
