  45 /* Behavior when the connection to the controller fails. */
  46 enum fail_mode {
  47     FAIL_OPEN,                  /* Act as learning switch. */
  48     FAIL_CLOSED                 /* Drop all packets. */
  49 };

  51 /* Maximum number of management connection listeners. */
  52 #define MAX_MGMT 8
  53 
  54 #define MAX_CONTROLLERS 3

  56 /* Settings that may be configured by the user. */
  57 struct settings {
  58     /* Overall mode of operation. */
  59     bool discovery;           /* Discover the controller automatically? */
  60     bool in_band;             /* Connect to controller in-band? */
  61 
  62     /* Related vconns and network devices. */
  63     const char *dp_name;        /* Local datapath. */
  64     int num_controllers;        /* Number of configured controllers. */
  65     const char *controller_names[MAX_CONTROLLERS]; /* Controllers (if not discovery mode). */
  66     const char *listener_names[MAX_MGMT]; /* Listen for mgmt connections. */
  67     size_t n_listeners;          /* Number of mgmt connection listeners. */
  68     const char *monitor_name;   /* Listen for traffic monitor connections. */
  69 
  70     /* Failure behavior. */
  71     enum fail_mode fail_mode; /* Act as learning switch if no controller? */
  72     int max_idle;             /* Idle time for flows in fail-open mode. */
  73     int probe_interval;       /* # seconds idle before sending echo request. */
  74     int max_backoff;          /* Max # seconds between connection attempts. */
  75 
  76     /* Packet-in rate-limiting. */
  77     int rate_limit;           /* Tokens added to bucket per second. */
  78     int burst_limit;          /* Maximum number token bucket size. */
  79 
  80     /* Discovery behavior. */
  81     regex_t accept_controller_regex;  /* Controller vconns to accept. */
  82     const char *accept_controller_re; /* String version of regex. */
  83     bool update_resolv_conf;          /* Update /etc/resolv.conf? */
  84 
  85     /* Spanning tree protocol. */
  86     bool enable_stp;
  87 
  88     /* Emergency flow protection/restoration behavior. */
  89     bool emerg_flow;
  90 };

  98 struct relay {
  99     struct list node;
 100 
 101 #define HALF_LOCAL 0
 102 #define HALF_REMOTE 1
 103     struct half halves[2];
 104 
 105     /* The secchan has a primary connection (relay) to an OpenFlow controller.
 106      * This primary connection actually makes two connections to the datapath:
 107      * one for OpenFlow requests and responses, and one that is only used for
 108      * receiving asynchronous events such as 'ofp_packet_in' events.  This
 109      * design keeps replies to OpenFlow requests from being dropped by the
 110      * kernel due to a flooded network device.
 111      *
 112      * The secchan may also have any number of secondary "management"
 113      * connections (relays).  These connections do not receive asychronous
 114      * events and thus have a null 'async_rconn'. */
 115     bool is_mgmt_conn;          /* Is this a management connection? */
 116     struct rconn *async_rconn;  /* For receiving asynchronous events. */
 117 };
