  14 void 
  15 myargs_usage(struct myargs options[], char * title, char * s1, char * s2, int exit_val)
  16 {
  17     #define AFMT "-20"
  18     #define OFMT "6"
  19     #define FMT "-40"
  20     struct myargs * optptr;
  21     if(s1)
  22         fprintf(stderr, "%s", s1);
  23     if(s2)
  24         fprintf(stderr, " %s", s2);
  25     if (s1 || s2)
  26         fprintf(stderr, "\n");
  27     if(title)
  28         fprintf(stderr, "%s\n", title);
  29     for( optptr = &options[0]; optptr->name != NULL ; optptr++)
  30     {
  31         fprintf(stderr, "   -%c/--%"AFMT"s  ", optptr->shortname, optptr->name);
  32         switch(optptr->type)
  33         {
  34             case MYARGS_NONE:
  35                 fprintf(stderr, " %"OFMT"s %"FMT"s\n", "",optptr->comment);
  36                 break;
  37             case MYARGS_INTEGER:
  38                 fprintf(stderr, " %"OFMT"s %"FMT"s (%d)\n", "<int>", optptr->comment, optptr->default_val.integer);
  39                 break;
  40             case MYARGS_FLAG:
  41                 fprintf(stderr, " %"OFMT"s %"FMT"s (%s)\n", "", optptr->comment, optptr->default_val.flag? "on" : "off");
  42                 break;
  43             case MYARGS_STRING:
  44                 fprintf(stderr, " %"OFMT"s %"FMT"s (\"%s\")\n", "<str>", optptr->comment, optptr->default_val.string);
  45                 break;
  46             case MYAGRS_DECIMAL:
  47                 fprintf(stderr, " %"OFMT"s %"FMT"s (%lf)\n", "<real>", optptr->comment, optptr->default_val.decimal);
  48                 break;
  49             default: 
  50                 fprintf(stderr, "--- unhandled argument type %d", optptr->type);
  51                 abort();
  52         };
  53     }
  54     fprintf(stderr, "\n");
  55     exit(exit_val);
  56 }

  58 const struct option *
  59 myargs_to_long(struct myargs options[])
  60 {
  61     struct option * longopts;
  62     int n = count_options(options);
  63     int i;
  64     longopts = malloc(sizeof(struct option) * (n+1));
  65     for(i=0;i<=n;i++)
  66     {
  67         if(options[i].name)
  68             longopts[i].name = strdup(options[i].name);
  69         else 
  70             longopts[i].name = NULL;
  71         if( options[i].type == MYARGS_NONE ) 
  72             longopts[i].has_arg = no_argument;
  73         else if ( options[i].type == MYARGS_FLAG)
  74             longopts[i].has_arg = optional_argument;
  75         else
  76             longopts[i].has_arg = required_argument;
  77         longopts[i].flag =  NULL;
  78         longopts[i].val  = options[i].shortname;
  79     }
  80     return longopts;
  81 }

  83 char * 
  84 myargs_to_short(struct myargs options[])
  85 {
  86     char * shortargs;
  87     int n = count_options(options);
  88     int i;
  89     int len=0;
  90     int max = n*2 + 1;
  91     shortargs = malloc(max);
  92     for(i=0; i< n; i++)
  93     {
  94         len+= snprintf(&shortargs[len], max-len, "%c", 
  95                 options[i].shortname);
  96         if(options[i].type != MYARGS_NONE) {
  97             if(options[i].type == MYARGS_FLAG)
  98                 len+= snprintf(&shortargs[len], max-len, "::");
  99             else
 100                 len+= snprintf(&shortargs[len], max-len, ":");
 101         }
 102     }
 103     shortargs[len]=0;
 104     return shortargs;
 105 }

 107 int 
 108 count_options(struct myargs options[])
 109 {
 110     int count = 0;
 111     struct myargs *opt;
 112     for ( opt = &options[0]; opt->name != NULL; opt++)
 113         count++;
 114     return count;
 115 }
 116 /************************************************************************/
 117 int 
 118 find_arg(struct myargs options[], char * arg)
 119 {
 120     int i = 0;
 121     struct myargs *opt;
 122     for ( opt = &options[0]; opt->name != NULL; opt++)
 123     {
 124         if(!strcmp(opt->name, arg))
 125             return i;
 126         i++;
 127     }
 128     return -1;
 129 }
