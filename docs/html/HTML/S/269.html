 252 void
 253 nf2_reset_card(struct net_device *dev)
 254 {
 255         volatile unsigned int val;
 256 
 257         if (dev == NULL) {
 258                 return;
 259         }
 260 
 261         /* If we are operating on a NetFPGA enabled box, reset the card */
 262         printk(KERN_INFO "openflowswitch-netfpga2: Resetting the NetFPGA.\n");
 263         nf2k_reg_read(dev, WDT_CPCI_REG_CTRL, (void *)&val);
 264         val |= 0x100;
 265         nf2k_reg_write(dev, WDT_CPCI_REG_CTRL, (void *)&val);
 266         printk(KERN_INFO "openflowswitch-netfpga2: Reset the NetFPGA.\n");
 267         ssleep(2);
 268 }

 270 void
 271 nf2_clear_watchdog(struct net_device *dev)
 272 {
 273         volatile unsigned int enable_status;
 274 
 275 #ifndef NF2_WATCHDOG
 276         return;
 277 #endif
 278         if (dev == NULL) {
 279                 return;
 280         }
 281 
 282         nf2k_reg_read(dev, WDT_ENABLE_FLG_REG, (void *)&enable_status);
 283         enable_status &= 0x1;
 284 
 285         if (enable_status == WATCHDOG_DISABLE) {
 286                 enable_status = WATCHDOG_ENABLE;
 287                 nf2k_reg_write(dev, WDT_ENABLE_FLG_REG, (void *)&enable_status);
 288         }
 289         return;
 290 }

 292 /* Write a wildcard entry to the specified device and row. The row consists of
 293  * the actual entry, its mask that specifies wildcards, as well as the action(s)
 294  * to be taken if the row is matched
 295  */
 296 int
 297 nf2_write_of_wildcard(struct net_device *dev, int row,
 298                       nf2_of_entry_wrap *entry, nf2_of_mask_wrap *mask,
 299                       nf2_of_action_wrap *action)
 300 {
 301         int i;
 302         int val;
 303         struct timeval t;
 304 
 305         NF2DEBUGMSG("** Begin wildcard entry write to row: %i\n", row);
 306         log_entry(entry);
 307         log_mask(mask);
 308         log_action(action);
 309         log_entry_raw(entry);
 310         log_mask_raw(mask);
 311         log_action_raw(action);
 312 
 313         for (i = 0; i < NF2_OF_ENTRY_WORD_LEN; ++i) {
 314                 nf2k_reg_write(dev,
 315                                OPENFLOW_WILDCARD_LOOKUP_CMP_0_REG
 316                                + (4 * i), &(entry->raw[i]));
 317         }
 318 
 319         for (i = 0; i < NF2_OF_MASK_WORD_LEN; ++i) {
 320                 nf2k_reg_write(dev,
 321                                OPENFLOW_WILDCARD_LOOKUP_CMP_MASK_0_REG
 322                                + (4 * i), &(mask->raw[i]));
 323         }
 324 
 325         for (i = 0; i < NF2_OF_ACTION_WORD_LEN; ++i) {
 326                 nf2k_reg_write(dev,
 327                                OPENFLOW_WILDCARD_LOOKUP_ACTION_0_REG
 328                                + (4 * i), &(action->raw[i]));
 329         }
 330 
 331         // Reset the stats for the row
 332         val = 0;
 333         nf2k_reg_write(dev,
 334                        OPENFLOW_WILDCARD_LOOKUP_BYTES_HIT_0_REG + (4 * row),
 335                        &val);
 336         nf2k_reg_write(dev, OPENFLOW_WILDCARD_LOOKUP_PKTS_HIT_0_REG + (4 * row),
 337                        &val);
 338         nf2k_reg_write(dev,
 339                        OPENFLOW_WILDCARD_LOOKUP_LAST_SEEN_TS_0_REG + (4 * row),
 340                        &val);
 341         nf2k_reg_write(dev, OPENFLOW_WILDCARD_LOOKUP_WRITE_ADDR_REG, &row);
 342 
 343         do_gettimeofday(&t);
 344         NF2DEBUGMSG("** End wildcard entry write to row: %i time: %i.%i\n",
 345                     row, (int)t.tv_sec, (int)t.tv_usec);
 346 
 347         return 0;
 348 }

 350 int
 351 nf2_write_of_exact(struct net_device *dev, int row,
 352                    nf2_of_entry_wrap *entry, nf2_of_action_wrap *action)
 353 {
 354         int i;
 355         int val;
 356         struct timeval t;
 357         unsigned int index = row << 7;
 358 
 359         NF2DEBUGMSG("** Begin exact match entry write to row: %i\n", row);
 360         log_entry(entry);
 361         log_action(action);
 362         log_entry_raw(entry);
 363         log_action_raw(action);
 364 
 365         for (i = 0; i < NF2_OF_ENTRY_WORD_LEN; ++i) {
 366                 nf2k_reg_write(dev, SRAM_BASE_ADDR + index
 367                                + (4 * i), &(entry->raw[i]));
 368         }
 369 
 370         // blank out the counters
 371         val = 0;
 372         for (i = 0; i < NF2_OF_EXACT_COUNTERS_WORD_LEN; ++i) {
 373                 nf2k_reg_write(dev, SRAM_BASE_ADDR + index
 374                                + sizeof(nf2_of_entry_wrap)
 375                                + (4 * i), &val);
 376         }
 377 
 378         // write the actions
 379         for (i = 0; i < NF2_OF_ACTION_WORD_LEN; ++i) {
 380                 nf2k_reg_write(dev, SRAM_BASE_ADDR + index
 381                                + sizeof(nf2_of_entry_wrap)
 382                                + sizeof(nf2_of_exact_counters_wrap)
 383                                + (4 * i), &(action->raw[i]));
 384         }
 385 
 386         do_gettimeofday(&t);
 387         NF2DEBUGMSG("** End exact match entry write to row: %i time: %i.%i\n",
 388                     row, (int)t.tv_sec, (int)t.tv_usec);
 389 
 390         return 0;
 391 }

 395 nf2_modify_write_of_wildcard(struct net_device *dev, int row,

 423                       &last_reg_val);
 424 
 425         for (i = 0; i < NF2_OF_ENTRY_WORD_LEN; ++i) {
 426                 nf2k_reg_write(dev, OPENFLOW_WILDCARD_LOOKUP_CMP_0_REG
 427                                + (4 * i), &(entry->raw[i]));
 428         }
 429 
 430         for (i = 0; i < NF2_OF_MASK_WORD_LEN; ++i) {
 431                 nf2k_reg_write(dev, OPENFLOW_WILDCARD_LOOKUP_CMP_MASK_0_REG

 433         }

 436                 nf2k_reg_write(dev, OPENFLOW_WILDCARD_LOOKUP_ACTION_0_REG
 437                                + (4 * i), &(action->raw[i]));
 438         }
 439 
 440         nf2k_reg_write(dev,
 441                        OPENFLOW_WILDCARD_LOOKUP_BYTES_HIT_0_REG + (4 * row),
 442                        &bytes_reg_val);
 443         nf2k_reg_write(dev, OPENFLOW_WILDCARD_LOOKUP_PKTS_HIT_0_REG + (4 * row),
 444                        &pkts_reg_val);
 445         nf2k_reg_write(dev,
 446                        OPENFLOW_WILDCARD_LOOKUP_LAST_SEEN_TS_0_REG + (4 * row),
 447                        &last_reg_val);
 448         nf2k_reg_write(dev, OPENFLOW_WILDCARD_LOOKUP_WRITE_ADDR_REG, &row);
 449 
 450         do_gettimeofday(&t);
 451         NF2DEBUGMSG
 452                 ("** End wildcard modified action write to row: %i time: %i.%i\n",
 453                  row, (int)t.tv_sec, (int)t.tv_usec);
 454         NF2DEBUGMSG("   Bytes hit count: %d\n", bytes_reg_val);
 455         NF2DEBUGMSG("   Pkts  hit count: %d\n", pkts_reg_val);
 456         NF2DEBUGMSG("   Last seen      : %d\n", last_reg_val);
 457 
 458         return 0;

 491 nf2_get_exact_packet_count(struct net_device *dev, int row)
 492 {
 493         unsigned int val = 0;
 494         unsigned int index = 0;
 495 
 496         /* TODO: Need to scrape data from all 4 registers
 497          * in the case of a wildcarded source port and
 498          * forward all action type
 499          */
 500         nf2_of_exact_counters_wrap counters;
 501         memset(&counters, 0, sizeof(nf2_of_exact_counters_wrap));
 502 
 503         // build the index to our counters
 504         index = row << 7;
 505 
 506         // Read the first word into our struct, to not disturb the byte count
 507         nf2k_reg_read(dev,
 508                       SRAM_BASE_ADDR + index + sizeof(nf2_of_entry_wrap),

 512         NF2DEBUGMSG("** Exact match packet count request row: %i count: %i\n",
 513                     row, val);

 635 struct nf2_device_info *
 636 nf2_get_device_info(struct net_device *dev)
 637 {
 638         struct nf2_device_info *nf2devinfo;
 639         int i;
 640 
 641         nf2devinfo = kzalloc(sizeof(struct nf2_device_info), GFP_KERNEL);
 642         if (nf2devinfo == NULL)
 643                 return NULL;
 644 
 645         // Read the version and revision
 646         nf2k_reg_read(dev, DEV_ID_DEVICE_ID_REG, &(nf2devinfo->nf2_device_id));
 647         nf2k_reg_read(dev, DEV_ID_REVISION_REG, &(nf2devinfo->nf2_device_rev));
 648 
 649         // Read the design name string
 650         for (i = 0; i < (DEVICE_STR_LEN / 4) - 2; i++) {
 651                 nf2k_reg_read(dev, DEV_ID_DEV_STR_0_REG + i * 4,
 652                               (uint32_t *)(nf2devinfo->nf2_device_str + i * 4));
 653                 *(uint32_t *)(nf2devinfo->nf2_device_str + i * 4)
 654                         = ntohl(*(uint32_t *)
 655                                 (nf2devinfo->nf2_device_str + i * 4));
 656         }
 657         nf2devinfo->nf2_device_str[DEVICE_STR_LEN - 1] = '\0';
 658 
 659         return nf2devinfo;
 660 }

 804 struct nf2_all_ports_info *
 805 nf2_get_all_ports_info(struct net_device *dev)
 806 {
 807         struct nf2_all_ports_info *nf2portinfo;
 808         struct nf2_all_ports_info_addr *nf2addr;
 809         int i;
 810 
 811         nf2portinfo = kzalloc(sizeof(struct nf2_all_ports_info), GFP_KERNEL);
 812         if (nf2portinfo == NULL)
 813                 return NULL;
 814         nf2addr = nf2_get_all_ports_info_addr();
 815         if (nf2addr == NULL)
 816                 return NULL;
 817 
 818         for (i = 0; i < NF2_PORT_NUM; i++) {
 819                 nf2k_reg_read(dev, nf2addr->rx_q_num_pkts_stored_reg[i],
 820                               &(nf2portinfo->port[i].rx_q_num_pkts_stored));
 821                 nf2k_reg_read(dev, nf2addr->rx_q_num_pkts_dropped_full_reg[i],
 822                               &(nf2portinfo
 823                                 ->port[i].rx_q_num_pkts_dropped_full));
 824                 nf2k_reg_read(dev, nf2addr->rx_q_num_pkts_dropped_bad_reg[i],
 825                               &(nf2portinfo
 826                                 ->port[i].rx_q_num_pkts_dropped_bad));
 827                 nf2k_reg_read(dev, nf2addr->rx_q_num_words_pushed_reg[i],
 828                               &(nf2portinfo->port[i].rx_q_num_words_pushed));
 829                 nf2k_reg_read(dev, nf2addr->rx_q_num_bytes_pushed_reg[i],
 830                               &(nf2portinfo->port[i].rx_q_num_bytes_pushed));
 831                 nf2k_reg_read(dev, nf2addr->rx_q_num_pkts_dequeued_reg[i],
 832                               &(nf2portinfo->port[i].rx_q_num_pkts_dequeued));
 833                 nf2k_reg_read(dev, nf2addr->rx_q_num_pkts_in_queue_reg[i],
 834                               &(nf2portinfo->port[i].rx_q_num_pkts_in_queue));
 835                 nf2k_reg_read(dev, nf2addr->tx_q_num_pkts_in_queue_reg[i],
 836                               &(nf2portinfo->port[i].tx_q_num_pkts_in_queue));
 837                 nf2k_reg_read(dev, nf2addr->tx_q_num_pkts_sent_reg[i],
