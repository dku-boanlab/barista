  41 /* Initializes 'b' as an empty ofpbuf that contains the 'allocated' bytes of
  42  * memory starting at 'base'.
  43  *
  44  * 'base' should ordinarily be the first byte of a region obtained from
  45  * malloc(), but in circumstances where it can be guaranteed that 'b' will
  46  * never need to be expanded or freed, it can be a pointer into arbitrary
  47  * memory. */
  48 void
  49 ofpbuf_use(struct ofpbuf *b, void *base, size_t allocated)
  50 {
  51     b->base = b->data = base;
  52     b->allocated = allocated;
  53     b->size = 0;
  54     b->l2 = b->l3 = b->l4 = b->l7 = NULL;
  55     b->next = NULL;
  56     b->private = NULL;
  57 }
  58 
  59 /* Initializes 'b' as an empty ofpbuf with an initial capacity of 'size'
  60  * bytes. */
  61 void
  62 ofpbuf_init(struct ofpbuf *b, size_t size)
  63 {
  64     ofpbuf_use(b, size ? xmalloc(size) : NULL, size);
  65 }

 118 /* Returns the number of bytes of headroom in 'b', that is, the number of bytes
 119  * of unused space in ofpbuf 'b' before the data that is in use.  (Most
 120  * commonly, the data in a ofpbuf is at its beginning, and thus the ofpbuf's
 121  * headroom is 0.) */
 122 size_t
 123 ofpbuf_headroom(struct ofpbuf *b) 
 124 {
 125     return (char*)b->data - (char*)b->base;
 126 }
 127 
 128 /* Returns the number of bytes that may be appended to the tail end of ofpbuf
 129  * 'b' before the ofpbuf must be reallocated. */
 130 size_t
 131 ofpbuf_tailroom(struct ofpbuf *b) 
 132 {
 133     return (char*)ofpbuf_end(b) - (char*)ofpbuf_tail(b);
 134 }

 136 /* Ensures that 'b' has room for at least 'size' bytes at its tail end,
 137  * reallocating and copying its data if necessary. */
 138 void
 139 ofpbuf_prealloc_tailroom(struct ofpbuf *b, size_t size) 
 140 {
 141     if (size > ofpbuf_tailroom(b)) {
 142         size_t new_allocated = b->allocated + MAX(size, 64);
 143         void *new_base = xmalloc(new_allocated);
 144         uintptr_t base_delta = (char*)new_base - (char*)b->base;
 145         memcpy(new_base, b->base, b->allocated);
 146         free(b->base);
 147         b->base = new_base;
 148         b->allocated = new_allocated;
 149         b->data = (char*)b->data + base_delta;
 150         if (b->l2) {
 151             b->l2 = (char*)b->l2 + base_delta;
 152         }
 153         if (b->l3) {
 154             b->l3 = (char*)b->l3 + base_delta;
 155         }
 156         if (b->l4) {
 157             b->l4 = (char*)b->l4 + base_delta;
 158         }
 159         if (b->l7) {
 160             b->l7 = (char*)b->l7 + base_delta;
 161         }
 162     }
 163 }

 165 void
 166 ofpbuf_prealloc_headroom(struct ofpbuf *b, size_t size) 
 167 {
 168     assert(size <= ofpbuf_headroom(b));
 169 }

 171 /* Appends 'size' bytes of data to the tail end of 'b', reallocating and
 172  * copying its data if necessary.  Returns a pointer to the first byte of the
 173  * new data, which is left uninitialized. */
 174 void *
 175 ofpbuf_put_uninit(struct ofpbuf *b, size_t size) 
 176 {
 177     void *p;
 178     ofpbuf_prealloc_tailroom(b, size);
 179     p = ofpbuf_tail(b);
 180     b->size += size;
 181     return p;
 182 }

 184 /* Appends 'size' zeroed bytes to the tail end of 'b'.  Data in 'b' is
 185  * reallocated and copied if necessary.  Returns a pointer to the first byte of
 186  * the data's location in the ofpbuf. */
 187 void *
 188 ofpbuf_put_zeros(struct ofpbuf *b, size_t size)
 189 {
 190     void *dst = ofpbuf_put_uninit(b, size);
 191     memset(dst, 0, size);
 192     return dst;
 193 }

 195 /* Appends the 'size' bytes of data in 'p' to the tail end of 'b'.  Data in 'b'
 196  * is reallocated and copied if necessary.  Returns a pointer to the first
 197  * byte of the data's location in the ofpbuf. */
 198 void *
 199 ofpbuf_put(struct ofpbuf *b, const void *p, size_t size)

 201     void *dst = ofpbuf_put_uninit(b, size);

 206 /* Reserves 'size' bytes of headroom so that they can be later allocated with
 207  * ofpbuf_push_uninit() without reallocating the ofpbuf. */
 208 void
 209 ofpbuf_reserve(struct ofpbuf *b, size_t size) 
 210 {
 211     assert(!b->size);
 212     ofpbuf_prealloc_tailroom(b, size);
 213     b->data = (char*)b->data + size;
 214 }

 216 void *
 217 ofpbuf_push_uninit(struct ofpbuf *b, size_t size) 
 218 {
 219     ofpbuf_prealloc_headroom(b, size);
 220     b->data = (char*)b->data - size;
 221     b->size += size;
 222     return b->data;
 223 }

 225 void *
 226 ofpbuf_push(struct ofpbuf *b, const void *p, size_t size) 
 227 {
 228     void *dst = ofpbuf_push_uninit(b, size);

 233 /* If 'b' contains at least 'offset + size' bytes of data, returns a pointer to
 234  * byte 'offset'.  Otherwise, returns a null pointer. */
 235 void *
 236 ofpbuf_at(const struct ofpbuf *b, size_t offset, size_t size) 
 237 {
 238     return offset + size <= b->size ? (char *) b->data + offset : NULL;
 239 }

 273 /* Removes 'size' bytes from the head end of 'b', which must contain at least
 274  * 'size' bytes of data.  Returns the first byte of data removed. */
 275 void *
 276 ofpbuf_pull(struct ofpbuf *b, size_t size) 
 277 {
 278     void *data = b->data;
 279     assert(b->size >= size);
 280     b->data = (char*)b->data + size;
 281     b->size -= size;
 282     return data;
 283 }
 284 
 285 /* If 'b' has at least 'size' bytes of data, removes that many bytes from the
 286  * head end of 'b' and returns the first byte removed.  Otherwise, returns a
 287  * null pointer without modifying 'b'. */
 288 void *
 289 ofpbuf_try_pull(struct ofpbuf *b, size_t size) 
 290 {
 291     return b->size >= size ? ofpbuf_pull(b, size) : NULL;
 292 }
