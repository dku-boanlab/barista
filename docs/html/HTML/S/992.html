  55 /**
  56  * \brief Function to check any conflicts between a new flow and existing flows
  57  * \param rule New flow
  58  * \return CONFLICT or NOT_CONFLICT
  59  */
  60 static int flow_rule_conflict(const flow_t *rule)
  61 {
  62     flow_t list[MAX_MATCH_LIST];
  63 
  64     int curr = 1;
  65     int conflict = NOT_CONFLICT;
  66 
  67     int action = DROP;
  68     int num_actions = rule->num_actions;
  69 
  70     memmove(&list[curr++], &rule, sizeof(flow_t));
  71 
  72     int i;
  73     for (i=0; i<num_actions; i++) {
  74         int type = rule->action[i].type;
  75 
  76         if (type == ACTION_OUTPUT) {
  77             action = FORWARD;
  78         } else if (type == ACTION_SET_VLAN_VID) {
  79             int j;
  80             for (j=0; j<(2 * (i + 1)); j++, curr++) {
  81                 memmove(&list[curr], &list[curr/2], sizeof(flow_t));
  82                 if ((j % 2) != 0)
  83                     list[curr].vlan_id = rule->action[i].vlan_id;
  84             }
  85         } else if (type == ACTION_SET_VLAN_PCP) {
  86             int j;
  87             for (j=0; j<(2 * (i + 1)); j++, curr++) {
  88                 memmove(&list[curr], &list[curr/2], sizeof(flow_t));
  89                 if ((j % 2) != 0)
  90                     list[curr].vlan_pcp = rule->action[i].vlan_pcp;
  91             }
  92         } else if (type == ACTION_SET_SRC_MAC) {
  93             int j;
  94             for (j=0; j<(2 * (i + 1)); j++, curr++) {
  95                 memmove(&list[curr], &list[curr/2], sizeof(flow_t));
  96                 if ((j % 2) != 0)
  97                     memmove(&list[curr].src_mac, rule->action[i].mac_addr, ETH_ALEN);
  98             }
  99         } else if (type == ACTION_SET_DST_MAC) {
 100             int j;
 101             for (j=0; j<(2 * (i + 1)); j++, curr++) {
 102                 memmove(&list[curr], &list[curr/2], sizeof(flow_t));
 103                 if ((j % 2) != 0)
 104                     memmove(&list[curr].dst_mac, rule->action[i].mac_addr, ETH_ALEN);
 105             }
 106         } else if (type == ACTION_SET_SRC_IP) {
 107             int j;
 108             for (j=0; j<(2 * (i + 1)); j++, curr++) {
 109                 memmove(&list[curr], &list[curr/2], sizeof(flow_t));
 110                 if ((j % 2) != 0)
 111                     list[curr].src_ip = rule->action[i].ip_addr;
 112             }
 113         } else if (type == ACTION_SET_DST_IP) {
 114             int j;
 115             for (j=0; j<(2 * (i + 1)); j++, curr++) {
 116                 memmove(&list[curr], &list[curr/2], sizeof(flow_t));
 117                 if ((j % 2) != 0)
 118                     list[curr].dst_ip = rule->action[i].ip_addr;
 119             }
 120         } else if (type == ACTION_SET_IP_TOS) {
 121             int j;
 122             for (j=0; j<(2 * (i + 1)); j++, curr++) {
 123                 memmove(&list[curr], &list[curr/2], sizeof(flow_t));
 124                 if ((j % 2) != 0)
 125                     list[curr].ip_tos = rule->action[i].ip_tos;
 126             }
 127         } else if (type == ACTION_SET_SRC_PORT) {
 128             int j;
 129             for (j=0; j<(2 * (i + 1)); j++, curr++) {
 130                 memmove(&list[curr], &list[curr/2], sizeof(flow_t));
 131                 if ((j % 2) != 0)
 132                     list[curr].src_port = rule->action[i].port;
 133             }
 134         } else if (type == ACTION_SET_DST_PORT) {
 135             int j;
 136             for (j=0; j<(2 * (i + 1)); j++, curr++) {
 137                 memmove(&list[curr], &list[curr/2], sizeof(flow_t));
 138                 if ((j % 2) != 0)
 139                     list[curr].dst_port = rule->action[i].port;
 140             }
 141         }
 142     }
 143 
 144     for (i=(curr/2); i<curr; i++) {
 145         rule_table_t *rule_tbl = &rule_table[rule->dpid % __DEFAULT_TABLE_SIZE];
 146         flow_t *target = &list[i];
 147 
 148         pthread_rwlock_rdlock(&rule_tbl->lock);
 149 
 150         flow_t *curr = rule_tbl->head;
 151         while (curr != NULL) {
 152             if (FLOW_COMPARE(curr, target)) {
 153                 int target_action = DROP;
 154 
 155                 int j;
 156                 for (j=0; j<curr->num_actions; j++) {
 157                     if (curr->action[j].type == ACTION_OUTPUT)
 158                         target_action = FORWARD;
 159                 }
 160 
 161                 if (action != target_action) {
 162                     conflict = CONFLICT;
 163                 }
 164 
 165                 break;
 166             }
 167 
 168             curr = curr->next;
 169         }
 170 
 171         pthread_rwlock_unlock(&rule_tbl->lock);
 172     }
 173 
 174     if (conflict == CONFLICT)
 175         return CONFLICT;
 176     else
 177         return NOT_CONFLICT;
 178 }

 182 /**
 183  * \brief The main function
 184  * \param activated The activation flag of this component
 185  * \param argc The number of arguments
 186  * \param argv Arguments

 188 int conflict_main(int *activated, int argc, char **argv)
 189 {
 190     LOG_INFO(CONFLICT_ID, "Init - Flow rule conflict resolution");
 191 
 192     num_rules = 0;
 193 
 194     rule_table = (rule_table_t *)CALLOC(__DEFAULT_TABLE_SIZE, sizeof(rule_table_t));
 195     if (rule_table == NULL) {
 196         PERROR("calloc");
 197         return -1;
 198     }
 199 
 200     int i;
 201     for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 202         pthread_rwlock_init(&rule_table[i].lock, NULL);
 203     }
 204 
 205     arr_q_init();
 206 
 207     activate();
 208 
 209     return 0;
 210 }

 212 /**
 213  * \brief The cleanup function
 214  * \param activated The activation flag of this component
 215  */
 216 int conflict_cleanup(int *activated)
 217 {
 218     LOG_INFO(CONFLICT_ID, "Clean up - Flow rule conflict resolution");
 219 
 220     deactivate();
 221 
 222     int i;
 223     for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 224         pthread_rwlock_wrlock(&rule_table[i].lock);
 225 
 226         flow_t *curr = rule_table[i].head;
 227         while (curr != NULL) {
 228             flow_t *tmp = curr;
 229             curr = curr->next;
 230             FREE(tmp);
 231         }
 232 
 233         pthread_rwlock_unlock(&rule_table[i].lock);
 234         pthread_rwlock_destroy(&rule_table[i].lock);
 235     }
 236 
 237     FREE(rule_table);
 238 
 239     arr_q_destroy();
 240 
 241     return 0;
 242 }

 254 /**
 255  * \brief The handler function
 256  * \param ev Read-only event
 257  * \param ev_out Read-write event (if this component has the write permission)
 258  */
 259 int conflict_handler(const event_t *ev, event_out_t *ev_out)
 260 {
 261     switch (ev->type) {
 262     case EV_DP_INSERT_FLOW:
 263         PRINT_EV("EV_DP_INSERT_FLOW\n");
 264         {
 265             const flow_t *rule = ev->flow;
 266 
 267             if (flow_rule_conflict(rule)) {
 268                 LOG_WARN(CONFLICT_ID, "Block - a new flow rule is conflict to existing flow rules");
 269                 return -1;
 270             }
 271         }
 272         break;
 273     case EV_FLOW_ADDED:
 274         PRINT_EV("EV_FLOW_ADDED\n");
 275         {
 276             const flow_t *flow = ev->flow;
 277             rule_table_t *rule_tbl = &rule_table[flow->dpid % __DEFAULT_TABLE_SIZE];
 278 
 279             flow_t *new = arr_dequeue();
 280             if (new == NULL) {
 281                 break;
 282             }
 283 
 284             memmove(new, flow, sizeof(flow_t));
 285 
 286             pthread_rwlock_wrlock(&rule_tbl->lock);
 287 
 288             if (rule_tbl->head == NULL) {
 289                 rule_tbl->head = new;
 290                 rule_tbl->tail = new;
 291             } else {
 292                 new->prev = rule_tbl->tail;
 293                 rule_tbl->tail->next = new;
 294                 rule_tbl->tail = new;
 295             }
 296 
 297             num_rules++;
 298 
 299             pthread_rwlock_unlock(&rule_tbl->lock);
 300         }
 301         break;
 302     case EV_FLOW_DELETED:
 303         PRINT_EV("EV_FLOW_DELETED\n");
 304         {
 305             const flow_t *flow = ev->flow;
 306             rule_table_t *rule_tbl = &rule_table[flow->dpid % __DEFAULT_TABLE_SIZE];
 307 
 308             rule_table_t tmp_list = {0};
 309 
 310             pthread_rwlock_wrlock(&rule_tbl->lock);
 311 
 312             flow_t *curr = rule_tbl->head;
 313             while (curr != NULL) {
 314                 if (FLOW_COMPARE(curr, flow)) {
 315                     if (tmp_list.head == NULL) {
 316                         tmp_list.head = curr;
 317                         tmp_list.tail = curr;
 318                         curr->r_next = NULL;
 319                     } else {
 320                         tmp_list.tail->r_next = curr;
 321                         tmp_list.tail = curr;
 322                         curr->r_next = NULL;
 323                     }
 324                 }
 325 
 326                 curr = curr->next;
 327             }
 328 
 329             curr = tmp_list.head;
 330             while (curr != NULL) {
 331                 flow_t *tmp = curr;
 332 
 333                 curr = curr->r_next;
 334 
 335                 if (tmp->prev != NULL && tmp->next != NULL) {
 336                     tmp->prev->next = tmp->next;
 337                     tmp->next->prev = tmp->prev;
 338                 } else if (tmp->prev == NULL && tmp->next != NULL) {
 339                     rule_tbl->head = tmp->next;
 340                     tmp->next->prev = NULL;
 341                 } else if (tmp->prev != NULL && tmp->next == NULL) {
 342                     rule_tbl->tail = tmp->prev;
 343                     tmp->prev->next = NULL;
 344                 } else if (tmp->prev == NULL && tmp->next == NULL) {
 345                     rule_tbl->head = NULL;
 346                     rule_tbl->tail = NULL;
 347                 }
 348 
 349                 num_rules--;
 350 
 351                 arr_enqueue(tmp);
 352             }
 353 
 354             pthread_rwlock_unlock(&rule_tbl->lock);
 355         }
 356         break;
 357     default:
 358         break;
 359     }
 360 
 361     return 0;
 362 }
