  96 /**
  97  * Key item used during the decryption process.
  98  */
  99 typedef struct _AIRPDCAP_KEY_ITEM {
 100         /**
 101          * Type of key. The type will remain unchanged during the
 102          * processing, even if some fields could be changed (e.g., WPA
 103          * fields).
 104          * @note
 105          * You can use constants AIRPDCAP_KEY_TYPE_xxx to indicate the
 106          * key type.
 107          */
 108         UINT8 KeyType;
 109 
 110         /**
 111          * Key data.
 112          * This field can be used for the following decryptographic
 113          * algorithms: WEP-40, with a key of 40 bits (10 hex-digits);
 114          * WEP-104, with a key of 104 bits (or 26 hex-digits); WPA or
 115          * WPA2.
 116          * @note
 117          * For WPA/WPA2, the PMK is calculated from the PSK, and the PSK
 118          * is calculated from the passphrase-SSID pair. You can enter one
 119          * of these 3 values and subsequent fields will be automatically
 120          * calculated.
 121          * @note
 122          * For WPA and WPA2 this implementation will use standards as
 123          * defined in 802.11i (2004) and 802.1X (2004).
 124          */
 125         union AIRPDCAP_KEY_ITEMDATA {
 126                 struct AIRPDCAP_KEY_ITEMDATA_WEP {
 127                         /**
 128                          * The binary value of the WEP key.
 129                          * @note
 130                          * It is accepted a key of length between
 131                          * AIRPDCAP_WEP_KEY_MINLEN and
 132                          * AIRPDCAP_WEP_KEY_MAXLEN. A WEP key
 133                          * standard-compliante should be either 40 bits
 134                          * (10 hex-digits, 5 bytes) for WEP-40 or 104 bits
 135                          * (26 hex-digits, 13 bytes) for WEP-104.
 136                          */
 137                         UCHAR WepKey[AIRPDCAP_WEP_KEY_MAXLEN];
 138                         /**
 139                          * The length of the WEP key. Acceptable range
 140                          * is [AIRPDCAP_WEP_KEY_MINLEN;AIRPDCAP_WEP_KEY_MAXLEN].
 141                          */
 142                         size_t WepKeyLen;
 143                 } Wep;
 144 
 145                 /**
 146                  * WPA/WPA2 key data. Note that the decryption process
 147                  * will use the PMK (equal to PSK), that is calculated
 148                  * from passphrase-SSID pair. You can define one of these
 149                  * three fields and necessary fields will be automatically
 150                  * calculated.
 151                  */
 152                 union AIRPDCAP_KEY_ITEMDATA_WPA {
 153 
 154                         UCHAR Psk[AIRPDCAP_WPA_PSK_LEN];
 155 
 156                         UCHAR Pmk[AIRPDCAP_WPA_PMK_LEN];
 157                 } Wpa;
 158         } KeyData;
 159 
 160         struct AIRPDCAP_KEY_ITEMDATA_PWD {
 161                 /**
 162                  * The string (null-terminated) value of
 163                  * the passphrase.
 164                  */
 165                 CHAR Passphrase[AIRPDCAP_WPA_PASSPHRASE_MAX_LEN+1];
 166                 /**
 167                  * The value of the SSID (up to
 168                  * AIRPDCAP_WPA_SSID_MAX_LEN octets).
 169                  * @note
 170                  * A zero-length SSID indicates broadcast.
 171                  */
 172                 CHAR Ssid[AIRPDCAP_WPA_SSID_MAX_LEN];
 173                 /**
 174                  *The length of the SSID
 175                  */
 176                 size_t SsidLen;
 177         } UserPwd;
 178 } AIRPDCAP_KEY_ITEM, *PAIRPDCAP_KEY_ITEM;

 200 /**
 201  * Returns the decryption_key_t struct given a string describing the key.
 202  * @param key_string [IN] Key string in one of the following formats:
 203  * - 0102030405 (40/64-bit WEP)
 204  * - 01:02:03:04:05 (40/64-bit WEP)
 205  * - 0102030405060708090a0b0c0d (104/128-bit WEP)
 206  * - 01:02:03:04:05:06:07:08:09:0a:0b:0c:0d (104/128-bit WEP)
 207  * - wep:01020304... (WEP)
 208  * - wep:01:02:03:04... (WEP)
 209  * - wpa-pwd:MyPassword (WPA + plaintext password + "wildcard" SSID)
 210  * - wpa-pwd:MyPassword:MySSID (WPA + plaintext password + specific SSID)
 211  * - wpa-psk:01020304... (WPA + 256-bit raw key)
 212  * @return A pointer to a freshly-g_malloc()ed decryption_key_t struct on
 213  *   success, or NULL on failure.
 214  * @see get_key_string()
 215  */
 216 decryption_key_t*
 217 parse_key_string(gchar* key_string);
 218 
 219 /**
 220  * Returns a newly allocated string representing the given decryption_key_t
 221  * struct.
 222  * @param dk [IN] Pointer to the key to be converted
 223  * @return A g_malloc()ed string representation of the key
 224  * @see parse_key_string()
 225  */
 226 gchar*
 227 get_key_string(decryption_key_t* dk);
