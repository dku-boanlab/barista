  45 /* Identification data for a flow.
  46    All fields are in network byte order.
  47    In decreasing order by size, so that flow structures can be hashed or
  48    compared bytewise. */
  49 struct flow {
  50     uint32_t nw_src;            /* IP source address. */
  51     uint32_t nw_dst;            /* IP destination address. */
  52     uint16_t in_port;           /* Input switch port. */
  53     uint16_t dl_vlan;           /* Input VLAN id. */
  54     uint16_t dl_type;           /* Ethernet frame type. */
  55     uint16_t tp_src;            /* TCP/UDP source port. */
  56     uint16_t tp_dst;            /* TCP/UDP destination port. */
  57     uint8_t dl_src[6];          /* Ethernet source address. */
  58     uint8_t dl_dst[6];          /* Ethernet destination address. */
  59     uint8_t dl_vlan_pcp;        /* Input VLAN priority. */
  60     uint8_t nw_tos;             /* IPv4 DSCP. */
  61     uint8_t nw_proto;           /* IP protocol. */
  62     uint8_t pad[3];
  63 };
  64 BUILD_ASSERT_DECL(sizeof(struct flow) == 36);

  74 static inline int
  75 flow_compare(const struct flow *a, const struct flow *b)

  80 static inline bool
  81 flow_equal(const struct flow *a, const struct flow *b)

  86 static inline size_t

  89     BUILD_ASSERT_DECL(!(sizeof *flow % sizeof(uint32_t)));
  90     return hash_words((const uint32_t *) flow,
  91                       sizeof *flow / sizeof(uint32_t), basis);
