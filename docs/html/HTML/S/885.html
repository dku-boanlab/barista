  67 /* Opens a socket for a local datapath, initializing 'dp'.  If
  68  * 'subscribe_dp_idx' is nonnegative, listens for asynchronous messages
  69  * (packet-in, etc.) from the datapath with that number; otherwise, 'dp' will
  70  * receive only replies to explicitly initiated requests. */
  71 int
  72 dpif_open(int subscribe_dp_idx, struct dpif *dp)
  73 {
  74     struct nl_sock *sock;
  75     int multicast_group = 0;
  76     int retval;
  77 
  78     retval = nl_lookup_genl_family(DP_GENL_FAMILY_NAME, &openflow_family);
  79     if (retval) {
  80         return retval;
  81     }
  82 
  83     if (subscribe_dp_idx >= 0) {
  84         retval = lookup_openflow_multicast_group(subscribe_dp_idx,
  85                                                  &multicast_group);
  86         if (retval) {
  87             return retval;
  88         }
  89     }
  90 
  91     /* Specify a large so_rcvbuf size because we occasionally need to be able
  92      * to retrieve large collections of flow records. */
  93     retval = nl_sock_create(NETLINK_GENERIC, multicast_group, 0,
  94                             4 * 1024u * 1024, &sock);
  95     if (retval) {
  96         return retval;
  97     }
  98 
  99     dp->sock = sock;
 100     return 0;
 101 }

 121 /* Tries to receive an openflow message from datapath 'dp_idx' on 'sock'.  If
 122  * successful, stores the received message into '*msgp' and returns 0.  The
 123  * caller is responsible for destroying the message with ofpbuf_delete().  On
 124  * failure, returns a positive errno value and stores a null pointer into
 125  * '*msgp'.
 126  *
 127  * Only Netlink messages with embedded OpenFlow messages are accepted.  Other
 128  * Netlink messages provoke errors.
 129  *
 130  * If 'wait' is true, dpif_recv_openflow waits for a message to be ready;
 131  * otherwise, returns EAGAIN if the 'sock' receive buffer is empty. */
 132 int
 133 dpif_recv_openflow(struct dpif *dp, int dp_idx, struct ofpbuf **bufferp,
 134                    bool wait)
 135 {
 136     struct nlattr *attrs[ARRAY_SIZE(openflow_policy)];
 137     struct ofpbuf *buffer;
 138     struct ofp_header *oh;
 139     uint16_t ofp_len;
 140 
 141     buffer = *bufferp = NULL;
 142     do {
 143         int retval;
 144 
 145         do {
 146             ofpbuf_delete(buffer);
 147             retval = nl_sock_recv(dp->sock, &buffer, wait);
 148         } while (retval == ENOBUFS
 149                  || (!retval
 150                      && (nl_msg_nlmsghdr(buffer)->nlmsg_type == NLMSG_DONE
 151                          || nl_msg_nlmsgerr(buffer, NULL))));
 152         if (retval) {
 153             if (retval != EAGAIN) {
 154                 VLOG_WARN_RL(&rl, "dpif_recv_openflow: %s", strerror(retval));
 155             }
 156             return retval;
 157         }
 158 
 159         if (nl_msg_genlmsghdr(buffer) == NULL) {
 160             VLOG_DBG_RL(&rl, "received packet too short for Generic Netlink");
 161             goto error;
 162         }
 163         if (nl_msg_nlmsghdr(buffer)->nlmsg_type != openflow_family) {
 164             VLOG_DBG_RL(&rl,
 165                         "received type (%"PRIu16") != openflow family (%d)",
 166                         nl_msg_nlmsghdr(buffer)->nlmsg_type, openflow_family);
 167             goto error;
 168         }
 169 
 170         if (!nl_policy_parse(buffer, NLMSG_HDRLEN + GENL_HDRLEN,
 171                              openflow_policy, attrs,
 172                              ARRAY_SIZE(openflow_policy))) {
 173             goto error;
 174         }
 175     } while (nl_attr_get_u32(attrs[DP_GENL_A_DP_IDX]) != dp_idx);
 176 
 177     oh = buffer->data = (void *) nl_attr_get(attrs[DP_GENL_A_OPENFLOW]);
 178     buffer->size = nl_attr_get_size(attrs[DP_GENL_A_OPENFLOW]);
 179     ofp_len = ntohs(oh->length);
 180     if (ofp_len != buffer->size) {
 181         VLOG_WARN_RL(&rl,
 182                      "ofp_header.length %"PRIu16" != attribute length %zu\n",
 183                      ofp_len, buffer->size);
 184         buffer->size = MIN(ofp_len, buffer->size);
 185     }
 186     *bufferp = buffer;
 187     return 0;
 188 
 189 error:
 190     ofpbuf_delete(buffer);
 191     return EPROTO;
 192 }

 194 /* Encapsulates 'msg', which must contain an OpenFlow message, in a Netlink
 195  * message, and sends it to the OpenFlow local datapath numbered 'dp_idx' via
 196  * 'sock'.
 197  *
 198  * Returns 0 if successful, otherwise a positive errno value.  Returns EAGAIN
 199  * if the 'sock' send buffer is full.
 200  *
 201  * If the send is successful, then the kernel module will receive it, but there
 202  * is no guarantee that any reply will not be dropped (see nl_sock_transact()
 203  * for details). 
 204  */
 205 int
 206 dpif_send_openflow(struct dpif *dp, int dp_idx, struct ofpbuf *buffer)
 207 {
 208     struct ofp_header *oh;
 209     unsigned int dump_flag;
 210     struct ofpbuf hdr;
 211     struct nlattr *nla;
 212     uint32_t fixed_buffer[64 / 4];
 213     struct iovec iov[3];
 214     int pad_bytes;
 215     int n_iov;
 216     int retval;
 217 
 218     /* The reply to OFPT_STATS_REQUEST may be multiple segments long, so we
 219      * need to specify NLM_F_DUMP in the request. */
 220     oh = ofpbuf_at_assert(buffer, 0, sizeof *oh);
 221     dump_flag = oh->type == OFPT_STATS_REQUEST ? NLM_F_DUMP : 0;
 222 
 223     ofpbuf_use(&hdr, fixed_buffer, sizeof fixed_buffer);
 224     nl_msg_put_genlmsghdr(&hdr, dp->sock, 32, openflow_family,
 225                           NLM_F_REQUEST | dump_flag, DP_GENL_C_OPENFLOW, 1);
 226     nl_msg_put_u32(&hdr, DP_GENL_A_DP_IDX, dp_idx);
 227     nla = ofpbuf_put_uninit(&hdr, sizeof *nla);
 228     nla->nla_len = sizeof *nla + buffer->size;
 229     nla->nla_type = DP_GENL_A_OPENFLOW;
 230     pad_bytes = NLA_ALIGN(nla->nla_len) - nla->nla_len;
 231     nl_msg_nlmsghdr(&hdr)->nlmsg_len = hdr.size + buffer->size + pad_bytes;
 232     n_iov = 2;
 233     iov[0].iov_base = hdr.data;
 234     iov[0].iov_len = hdr.size;
 235     iov[1].iov_base = buffer->data;
 236     iov[1].iov_len = buffer->size;
 237     if (pad_bytes) {
 238         static char zeros[NLA_ALIGNTO];
 239         n_iov++;
 240         iov[2].iov_base = zeros;
 241         iov[2].iov_len = pad_bytes; 
 242     }
 243     retval = nl_sock_sendv(dp->sock, iov, n_iov, false);
 244     if (retval && retval != EAGAIN) {
 245         VLOG_WARN_RL(&rl, "dpif_send_openflow: %s", strerror(retval));
 246     }
 247     return retval;
 248 }

 287 static const struct nl_policy openflow_multicast_policy[] = {
 288     [DP_GENL_A_DP_IDX] = { .type = NL_A_U32 },
 289     [DP_GENL_A_DP_NAME] = { .type = NL_A_STRING },
 290     [DP_GENL_A_MC_GROUP] = { .type = NL_A_U32 },
 291 };

 293 /* Looks up the Netlink multicast group and datapath index of a datapath
 294  * by either the datapath index or name.  If 'dp_idx' points to a value 
 295  * of '-1', then 'dp_name' is used to lookup the datapath.  If successful, 
 296  * stores the multicast group in '*multicast_group' and the index in
 297  * '*dp_idx' and returns 0. Otherwise, returns a positive errno value. */
 298 static int
 299 query_datapath(int *dp_idx, int *multicast_group, const char *dp_name)
 300 {
 301     struct nl_sock *sock;
 302     struct ofpbuf request, *reply;
 303     struct nlattr *attrs[ARRAY_SIZE(openflow_multicast_policy)];
 304     int retval;
 305 
 306     retval = nl_sock_create(NETLINK_GENERIC, 0, 0, 0, &sock);
 307     if (retval) {
 308         return retval;
 309     }
 310     ofpbuf_init(&request, 0);
 311     nl_msg_put_genlmsghdr(&request, sock, 0, openflow_family, NLM_F_REQUEST,
 312                           DP_GENL_C_QUERY_DP, 1);
 313     if (*dp_idx != -1) {
 314         nl_msg_put_u32(&request, DP_GENL_A_DP_IDX, *dp_idx);
 315     }
 316     if (dp_name) {
 317         nl_msg_put_string(&request, DP_GENL_A_DP_NAME, dp_name);
 318     }
 319     retval = nl_sock_transact(sock, &request, &reply);
 320     ofpbuf_uninit(&request);
 321     if (retval) {
 322         nl_sock_destroy(sock);
 323         return retval;
 324     }
 325     if (!nl_policy_parse(reply, NLMSG_HDRLEN + GENL_HDRLEN,
 326                          openflow_multicast_policy, attrs,
 327                          ARRAY_SIZE(openflow_multicast_policy))) {
 328         nl_sock_destroy(sock);
 329         ofpbuf_delete(reply);
 330         return EPROTO;
 331     }
 332     *dp_idx = nl_attr_get_u32(attrs[DP_GENL_A_DP_IDX]);
 333     *multicast_group = nl_attr_get_u32(attrs[DP_GENL_A_MC_GROUP]);
 334     nl_sock_destroy(sock);
 335     ofpbuf_delete(reply);
 336 
 337     return 0;
 338 }

 349 /* Looks up the datatpath index based on the name.  Returns the index, or
 350  * -1 on error. */
 351 int
 352 dpif_get_idx(const char *name)
 353 {
 354     int dp_idx = -1;
 355     int mc_group = 0;
 356 
 357     if (query_datapath(&dp_idx, &mc_group, name)) {
 358         return -1;
 359     }
 360 
 361     return dp_idx;
 362 }

 364 /* Sends the given 'command' to datapath 'dp', related to the local datapath
 365  * numbered 'dp_idx'.  If 'arg' is nonnull, adds it to the command as the
 366  * datapath or port name attribute depending on the requested operation.  
 367  * Returns 0 if successful, otherwise a positive errno value. */
 368 static int
 369 send_mgmt_command(struct dpif *dp, int dp_idx, int command, const char *arg)
 370 {
 371     struct ofpbuf request, *reply;
 372     int retval;
 373 
 374     ofpbuf_init(&request, 0);
 375     nl_msg_put_genlmsghdr(&request, dp->sock, 32, openflow_family,
 376                           NLM_F_REQUEST | NLM_F_ACK, command, 1);
 377     if (dp_idx != -1) {
 378         nl_msg_put_u32(&request, DP_GENL_A_DP_IDX, dp_idx);
 379     }
 380     if (arg) {
 381         if ((command == DP_GENL_C_ADD_DP) || (command == DP_GENL_C_DEL_DP)) {
 382             nl_msg_put_string(&request, DP_GENL_A_DP_NAME, arg);
 383         } else {
 384             nl_msg_put_string(&request, DP_GENL_A_PORTNAME, arg);
 385         }
 386     }
 387     retval = nl_sock_transact(dp->sock, &request, &reply);
 388     ofpbuf_uninit(&request);
 389     ofpbuf_delete(reply);
 390 
 391     return retval;
 392 }
