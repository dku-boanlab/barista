 283 /* Macros to byte-swap 64-bit, 32-bit and 16-bit quantities. */
 284 #define BSWAP64(x) \
 285         ((((x)&G_GINT64_CONSTANT(0xFF00000000000000U))>>56) |   \
 286          (((x)&G_GINT64_CONSTANT(0x00FF000000000000U))>>40) |   \
 287          (((x)&G_GINT64_CONSTANT(0x0000FF0000000000U))>>24) |   \
 288          (((x)&G_GINT64_CONSTANT(0x000000FF00000000U))>>8) |    \
 289          (((x)&G_GINT64_CONSTANT(0x00000000FF000000U))<<8) |    \
 290          (((x)&G_GINT64_CONSTANT(0x0000000000FF0000U))<<24) |   \
 291          (((x)&G_GINT64_CONSTANT(0x000000000000FF00U))<<40) |   \
 292          (((x)&G_GINT64_CONSTANT(0x00000000000000FFU))<<56))
 293 #define BSWAP32(x) \
 294         ((((x)&0xFF000000)>>24) | \
 295          (((x)&0x00FF0000)>>8) | \
 296          (((x)&0x0000FF00)<<8) | \
 297          (((x)&0x000000FF)<<24))
 298 #define BSWAP16(x) \
 299          ((((x)&0xFF00)>>8) | \
 300           (((x)&0x00FF)<<8))

 302 /* Macros to byte-swap possibly-unaligned 32-bit and 16-bit quantities;
 303  * they take a pointer to the quantity, and byte-swap it in place.
 304  */
 305 #define PBSWAP32(p) \
 306         {                       \
 307         guint8 tmp;             \
 308         tmp = (p)[3];           \
 309         (p)[3] = (p)[0];        \
 310         (p)[0] = tmp;           \
 311         tmp = (p)[2];           \
 312         (p)[2] = (p)[1];        \
 313         (p)[1] = tmp;           \
 314         }
 315 #define PBSWAP16(p) \
 316         {                       \
 317         guint8 tmp;             \
 318         tmp = (p)[1];           \
 319         (p)[1] = (p)[0];        \
 320         (p)[0] = tmp;           \
 321         }

 328 /* Pointer versions of ntohs and ntohl.  Given a pointer to a member of a
 329  * byte array, returns the value of the two or four bytes at the pointer.
 330  * The pletoh[sl] versions return the little-endian representation.
 331  * We also provide pntohll and pletohll, which extract 64-bit integral
 332  * quantities.
 333  *
 334  * These will work regardless of the byte alignment of the pointer.
 335  */
 336 
 337 #ifndef pntohs
 338 #define pntohs(p)  ((guint16)                       \
 339                     ((guint16)*((const guint8 *)(p)+0)<<8|  \
 340                      (guint16)*((const guint8 *)(p)+1)<<0))
 341 #endif
 342 
 343 #ifndef pntoh24
 344 #define pntoh24(p)  ((guint32)*((const guint8 *)(p)+0)<<16| \
 345                      (guint32)*((const guint8 *)(p)+1)<<8|  \
 346                      (guint32)*((const guint8 *)(p)+2)<<0)
 347 #endif
 348 
 349 #ifndef pntohl
 350 #define pntohl(p)  ((guint32)*((const guint8 *)(p)+0)<<24|  \
 351                     (guint32)*((const guint8 *)(p)+1)<<16|  \
 352                     (guint32)*((const guint8 *)(p)+2)<<8|   \
 353                     (guint32)*((const guint8 *)(p)+3)<<0)
 354 #endif
 355 
 356 #ifndef pntohll
 357 #define pntohll(p)  ((guint64)*((const guint8 *)(p)+0)<<56|  \
 358                      (guint64)*((const guint8 *)(p)+1)<<48|  \
 359                      (guint64)*((const guint8 *)(p)+2)<<40|  \
 360                      (guint64)*((const guint8 *)(p)+3)<<32|  \
 361                      (guint64)*((const guint8 *)(p)+4)<<24|  \
 362                      (guint64)*((const guint8 *)(p)+5)<<16|  \
 363                      (guint64)*((const guint8 *)(p)+6)<<8|   \
 364                      (guint64)*((const guint8 *)(p)+7)<<0)
 365 #endif
 366

 369 #define pletohs(p) ((guint16)                       \

 371                      (guint16)*((const guint8 *)(p)+0)<<0))
 372 #endif

 450 #define wtap_file_read_unknown_bytes(target, num_bytes, fh, err) \
 451         G_STMT_START \
 452         { \
 453                 int _bytes_read; \
 454                 _bytes_read = file_read((target), 1, (num_bytes), (fh)); \
 455                 if (_bytes_read != (int) (num_bytes)) { \
 456                         *(err) = file_error((fh)); \
 457                         return FALSE; \
 458                 } \
 459         } \
 460         G_STMT_END
 461 
 462 #define wtap_file_read_expected_bytes(target, num_bytes, fh, err) \
 463         G_STMT_START \
 464         { \
 465                 int _bytes_read; \
 466                 _bytes_read = file_read((target), 1, (num_bytes), (fh)); \
 467                 if (_bytes_read != (int) (num_bytes)) { \
 468                         *(err) = file_error((fh)); \
 469                         if (*(err) == 0 && _bytes_read > 0) { \
 470                                 *(err) = WTAP_ERR_SHORT_READ; \
 471                         } \
 472                         return FALSE; \
 473                 } \
 474         } \
 475         G_STMT_END
