  49 /* An event that will wake the following call to poll_block(). */
  50 struct poll_waiter {
  51     /* Set when the waiter is created. */
  52     struct list node;           /* Element in global waiters list. */
  53     int fd;                     /* File descriptor. */
  54     short int events;           /* Events to wait for (POLLIN, POLLOUT). */
  55     poll_fd_func *function;     /* Callback function, if any, or null. */
  56     void *aux;                  /* Argument to callback function. */
  57     struct backtrace *backtrace; /* Optionally, event that created waiter. */
  58 
  59     /* Set only when poll_block() is called. */
  60     struct pollfd *pollfd;      /* Pointer to element of the pollfds array
  61                                    (null if added from a callback). */
  62 };

  85 /* Registers 'fd' as waiting for the specified 'events' (which should be POLLIN
  86  * or POLLOUT or POLLIN | POLLOUT).  The following call to poll_block() will
  87  * wake up when 'fd' becomes ready for one or more of the requested events.
  88  *
  89  * The event registration is one-shot: only the following call to poll_block()
  90  * is affected.  The event will need to be re-registered after poll_block() is
  91  * called if it is to persist. */
  92 struct poll_waiter *
  93 poll_fd_wait(int fd, short int events)
  94 {
  95     return new_waiter(fd, events);
  96 }

  98 /* Causes the following call to poll_block() to block for no more than 'msec'
  99  * milliseconds.  If 'msec' is nonpositive, the following call to poll_block()
 100  * will not block at all.
 101  *
 102  * The timer registration is one-shot: only the following call to poll_block()
 103  * is affected.  The timer will need to be re-registered after poll_block() is
 104  * called if it is to persist. */
 105 void
 106 poll_timer_wait(int msec)
 107 {
 108     if (timeout < 0 || msec < timeout) {
 109         timeout = MAX(0, msec);
 110         if (VLOG_IS_DBG_ENABLED()) {
 111             backtrace_capture(&timeout_backtrace);
 112         }
 113     }
 114 }
 115 
 116 /* Causes the following call to poll_block() to wake up immediately, without
 117  * blocking. */
 118 void
 119 poll_immediate_wake(void)
 120 {
 121     poll_timer_wait(0);
 122 }

 124 static void PRINTF_FORMAT(2, 3)
 125 log_wakeup(const struct backtrace *backtrace, const char *format, ...)
 126 {
 127     struct ds ds;
 128     va_list args;
 129 
 130     ds_init(&ds);
 131     va_start(args, format);
 132     ds_put_format_valist(&ds, format, args);
 133     va_end(args);
 134 
 135     if (backtrace) {
 136         int i;
 137 
 138         ds_put_char(&ds, ':');
 139         for (i = 0; i < backtrace->n_frames; i++) {
 140             ds_put_format(&ds, " 0x%x", backtrace->frames[i]);
 141         }
 142     }
 143     VLOG_DBG("%s", ds_cstr(&ds));
 144     ds_destroy(&ds);
 145 }

 147 /* Blocks until one or more of the events registered with poll_fd_wait()
 148  * occurs, or until the minimum duration registered with poll_timer_wait()
 149  * elapses, or not at all if poll_immediate_wake() has been called.
 150  *
 151  * Also executes any autonomous subroutines registered with poll_fd_callback(),
 152  * if their file descriptors have become ready. */
 153 void
 154 poll_block(void)
 155 {
 156     static struct pollfd *pollfds;
 157     static size_t max_pollfds;
 158 
 159     struct poll_waiter *pw;
 160     struct list *node;
 161     int n_pollfds;
 162     int retval;
 163 
 164     assert(!running_cb);
 165     if (max_pollfds < n_waiters) {
 166         max_pollfds = n_waiters;
 167         pollfds = xrealloc(pollfds, max_pollfds * sizeof *pollfds);
 168     }
 169 
 170     n_pollfds = 0;
 171     LIST_FOR_EACH (pw, struct poll_waiter, node, &waiters) {
 172         pw->pollfd = &pollfds[n_pollfds];
 173         pollfds[n_pollfds].fd = pw->fd;
 174         pollfds[n_pollfds].events = pw->events;
 175         pollfds[n_pollfds].revents = 0;
 176         n_pollfds++;
 177     }
 178 
 179     retval = time_poll(pollfds, n_pollfds, timeout);
 180     if (retval < 0) {
 181         static struct vlog_rate_limit rl = VLOG_RATE_LIMIT_INIT(1, 5);
 182         VLOG_ERR_RL(&rl, "poll: %s", strerror(-retval));
 183     } else if (!retval && VLOG_IS_DBG_ENABLED()) {
 184         log_wakeup(&timeout_backtrace, "%d-ms timeout", timeout);
 185     }
 186 
 187     for (node = waiters.next; node != &waiters; ) {
 188         pw = CONTAINER_OF(node, struct poll_waiter, node);
 189         if (!pw->pollfd || !pw->pollfd->revents) {
 190             if (pw->function) {
 191                 node = node->next;
 192                 continue;
 193             }
 194         } else {
 195             if (VLOG_IS_DBG_ENABLED()) {
 196                 log_wakeup(pw->backtrace, "%s%s%s%s%s on fd %d",
 197                            pw->pollfd->revents & POLLIN ? "[POLLIN]" : "",
 198                            pw->pollfd->revents & POLLOUT ? "[POLLOUT]" : "",
 199                            pw->pollfd->revents & POLLERR ? "[POLLERR]" : "",
 200                            pw->pollfd->revents & POLLHUP ? "[POLLHUP]" : "",
 201                            pw->pollfd->revents & POLLNVAL ? "[POLLNVAL]" : "",
 202                            pw->fd);
 203             }
 204 
 205             if (pw->function) {
 206 #ifndef NDEBUG
 207                 running_cb = pw;
 208 #endif
 209                 pw->function(pw->fd, pw->pollfd->revents, pw->aux);
 210 #ifndef NDEBUG
 211                 running_cb = NULL;
 212 #endif
 213             }
 214         }
 215         node = node->next;
 216         poll_cancel(pw);
 217     }
 218 
 219     timeout = -1;
 220     timeout_backtrace.n_frames = 0;
 221 }

 223 /* Registers 'function' to be called with argument 'aux' by poll_block() when
 224  * 'fd' becomes ready for one of the events in 'events', which should be POLLIN
 225  * or POLLOUT or POLLIN | POLLOUT.
 226  *
 227  * The callback registration persists until the event actually occurs.  At that
 228  * point, it is automatically de-registered.  The callback function must
 229  * re-register the event by calling poll_fd_callback() again within the
 230  * callback, if it wants to be called back again later. */
 231 struct poll_waiter *
 232 poll_fd_callback(int fd, short int events, poll_fd_func *function, void *aux)
 233 {
 234     struct poll_waiter *pw = new_waiter(fd, events);
 235     pw->function = function;
 236     pw->aux = aux;
 237     return pw;
 238 }

 240 /* Cancels the file descriptor event registered with poll_fd_wait() or
 241  * poll_fd_callback().  'pw' must be the struct poll_waiter returned by one of
 242  * those functions.
 243  *
 244  * An event registered with poll_fd_wait() may be canceled from its time of
 245  * registration until the next call to poll_block().  At that point, the event
 246  * is automatically canceled by the system and its poll_waiter is freed.
 247  *
 248  * An event registered with poll_fd_callback() may be canceled from its time of
 249  * registration until its callback is actually called.  At that point, the
 250  * event is automatically canceled by the system and its poll_waiter is
 251  * freed. */
 252 void
 253 poll_cancel(struct poll_waiter *pw)
 254 {
 255     if (pw) {
 256         assert(pw != running_cb);
 257         list_remove(&pw->node);
 258         free(pw->backtrace);
 259         free(pw);
 260         n_waiters--;
 261     }
 262 }

 264 /* Creates and returns a new poll_waiter for 'fd' and 'events'. */
 265 static struct poll_waiter *
 266 new_waiter(int fd, short int events)
 267 {
 268     struct poll_waiter *waiter = xcalloc(1, sizeof *waiter);
 269     assert(fd >= 0);
 270     waiter->fd = fd;
 271     waiter->events = events;
 272     if (VLOG_IS_DBG_ENABLED()) {
 273         waiter->backtrace = xmalloc(sizeof *waiter->backtrace);
 274         backtrace_capture(waiter->backtrace);
 275     }
 276     list_push_back(&waiters, &waiter->node);
 277     n_waiters++;
 278     return waiter;
 279 }
