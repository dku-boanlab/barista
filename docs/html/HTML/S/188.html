 102 int
 103 udatapath_cmd(int argc, char *argv[])
 104 {
 105     int n_listeners;
 106     int error;
 107     int i;
 108 
 109     set_program_name(argv[0]);
 110     register_fault_handlers();
 111     time_init();
 112     vlog_init();
 113     parse_options(argc, argv);
 114     signal(SIGPIPE, SIG_IGN);
 115 
 116     if (argc - optind < 1) {
 117         OFP_FATAL(0, "at least one listener argument is required; "
 118           "use --help for usage");
 119     }
 120 
 121     error = dp_new(&dp, dpid);
 122 
 123     n_listeners = 0;
 124     for (i = optind; i < argc; i++) {
 125         const char *pvconn_name = argv[i];
 126         struct pvconn *pvconn;
 127         int retval;
 128 
 129         retval = pvconn_open(pvconn_name, &pvconn);
 130         if (!retval || retval == EAGAIN) {
 131             dp_add_pvconn(dp, pvconn);
 132             n_listeners++;
 133         } else {
 134             ofp_error(retval, "opening %s", pvconn_name);
 135         }
 136     }
 137     if (!n_listeners) {
 138         OFP_FATAL(0, "could not listen for any connections");
 139     }
 140 
 141     if (port_list) {
 142         add_ports(dp, port_list);
 143     }
 144     if (local_port) {
 145         error = dp_add_local_port(dp, local_port, 0);
 146         if (error) {
 147             OFP_FATAL(error, "failed to add local port %s", local_port);
 148         }
 149     }
 150 
 151     error = vlog_server_listen(NULL, NULL);
 152     if (error) {
 153         OFP_FATAL(error, "could not listen for vlog connections");
 154     }
 155 
 156     die_if_already_running();
 157     daemonize();
 158 
 159     for (;;) {
 160         dp_run(dp);
 161         dp_wait(dp);
 162         poll_block();
 163     }
 164 
 165     return 0;
 166 }

 168 static void
 169 add_ports(struct datapath *dp, char *port_list)
 170 {
 171     char *port, *save_ptr;
 172 
 173     /* Glibc 2.7 has a bug in strtok_r when compiling with optimization that
 174      * can cause segfaults here:
 175      * http://sources.redhat.com/bugzilla/show_bug.cgi?id=5614.
 176      * Using ",," instead of the obvious "," works around it. */
 177     for (port = strtok_r(port_list, ",,", &save_ptr); port;
 178          port = strtok_r(NULL, ",,", &save_ptr)) {
 179         int error = dp_add_port(dp, port, num_queues);
 180         if (error) {
 181             ofp_fatal(error, "failed to add port %s", port);
 182         }
 183     }
 184 }

 186 static void
 187 parse_options(int argc, char *argv[])
 188 {
 189     enum {
 190         OPT_MFR_DESC = UCHAR_MAX + 1,
 191         OPT_HW_DESC,
 192         OPT_SW_DESC,
 193         OPT_DP_DESC,
 194         OPT_SERIAL_NUM,
 195         OPT_BOOTSTRAP_CA_CERT,
 196         OPT_NO_LOCAL_PORT,
 197         OPT_NO_SLICING
 198     };
 199 
 200     static struct option long_options[] = {
 201         {"interfaces",  required_argument, 0, 'i'},
 202         {"local-port",  required_argument, 0, 'L'},
 203         {"no-local-port", no_argument, 0, OPT_NO_LOCAL_PORT},
 204         {"datapath-id", required_argument, 0, 'd'},
 205         {"verbose",     optional_argument, 0, 'v'},
 206         {"help",        no_argument, 0, 'h'},
 207         {"version",     no_argument, 0, 'V'},
 208         {"no-slicing",  no_argument, 0, OPT_NO_SLICING},
 209         {"mfr-desc",    required_argument, 0, OPT_MFR_DESC},
 210         {"hw-desc",     required_argument, 0, OPT_HW_DESC},
 211         {"sw-desc",     required_argument, 0, OPT_SW_DESC},
 212         {"dp_desc",  required_argument, 0, OPT_DP_DESC},
 213         {"serial_num",  required_argument, 0, OPT_SERIAL_NUM},
 214         DAEMON_LONG_OPTIONS,
 215 #ifdef HAVE_OPENSSL
 216         VCONN_SSL_LONG_OPTIONS
 217         {"bootstrap-ca-cert", required_argument, 0, OPT_BOOTSTRAP_CA_CERT},
 218 #endif
 219         {0, 0, 0, 0},
 220     };
 221     char *short_options = long_options_to_short_options(long_options);
 222 
 223     for (;;) {
 224         int indexptr;
 225         int c;
 226 
 227         c = getopt_long(argc, argv, short_options, long_options, &indexptr);
 228         if (c == -1) {
 229             break;
 230         }
 231 
 232         switch (c) {
 233         case 'd':
 234             if (strlen(optarg) != 12
 235                 || strspn(optarg, "0123456789abcdefABCDEF") != 12) {
 236                 ofp_fatal(0, "argument to -d or --datapath-id must be "
 237                           "exactly 12 hex digits");
 238             }
 239             dpid = strtoll(optarg, NULL, 16);
 240             if (!dpid) {
 241                 ofp_fatal(0, "argument to -d or --datapath-id must "
 242                           "be nonzero");
 243             }
 244             break;
 245 
 246         case 'h':
 247             usage();
 248 
 249         case 'V':
 250             printf("%s %s compiled "__DATE__" "__TIME__"\n",
 251                    program_name, VERSION BUILDNR);
 252             exit(EXIT_SUCCESS);
 253 
 254         case 'v':
 255             vlog_set_verbosity(optarg);
 256             break;
 257 
 258         case 'i':
 259             if (!port_list) {
 260                 port_list = optarg;
 261             } else {
 262                 port_list = xasprintf("%s,%s", port_list, optarg);
 263             }
 264             break;
 265 
 266         case 'L':
 267             local_port = optarg;
 268             break;
 269 
 270         case OPT_NO_LOCAL_PORT:
 271             local_port = NULL;
 272             break;
 273 
 274         case OPT_MFR_DESC:
 275             strncpy(mfr_desc, optarg, sizeof mfr_desc);
 276             break;
 277 
 278         case OPT_HW_DESC:
 279             strncpy(hw_desc, optarg, sizeof hw_desc);
 280             break;
 281 
 282         case OPT_SW_DESC:
 283             strncpy(sw_desc, optarg, sizeof sw_desc);
 284             break;
 285 
 286         case OPT_DP_DESC:
 287             strncpy(dp_desc, optarg, sizeof dp_desc);
 288             break;
 289 
 290         case OPT_SERIAL_NUM:
 291             strncpy(serial_num, optarg, sizeof serial_num);
 292             break;
 293 
 294         case OPT_NO_SLICING:
 295             num_queues = 0;
 296             break;
 297 
 298         DAEMON_OPTION_HANDLERS
 299 
 300 #ifdef HAVE_OPENSSL
 301         VCONN_SSL_OPTION_HANDLERS
 302 
 303         case OPT_BOOTSTRAP_CA_CERT:
 304             vconn_ssl_set_ca_cert_file(optarg, true);
 305             break;
 306 #endif
 307 
 308         case '?':
 309             exit(EXIT_FAILURE);
 310 
 311         default:
 312             abort();
 313         }
 314     }
 315     free(short_options);
 316 }

 318 static void
 319 usage(void)
 320 {
 321     printf("%s: userspace OpenFlow datapath\n"
 322            "usage: %s [OPTIONS] LISTEN...\n"
 323            "where LISTEN is a passive OpenFlow connection method on which\n"
 324        "to listen for incoming connections from the secure channel.\n",
 325            program_name, program_name);
 326     vconn_usage(false, true, false);
 327     printf("\nConfiguration options:\n"
 328            "  -i, --interfaces=NETDEV[,NETDEV]...\n"
 329            "                          add specified initial switch ports\n"
 330            "  -L, --local-port=NETDEV set network device for local port\n"
 331            "  --no-local-port         disable local port\n"
 332            "  -d, --datapath-id=ID    Use ID as the OpenFlow switch ID\n"
 333            "                          (ID must consist of 12 hex digits)\n"
 334            "  --no-slicing            disable slicing\n"
 335            "\nOther options:\n"
 336            "  -D, --detach            run in background as daemon\n"
 337            "  -P, --pidfile[=FILE]    create pidfile (default: %s/ofdatapath.pid)\n"
 338            "  -f, --force             with -P, start even if already running\n"
 339            "  -v, --verbose=MODULE[:FACILITY[:LEVEL]]  set logging levels\n"
 340            "  -v, --verbose           set maximum verbosity level\n"
 341            "  -h, --help              display this help message\n"
 342            "  -V, --version           display version information\n",
 343         ofp_rundir);
 344     exit(EXIT_SUCCESS);
 345 }
