  79 /* Usage:
  80  *
  81  * TRY {
  82  *      code;
  83  * }
  84  *
  85  * CATCH(exception) {
  86  *      code;
  87  * }
  88  *
  89  * CATCH2(exception1, exception2) {
  90  *      code;
  91  * }
  92  *
  93  * CATCH_ALL {
  94  *      code;
  95  * }
  96  *
  97  * FINALLY {
  98  *      code;
  99  * }
 100  *
 101  * ENDTRY;
 102  *
 103  * ********* Never use 'goto' or 'return' inside the TRY, CATCH, CATCH_ALL,
 104  * ********* or FINALLY blocks. Execution must proceed through ENDTRY before
 105  * ********* branching out.
 106  *
 107  * This is really something like:
 108  *
 109  * {
 110  *      caught = FALSE:
 111  *      x = setjmp();
 112  *      if (x == 0) {
 113  *              <TRY code>
 114  *      }
 115  *      if (!caught && x == 1) {
 116  *              caught = TRUE;
 117  *              <CATCH(1) code>
 118  *      }
 119  *      if (!caught && x == 2) {
 120  *              caught = TRUE;
 121  *              <CATCH(2) code>
 122  *      }
 123  *      if (!caught && (x == 3 || x == 4)) {
 124  *              caught = TRUE;
 125  *              <CATCH2(3,4) code>
 126  *      }
 127  *      if (!caught && x != 0) {
 128  *              caught = TRUE;
 129  *              <CATCH_ALL code>
 130  *      }
 131  *      <FINALLY code>
 132  *      if(!caught) {
 133  *              RETHROW(x)
 134  *      }
 135  * }<ENDTRY tag>
 136  *
 137  * All CATCH's must precede a CATCH_ALL.
 138  * FINALLY must occur after any CATCH or CATCH_ALL.
 139  * ENDTRY marks the end of the TRY code.
 140  * TRY and ENDTRY are the mandatory parts of a TRY block.
 141  * CATCH, CATCH_ALL, and FINALLY are all optional (although
 142  * you'll probably use at least one, otherwise why "TRY"?)
 143  *
 144  * GET_MESSAGE  returns string ptr to exception message
 145  *              when exception is thrown via THROW_MESSAGE()
 146  *
 147  * To throw/raise an exception.
 148  *
 149  * THROW(exception)
 150  * RETHROW                              rethrow the caught exception
 151  *
 152  * A cleanup callback is a function called in case an exception occurs
 153  * and is not caught. It should be used to free any dynamically-allocated data.
 154  * A pop or call_and_pop should occur at the same statement-nesting level
 155  * as the push.
 156  *
 157  * CLEANUP_CB_PUSH(func, data)
 158  * CLEANUP_CB_POP
 159  * CLEANUP_CB_CALL_AND_POP
 160  */
 161 
 162 /* we do up to three passes through the bit of code after except_try_push(),
 163  * and except_state is used to keep track of where we are.
 164  */
 165 #define EXCEPT_CAUGHT   1 /* exception has been caught, no need to rethrow at
 166                            * END_TRY */
 167 
 168 #define EXCEPT_RETHROWN 2 /* the exception was rethrown from a CATCH
 169                            * block. Don't reenter the CATCH blocks, but do
 170                            * execute FINALLY and rethrow at END_TRY */
 171 
 172 #define EXCEPT_FINALLY  4 /* we've entered the FINALLY block - don't allow
 173                            * RETHROW, and don't reenter FINALLY if a
 174                            * different exception is thrown */
 175 
 176 #define TRY \
 177 {\
 178         except_t *exc; \
 179         volatile int except_state = 0; \
 180         static const except_id_t catch_spec[] = { \
 181                 { XCEPT_GROUP_WIRESHARK, XCEPT_CODE_ANY } }; \
 182         except_try_push(catch_spec, 1, &exc); \
 183                                                        \
 184         if(except_state & EXCEPT_CAUGHT)               \
 185             except_state |= EXCEPT_RETHROWN;           \
 186         except_state &= ~EXCEPT_CAUGHT;                \
 187                                                        \
 188         if (except_state == 0 && exc == 0)             \
 189                 /* user's code goes here */
 190 
 191 #define ENDTRY \
 192         /* rethrow the exception if necessary */ \
 193         if(!(except_state&EXCEPT_CAUGHT) && exc != 0)  \
 194             except_rethrow(exc);                 \
 195         except_try_pop();\
 196 }
 197 
 198 /* the (except_state |= EXCEPT_CAUGHT) in the below is a way of setting
 199  * except_state before the user's code, without disrupting the user's code if
 200  * it's a one-liner.
 201  */
 202 #define CATCH(x) \
 203         if (except_state == 0 && exc != 0 && exc->except_id.except_code == (x) && \
 204             (except_state |= EXCEPT_CAUGHT))                                      \
 205                 /* user's code goes here */
 206 
 207 #define CATCH2(x,y) \
 208         if (except_state == 0 && exc != 0 && \
 209             (exc->except_id.except_code == (x) || exc->except_id.except_code == (y)) && \
 210             (except_state|=EXCEPT_CAUGHT))                                             \
 211                 /* user's code goes here */
 212 
 213 #define CATCH_ALL \
 214         if (except_state == 0 && exc != 0 && \
 215             (except_state|=EXCEPT_CAUGHT))                                             \
 216                 /* user's code goes here */
 217 
 218 
 219 #define FINALLY \
 220         if( !(except_state & EXCEPT_FINALLY) && (except_state|=EXCEPT_FINALLY)) \
 221                 /* user's code goes here */
 222 
 223 #define THROW(x) \
 224         except_throw(XCEPT_GROUP_WIRESHARK, (x), NULL)
 225 
 226 #define THROW_MESSAGE(x, y) \
 227         except_throw(XCEPT_GROUP_WIRESHARK, (x), (y))
 228 
 229 #define GET_MESSAGE                     except_message(exc)
 230 
 231 #define RETHROW                                     \
 232     {                                               \
 233         /* check we're in a catch block */          \
 234         g_assert(except_state == EXCEPT_CAUGHT);    \
 235         /* we can't use except_rethrow here, as that pops a catch block \
 236          * off the stack, and we don't want to do that, because we want to \
 237          * excecute the FINALLY {} block first.     \
 238          * except_throw doesn't provide an interface to rethrow an existing \
 239          * exception; however, longjmping back to except_try_push() has the \
 240          * desired effect.                          \
 241          *                                          \
 242          * Note also that THROW and RETHROW should provide much the same \
 243          * functionality in terms of which blocks to enter, so any messing \ 
 244          * about with except_state in here would indicate that THROW is \
 245          * doing the wrong thing.                   \
 246          */                                         \
 247         longjmp(except_ch.except_jmp,1);            \
 248     }

 252 /* Register cleanup functions in case an exception is thrown and not caught.
 253  * From the Kazlib documentation, with modifications for use with the
 254  * Wireshark-specific macros:
 255  *
 256  * CLEANUP_PUSH(func, arg)
 257  *
 258  *  The call to CLEANUP_PUSH shall be matched with a call to
 259  *  CLEANUP_CALL_AND_POP or CLEANUP_POP which must occur in the same
 260  *  statement block at the same level of nesting. This requirement allows
 261  *  an implementation to provide a CLEANUP_PUSH macro which opens up a
 262  *  statement block and a CLEANUP_POP which closes the statement block.
 263  *  The space for the registered pointers can then be efficiently
 264  *  allocated from automatic storage.
 265  *
 266  *  The CLEANUP_PUSH macro registers a cleanup handler that will be
 267  *  called if an exception subsequently occurs before the matching
 268  *  CLEANUP_[CALL_AND_]POP is executed, and is not intercepted and
 269  *  handled by a try-catch region that is nested between the two.
 270  *
 271  *  The first argument to CLEANUP_PUSH is a pointer to the cleanup
 272  *  handler, a function that returns nothing and takes a single
 273  *  argument of type void*. The second argument is a void* value that
 274  *  is registered along with the handler.  This value is what is passed
 275  *  to the registered handler, should it be called.
 276  *
 277  *  Cleanup handlers are called in the reverse order of their nesting:
 278  *  inner handlers are called before outer handlers.
 279  *
 280  *  The program shall not leave the cleanup region between
 281  *  the call to the macro CLEANUP_PUSH and the matching call to
 282  *  CLEANUP_[CALL_AND_]POP by means other than throwing an exception,
 283  *  or calling CLEANUP_[CALL_AND_]POP.
 284  *
 285  *  Within the call to the cleanup handler, it is possible that new
 286  *  exceptions may happen.  Such exceptions must be handled before the
 287  *  cleanup handler terminates. If the call to the cleanup handler is
 288  *  terminated by an exception, the behavior is undefined. The exception
 289  *  which triggered the cleanup is not yet caught; thus the program
 290  *  would be effectively trying to replace an exception with one that
 291  *  isn't in a well-defined state.
 292  *
 293  *
 294  * CLEANUP_POP and CLEANUP_CALL_AND_POP
 295  *
 296  *  A call to the CLEANUP_POP or CLEANUP_CALL_AND_POP macro shall match
 297  *  each call to CLEANUP_PUSH which shall be in the same statement block
 298  *  at the same nesting level.  It shall match the most recent such a
 299  *  call that is not matched by a previous CLEANUP_[CALL_AND_]POP at
 300  *  the same level.
 301  *
 302  *  These macros causes the registered cleanup handler to be removed. If
 303  *  CLEANUP_CALL_AND_POP is called, the cleanup handler is called.
 304  *  In that case, the registered context pointer is passed to the cleanup
 305  *  handler. If CLEANUP_POP is called, the cleanup handler is not called.
 306  *
 307  *  The program shall not leave the region between the call to the
 308  *  macro CLEANUP_PUSH and the matching call to CLEANUP_[CALL_AND_]POP
 309  *  other than by throwing an exception, or by executing the
 310  *  CLEANUP_CALL_AND_POP.
 311  *
 312  */
 313
