 123 /**
 124  * \brief Function to handle pktin messages
 125  * \param pktin Pktin message
 126  */
 127 static int l2_learning(const pktin_t *pktin)
 128 {
 129 #ifdef __ENABLE_CBENCH
 130     send_packet(pktin, PORT_FLOOD);
 131     return 0;
 132 #else /* !__ENABLE_CBENCH */
 133     if ((pktin->proto & (PROTO_ARP | PROTO_IPV4)) == 0) {
 134         discard_packet(pktin);
 135         return -1;
 136     }
 137 
 138     uint64_t mac = mac2int(pktin->src_mac);
 139     uint32_t mkey = hash_func((uint32_t *)&mac, 2) % MAC_HASH_SIZE;
 140 
 141     pthread_rwlock_rdlock(&mac_table[mkey].lock);
 142 
 143     // check source MAC
 144     mac_entry_t *curr = mac_table[mkey].head;
 145     while (curr != NULL) {
 146         if (curr->dpid == pktin->dpid && curr->mac == mac) {
 147             break;
 148         }
 149         curr = curr->next;
 150     }
 151 
 152     pthread_rwlock_unlock(&mac_table[mkey].lock);
 153 
 154     // new MAC?
 155     if (curr == NULL) {
 156         curr = mac_dequeue();
 157         if (curr == NULL) {
 158             discard_packet(pktin);
 159             return -1;
 160         }
 161 
 162         curr->dpid = pktin->dpid;
 163         curr->port = pktin->port;
 164 
 165         curr->ip = pktin->src_ip;
 166         curr->mac = mac;
 167 
 168         curr->prev = curr->next = NULL;
 169 
 170         pthread_rwlock_wrlock(&mac_table[mkey].lock);
 171 
 172         if (mac_table[mkey].head == NULL) {
 173             mac_table[mkey].head = curr;
 174             mac_table[mkey].tail = curr;
 175         } else {
 176             curr->prev = mac_table[mkey].tail;
 177             mac_table[mkey].tail->next = curr;
 178             mac_table[mkey].tail = curr;
 179         }
 180 
 181         pthread_rwlock_unlock(&mac_table[mkey].lock);
 182     }
 183 
 184     // get destination MAC
 185     mac = mac2int(pktin->dst_mac);
 186 
 187     // broadcast?
 188     if (mac == 0xffffffffffff) {
 189         send_packet(pktin, PORT_FLOOD);
 190         return 0;
 191     }
 192 
 193     uint16_t port = 0;
 194     mkey = hash_func((uint32_t *)&mac, 2) % MAC_HASH_SIZE;
 195 
 196     pthread_rwlock_rdlock(&mac_table[mkey].lock);
 197 
 198     // where to forward?
 199     curr = mac_table[mkey].head;
 200     while (curr != NULL) {
 201         if (curr->dpid == pktin->dpid && curr->mac == mac) {
 202             port = curr->port;
 203             break;
 204         }
 205         curr = curr->next;
 206     }
 207 
 208     pthread_rwlock_unlock(&mac_table[mkey].lock);
 209 
 210     // unknown port?
 211     if (port == 0) {
 212         send_packet(pktin, PORT_FLOOD);
 213         return 0;
 214     }
 215 
 216     if (pktin->proto & PROTO_IPV4) // IPv4
 217         insert_flow(pktin, port);
 218     else // ARP
 219         send_packet(pktin, port);
 220 
 221     return 0;
 222 #endif /* !__ENABLE_CBENCH */
 223 }

 227 /**
 228  * \brief The main function
 229  * \param activated The activation flag of this application
 230  * \param argc The number of arguments
 231  * \param argv Arguments
 232  */
 233 int l2_learning_main(int *activated, int argc, char **argv)
 234 {
 235     ALOG_INFO(L2_LEARNING_ID, "Init - L2 learning");
 236 
 237     mac_table = (mac_table_t *)CALLOC(MAC_HASH_SIZE, sizeof(mac_table_t));
 238     if (mac_table == NULL) {
 239         PERROR("calloc");
 240         return -1;
 241     }
 242 
 243     int i;
 244     for (i=0; i<MAC_HASH_SIZE; i++) {
 245         pthread_rwlock_init(&mac_table[i].lock, NULL);
 246     }
 247 
 248     mac_q_init();
 249 
 250     activate();
 251 
 252     return 0;
 253 }

 255 /**
 256  * \brief The cleanup function
 257  * \param activated The activation flag of this application
 258  */
 259 int l2_learning_cleanup(int *activated)
 260 {
 261     ALOG_INFO(L2_LEARNING_ID, "Clean up - L2 learning");
 262 
 263     deactivate();
 264 
 265     int i;
 266     for (i=0; i<MAC_HASH_SIZE; i++) {
 267         pthread_rwlock_wrlock(&mac_table[i].lock);
 268 
 269         mac_entry_t *curr = mac_table[i].head;
 270         while (curr != NULL) {
 271             mac_entry_t *tmp = curr;
 272             curr = curr->next;
 273             FREE(tmp);
 274         }
 275 
 276         pthread_rwlock_unlock(&mac_table[i].lock);
 277         pthread_rwlock_destroy(&mac_table[i].lock);
 278     }
 279 
 280     mac_q_destroy();
 281 
 282     FREE(mac_table);
 283 
 284     return 0;
 285 }

 287 /**
 288  * \brief Function to list up all MAC tables
 289  * \param cli The CLI pointer
 290  */
 291 static int list_all_entries(cli_t *cli)
 292 {
 293     cli_print(cli, "<MAC Tables>");
 294 
 295     int i, cnt = 0;
 296     for (i=0; i<MAC_HASH_SIZE; i++) {
 297         pthread_rwlock_rdlock(&mac_table[i].lock);
 298 
 299         mac_entry_t *curr = mac_table[i].head;
 300         while (curr != NULL) {
 301             uint8_t mac[ETH_ALEN];
 302             int2mac(curr->mac, mac);
 303 
 304             char macaddr[__CONF_WORD_LEN] = {0};
 305             mac2str(mac, macaddr);
 306 
 307             struct in_addr ip_addr;
 308             ip_addr.s_addr = curr->ip;
 309 
 310             cli_print(cli, "  %lu: %s (%s), %u", curr->dpid, macaddr, inet_ntoa(ip_addr), curr->port);
 311 
 312             cnt++;
 313 
 314             curr = curr->next;
 315         }
 316 
 317         pthread_rwlock_unlock(&mac_table[i].lock);
 318     }
 319 
 320     if (!cnt)
 321         cli_print(cli, "  No entry");
 322 
 323     return 0;
 324 }

 488 /**
 489  * \brief The handler function
 490  * \param av Read-only app event
 491  * \param av_out Read-write app event (if this application has the write permission)
 492  */
 493 int l2_learning_handler(const app_event_t *av, app_event_out_t *av_out)
 494 {
 495     switch (av->type) {
 496     case AV_DP_RECEIVE_PACKET:
 497         PRINT_EV("AV_DP_RECEIVE_PACKET\n");
 498         {
 499             const pktin_t *pktin = av->pktin;
 500 
 501             l2_learning(pktin);
 502         }
 503         break;
 504     case AV_DP_PORT_DELETED:
 505         PRINT_EV("AV_DP_PORT_DELETED\n");
 506         {
 507             const port_t *port = av->port;
 508 
 509             int i;
 510             for (i=0; i<MAC_HASH_SIZE; i++) {
 511                 mac_table_t tmp_list = {0};
 512 
 513                 pthread_rwlock_wrlock(&mac_table[i].lock);
 514 
 515                 mac_entry_t *curr = mac_table[i].head;
 516                 while (curr != NULL) {
 517                     if (curr->dpid == port->dpid && curr->port == port->port) {
 518                         if (tmp_list.head == NULL) {
 519                             tmp_list.head = curr;
 520                             tmp_list.tail = curr;
 521                             curr->r_next = NULL;
 522                         } else {
 523                             tmp_list.tail->r_next = curr;
 524                             tmp_list.tail = curr;
 525                             curr->r_next = NULL;
 526                         }
 527                     }
 528 
 529                     curr = curr->next;
 530                 }
 531 
 532                 curr = tmp_list.head;
 533                 while (curr != NULL) {
 534                     mac_entry_t *tmp = curr;
 535 
 536                     curr = curr->r_next;
 537 
 538                     if (tmp->prev != NULL && tmp->next != NULL) {
 539                         tmp->prev->next = tmp->next;
 540                         tmp->next->prev = tmp->prev;
 541                     } else if (tmp->prev == NULL && tmp->next != NULL) {
 542                         mac_table[i].head = tmp->next;
 543                         tmp->next->prev = NULL;
 544                     } else if (tmp->prev != NULL && tmp->next == NULL) {
 545                         mac_table[i].tail = tmp->prev;
 546                         tmp->prev->next = NULL;
 547                     } else if (tmp->prev == NULL && tmp->next == NULL) {
 548                         mac_table[i].head = NULL;
 549                         mac_table[i].tail = NULL;
 550                     }
 551 
 552                     mac_enqueue(tmp);
 553                 }
 554 
 555                 pthread_rwlock_unlock(&mac_table[i].lock);
 556             }
 557         }
 558         break;
 559     case AV_SW_DISCONNECTED:
 560         PRINT_EV("AV_SW_DISCONNECTED\n");
 561         {
 562             const switch_t *sw = av->sw;
 563 
 564             int i;
 565             for (i=0; i<MAC_HASH_SIZE; i++) {
 566                 mac_table_t tmp_list = {0};
 567 
 568                 pthread_rwlock_wrlock(&mac_table[i].lock);
 569 
 570                 mac_entry_t *curr = mac_table[i].head;
 571                 while (curr != NULL) {
 572                     if (curr->dpid == sw->dpid) {
 573                         if (tmp_list.head == NULL) {
 574                             tmp_list.head = curr;
 575                             tmp_list.tail = curr;
 576                             curr->r_next = NULL;
 577                         } else {
 578                             tmp_list.tail->r_next = curr;
 579                             tmp_list.tail = curr;
 580                             curr->r_next = NULL;
 581                         }
 582                     }
 583 
 584                     curr = curr->next;
 585                 }
 586 
 587                 curr = tmp_list.head;
 588                 while (curr != NULL) {
 589                     mac_entry_t *tmp = curr;
 590 
 591                     curr = curr->next;
 592 
 593                     if (tmp->prev != NULL && tmp->next != NULL) {
 594                         tmp->prev->next = tmp->next;
 595                         tmp->next->prev = tmp->prev;
 596                     } else if (tmp->prev == NULL && tmp->next != NULL) {
 597                         mac_table[i].head = tmp->next;
 598                         tmp->next->prev = NULL;
 599                     } else if (tmp->prev != NULL && tmp->next == NULL) {
 600                         mac_table[i].tail = tmp->prev;
 601                         tmp->prev->next = NULL;
 602                     } else if (tmp->prev == NULL && tmp->next == NULL) {
 603                         mac_table[i].head = NULL;
 604                         mac_table[i].tail = NULL;
 605                     }
 606 
 607                     mac_enqueue(tmp);
 608                 }
 609 
 610                 pthread_rwlock_unlock(&mac_table[i].lock);
 611             }
 612         }
 613         break;
 614     default:
 615         break;
 616     }
 617 
 618     return 0;
 619 }
