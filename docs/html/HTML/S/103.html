 113 int start(struct oflops_context * ctx) {
 114   struct timeval now;
 115   gettimeofday(&now, NULL);
 116   char *data;
 117   char msg[1024];
 118   int res;
 119 
 120   //init measurement queue
 121   TAILQ_INIT(&head); 
 122 
 123   //Initialize pcap-based  tcp flow reassembler for the communication 
 124   //channel
 125   msg_init();
 126   snprintf(msg, 1024,  "Intializing module %s", name());
 127 
 128   //log when I start module
 129   gettimeofday(&now, NULL);
 130   oflops_log(now, GENERIC_MSG, msg);
 131   oflops_log(now, GENERIC_MSG, cli_param);
 132 
 133   //start openflow session with switch
 134   make_ofp_hello((void *)&data);
 135   res = oflops_send_of_mesgs(ctx, data, sizeof(struct ofp_hello));
 136   free(data);  
 137 
 138   //send a message to clean up flow tables. 
 139   printf("cleaning up flow table...\n");
 140   res = make_ofp_flow_del((void *)&data);
 141   res = oflops_send_of_mesg(ctx, (void *)data);  
 142   free(data);
 143 
 144   //get port and cpu status from switch 
 145   gettimeofday(&now, NULL);
 146   add_time(&now, 1, 0);
 147   oflops_schedule_timer_event(ctx,&now, SNMPGET);
 148  
 149   //Schedule end
 150   gettimeofday(&now, NULL);
 151   add_time(&now, 1, 0);
 152   oflops_schedule_timer_event(ctx,&now, SND_PKT);
 153  
 154   //Schedule end
 155   gettimeofday(&now, NULL);
 156   add_time(&now, 61, 0);
 157   oflops_schedule_timer_event(ctx,&now, BYESTR);
 158 
 159   return 0;
 160 }

 162 /** Handle timer  
 163  * @param ctx pointer to opaque context
 164  * @param te pointer to timer event
 165  */
 166 int handle_timer_event(struct oflops_context * ctx, struct timer_event *te)
 167 {
 168   struct timeval now;
 169   char * str;
 170   int i;
 171 
 172   gettimeofday(&now,NULL);
 173   str = (char *) te->arg;
 174  
 175   if(!strcmp(str,SNMPGET)) {
 176     for(i=0;i<ctx->cpuOID_count;i++) {
 177       oflops_snmp_get(ctx, ctx->cpuOID[i], ctx->cpuOID_len[i]);
 178     }
 179     for(i=0;i<ctx->n_channels;i++) {
 180       oflops_snmp_get(ctx, ctx->channels[i].inOID, ctx->channels[i].inOID_len);
 181       oflops_snmp_get(ctx, ctx->channels[i].outOID, ctx->channels[i].outOID_len);
 182     }  
 183     gettimeofday(&now, NULL);
 184     add_time(&now, 1, 0);
 185     oflops_schedule_timer_event(ctx,&now, SNMPGET);
 186   } else if(!strcmp(str,BYESTR)) {
 187     oflops_end_test(ctx,1);
 188   } else if(!strcmp(str,SND_PKT)) {
 189     generate_pkt_out(ctx, &te->sched_time);
 190     write(ctx->control_fd, b, b_len);
 191     add_time(&now, probe_snd_interval/sec_to_usec, probe_snd_interval%sec_to_usec);
 192     oflops_schedule_timer_event(ctx,&now, SND_PKT);    
 193   } else
 194     fprintf(stderr, "Unknown timer event: %s", str);
 195   return 0;
 196 }

 198 int 
 199 destroy(oflops_context *ctx) {
 200   struct entry *np;
 201   uint32_t mean, median, variance, i;
 202   float loss;
 203   char msg[1024];
 204   double *data;
 205   struct timeval now;
 206 
 207   gettimeofday(&now, NULL);
 208 
 209   data = xmalloc(rcv_pkt_count*sizeof(double));
 210   i=0;
 211   for (np = head.tqh_first; np != NULL; np = np->entries.tqe_next) {
 212     data[i++] = (double)time_diff(&np->snd, &np->rcv);
 213     if(print) {
 214       snprintf(msg, 1024, "%lu.%06lu:%lu.%06lu:%d",
 215                np->snd.tv_sec, np->snd.tv_usec,
 216                np->rcv.tv_sec, np->rcv.tv_usec,
 217                np->id); 
 218       oflops_log(now, OFPT_PACKET_IN_MSG, msg);
 219     }
 220     free(np);
 221   }
 222   
 223   if(i > 0) {
 224     gsl_sort (data, 1, i);
 225     
 226     //calculating statistical measures
 227     mean = (uint32_t)gsl_stats_mean(data, 1, i);
 228     variance = (uint32_t)gsl_stats_variance(data, 1, i);
 229     median = (uint32_t)gsl_stats_median_from_sorted_data (data, 1, i);
 230     loss = (float)i/(float)pkt_counter;
 231 
 232     snprintf(msg, 1024, "statistics:%lu:%lu:%lu:%f:%d", (long unsigned)mean, (long unsigned)median, 
 233              (long unsigned)sqrt(variance), loss, i);
 234     printf("statistics:%lu:%lu:%lu:%f:%d", (long unsigned)mean, (long unsigned)median, 
 235            (long unsigned)variance, loss, i);
 236     oflops_log(now, GENERIC_MSG, msg);
 237   }
 238   return 0;
 239 }

 242 handle_snmp_event(struct oflops_context * ctx, struct snmp_event * se) {
 243   netsnmp_variable_list *vars;
 244   int i;
 245   char msg[1024], log_buf[1024];
 246   struct timeval now;
 247 
 248   for(vars = se->pdu->variables; vars; vars = vars->next_variable)  {
 249     snprint_value(msg, 1024, vars->name, vars->name_length, vars);
 250 
 251 
 252     for (i = 0; i < ctx->cpuOID_count; i++) {
 253       if((vars->name_length == ctx->cpuOID_len[i]) &&
 254          (memcmp(vars->name, ctx->cpuOID[i],  ctx->cpuOID_len[i] * sizeof(oid)) == 0) ) {
 255         snprintf(log_buf, 1024, "cpu:%ld:%ld:%s",
 256                  se->pdu->reqid, 
 257                  vars->name[ vars->name_length - 1],msg);
 258         oflops_log(now, SNMP_MSG, log_buf);
 259       }
 260     } 
 261       
 262     for(i=0;i<ctx->n_channels;i++) {
 263       if((vars->name_length == ctx->channels[i].inOID_len) &&
 264          (memcmp(vars->name, ctx->channels[i].inOID,  
 265                  ctx->channels[i].inOID_len * sizeof(oid)) == 0) ) {
 266         snprintf(log_buf, 1024, "port:rx:%ld:%d:%s",  
 267                  se->pdu->reqid, 
 268                  (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 269         oflops_log(now, SNMP_MSG, log_buf);
 270         break;
 271       }
 272         
 273       if((vars->name_length == ctx->channels[i].outOID_len) &&
 274          (memcmp(vars->name, ctx->channels[i].outOID,  
 275                  ctx->channels[i].outOID_len * sizeof(oid))==0) ) {
 276         snprintf(log_buf, 1024, "port:tx:%ld:%d:%s",  
 277                  se->pdu->reqid, 
 278                  (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 279         oflops_log(now, SNMP_MSG, log_buf);
 280         break;
 281       }
 282     } //for
 283   }
 284   return 0;
 285 }

 288  * Initialization code with parameters
 289  * @param ctx 
 290  */
 291 int init(struct oflops_context *ctx, char * config_str) {
 292   char *pos = NULL;
 293   char *param = config_str;
 294   char *value = NULL;
 295   struct timeval now;
 296 
 297   //init counters
 298   finished = 0;
 299   gettimeofday(&now, NULL);
 300   cli_param = strdup(config_str);
 301 
 302   while(*config_str == ' ') {
 303     config_str++;
 304   }
 305   param = config_str;
 306   while(1) {
 307     pos = index(param, ' ');
 308 
 309     if((pos == NULL)) {
 310       if (*param != '\0') {
 311         pos = param + strlen(param) + 1;
 312       } else
 313         break;
 314     }
 315     *pos='\0';
 316     pos++;
 317     value = index(param,'=');
 318     *value = '\0';
 319     value++;
 320     //fprintf(stderr, "param = %s, value = %s\n", param, value);
 321     if(value != NULL) {
 322       if(strcmp(param, "pkt_size") == 0) {
 323         //parse int to get pkt size
 324         pkt_size = strtol(value, NULL, 0);
 325         if((pkt_size < MIN_PKT_SIZE) && (pkt_size > MAX_PKT_SIZE))
 326           perror_and_exit("Invalid packet size value", 1);
 327       }  else if(strcmp(param, "probe_rate") == 0) {
 328         //parse int to get measurement probe rate
 329         proberate = strtol(value, NULL, 0);
 330         if((proberate <= 0) || (proberate >= 1010)) 
 331           perror_and_exit("Invalid probe rate param(Value between 1 and 1010)", 1);
 332       } else if(strcmp(param, "flows") == 0) {
 333         //parse int to get pkt size
 334         flows = strtol(value, NULL, 0);
 335         if(flows <= 0)  
 336           perror_and_exit("Invalid flow number", 1);
 337       } else if(strcmp(param, "print") == 0) {
 338         //parse int to get pkt size
 339         print = strtol(value, NULL, 0);
 340       } else 
 341         fprintf(stderr, "Invalid parameter:%s\n", param);
 342       param = pos;
 343     }
 344   } 
 345 
 346   //calculate sendind interval
 347   probe_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (proberate * mbits_to_bits);
 348   fprintf(stderr, "Sending probe interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 349           (uint32_t)probe_snd_interval, (uint32_t)pkt_size, (uint32_t)proberate);
 350   return 0;
 351 }

 353 int
 354 generate_pkt_out(struct oflops_context * ctx,struct timeval *now) {
 355   struct ofp_action_output *act_out;
 356   if(b == NULL) {
 357     b_len = sizeof(struct ofp_packet_out) + sizeof(struct ofp_action_output) + pkt_size;
 358     b = (char *)xmalloc(b_len*sizeof(char));
 359     
 360     //setting up openflow packet out fields
 361     pkt_out = (struct ofp_packet_out *)b;
 362     pkt_out->header.version = OFP_VERSION;
 363     pkt_out->header.type =  OFPT_PACKET_OUT;
 364     pkt_out->header.length =  htons(b_len);
 365     pkt_out->header.xid = 0;
 366     
 367     pkt_out->buffer_id = htonl(-1);
 368     pkt_out->in_port = htons(OFPP_NONE);
 369     pkt_out->actions_len = htons(sizeof(struct ofp_action_output));
 370     
 371     //pkt_out->actions = xmalloc(sizeof(struct ofp_action_output));
 372     act_out = (struct ofp_action_output *) pkt_out->actions;
 373     act_out->type = htons(OFPAT_OUTPUT);
 374     act_out->len = htons(8);      
 375     act_out->port = htons(ctx->channels[OFLOPS_DATA1].of_port);
 376     act_out->max_len = htons(2000);
 377     
 378     //setting up the ethernet header
 379     ether = (struct ether_header *)(b + sizeof(struct ofp_packet_out) + 
 380                                     sizeof(struct ofp_action_output));
 381     memcpy(ether->ether_shost, "\x00\x1e\x68\x9a\xc5\x75", ETH_ALEN);
 382     memcpy(ether->ether_dhost, local_mac, ETH_ALEN);
 383     ether->ether_type = htons(ETHERTYPE_IP);
 384     
 385     //setting up the ip header
 386     ip = (struct iphdr *)(b + sizeof(struct ofp_packet_out) + 
 387                           sizeof(struct ofp_action_output) + sizeof(struct ether_header));
 388     ip->protocol=1;
 389     ip->ihl=5;
 390     ip->version=4;
 391     ip->ttl = 100;
 392     ip->protocol = IPPROTO_UDP; //udp protocol
 393     ip->saddr = inet_addr("10.1.1.1"); 
 394     ip->daddr = inet_addr("192.168.3.0"); //test.nw_dst;
 395     ip->tot_len = htons(pkt_size - sizeof(struct ether_header));
 396     
 397     //setting up the udp header
 398     udp = (struct udphdr *)(b + sizeof(struct ofp_packet_out) + 
 399                             sizeof(struct ofp_action_output) + sizeof(struct ether_header) +
 400                             sizeof(struct iphdr));
 401     udp->source = htons(8080);
 402     udp->dest = htons(8080);
 403     udp->len = htons(pkt_size - sizeof(struct ether_header) 
 404                       - sizeof(struct iphdr));
 405 
 406     //setting up pktgen header
 407     pktgen = (struct pktgen_hdr *)(b + sizeof(struct ofp_packet_out) + 
 408                                    sizeof(struct ofp_action_output) + sizeof(struct ether_header) +
 409                                    sizeof(struct iphdr) + sizeof(struct udphdr));
 410     pktgen->magic = 0xbe9be955;
 411   }
 412 
 413   
 414   ip->daddr = htonl(ntohl(inet_addr("192.168.3.0")) + (rand()%(flows)) ); //test.nw_dst;
 415   ip->check=ip_sum_calc(20, (void *)ip);
 416   pktgen->tv_sec = htonl(now->tv_sec);
 417   pktgen->tv_usec = htonl(now->tv_usec);
 418   pktgen->seq_num = htonl(++pkt_counter);
 419 
 420   return 1;
 421 }

 438 /** Handle pcap event.
 439  * @param ctx pointer to opaque context
 440  * @param pe pcap event
 441  * @param ch enumeration of channel that pcap event is triggered
 442  */
 443 int handle_pcap_event(struct oflops_context *ctx, struct pcap_event *pe,
 444                       oflops_channel_name ch) {
 445   struct flow fl;
 446   if (ch == OFLOPS_DATA1) {
 447     struct pktgen_hdr *pkt;
 448     pkt = extract_pktgen_pkt((unsigned char *)pe->data, pe->pcaphdr.caplen, &fl);
 449     if(pktgen == NULL) //skip non IP packets
 450       return 0;
 451     
 452     struct entry *n1 = malloc(sizeof(struct entry));
 453     n1->snd.tv_sec = htonl(pkt->tv_sec);
 454     n1->snd.tv_usec = htonl(pkt->tv_usec);
 455     memcpy(&n1->rcv, &pe->pcaphdr.ts, sizeof(struct timeval));
 456     n1->ip = fl.nw_src;
 457     n1->id = ntohl(pkt->seq_num);
 458     rcv_pkt_count++;
 459     TAILQ_INSERT_TAIL(&head, n1, entries);
 460   }
 461   return 0;
 462 }
