 107 void
 108 leak_checker_start(const char *file_name)
 109 {
 110     if (!file) {
 111         file = fopen(file_name, "w");
 112         if (!file) {
 113             VLOG_WARN("failed to create \"%s\": %s",
 114                       file_name, strerror(errno));
 115             return;
 116         }
 117         setvbuf(file, NULL, _IONBF, 0);
 118         VLOG_WARN("enabled memory leak logging to \"%s\"", file_name);
 119         get_hooks(&libc_hooks);
 120         set_hooks(&our_hooks);
 121     }
 122 }
 123 
 124 void
 125 leak_checker_stop(void)
 126 {
 127     if (file) {
 128         fclose(file);
 129         file = NULL;
 130         set_hooks(&libc_hooks);
 131         VLOG_WARN("disabled memory leak logging");
 132     }
 133 }

 147 static void PRINTF_FORMAT(1, 2)
 148 log_callers(const char *format, ...)
 149 {
 150     struct backtrace backtrace;
 151     va_list args;
 152     int i;
 153 
 154     va_start(args, format);
 155     vfprintf(file, format, args);
 156     va_end(args);
 157 
 158     putc(':', file);
 159     backtrace_capture(&backtrace);
 160     for (i = 0; i < backtrace.n_frames; i++) {
 161         fprintf(file, " 0x%x", backtrace.frames[i]);
 162     }
 163     putc('\n', file);
 164 }

 166 static void
 167 reset_hooks(void)
 168 {
 169     static int count;
 170 
 171     if (count++ >= 100 && limit && file) {
 172         struct stat s;
 173         count = 0;
 174         if (fstat(fileno(file), &s) < 0) {
 175             VLOG_WARN("cannot fstat leak checker log file: %s",
 176                       strerror(errno));
 177             return;
 178         }
 179         if (s.st_size > limit) {
 180             VLOG_WARN("leak checker log file size exceeded limit");
 181             leak_checker_stop();
 182             return;
 183         }
 184     }
 185     if (file) {
 186         set_hooks(&our_hooks);
 187     }
 188 }

 190 static void *
 191 hook_malloc(size_t size, const void *caller UNUSED)
 192 {
 193     void *p;
 194 
 195     set_hooks(&libc_hooks);
 196     p = malloc(size);
 197     get_hooks(&libc_hooks);
 198 
 199     log_callers("malloc(%zu) -> %p", size, p);
 200 
 201     reset_hooks();
 202     return p;
 203 }

 205 void
 206 leak_checker_claim(const void *p)
 207 {
 208     if (!file) {
 209         return;
 210     }
 211 
 212     if (p) {
 213         set_hooks(&libc_hooks);
 214         log_callers("claim(%p)", p);
 215         reset_hooks();
 216     }
 217 }

 219 static void
 220 hook_free(void *p, const void *caller UNUSED)
 221 {
 222     if (!p) {
 223         return;

 225

 235 static void *
 236 hook_realloc(void *p, size_t size, const void *caller UNUSED)
 237 {
 238     void *q;
 239 
 240     set_hooks(&libc_hooks);
 241     q = realloc(p, size);
 242     get_hooks(&libc_hooks);
 243 
 244     if (p != q) {
 245         log_callers("realloc(%p, %zu) -> %p", p, size, q);
 246     }
 247 
 248     reset_hooks();
 249 
 250     return q;
 251 }
