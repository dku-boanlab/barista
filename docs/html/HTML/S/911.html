 131 void
 132 svec_unique(struct svec *svec)
 133 {
 134     assert(svec_is_sorted(svec));
 135     if (svec->n > 1) {
 136         /* This algorithm is lazy and sub-optimal, but it's "obviously correct"
 137          * and asymptotically optimal . */
 138         struct svec tmp;
 139         size_t i;
 140 
 141         svec_init(&tmp);
 142         svec_add(&tmp, svec->names[0]);
 143         for (i = 1; i < svec->n; i++) {
 144             if (strcmp(svec->names[i - 1], svec->names[i])) {
 145                 svec_add(&tmp, svec->names[i]);
 146             }
 147         }
 148         svec_swap(&tmp, svec);
 149         svec_destroy(&tmp);
 150     }
 151 }

 153 void
 154 svec_diff(const struct svec *a, const struct svec *b,
 155           struct svec *a_only, struct svec *both, struct svec *b_only)
 156 {
 157     size_t i, j;
 158 
 159     assert(svec_is_sorted(a));
 160     assert(svec_is_sorted(b));
 161     if (a_only) {
 162         svec_init(a_only);
 163     }
 164     if (both) {
 165         svec_init(both);
 166     }
 167     if (b_only) {
 168         svec_init(b_only);
 169     }
 170     for (i = j = 0; i < a->n && j < b->n; ) {
 171         int cmp = strcmp(a->names[i], b->names[j]);
 172         if (cmp < 0) {
 173             if (a_only) {
 174                 svec_add(a_only, a->names[i]);
 175             }
 176             i++;
 177         } else if (cmp > 0) {
 178             if (b_only) {
 179                 svec_add(b_only, b->names[j]);
 180             }
 181             j++;
 182         } else {
 183             if (both) {
 184                 svec_add(both, a->names[i]);
 185             }
 186             i++;
 187             j++;
 188         }
 189     }
 190     if (a_only) {
 191         for (; i < a->n; i++) {
 192             svec_add(a_only, a->names[i]);
 193         }
 194     }
 195     if (b_only) {
 196         for (; j < b->n; j++) {
 197             svec_add(b_only, b->names[j]);
 198         }
 199     }
 200 }

 208 size_t
 209 svec_find(const struct svec *svec, const char *name)
 210 {
 211     char **p;
 212 
 213     assert(svec_is_sorted(svec));
 214     p = bsearch(&name, svec->names, svec->n, sizeof *svec->names,
 215                 compare_strings);
 216     return p ? p - svec->names : SIZE_MAX;
 217 }

 219 bool
 220 svec_is_sorted(const struct svec *svec)
 221 {
 222     size_t i;
 223 
 224     for (i = 1; i < svec->n; i++) {
 225         if (strcmp(svec->names[i - 1], svec->names[i]) > 0) {
 226             return false;
 227         }
 228     }
 229     return true;
 230 }

 238 const char *
 239 svec_get_duplicate(const struct svec *svec)
 240 {
 241     assert(svec_is_sorted(svec));
 242     if (svec->n > 1) {
 243         size_t i;
 244         for (i = 1; i < svec->n; i++) {
 245             if (!strcmp(svec->names[i - 1], svec->names[i])) {
 246                 return svec->names[i];
 247             }
 248         }
 249     }
 250     return NULL;
 251 }

 272 /* Breaks 'words' into words at white space, respecting shell-like quoting
 273  * conventions, and appends the words to 'svec'. */
 274 void
 275 svec_parse_words(struct svec *svec, const char *words)
 276 {
 277     struct ds word = DS_EMPTY_INITIALIZER;
 278     const char *p, *q;
 279 
 280     for (p = words; *p != '\0'; p = q) {
 281         int quote = 0;
 282 
 283         while (isspace((unsigned char) *p)) {
 284             p++;
 285         }
 286         if (*p == '\0') {
 287             break;
 288         }
 289 
 290         ds_clear(&word);
 291         for (q = p; *q != '\0'; q++) {
 292             if (*q == quote) {
 293                 quote = 0;
 294             } else if (*q == '\'' || *q == '"') {
 295                 quote = *q;
 296             } else if (*q == '\\' && (!quote || quote == '"')) {
 297                 q++;
 298                 if (*q == '\0') {
 299                     VLOG_WARN("%s: ends in trailing backslash", words);
 300                     break;
 301                 }
 302                 ds_put_char(&word, *q);
 303             } else if (isspace((unsigned char) *q) && !quote) {
 304                 q++;
 305                 break;
 306             } else {
 307                 ds_put_char(&word, *q);
 308             }
 309         }
 310         svec_add(svec, ds_cstr(&word));
 311         if (quote) {
 312             VLOG_WARN("%s: word ends inside quoted string", words);
 313         }
 314     }
 315     ds_destroy(&word);
 316 }

 318 bool
 319 svec_equal(const struct svec *a, const struct svec *b)
 320 {
 321     size_t i;
 322 
 323     if (a->n != b->n) {
 324         return false;
 325     }
 326     for (i = 0; i < a->n; i++) {
 327         if (strcmp(a->names[i], b->names[i])) {
 328             return false;
 329         }
 330     }
 331     return true;
 332 }

 334 char *
 335 svec_join(const struct svec *svec, const char *delimiter)
 336 {
 337     struct ds ds;
 338     size_t i;
 339 
 340     ds_init(&ds);
 341     for (i = 0; i < svec->n; i++) {
 342         if (i) {
 343             ds_put_cstr(&ds, delimiter);
 344         }
 345         ds_put_cstr(&ds, svec->names[i]);
 346     }
 347     return ds_cstr(&ds);
 348 }
