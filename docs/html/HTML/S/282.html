  29 /* 'skb' was received on port 'p', which may be a physical switch port, the
  30  * local port, or a null pointer.  Process it according to 'chain'.  Returns 0
  31  * if successful, in which case 'skb' is destroyed, or -ESRCH if there is no
  32  * matching flow, in which case 'skb' still belongs to the caller. */
  33 int run_flow_through_tables(struct sw_chain *chain, struct sk_buff *skb,
  34                             struct net_bridge_port *p)
  35 {
  36         /* Ethernet address used as the destination for STP frames. */
  37         static const uint8_t stp_eth_addr[ETH_ALEN]
  38                 = { 0x01, 0x80, 0xC2, 0x00, 0x00, 0x01 };
  39         struct sw_flow_key key;
  40         struct sw_flow *flow;
  41 
  42         if (flow_extract(skb, p ? p->port_no : OFPP_NONE, &key)
  43             && (chain->dp->flags & OFPC_FRAG_MASK) == OFPC_FRAG_DROP) {
  44                 /* Drop fragment. */
  45                 kfree_skb(skb);
  46                 return 0;
  47         }
  48         if (p && p->config & (OFPPC_NO_RECV | OFPPC_NO_RECV_STP) &&
  49             p->config & (compare_ether_addr(key.dl_dst, stp_eth_addr)
  50                         ? OFPPC_NO_RECV : OFPPC_NO_RECV_STP)) {
  51                 kfree_skb(skb);
  52                 return 0;
  53         }
  54 
  55         flow = chain_lookup(chain, &key, 0);
  56         if (likely(flow != NULL)) {
  57                 struct sw_flow_actions *sf_acts = rcu_dereference(flow->sf_acts);
  58                 flow_used(flow, skb);
  59                 execute_actions(chain->dp, skb, &key,
  60                                 sf_acts->actions, sf_acts->actions_len, 0);
  61                 return 0;
  62         } else {
  63                 return -ESRCH;
  64         }
  65 }

  67 /* 'skb' was received on port 'p', which may be a physical switch port, the
  68  * local port, or a null pointer.  Process it according to 'chain', sending it
  69  * up to the controller if no flow matches.  Takes ownership of 'skb'. */
  70 void fwd_port_input(struct sw_chain *chain, struct sk_buff *skb,
  71                     struct net_bridge_port *p)
  72 {
  73         WARN_ON_ONCE(skb_shared(skb));
  74         WARN_ON_ONCE(skb->destructor);
  75         if (run_flow_through_tables(chain, skb, p))
  76                 dp_output_control(chain->dp, skb, chain->dp->miss_send_len,
  77                                   OFPR_NO_MATCH);
  78 }

 127 static int
 128 recv_packet_out(struct sw_chain *chain, const struct sender *sender,
 129                 const void *msg)
 130 {
 131         const struct ofp_packet_out *opo = msg;
 132         struct sk_buff *skb;
 133         uint16_t v_code;
 134         struct sw_flow_key key;
 135         size_t actions_len = ntohs(opo->actions_len);
 136 
 137         if (actions_len > (ntohs(opo->header.length) - sizeof *opo)) {
 138                 if (net_ratelimit()) 
 139                         printk(KERN_NOTICE "%s: message too short for number "
 140                                "of actions\n", chain->dp->netdev->name);
 141                 return -EINVAL;
 142         }
 143 
 144         if (ntohl(opo->buffer_id) == (uint32_t) -1) {
 145                 int data_len = ntohs(opo->header.length) - sizeof *opo - actions_len;
 146 
 147                 /* FIXME: there is likely a way to reuse the data in msg. */
 148                 skb = alloc_skb(data_len, GFP_ATOMIC);
 149                 if (!skb)
 150                         return -ENOMEM;
 151 
 152                 /* FIXME?  We don't reserve NET_IP_ALIGN or NET_SKB_PAD since
 153                  * we're just transmitting this raw without examining anything
 154                  * at those layers. */
 155                 skb_put(skb, data_len);
 156                 skb_copy_to_linear_data(skb,
 157                                         (uint8_t *)opo->actions + actions_len, 
 158                                         data_len);
 159                 skb_reset_mac_header(skb);
 160         } else {
 161                 skb = retrieve_skb(ntohl(opo->buffer_id));
 162                 if (!skb)
 163                         return -ESRCH;
 164         }
 165 
 166         dp_set_origin(chain->dp, ntohs(opo->in_port), skb);
 167 
 168         flow_extract(skb, ntohs(opo->in_port), &key);
 169 
 170         v_code = validate_actions(chain->dp, &key, opo->actions, actions_len);
 171         if (v_code != ACT_VALIDATION_OK) {
 172                 dp_send_error_msg(chain->dp, sender, OFPET_BAD_ACTION, v_code,
 173                                   msg, ntohs(opo->header.length));
 174                 goto error;
 175         }
 176 
 177         execute_actions(chain->dp, skb, &key, opo->actions, actions_len, 1);
 178 
 179         return 0;
 180 
 181 error:
 182         kfree_skb(skb);
 183         return -EINVAL;
 184 }

 211 static int
 212 add_flow(struct sw_chain *chain, const struct sender *sender,
 213                 const struct ofp_flow_mod *ofm)
 214 {
 215         int error = -ENOMEM;
 216         uint16_t v_code;
 217         struct sw_flow *flow;
 218         size_t actions_len = ntohs(ofm->header.length) - sizeof *ofm;
 219         int overlap;
 220 
 221         /* Allocate memory. */
 222         flow = flow_alloc(actions_len, GFP_ATOMIC);
 223         if (flow == NULL)
 224                 goto error;
 225 
 226         flow_extract_match(&flow->key, &ofm->match);
 227 
 228         v_code = validate_actions(chain->dp, &flow->key, ofm->actions, actions_len);
 229         if (v_code != ACT_VALIDATION_OK) {
 230                 dp_send_error_msg(chain->dp, sender, OFPET_BAD_ACTION, v_code,
 231                                   ofm, ntohs(ofm->header.length));
 232                 goto error_free_flow;
 233         }
 234 
 235         flow->priority = flow->key.wildcards ? ntohs(ofm->priority) : -1;
 236 
 237         if (ntohs(ofm->flags) & OFPFF_CHECK_OVERLAP) {
 238                 /* check whether there is any conflict */
 239                 overlap = chain_has_conflict(chain, &flow->key, flow->priority,
 240                                              false);
 241                 if (overlap){
 242                         dp_send_error_msg(chain->dp, sender,
 243                                           OFPET_FLOW_MOD_FAILED,
 244                                           OFPFMFC_OVERLAP,
 245                                           ofm, ntohs(ofm->header.length));
 246                         goto error_free_flow;
 247                 }
 248         }
 249 
 250         if (ntohs(ofm->flags) & OFPFF_EMERG) {
 251                 if (ntohs(ofm->idle_timeout) != OFP_FLOW_PERMANENT
 252                     || ntohs(ofm->hard_timeout) != OFP_FLOW_PERMANENT) {
 253                         dp_send_error_msg(chain->dp, sender,
 254                                           OFPET_FLOW_MOD_FAILED,
 255                                           OFPFMFC_BAD_EMERG_TIMEOUT,
 256                                           ofm, ntohs(ofm->header.length));
 257                         goto error_free_flow;
 258                 }
 259         }
 260 
 261         /* Fill out flow. */
 262         flow->idle_timeout = ntohs(ofm->idle_timeout);
 263         flow->hard_timeout = ntohs(ofm->hard_timeout);
 264         flow->send_flow_rem = (ntohs(ofm->flags) & OFPFF_SEND_FLOW_REM) ? 1 : 0;
 265         flow->emerg_flow = (ntohs(ofm->flags) & OFPFF_EMERG) ? 1 : 0;
 266         flow_setup_actions(flow, ofm->actions, actions_len);
 267 
 268         /* Act. */
 269         error = chain_insert(chain, flow,
 270                              (ntohs(ofm->flags) & OFPFF_EMERG) ? 1 : 0);
 271         if (error == -ENOBUFS) {
 272                 dp_send_error_msg(chain->dp, sender, OFPET_FLOW_MOD_FAILED, 
 273                                 OFPFMFC_ALL_TABLES_FULL, ofm, ntohs(ofm->header.length));
 274                 goto error_free_flow;
 275         } else if (error)
 276                 goto error_free_flow;
 277 
 278         error = 0;
 279         if (ntohl(ofm->buffer_id) != (uint32_t) -1) {
 280                 struct sk_buff *skb = retrieve_skb(ntohl(ofm->buffer_id));
 281                 if (skb) {
 282                         struct sw_flow_key key;
 283                         flow_used(flow, skb);
 284                         dp_set_origin(chain->dp, ntohs(ofm->match.in_port), skb);
 285                         flow_extract(skb, ntohs(ofm->match.in_port), &key);
 286                         execute_actions(chain->dp, skb, &key, ofm->actions, actions_len, 0);
 287                 }
 288                 else
 289                         error = -ESRCH;
 290         }
 291         return error;
 292 
 293 error_free_flow:
 294         flow_free(flow);
 295 error:
 296         if (ntohl(ofm->buffer_id) != (uint32_t) -1)
 297                 discard_skb(ntohl(ofm->buffer_id));
 298         return error;
 299 }

 301 static int
 302 mod_flow(struct sw_chain *chain, const struct sender *sender,
 303                 const struct ofp_flow_mod *ofm)
 304 {
 305         int error = -ENOMEM;
 306         uint16_t v_code;
 307         size_t actions_len = ntohs(ofm->header.length) - sizeof *ofm;
 308         struct sw_flow *flow;
 309         int strict;
 310 
 311         /* Allocate memory. */
 312         flow = flow_alloc(actions_len,GFP_ATOMIC);
 313         if (flow == NULL)
 314                 goto error;
 315 
 316         flow_extract_match(&flow->key, &ofm->match);
 317 
 318         v_code = validate_actions(chain->dp, &flow->key, ofm->actions, actions_len);
 319         if (v_code != ACT_VALIDATION_OK) {
 320                 dp_send_error_msg(chain->dp, sender, OFPET_BAD_ACTION, v_code,
 321                                   ofm, ntohs(ofm->header.length));
 322                 goto error_free_flow;
 323         }
 324 
 325         flow->priority = flow->key.wildcards ? ntohs(ofm->priority) : -1;
 326         strict = (ofm->command == htons(OFPFC_MODIFY_STRICT)) ? 1 : 0;
 327 
 328         /* First try to modify existing flows if any */
 329         /* if there is no matching flow, add it */ 
 330         if (!chain_modify(chain, &flow->key, flow->priority, strict,
 331                           ofm->actions, actions_len,
 332                           (ntohs(ofm->flags) & OFPFF_EMERG) ? 1 : 0)) {
 333                 /* Fill out flow. */
 334                 flow->idle_timeout = ntohs(ofm->idle_timeout);
 335                 flow->hard_timeout = ntohs(ofm->hard_timeout);
 336                 flow->send_flow_rem = (ntohs(ofm->flags) & OFPFF_SEND_FLOW_REM)
 337                         ? 1 : 0;
 338                 flow->emerg_flow = (ntohs(ofm->flags) & OFPFF_EMERG) ? 1 : 0;
 339                 flow_setup_actions(flow, ofm->actions, actions_len);
 340                 error = chain_insert(chain, flow,
 341                                      (ntohs(ofm->flags) & OFPFF_EMERG) ? 1 : 0);
 342                 if (error == -ENOBUFS) {
 343                         dp_send_error_msg(chain->dp, sender,
 344                                           OFPET_FLOW_MOD_FAILED,
 345                                           OFPFMFC_ALL_TABLES_FULL,
 346                                           ofm, ntohs(ofm->header.length));
 347                         goto error_free_flow;
 348                 } else if (error) {
 349                         goto error_free_flow;
 350                 }
 351         }
 352 
 353         error = 0;
 354         if (ntohl(ofm->buffer_id) != (uint32_t) -1) {
 355                 struct sk_buff *skb = retrieve_skb(ntohl(ofm->buffer_id));
 356                 if (skb) {
 357                         struct sw_flow_key skb_key;
 358                         flow_extract(skb, ntohs(ofm->match.in_port), &skb_key);
 359                         execute_actions(chain->dp, skb, &skb_key, 
 360                                         ofm->actions, actions_len, 0);
 361                 }
 362                 else
 363                         error = -ESRCH;
 364         }
 365         return error;
 366 
 367 error_free_flow:
 368         flow_free(flow);
 369 error:
 370         if (ntohl(ofm->buffer_id) != (uint32_t) -1)
 371                 discard_skb(ntohl(ofm->buffer_id));
 372         return error;
 373 }

 375 static int
 376 recv_flow(struct sw_chain *chain, const struct sender *sender, const void *msg)
 377 {
 378         const struct ofp_flow_mod *ofm = msg;
 379         uint16_t command = ntohs(ofm->command);
 380 
 381         if (command == OFPFC_ADD) {
 382                 return add_flow(chain, sender, ofm);
 383         } else if ((command == OFPFC_MODIFY) || (command == OFPFC_MODIFY_STRICT)) {
 384                 return mod_flow(chain, sender, ofm);
 385         }  else if (command == OFPFC_DELETE) {
 386                 struct sw_flow_key key;
 387                 flow_extract_match(&key, &ofm->match);
 388                 return chain_delete(chain, &key, ofm->out_port, 0, 0,
 389                                     (ntohs(ofm->flags) & OFPFF_EMERG) ? 1 : 0)
 390                         ? 0 : -ESRCH;
 391         } else if (command == OFPFC_DELETE_STRICT) {
 392                 struct sw_flow_key key;
 393                 uint16_t priority;
 394                 flow_extract_match(&key, &ofm->match);
 395                 priority = key.wildcards ? ntohs(ofm->priority) : -1;
 396                 return chain_delete(chain, &key, ofm->out_port, priority, 1,
 397                                     (ntohs(ofm->flags) & OFPFF_EMERG) ? 1 : 0)
 398                         ? 0 : -ESRCH;
 399         } else {
 400                 return -ENOTSUPP;
 401         }
 402 }

 426 /* 'msg', which is 'length' bytes long, was received across Netlink from
 427  * 'sender'.  Apply it to 'chain'. */
 428 int
 429 fwd_control_input(struct sw_chain *chain, const struct sender *sender,
 430                   const void *msg, size_t length)
 431 {
 432 
 433         struct openflow_packet {
 434                 size_t min_size;
 435                 int (*handler)(struct sw_chain *, const struct sender *,
 436                                const void *);
 437         };
 438 
 439         static const struct openflow_packet packets[] = {
 440                 [OFPT_HELLO] = {
 441                         sizeof (struct ofp_header),
 442                         recv_hello,
 443                 },
 444                 [OFPT_BARRIER_REQUEST] = {
 445                         sizeof (struct ofp_header),
 446                         recv_barrier_request,
 447                 },
 448                 [OFPT_ECHO_REQUEST] = {
 449                         sizeof (struct ofp_header),
 450                         recv_echo_request,
 451                 },
 452                 [OFPT_ECHO_REPLY] = {
 453                         sizeof (struct ofp_header),
 454                         recv_echo_reply,
 455                 },
 456                 [OFPT_VENDOR] = {
 457                         sizeof (struct ofp_vendor_header),
 458                         recv_vendor,
 459                 },
 460                 [OFPT_FEATURES_REQUEST] = {
 461                         sizeof (struct ofp_header),
 462                         recv_features_request,
 463                 },
 464                 [OFPT_GET_CONFIG_REQUEST] = {
 465                         sizeof (struct ofp_header),
 466                         recv_get_config_request,
 467                 },
 468                 [OFPT_SET_CONFIG] = {
 469                         sizeof (struct ofp_switch_config),
 470                         recv_set_config,
 471                 },
 472                 [OFPT_PACKET_OUT] = {
 473                         sizeof (struct ofp_packet_out),
 474                         recv_packet_out,
 475                 },
 476                 [OFPT_FLOW_MOD] = {
 477                         sizeof (struct ofp_flow_mod),
 478                         recv_flow,
 479                 },
 480                 [OFPT_PORT_MOD] = {
 481                         sizeof (struct ofp_port_mod),
 482                         recv_port_mod,
 483                 }
 484         };
 485 
 486         struct ofp_header *oh;
 487 
 488         oh = (struct ofp_header *) msg;
 489         if (oh->version != OFP_VERSION
 490             && oh->type != OFPT_HELLO
 491             && oh->type != OFPT_ERROR
 492             && oh->type != OFPT_ECHO_REQUEST
 493             && oh->type != OFPT_ECHO_REPLY
 494             && oh->type != OFPT_VENDOR)
 495         {
 496                 dp_send_error_msg(chain->dp, sender, OFPET_BAD_REQUEST,
 497                                   OFPBRC_BAD_VERSION, msg, length);
 498                 return -EINVAL;
 499         }
 500         if (ntohs(oh->length) != length) {
 501                 if (net_ratelimit())
 502                         printk(KERN_NOTICE "%s: received message length "
 503                                "wrong: %d/%d\n", chain->dp->netdev->name,
 504                                ntohs(oh->length), length);
 505                 return -EINVAL;
 506         }
 507 
 508         if (oh->type < ARRAY_SIZE(packets)) {
 509                 const struct openflow_packet *pkt = &packets[oh->type];
 510                 if (pkt->handler) {
 511                         if (length < pkt->min_size)
 512                                 return -EFAULT;
 513                         return pkt->handler(chain, sender, msg);
 514                 }
 515         }
 516         dp_send_error_msg(chain->dp, sender, OFPET_BAD_REQUEST,
 517                           OFPBRC_BAD_TYPE, msg, length);
 518         return -EINVAL;
 519 }

 521 /* Packet buffering. */

 523 #define OVERWRITE_SECS  1
 524 #define OVERWRITE_JIFFIES (OVERWRITE_SECS * HZ)
 525 
 526 struct packet_buffer {
 527         struct sk_buff *skb;
 528         uint32_t cookie;
 529         unsigned long exp_jiffies;
 530 };
 531 
 532 static struct packet_buffer buffers[N_PKT_BUFFERS];
 533 static unsigned int buffer_idx;
 534 static DEFINE_SPINLOCK(buffer_lock);
 535 
 536 uint32_t fwd_save_skb(struct sk_buff *skb)
 537 {
 538         struct sk_buff *old_skb = NULL;
 539         struct packet_buffer *p;
 540         unsigned long int flags;
 541         uint32_t id;
 542 
 543         /* FIXME: Probably just need a skb_clone() here. */
 544         skb = skb_copy(skb, GFP_ATOMIC);
 545         if (!skb)
 546                 return -1;
 547 
 548         spin_lock_irqsave(&buffer_lock, flags);
 549         buffer_idx = (buffer_idx + 1) & PKT_BUFFER_MASK;
 550         p = &buffers[buffer_idx];
 551         if (p->skb) {
 552                 /* Don't buffer packet if existing entry is less than
 553                  * OVERWRITE_SECS old. */
 554                 if (time_before(jiffies, p->exp_jiffies)) {
 555                         spin_unlock_irqrestore(&buffer_lock, flags);
 556                         kfree_skb(skb);
 557                         return -1;
 558                 } else {
 559                         /* Defer kfree_skb() until interrupts re-enabled.
 560                          * FIXME: we only need to do that if it has a
 561                          * destructor, but it never should since we orphan
 562                          * sk_buffs on entry. */
 563                         old_skb = p->skb;
 564                 }
 565         }
 566         /* Don't use maximum cookie value since the all-bits-1 id is
 567          * special. */
 568         if (++p->cookie >= (1u << PKT_COOKIE_BITS) - 1)
 569                 p->cookie = 0;
 570         p->skb = skb;
 571         p->exp_jiffies = jiffies + OVERWRITE_JIFFIES;
 572         id = buffer_idx | (p->cookie << PKT_BUFFER_BITS);
 573         spin_unlock_irqrestore(&buffer_lock, flags);
 574 
 575         if (old_skb)
 576                 kfree_skb(old_skb);
 577 
 578         return id;
 579 }

 581 static struct sk_buff *retrieve_skb(uint32_t id)
 582 {
 583         unsigned long int flags;
 584         struct sk_buff *skb = NULL;
 585         struct packet_buffer *p;
 586 
 587         spin_lock_irqsave(&buffer_lock, flags);
 588         p = &buffers[id & PKT_BUFFER_MASK];
 589         if (p->cookie == id >> PKT_BUFFER_BITS) {
 590                 skb = p->skb;
 591                 p->skb = NULL;
 592         } else {
 593                 if (net_ratelimit())
 594                         printk(KERN_NOTICE "cookie mismatch: %x != %x\n",
 595                                id >> PKT_BUFFER_BITS, p->cookie);
 596         }
 597         spin_unlock_irqrestore(&buffer_lock, flags);
 598 
 599         return skb;
 600 }

 602 void fwd_discard_all(void) 
 603 {
 604         int i;
 605 
 606         for (i = 0; i < N_PKT_BUFFERS; i++) {
 607                 struct sk_buff *skb;
 608                 unsigned long int flags;
 609 
 610                 /* Defer kfree_skb() until interrupts re-enabled. */
 611                 spin_lock_irqsave(&buffer_lock, flags);
 612                 skb = buffers[i].skb;
 613                 buffers[i].skb = NULL;
 614                 spin_unlock_irqrestore(&buffer_lock, flags);
 615 
 616                 kfree_skb(skb);
 617         }
 618 }

 620 static void discard_skb(uint32_t id)
 621 {
 622         struct sk_buff *old_skb = NULL;
 623         unsigned long int flags;
 624         struct packet_buffer *p;
 625 
 626         spin_lock_irqsave(&buffer_lock, flags);
 627         p = &buffers[id & PKT_BUFFER_MASK];
 628         if (p->cookie == id >> PKT_BUFFER_BITS) {
 629                 /* Defer kfree_skb() until interrupts re-enabled. */
 630                 old_skb = p->skb;
 631                 p->skb = NULL;
 632         }
 633         spin_unlock_irqrestore(&buffer_lock, flags);
 634 
 635         if (old_skb)
 636                 kfree_skb(old_skb);
 637 }
