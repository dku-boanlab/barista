  83 /**
  84  * \brief Function to calculate shortest paths for all pairs
  85  * \return None
  86  */
  87 static int floyd_warshall(void)
  88 {
  89     // initialze dist and next
  90     int i;
  91     for (i=0; i<MAX_NUM_SWITCHES; i++) {
  92         int j;
  93         for (j=0; j<MAX_NUM_SWITCHES; j++) {
  94             dist[i][j] = path[i][j];
  95 
  96             if (i != j) {
  97                 next[i][j] = j + 1;
  98             } else {
  99                 next[i][j] = 0;
 100             }
 101         }
 102     }
 103 
 104     // update dist and next
 105     int k;
 106     for (k=0; k<MAX_NUM_SWITCHES; k++) {
 107         int i;
 108         for (i=0; i<MAX_NUM_SWITCHES; i++) {
 109             int j;
 110             for (j=0; j<MAX_NUM_SWITCHES; j++) {
 111                 if (dist[i][j] > dist[i][k] + dist[k][j]) {
 112                     dist[i][j] = dist[i][k] + dist[k][j];
 113                     next[i][j] = next[i][k];
 114                 }
 115             }
 116         }
 117     }
 118 
 119     return 0;
 120 }

 126 static int add_switch(const switch_t *sw)
 127 {
 128     uint64_t dpid = sw->dpid;
 129 
 130     pthread_rwlock_wrlock(&path_lock);
 131 
 132     int i;
 133     for (i=1; i<MAX_NUM_SWITCHES; i++) {
 134         if (dpid_list[i] == dpid) {
 135             pthread_rwlock_unlock(&path_lock);
 136             return -1;
 137         }
 138     }
 139 
 140     for (i=1; i<MAX_NUM_SWITCHES; i++) {
 141         if (dpid_list[i] == 0) {
 142             dpid_list[i] = dpid;
 143             DEBUG("add -> dpid_list[%d] = %lu\n", i, dpid);
 144             break;
 145         }
 146     }
 147 
 148     pthread_rwlock_unlock(&path_lock);
 149 
 150     return 0;
 151 }

 245 static int shortest_path(int *route, uint64_t src_dpid, uint64_t dst_dpid)
 246 {

 248 
 249     int src = get_index_from_dpid(src_dpid);

 258     }
 259 
 260     int hop = 0;

 263     DEBUG("%d->%d %2d %d", u-1, v-1, dist[src][dst], u-1);

 270     } while (u != v);
 271 #endif
 272 
 273     DEBUG("\n");
 274 
 275     pthread_rwlock_unlock(&path_lock);

 394     FREE(dist);

 396

 541 /**
 542  * \brief Function to handle pktin messages
 543  * \param pktin Pktin message
 544  */
 545 static int l2_shortest(const pktin_t *pktin)
 546 {
 547 #ifdef __ENABLE_CBENCH
 548     send_packet(pktin, PORT_FLOOD);
 549     return 0;
 550 #else /* !__ENABLE_CBENCH */
 551     if ((pktin->proto & (PROTO_ARP | PROTO_IPV4)) == 0) {
 552         discard_packet(pktin);
 553         return -1;
 554     }
 555 
 556     uint64_t mac = mac2int(pktin->src_mac);
 557     uint32_t mkey = hash_func((uint32_t *)&mac, 2) % MAC_HASH_SIZE;
 558 
 559     pthread_rwlock_rdlock(&mac_table[mkey].lock);
 560 
 561     // check source MAC
 562     mac_entry_t *curr = mac_table[mkey].head;
 563     while (curr != NULL) {
 564         if (curr->mac == mac) {
 565             break;
 566         }
 567         curr = curr->next;
 568     }
 569 
 570     pthread_rwlock_unlock(&mac_table[mkey].lock);
 571 
 572     // new MAC?
 573     if (curr == NULL) {
 574         curr = mac_dequeue();
 575         if (curr == NULL) {
 576             discard_packet(pktin);
 577             return -1;
 578         }
 579 
 580         curr->dpid = pktin->dpid;
 581         curr->port = pktin->port;
 582 
 583         curr->ip = pktin->src_ip;
 584         curr->mac = mac;
 585 
 586         curr->prev = curr->next = NULL;
 587 
 588         pthread_rwlock_wrlock(&mac_table[mkey].lock);
 589 
 590         if (mac_table[mkey].head == NULL) {
 591             mac_table[mkey].head = curr;
 592             mac_table[mkey].tail = curr;
 593         } else {
 594             curr->prev = mac_table[mkey].tail;
 595             mac_table[mkey].tail->next = curr;
 596             mac_table[mkey].tail = curr;
 597         }
 598 
 599         pthread_rwlock_unlock(&mac_table[mkey].lock);
 600     }
 601 
 602     // get destination MAC
 603     mac = mac2int(pktin->dst_mac);
 604 
 605     uint64_t dpid = 0;
 606     uint16_t port = 0;
 607 
 608     // broadcast?
 609     if (mac == 0xffffffffffff) {
 610         int i;
 611         for (i=0; i<MAC_HASH_SIZE; i++) {
 612             pthread_rwlock_rdlock(&mac_table[i].lock);
 613 
 614             mac_entry_t *curr = mac_table[i].head;
 615             while (curr != NULL) {
 616                 if (curr->ip == pktin->dst_ip) {
 617                     dpid = curr->dpid;
 618                     port = curr->port;
 619                     break;
 620                 }
 621                 curr = curr->next;
 622             }
 623 
 624             pthread_rwlock_unlock(&mac_table[i].lock);
 625 
 626             if (dpid)
 627                 break;
 628         }
 629 
 630         if (!dpid) {
 631             send_packet(pktin, PORT_FLOOD);
 632             return 0;
 633         }
 634     } else {
 635         mkey = hash_func((uint32_t *)&mac, 2) % MAC_HASH_SIZE;
 636 
 637         pthread_rwlock_rdlock(&mac_table[mkey].lock);
 638 
 639         // known destination?
 640         curr = mac_table[mkey].head;
 641         while (curr != NULL) {
 642             if (curr->mac == mac) {
 643                 dpid = curr->dpid;
 644                 port = curr->port;
 645                 break;
 646             }
 647             curr = curr->next;
 648         }
 649 
 650         pthread_rwlock_unlock(&mac_table[mkey].lock);
 651     }
 652 
 653     // unknown destination?
 654     if (port == 0) {
 655         if (pktin->buffer_id == (uint32_t)-1)
 656             send_packet_w_payload(pktin, PORT_FLOOD);
 657         else
 658             send_packet(pktin, PORT_FLOOD);
 659         return 0;
 660     }
 661 
 662     // source dpid is equal to destination dpid?
 663     if (pktin->dpid == dpid) {
 664         if (pktin->proto & PROTO_ARP) {
 665             if (pktin->buffer_id == (uint32_t)-1)
 666                 send_packet_w_payload(pktin, port);
 667             else
 668                 send_packet(pktin, port);
 669         } else {
 670             if (pktin->buffer_id == (uint32_t)-1)
 671                 send_packet_w_payload(pktin, port);
 672             else
 673                 insert_flow(pktin, port);
 674         }
 675         return 0;
 676     } else {
 677         // get the shortest path from src to dst
 678         int route[MAX_HOP_COUNT] = {0};
 679         int hop = shortest_path(route, pktin->dpid, dpid);
 680 
 681         if (hop > 0) {
 682             if (pktin->buffer_id == (uint32_t)-1)
 683                 send_packet_4_next_w_payload(pktin, hop, route, port);
 684             else
 685                 insert_flow_4_next(pktin, hop, route, port);
 686         } else {
 687             discard_packet(pktin);
 688         }
 689     }
 690 
 691     return 0;
 692 #endif /* !__ENABLE_CBENCH */
 693 }

 727 /**
 728  * \brief The cleanup function
 729  * \param activated The activation flag of this application
 730  */
 731 int l2_shortest_cleanup(int *activated)
 732 {
 733     ALOG_INFO(L2_SHORTEST_ID, "Clean up - L2 shortest-path");
 734 
 735     deactivate();
 736 
 737     int i;
 738     for (i=0; i<MAC_HASH_SIZE; i++) {
 739         pthread_rwlock_wrlock(&mac_table[i].lock);
 740 
 741         mac_entry_t *curr = mac_table[i].head;
 742         while (curr != NULL) {
 743             mac_entry_t *tmp = curr;
 744             curr = curr->next;
 745             FREE(tmp);
 746         }
 747 
 748         pthread_rwlock_unlock(&mac_table[i].lock);
 749         pthread_rwlock_destroy(&mac_table[i].lock);
 750     }
 751 
 752     mac_q_destroy();
 753 
 754     path_destroy();
 755 
 756     FREE(mac_table);
 757 
 758     return 0;
 759 }

 935 int l2_shortest_cli(cli_t *cli, char **args)
 936 {

 947             show_entry_mac(cli, args[2]);
 948             return 0;

 950             show_entry_ip(cli, args[2]);
 951             return 0;
 952         }
 953     }
 954 
 955     cli_print(cli, "<Available Commands>");
 956     cli_print(cli, "  l2_shortest list tables");

 964 /**
 965  * \brief The handler function
 966  * \param av Read-only app event
 967  * \param av_out Read-write app event (if this application has the write permission)
 968  */
 969 int l2_shortest_handler(const app_event_t *av, app_event_out_t *av_out)
 970 {
 971     switch (av->type) {
 972     case AV_DP_RECEIVE_PACKET:
 973         PRINT_EV("AV_DP_RECEIVE_PACKET\n");
 974         {
 975             const pktin_t *pktin = av->pktin;
 976 
 977             l2_shortest(pktin);
 978         }
 979         break;
 980     case AV_DP_PORT_DELETED:
 981         PRINT_EV("AV_DP_PORT_DELETED\n");
 982         {
 983             const port_t *port = av->port;
 984 
 985             int i;
 986             for (i=0; i<MAC_HASH_SIZE; i++) {
 987                 mac_table_t tmp_list = {0};
 988 
 989                 pthread_rwlock_wrlock(&mac_table[i].lock);
 990 
 991                 mac_entry_t *curr = mac_table[i].head;
 992                 while (curr != NULL) {
 993                     if (curr->dpid == port->dpid && curr->port == port->port) {
 994                         if (tmp_list.head == NULL) {
 995                             tmp_list.head = curr;
 996                             tmp_list.tail = curr;
 997                             curr->r_next = NULL;
 998                         } else {
 999                             tmp_list.tail->r_next = curr;
1000                             tmp_list.tail = curr;
1001                             curr->r_next = NULL;
1002                         }
1003                     }
1004 
1005                     curr = curr->next;
1006                 }
1007 
1008                 curr = tmp_list.head;
1009                 while (curr != NULL) {
1010                     mac_entry_t *tmp = curr;
1011 
1012                     curr = curr->r_next;
1013 
1014                     if (tmp->prev != NULL && tmp->next != NULL) {
1015                         tmp->prev->next = tmp->next;
1016                         tmp->next->prev = tmp->prev;
1017                     } else if (tmp->prev == NULL && tmp->next != NULL) {
1018                         mac_table[i].head = tmp->next;
1019                         tmp->next->prev = NULL;
1020                     } else if (tmp->prev != NULL && tmp->next == NULL) {
1021                         mac_table[i].tail = tmp->prev;
1022                         tmp->prev->next = NULL;
1023                     } else if (tmp->prev == NULL && tmp->next == NULL) {
1024                         mac_table[i].head = NULL;
1025                         mac_table[i].tail = NULL;
1026                     }
1027 
1028                     mac_enqueue(tmp);
1029                 }
1030 
1031                 pthread_rwlock_unlock(&mac_table[i].lock);
1032             }
1033         }
1034         break;
1035     case AV_SW_CONNECTED:
1036         PRINT_EV("AV_SW_CONNECTED\n");
1037         {
1038             const switch_t *sw = av->sw;
1039 
1040             add_switch(sw);
1041         }
1042         break;
1043     case AV_SW_DISCONNECTED:
1044         PRINT_EV("AV_SW_DISCONNECTED\n");
1045         {
1046             const switch_t *sw = av->sw;
1047 
1048             int i;
1049             for (i=0; i<MAC_HASH_SIZE; i++) {
1050                 mac_table_t tmp_list = {0};
1051 
1052                 pthread_rwlock_wrlock(&mac_table[i].lock);
1053 
1054                 mac_entry_t *curr = mac_table[i].head;
1055                 while (curr != NULL) {
1056                     if (curr->dpid == sw->dpid) {
1057                         if (tmp_list.head == NULL) {
1058                             tmp_list.head = curr;
1059                             tmp_list.tail = curr;
1060                             curr->r_next = NULL;
1061                         } else {
1062                             tmp_list.tail->r_next = curr;
1063                             tmp_list.tail = curr;
1064                             curr->r_next = NULL;
1065                         }
1066                     }
1067 
1068                     curr = curr->next;
1069                 }
1070 
1071                 curr = tmp_list.head;
1072                 while (curr != NULL) {
1073                     mac_entry_t *tmp = curr;
1074 
1075                     curr = curr->next;
1076 
1077                     if (tmp->prev != NULL && tmp->next != NULL) {
1078                         tmp->prev->next = tmp->next;
1079                         tmp->next->prev = tmp->prev;
1080                     } else if (tmp->prev == NULL && tmp->next != NULL) {
1081                         mac_table[i].head = tmp->next;
1082                         tmp->next->prev = NULL;
1083                     } else if (tmp->prev != NULL && tmp->next == NULL) {
1084                         mac_table[i].tail = tmp->prev;
1085                         tmp->prev->next = NULL;
1086                     } else if (tmp->prev == NULL && tmp->next == NULL) {
1087                         mac_table[i].head = NULL;
1088                         mac_table[i].tail = NULL;
1089                     }
1090 
1091                     mac_enqueue(tmp);
1092                 }
1093 
1094                 pthread_rwlock_unlock(&mac_table[i].lock);
1095             }
1096 
1097             del_switch(sw);
1098         }
1099         break;
1100     case AV_LINK_ADDED:
1101         PRINT_EV("AV_LINK_ADDED\n");
1102         {
1103             const port_t *link = av->port;
1104 
1105             add_link(link);
1106         }
1107         break;
1108     case AV_LINK_DELETED:
1109         PRINT_EV("AV_LINK_DELETED\n");
1110         {
1111             const port_t *link = av->port;
1112 
1113             del_link(link);
1114         }
1115         break;
1116     default:
1117         break;
1118     }
1119 
1120     return 0;
1121 }
