  21 #ifndef __MD5_H__
  22 #define __MD5_H__
  23 
  24 /**
  25  * @file md5.h
  26  * @brief MD5 Functions
  27  */
  28 
  29 /**
  30  * @defgroup PJLIB_UTIL_MD5 MD5 Functions
  31  * @ingroup PJLIB_UTIL
  32  * @{
  33  */
  34 
  35 #define md5_byte_t guint8
  36 
  37 /** MD5 context. */
  38 typedef struct md5_state_s
  39 {
  40         guint32 buf[4];
  41         guint32 bits[2];
  42         guint32 in[16];
  43 } md5_state_t;
  44 
  45 /** Initialize the algorithm. 
  46  *  @param pms          MD5 context.
  47  */
  48 void md5_init(md5_state_t *pms);
  49 
  50 /** Append a string to the message. 
  51  *  @param pms          MD5 context.
  52  *  @param data         Data.
  53  *  @param nbytes       Length of data.
  54  */
  55 void md5_append( md5_state_t *pms, 
  56                              const guint8 *data, guint nbytes);
  57 
  58 /** Finish the message and return the digest. 
  59  *  @param pms          MD5 context.
  60  *  @param digest       16 byte digest.
  61  */
  62 void md5_finish(md5_state_t *pms, guint8 digest[16]);
  63 
  64 
  65 void md5_hmac(const guint8* text, gint text_len, const guint8* key, gint key_len, guint8 digest[16]);
  66 
  67 /**
  68  * @}
  69  */
  70 
  71 #endif  /* _CRYPT_MD5_H__ */
