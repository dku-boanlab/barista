  25 #ifndef __NSTIME_H__
  26 #define __NSTIME_H__
  27 
  28 #include <time.h>
  29 
  30 #include <wiretap/wtap.h>
  31 
  32 typedef struct {
  33         time_t  secs;
  34         int     nsecs;
  35 } nstime_t;
  36 
  37 /* functions */
  38 
  39 /* set the given nstime_t to zero */
  40 extern void nstime_set_zero(nstime_t *nstime);
  41 
  42 /* is the given nstime_t currently zero? */
  43 extern gboolean nstime_is_zero(nstime_t *nstime);
  44 
  45 /* set the given nstime_t to (0,maxint) to mark it as "unset"
  46  * That way we can find the first frame even when a timestamp
  47  * is zero (fix for bug 1056)
  48  */
  49 extern void nstime_set_unset(nstime_t *nstime);
  50 
  51 /* is the given nstime_t currently (0,maxint)? */
  52 extern gboolean nstime_is_unset(nstime_t *nstime);
  53 
  54 /* calculate the delta between two times (can be negative!)
  55  *
  56  * delta = b-a
  57  *
  58  * Note that it is acceptable for two or more of the arguments to point at the
  59  * same structure.
  60  */
  61 extern void nstime_delta(nstime_t *delta, const nstime_t *b, const nstime_t *a );
  62 
  63 /* calculate the sum of two times
  64  *
  65  * sum = a+b
  66  *
  67  * Note that it is acceptable for two or more of the arguments to point at the
  68  * same structure.
  69  */
  70 extern void nstime_sum(nstime_t *sum, const nstime_t *b, const nstime_t *a );
  71 
  72 /* sum += a */
  73 #define nstime_add(sum, a) nstime_sum(sum, sum, a)
  74 
  75 /* compare two times are return a value similar to memcmp() or strcmp().
  76  *
  77  * a > b : > 0
  78  * a = b : 0
  79  * a < b : < 0
  80  */
  81 extern int nstime_cmp(nstime_t *a, const nstime_t *b );
  82 
  83 /* converts nstime to double, time base is milli seconds */
  84 extern double nstime_to_msec(const nstime_t *time);
  85 
  86 /* converts nstime to double, time base is seconds */
  87 extern double nstime_to_sec(const nstime_t *time);
  88 
  89 /* converts wtap_nstime to double, time base is seconds */
  90 extern double wtap_nstime_to_sec(const struct wtap_nstime *time);
  91 
  92 #endif /* __NSTIME_H__  */
