  38 static struct port_array_l2 l2_sentinel;
  39 static struct port_array_l3 l3_sentinel;
  40 static bool inited;

  42 /* Initializes 'pa' as an empty port_array. */
  43 void
  44 port_array_init(struct port_array *pa)
  45 {
  46     size_t i;
  47     if (!inited) {
  48         inited = true;
  49         for (i = 0; i < PORT_ARRAY_L2_SIZE; i++) {
  50             l2_sentinel.l2[i] = &l3_sentinel;
  51         }
  52     }
  53     for (i = 0; i < PORT_ARRAY_L1_SIZE; i++) {
  54         pa->l1[i] = &l2_sentinel;
  55     }
  56 }
  57 
  58 /* Frees all the memory allocated for 'pa'.  It is the client's responsibility
  59  * to free memory that 'pa' elements point to. */
  60 void
  61 port_array_destroy(struct port_array *pa)
  62 {
  63     unsigned int l1_idx;
  64 
  65     for (l1_idx = 0; l1_idx < PORT_ARRAY_L1_SIZE; l1_idx++) {
  66         struct port_array_l2 *l2 = pa->l1[l1_idx];
  67 
  68         if (l2 != &l2_sentinel) {
  69             unsigned int l2_idx;
  70 
  71             for (l2_idx = 0; l2_idx < PORT_ARRAY_L2_SIZE; l2_idx++) {
  72                 struct port_array_l3 *l3 = l2->l2[l2_idx];
  73                 if (l3 != &l3_sentinel) {
  74                     free(l3);
  75                 }
  76             }
  77             free(l2);
  78         }
  79     }
  80 }
  81 
  82 /* Clears all elements of 'pa' to null pointers. */
  83 void
  84 port_array_clear(struct port_array *pa)
  85 {
  86     port_array_destroy(pa);
  87     port_array_init(pa);
  88 }

  90 /* Sets 'pa' element numbered 'idx' to 'p'. */
  91 void
  92 port_array_set(struct port_array *pa, uint16_t idx, void *p)
  93 {
  94     struct port_array_l2 **l2p, *l2;
  95     struct port_array_l3 **l3p, *l3;
  96 
  97     /* Traverse level 1. */
  98     l2p = &pa->l1[PORT_ARRAY_L1(idx)];
  99     if (*l2p == &l2_sentinel) {
 100         *l2p = xmemdup(&l2_sentinel, sizeof l2_sentinel);
 101     }
 102     l2 = *l2p;
 103 
 104     /* Traverse level 2. */
 105     l3p = &l2->l2[PORT_ARRAY_L2(idx)];
 106     if (*l3p == &l3_sentinel) {
 107         *l3p = xmemdup(&l3_sentinel, sizeof l3_sentinel);
 108     }
 109     l3 = *l3p;
 110 
 111     /* Set level 3. */
 112     l3->l3[PORT_ARRAY_L3(idx)] = p;
 113 }

 115 static void *
 116 next(const struct port_array *pa, unsigned int *idxp)
 117 {
 118     unsigned int idx = *idxp;
 119 
 120     /* Using shift-right directly here, instead of PORT_ARRAY_L1(idx), ensures
 121      * that with an initially too-big value of '*idxp' we will skip the outer
 122      * loop and return NULL. */
 123     unsigned int l1_idx = idx >> PORT_ARRAY_L1_SHIFT;
 124     unsigned int l2_idx = PORT_ARRAY_L2(idx);
 125     unsigned int l3_idx = PORT_ARRAY_L3(idx);
 126     while (l1_idx < PORT_ARRAY_L1_SIZE) {
 127         struct port_array_l2 *l2 = pa->l1[l1_idx];
 128         if (l2 != &l2_sentinel) {
 129             while (l2_idx < PORT_ARRAY_L2_SIZE) {
 130                 struct port_array_l3 *l3 = l2->l2[l2_idx];
 131                 if (l3 != &l3_sentinel) {
 132                     while (l3_idx < PORT_ARRAY_L3_SIZE) {
 133                         void *p = l3->l3[l3_idx];
 134                         if (p) {
 135                             *idxp = ((l1_idx << PORT_ARRAY_L1_SHIFT)
 136                                      | (l2_idx << PORT_ARRAY_L2_SHIFT)
 137                                      | (l3_idx << PORT_ARRAY_L3_SHIFT));
 138                             return p;
 139                         }
 140                         l3_idx++;
 141                     }
 142                 }
 143                 l2_idx++;
 144                 l3_idx = 0;
 145             }
 146         }
 147         l1_idx++;
 148         l2_idx = 0;
 149         l3_idx = 0;
 150     }
 151     *idxp = PORT_ARRAY_SIZE;
 152     return NULL;
 153 }

 155 /* Returns the value of the lowest-numbered non-empty element of 'pa', and sets
 156  * '*idxp' to that element's index.  If 'pa' is entirely empty, returns a null
 157  * pointer and sets '*idxp' to 65536.  */
 158 void *
 159 port_array_first(const struct port_array *pa, unsigned int *idxp)
 160 {
 161     *idxp = 0;
 162     return next(pa, idxp);
 163 }
 164 
 165 /* Returns the value of the lowest-numbered non-empty element of 'pa' greater
 166  * than the initial value of '*idxp', and sets '*idxp' to that element's index.
 167  * If 'pa' contains no non-empty elements with indexes greater than the initial
 168  * value of '*idxp', returns a null pointer and sets '*idxp' to 65536.  */
 169 void *
 170 port_array_next(const struct port_array *pa, unsigned int *idxp)
 171 {
 172     ++*idxp;
 173     return next(pa, idxp);
 174 }
