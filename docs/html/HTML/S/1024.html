 227 /**
 228  * \brief Function to process meta events
 229  * \return NULL
 230  */
 231 static void *meta_events(void *null)
 232 {
 233     int *num_events = ev_ctx->num_events;
 234     meta_event_t *meta = ev_ctx->meta_event;
 235 
 236     while (1) {
 237         int ev_id;
 238         for (ev_id=0; ev_id<__MAX_EVENTS; ev_id++) {
 239             int num_event = num_events[ev_id];
 240 
 241             int j;
 242             for (j=0; j<__MAX_META_EVENTS; j++) {
 243                 if (meta[j].event == ev_id) {
 244                     switch (meta[j].condition) {
 245                     case META_GT: // >
 246                         if (num_event > meta[j].threshold) {
 247                             //char **args = split_line(meta[j].cmd);
 248                             //execute(args);
 249                             //FREE(args);
 250                         }
 251                         break;
 252                     case META_GTE: // >=
 253                         if (num_event >= meta[j].threshold) {
 254                             //char **args = split_line(meta[j].cmd);
 255                             //execute(args);
 256                             //FREE(args);
 257                         }
 258                         break;
 259                     case META_LT: // <
 260                         if (num_event < meta[j].threshold) {
 261                             //char **args = split_line(meta[j].cmd);
 262                             //execute(args);
 263                             //FREE(args);
 264                         }
 265                         break;
 266                     case META_LTE: // <=
 267                         if (num_event <= meta[j].threshold) {
 268                             //char **args = split_line(meta[j].cmd);
 269                             //execute(args);
 270                             //FREE(args);
 271                         }
 272                         break;
 273                     case META_EQ: // ==
 274                         if (num_event == meta[j].threshold) {
 275                             //char **args = split_line(meta[j].cmd);
 276                             //execute(args);
 277                             //FREE(args);
 278                         }
 279                         break;
 280                     default:
 281                         break;
 282                     }
 283                 }
 284             }
 285 
 286             num_events[ev_id] = 0;
 287         }
 288 
 289         waitsec(1, 0);
 290     }
 291 
 292     return NULL;
 293 }

 297 /**
 298  * \brief Function to initialize the event handler
 299  * \param ctx The context of the Barista NOS
 300  */
 301 int event_init(ctx_t *ctx)
 302 {
 303     if (event_is_set == FALSE) {
 304         ev_ctx = ctx;
 305         event_is_set = TRUE;
 306     } else {
 307         return -1;
 308     }
 309 
 310     pthread_t thread;
 311     if (pthread_create(&thread, NULL, &meta_events, NULL) < 0) {
 312         PERROR("pthread_create");
 313         return -1;
 314     }
 315 
 316     return 0;
 317 }
