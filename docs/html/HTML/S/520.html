  43 /*  Win32: Since GLib2.6, we use UTF8 throughout the code, so file functions
  44  *  must tweak a given filename from UTF8 to UTF16 as we use NT Unicode (Win9x
  45  *  - now unsupported - used locale based encoding here).
  46  */
  47 #if defined _WIN32 && (GLIB_MAJOR_VERSION > 2 || (GLIB_MAJOR_VERSION == 2 && GLIB_MINOR_VERSION >= 6))
  48 #include <stdio.h>
  49 
  50 extern int eth_stdio_open (const gchar *filename, int flags, int mode);
  51 extern int eth_stdio_rename (const gchar *oldfilename, const gchar *newfilename);
  52 extern int eth_stdio_mkdir (const gchar *filename, int mode);
  53 extern int eth_stdio_stat (const gchar *filename, struct stat *buf);
  54 extern int eth_stdio_unlink (const gchar *filename);
  55 extern int eth_stdio_remove (const gchar *filename);
  56 extern FILE * eth_stdio_fopen (const gchar *filename, const gchar *mode);
  57 extern FILE * eth_stdio_freopen (const gchar *filename, const gchar *mode, FILE *stream);
  58 
  59 #define eth_open        eth_stdio_open
  60 #define eth_rename      eth_stdio_rename
  61 #define eth_mkdir       eth_stdio_mkdir
  62 #define eth_stat        eth_stdio_stat
  63 #define eth_unlink      eth_stdio_unlink
  64 #define eth_remove      eth_stdio_remove
  65 #define eth_fopen       eth_stdio_fopen
  66 #define eth_freopen     eth_stdio_freopen
  67 
  68 #else   /* _WIN32 && GLIB_MAJOR_VERSION */
  69

  76 #else
  77 #define eth_open        open
  78 #define eth_stat        stat
  79 #define eth_unlink      unlink
  80 #define eth_mkdir(dir,mode)     mkdir(dir,mode)
  81 #endif /* _WIN32 */
  82 
  83 #define eth_rename      rename

  86 #define eth_freopen     freopen

  88 #endif  /* _WIN32 && GLIB_MAJOR_VERSION */

  91 /* some common file function differences between UNIX and WIN32 */
  92 #ifdef _WIN32
  93 /* the Win32 API prepends underscores for whatever reasons */
  94 #define eth_read  _read
  95 #define eth_write _write
  96 #define eth_close _close
  97 #define eth_dup   _dup
  98 #define eth_lseek _lseek
  99 #else
 100 #define eth_read  read
 101 #define eth_write write
 102 #define eth_close close
 103 #define eth_dup   dup
 104 #define eth_lseek lseek
 105 #define O_BINARY        0               /* Win32 needs the O_BINARY flag for open() */
 106 #endif /* _WIN32 */
