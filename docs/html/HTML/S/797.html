  63 static bool
  64 stp_local_packet_cb(struct relay *r, void *stp_)
  65 {
  66     struct ofpbuf *msg = r->halves[HALF_LOCAL].rxbuf;
  67     struct ofp_header *oh;
  68     struct stp_data *stp = stp_;
  69     struct ofp_packet_in *opi;
  70     struct eth_header *eth;
  71     struct llc_header *llc;
  72     struct ofpbuf payload;
  73     uint16_t port_no;
  74     struct flow flow;
  75 
  76     oh = msg->data;
  77     if (oh->type == OFPT_FEATURES_REPLY
  78         && msg->size >= offsetof(struct ofp_switch_features, ports)) {
  79         struct ofp_switch_features *osf = msg->data;
  80         osf->capabilities |= htonl(OFPC_STP);
  81         return false;
  82     }
  83 
  84     if (!get_ofp_packet_eth_header(r, &opi, &eth)
  85         || !eth_addr_equals(eth->eth_dst, stp_eth_addr)) {
  86         return false;
  87     }
  88 
  89     port_no = ntohs(opi->in_port);
  90     if (port_no >= STP_MAX_PORTS) {
  91         /* STP only supports 255 ports. */
  92         return false;
  93     }
  94     if (port_watcher_get_config(stp->pw, port_no) & OFPPC_NO_STP) {
  95         /* We're not doing STP on this port. */
  96         return false;
  97     }
  98 
  99     if (opi->reason == OFPR_ACTION) {
 100         /* The controller set up a flow for this, so we won't intercept it. */
 101         return false;
 102     }
 103 
 104     get_ofp_packet_payload(opi, &payload);
 105     flow_extract(&payload, port_no, &flow);
 106     if (flow.dl_type != htons(OFP_DL_TYPE_NOT_ETH_TYPE)) {
 107         VLOG_DBG("non-LLC frame received on STP multicast address");
 108         return false;
 109     }
 110     llc = ofpbuf_at_assert(&payload, sizeof *eth, sizeof *llc);
 111     if (llc->llc_dsap != STP_LLC_DSAP) {
 112         VLOG_DBG("bad DSAP 0x%02"PRIx8" received on STP multicast address",
 113                  llc->llc_dsap);
 114         return false;
 115     }
 116 
 117     /* Trim off padding on payload. */
 118     if (payload.size > ntohs(eth->eth_type) + ETH_HEADER_LEN) {
 119         payload.size = ntohs(eth->eth_type) + ETH_HEADER_LEN;
 120     }
 121     if (ofpbuf_try_pull(&payload, ETH_HEADER_LEN + LLC_HEADER_LEN)) {
 122         struct stp_port *p = stp_get_port(stp->stp, port_no);
 123         stp_received_bpdu(p, payload.data, payload.size);
 124     }
 125 
 126     return true;
 127 }

 129 static void
 130 stp_periodic_cb(void *stp_)
 131 {
 132     struct stp_data *stp = stp_;
 133     long long int now = time_msec();
 134     long long int elapsed = now - stp->last_tick;
 135     struct stp_port *p;
 136 
 137     if (!port_watcher_is_ready(stp->pw)) {
 138         /* Can't start STP until we know port flags, because port flags can
 139          * disable STP. */
 140         return;
 141     }
 142     if (elapsed <= 0) {
 143         return;
 144     }
 145 
 146     stp_tick(stp->stp, MIN(INT_MAX, elapsed));
 147     stp->last_tick = now;
 148 
 149     while (stp_get_changed_port(stp->stp, &p)) {
 150         int port_no = stp_port_no(p);
 151         enum stp_state s_state = stp_port_get_state(p);
 152 
 153         if (s_state != STP_DISABLED) {
 154             VLOG_INFO("STP: Port %d entered %s state",
 155                       port_no, stp_state_name(s_state));
 156         }
 157         if (!(port_watcher_get_config(stp->pw, port_no) & OFPPC_NO_STP)) {
 158             uint32_t p_config = 0;
 159             uint32_t p_state;
 160             switch (s_state) {
 161             case STP_LISTENING:
 162                 p_state = OFPPS_STP_LISTEN;
 163                 break;
 164             case STP_LEARNING:
 165                 p_state = OFPPS_STP_LEARN;
 166                 break;
 167             case STP_DISABLED:
 168             case STP_FORWARDING:
 169                 p_state = OFPPS_STP_FORWARD;
 170                 break;
 171             case STP_BLOCKING:
 172                 p_state = OFPPS_STP_BLOCK;
 173                 break;
 174             default:
 175                 VLOG_DBG_RL(&rl, "STP: Port %d has bad state %x",
 176                             port_no, s_state);
 177                 p_state = OFPPS_STP_FORWARD;
 178                 break;
 179             }
 180             if (!stp_forward_in_state(s_state)) {
 181                 p_config = OFPPC_NO_FLOOD;
 182             }
 183             port_watcher_set_flags(stp->pw, port_no, 
 184                                    p_config, OFPPC_NO_FLOOD,
 185                                    p_state, OFPPS_STP_MASK);
 186         } else {
 187             /* We don't own those flags. */
 188         }
 189     }
 190 }

 198 static void
 199 send_bpdu(struct ofpbuf *pkt, int port_no, void *stp_)
 200 {
 201     struct stp_data *stp = stp_;
 202     const uint8_t *port_mac = port_watcher_get_hwaddr(stp->pw, port_no);
 203     if (port_mac) {
 204         struct eth_header *eth = pkt->l2;
 205         struct ofpbuf *opo;
 206 
 207         memcpy(eth->eth_src, port_mac, ETH_ADDR_LEN);
 208         opo = make_unbuffered_packet_out(pkt, OFPP_NONE, port_no);
 209 
 210         rconn_send_with_limit(stp->local_rconn, opo, &stp->n_txq, OFPP_MAX);
 211     } else {
 212         VLOG_WARN_RL(&rl, "cannot send BPDU on missing port %d", port_no);
 213     }
 214     ofpbuf_delete(pkt);
 215 }

 223 static void
 224 stp_port_changed_cb(uint16_t port_no,
 225                     const struct ofp_phy_port *old UNUSED,
 226                     const struct ofp_phy_port *new,
 227                     void *stp_)
 228 {
 229     struct stp_data *stp = stp_;
 230     struct stp_port *p;
 231 
 232     if (!stp_is_port_supported(port_no)) {
 233         return;
 234     }
 235 
 236     p = stp_get_port(stp->stp, port_no);
 237     if (!new
 238         || new->config & htonl(OFPPC_NO_STP | OFPPC_PORT_DOWN)
 239         || new->state & htonl(OFPPS_LINK_DOWN)) {
 240         stp_port_disable(p);
 241     } else {
 242         int speed = 0;
 243         stp_port_enable(p);
 244         if (new->curr & (OFPPF_10MB_HD | OFPPF_10MB_FD)) {
 245             speed = 10;
 246         } else if (new->curr & (OFPPF_100MB_HD | OFPPF_100MB_FD)) {
 247             speed = 100;
 248         } else if (new->curr & (OFPPF_1GB_HD | OFPPF_1GB_FD)) {
 249             speed = 1000;
 250         } else if (new->curr & OFPPF_10GB_FD) {
 251             speed = 10000;
 252         }
 253         stp_port_set_speed(p, speed);
 254     }
 255 }

 274 void
 275 stp_start(struct secchan *secchan, struct port_watcher *pw,
 276           struct rconn *local, struct rconn *remote)
 277 {
 278     uint8_t dpid[ETH_ADDR_LEN];
 279     struct stp_data *stp;
 280 
 281     stp = xcalloc(1, sizeof *stp);
 282     eth_addr_random(dpid);
 283     stp->stp = stp_create("stp", eth_addr_to_uint64(dpid), send_bpdu, stp);
 284     stp->pw = pw;
 285     stp->local_rconn = local;
 286     stp->remote_rconn = remote;
 287     stp->last_tick = time_msec();
 288 
 289     port_watcher_register_callback(pw, stp_port_changed_cb, stp);
 290     port_watcher_register_local_port_callback(pw, stp_local_port_changed_cb,
 291                                               stp);
 292     add_hook(secchan, &stp_hook_class, stp);
 293 }
