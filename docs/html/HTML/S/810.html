  83 /* Returns the number of fields that differ from 'a' to 'b'. */
  84 static int
  85 opp_differs(const struct ofp_phy_port *a, const struct ofp_phy_port *b)
  86 {
  87     BUILD_ASSERT_DECL(sizeof *a == 48); /* Trips when we add or remove fields. */
  88     return ((a->port_no != b->port_no)
  89             + (memcmp(a->hw_addr, b->hw_addr, sizeof a->hw_addr) != 0)
  90             + (memcmp(a->name, b->name, sizeof a->name) != 0)
  91             + (a->config != b->config)
  92             + (a->state != b->state)
  93             + (a->curr != b->curr)
  94             + (a->advertised != b->advertised)
  95             + (a->supported != b->supported)
  96             + (a->peer != b->peer));
  97 }

  99 static void
 100 sanitize_opp(struct ofp_phy_port *opp)
 101 {
 102     size_t i;
 103 
 104     for (i = 0; i < sizeof opp->name; i++) {
 105         char c = opp->name[i];
 106         if (c && (c < 0x20 || c > 0x7e)) {
 107             opp->name[i] = '.';
 108         }
 109     }
 110     opp->name[sizeof opp->name - 1] = '\0';
 111 }

 125 void
 126 get_port_name(const struct ofp_phy_port *port, char *name, size_t name_size)
 127 {
 128     char *p;
 129 
 130     memcpy(name, port->name, MIN(name_size, sizeof port->name));
 131     name[name_size - 1] = '\0';
 132     for (p = name; *p != '\0'; p++) {
 133         if (*p < 32 || *p > 126) {
 134             *p = '.';
 135         }
 136     }

 145 static void
 146 call_local_port_changed_callbacks(struct port_watcher *pw)
 147 {
 148     char name[OFP_MAX_PORT_NAME_LEN + 1];
 149     const struct ofp_phy_port *port;
 150     int i;
 151 
 152     /* Pass the local port to the callbacks, if it exists.
 153        Pass a null pointer if there is no local port. */
 154     port = lookup_port(pw, OFPP_LOCAL);
 155 
 156     /* Log the name of the local port. */
 157     if (port) {
 158         get_port_name(port, name, sizeof name);
 159     } else {
 160         name[0] = '\0';
 161     }
 162     if (strcmp(pw->local_port_name, name)) {
 163         if (name[0]) {
 164             VLOG_INFO("Identified data path local port as \"%s\".", name);
 165         } else {
 166             VLOG_WARN("Data path has no local port.");
 167         }
 168         strcpy(pw->local_port_name, name);
 169     }
 170 
 171     /* Invoke callbacks. */
 172     for (i = 0; i < pw->n_local_cbs; i++) {
 173         local_port_changed_cb_func *cb = pw->local_cbs[i].local_port_changed;
 174         (cb)(port, pw->local_cbs[i].aux);
 175     }
 176 }

 178 static void
 179 update_phy_port(struct port_watcher *pw, struct ofp_phy_port *opp,
 180                 uint8_t reason)
 181 {
 182     struct ofp_phy_port *old;
 183     uint16_t port_no;
 184 
 185     port_no = ntohs(opp->port_no);
 186     old = lookup_port(pw, port_no);
 187 
 188     if (reason == OFPPR_DELETE && old) {
 189         call_port_changed_callbacks(pw, port_no, old, NULL);
 190         free(old);
 191         port_array_set(&pw->ports, port_no, NULL);
 192     } else if (reason == OFPPR_MODIFY || reason == OFPPR_ADD) {
 193         if (old) {
 194             uint32_t s_mask = htonl(OFPPS_STP_MASK);
 195             opp->state = (opp->state & ~s_mask) | (old->state & s_mask);
 196         }
 197         if (!old || opp_differs(opp, old)) {
 198             struct ofp_phy_port new = *opp;
 199             sanitize_opp(&new);
 200             call_port_changed_callbacks(pw, port_no, old, &new);
 201             if (old) {
 202                 *old = new;
 203             } else {
 204                 port_array_set(&pw->ports, port_no, xmemdup(&new, sizeof new));
 205             }
 206         }
 207     }
 208 }

 210 static void
 211 update_netdev_monitor_devices(struct port_watcher *pw)
 212 {
 213     struct ofp_phy_port *p;
 214     struct svec netdevs;
 215     unsigned int port_no;
 216 
 217     svec_init(&netdevs);
 218     shash_clear(&pw->port_by_name);
 219     for (p = port_array_first(&pw->ports, &port_no); p;
 220          p = port_array_next(&pw->ports, &port_no)) {
 221         const char *name = (const char *) p->name;
 222         svec_add(&netdevs, name);
 223         shash_add(&pw->port_by_name, name, p);
 224     }
 225     netdev_monitor_set_devices(pw->mon, netdevs.names, netdevs.n);
 226     svec_destroy(&netdevs);
 227 }

 229 static bool
 230 port_watcher_local_packet_cb(struct relay *r, void *pw_)
 231 {
 232     struct port_watcher *pw = pw_;
 233     struct ofpbuf *msg = r->halves[HALF_LOCAL].rxbuf;
 234     struct ofp_header *oh = msg->data;
 235 
 236     if (oh->type == OFPT_FEATURES_REPLY
 237         && msg->size >= offsetof(struct ofp_switch_features, ports)) {
 238         struct ofp_switch_features *osf = msg->data;
 239         bool seen[PORT_ARRAY_SIZE];
 240         struct ofp_phy_port *p;
 241         unsigned int port_no;
 242         size_t n_ports;
 243         size_t i;
 244 
 245         pw->got_feature_reply = true;
 246         if (pw->datapath_id != osf->datapath_id) {
 247             pw->datapath_id = osf->datapath_id;
 248             VLOG_INFO("Datapath id is %012"PRIx64, ntohll(pw->datapath_id));
 249         }
 250 
 251         /* Update each port included in the message. */
 252         memset(seen, false, sizeof seen);
 253         n_ports = ((msg->size - offsetof(struct ofp_switch_features, ports))
 254                    / sizeof *osf->ports);
 255         for (i = 0; i < n_ports; i++) {
 256             struct ofp_phy_port *opp = &osf->ports[i];
 257             update_phy_port(pw, opp, OFPPR_MODIFY);
 258             seen[ntohs(opp->port_no)] = true;
 259         }
 260 
 261         /* Delete all the ports not included in the message. */
 262         for (p = port_array_first(&pw->ports, &port_no); p;
 263              p = port_array_next(&pw->ports, &port_no)) {
 264             if (!seen[port_no]) {
 265                 update_phy_port(pw, p, OFPPR_DELETE);
 266             }
 267         }
 268 
 269         update_netdev_monitor_devices(pw);
 270 
 271         call_local_port_changed_callbacks(pw);
 272     } else if (oh->type == OFPT_PORT_STATUS
 273                && msg->size >= sizeof(struct ofp_port_status)) {
 274         struct ofp_port_status *ops = msg->data;
 275         update_phy_port(pw, &ops->desc, ops->reason);
 276         if (ops->desc.port_no == htons(OFPP_LOCAL)) {
 277             call_local_port_changed_callbacks(pw);
 278         }
 279         if (ops->reason == OFPPR_ADD || OFPPR_DELETE) {
 280             update_netdev_monitor_devices(pw);
 281         }
 282     }
 283     return false;
 284 }

 286 static void
 287 bring_netdev_up_or_down(const char *name, bool down)
 288 {
 289     static struct vlog_rate_limit rl = VLOG_RATE_LIMIT_INIT(1, 5);
 290     struct netdev *netdev;
 291     int retval;
 292 
 293     retval = netdev_open(name, NETDEV_ETH_TYPE_NONE, &netdev);
 294     if (!retval) {
 295         if (down) {
 296             retval = netdev_turn_flags_off(netdev, NETDEV_UP, true);
 297         } else {
 298             retval = netdev_turn_flags_on(netdev, NETDEV_UP, true);
 299         }
 300         if (retval) {
 301             VLOG_WARN_RL(&rl, "failed to bring network device %s %s: %s",
 302                          name, down ? "down" : "up", strerror(retval));
 303         }
 304         netdev_close(netdev);
 305     } else {
 306         VLOG_WARN_RL(&rl, "failed to open network device %s: %s",
 307                      name, strerror(retval));
 308     }
 309 }

 311 static bool
 312 port_watcher_remote_packet_cb(struct relay *r, void *pw_)
 313 {
 314     struct port_watcher *pw = pw_;
 315     struct ofpbuf *msg = r->halves[HALF_REMOTE].rxbuf;
 316     struct ofp_header *oh = msg->data;
 317 
 318     if (oh->type == OFPT_PORT_MOD
 319         && msg->size >= sizeof(struct ofp_port_mod)) {
 320         struct ofp_port_mod *opm = msg->data;
 321         uint16_t port_no = ntohs(opm->port_no);
 322         struct ofp_phy_port *pw_opp = lookup_port(pw, port_no);
 323         if (pw_opp->port_no != htons(OFPP_NONE)) {
 324             struct ofp_phy_port old = *pw_opp;
 325             pw_opp->config = ((pw_opp->config & ~opm->mask)
 326                               | (opm->config & opm->mask));
 327             call_port_changed_callbacks(pw, port_no, &old, pw_opp);
 328             if (pw_opp->port_no == htons(OFPP_LOCAL)) {
 329                 call_local_port_changed_callbacks(pw);
 330             }
 331 
 332             if (opm->mask & htonl(OFPPC_PORT_DOWN)) {
 333                 bring_netdev_up_or_down((const char *) pw_opp->name,
 334                                         opm->config & htonl(OFPPC_PORT_DOWN));
 335             }
 336         }
 337     }
 338     return false;
 339 }

 352 static void
 353 port_watcher_periodic_cb(void *pw_)
 354 {
 355     struct port_watcher *pw = pw_;
 356     const char *name;
 357 
 358     if (!pw->got_feature_reply
 359         && time_now() >= pw->last_feature_request + 5
 360         && rconn_is_connected(pw->local_rconn)) {
 361         struct ofpbuf *b;
 362         make_openflow(sizeof(struct ofp_header), OFPT_FEATURES_REQUEST, &b);
 363         rconn_send_with_limit(pw->local_rconn, b, &pw->n_txq, 1);
 364         pw->last_feature_request = time_now();
 365     }
 366 
 367     netdev_monitor_run(pw->mon);
 368     while ((name = netdev_monitor_poll(pw->mon)) != NULL) {
 369         struct ofp_phy_port *opp;
 370         struct ofp_phy_port new_opp;
 371         enum netdev_flags flags;
 372         int retval;
 373 
 374         opp = shash_find_data(&pw->port_by_name, name);
 375         if (!opp) {
 376             continue;
 377         }
 378 
 379         retval = netdev_nodev_get_flags(name, &flags);
 380         if (retval) {
 381             static struct vlog_rate_limit rl = VLOG_RATE_LIMIT_INIT(1, 5);
 382             VLOG_WARN_RL(&rl, "could not get flags for %s", name);
 383             continue;
 384         }
 385 
 386         new_opp = *opp;
 387         set_bit(htonl(OFPPC_PORT_DOWN), ~flags & NETDEV_UP, &new_opp.config);
 388         set_bit(htonl(OFPPS_LINK_DOWN), ~flags & NETDEV_CARRIER,
 389                 &new_opp.state);
 390         if (opp->config != new_opp.config || opp->state != new_opp.state) {
 391             struct ofp_port_status *ops;
 392             struct ofpbuf *b;
 393 
 394             /* Notify other secchan modules. */
 395             update_phy_port(pw, &new_opp, OFPPR_MODIFY);
 396             if (new_opp.port_no == htons(OFPP_LOCAL)) {
 397                 call_local_port_changed_callbacks(pw);
 398             }
 399 
 400             /* Notify the controller that the flags changed. */
 401             ops = make_openflow(sizeof *ops, OFPT_PORT_STATUS, &b);
 402             ops->reason = OFPPR_MODIFY;
 403             ops->desc = new_opp;
 404             rconn_send(pw->remote_rconn, b, NULL);
 405         }
 406     }
 407 }

 409 static void
 410 port_watcher_wait_cb(void *pw_)
 411 {
 412     struct port_watcher *pw = pw_;
 413     if (!pw->got_feature_reply && rconn_is_connected(pw->local_rconn)) {
 414         if (pw->last_feature_request != TIME_MIN) {
 415             poll_timer_wait(pw->last_feature_request + 5 - time_now());
 416         } else {
 417             poll_immediate_wake();
 418         }
 419     }
 420     netdev_monitor_wait(pw->mon);
 421 }

 438 static void
 439 put_features(struct ds *ds, const char *name, uint32_t features)
 440 {
 441     if (features & (OFPPF_10MB_HD | OFPPF_10MB_FD
 442                     | OFPPF_100MB_HD | OFPPF_100MB_FD
 443                     | OFPPF_1GB_HD | OFPPF_1GB_FD | OFPPF_10GB_FD)) {
 444         ds_put_cstr(ds, name);
 445         put_duplexes(ds, "10M", features, OFPPF_10MB_HD, OFPPF_10MB_FD);
 446         put_duplexes(ds, "100M", features,
 447                      OFPPF_100MB_HD, OFPPF_100MB_FD);
 448         put_duplexes(ds, "1G", features, OFPPF_1GB_HD, OFPPF_1GB_FD);
 449         if (features & OFPPF_10GB_FD) {
 450             ds_put_cstr(ds, " 10G");
 451         }
 452         if (features & OFPPF_AUTONEG) {
 453             ds_put_cstr(ds, " AUTO_NEG");
 454         }
 455         if (features & OFPPF_PAUSE) {
 456             ds_put_cstr(ds, " PAUSE");
 457         }
 458         if (features & OFPPF_PAUSE_ASYM) {
 459             ds_put_cstr(ds, " PAUSE_ASYM");
 460         }
 461     }
 462 }

 464 static void
 465 log_port_status(uint16_t port_no,
 466                 const struct ofp_phy_port *old,
 467                 const struct ofp_phy_port *new,
 468                 void *aux UNUSED)
 469 {
 470     if (VLOG_IS_DBG_ENABLED()) {
 471         if (old && new && (opp_differs(old, new)
 472                            == ((old->config != new->config)
 473                                + (old->state != new->state))))
 474         {
 475             /* Don't care if only state or config changed. */
 476         } else if (!new) {
 477             if (old) {
 478                 VLOG_DBG("Port %d deleted", port_no);
 479             }
 480         } else {
 481             struct ds ds = DS_EMPTY_INITIALIZER;
 482             uint32_t curr = ntohl(new->curr);
 483             uint32_t supported = ntohl(new->supported);
 484             ds_put_format(&ds, "\"%s\", "ETH_ADDR_FMT, new->name,
 485                           ETH_ADDR_ARGS(new->hw_addr));
 486             if (curr) {
 487                 put_features(&ds, ", current", curr);
 488             }
 489             if (supported) {
 490                 put_features(&ds, ", supports", supported);
 491             }
 492             VLOG_DBG("Port %d %s: %s",
 493                      port_no, old ? "changed" : "added", ds_cstr(&ds));
 494             ds_destroy(&ds);
 495         }
 496     }
 497 }

 542 void
 543 port_watcher_set_flags(struct port_watcher *pw, uint16_t port_no, 
 544                        uint32_t config, uint32_t c_mask,
 545                        uint32_t state, uint32_t s_mask)
 546 {
 547     struct ofp_phy_port old;
 548     struct ofp_phy_port *p;
 549     struct ofp_port_mod *opm;
 550     struct ofp_port_status *ops;
 551     struct ofpbuf *b;
 552 
 553     p = lookup_port(pw, port_no);
 554     if (!p) {
 555         return;
 556     }
 557 
 558     if (!((ntohl(p->state) ^ state) & s_mask) 
 559             && (!((ntohl(p->config) ^ config) & c_mask))) {
 560         return;
 561     }
 562     old = *p;
 563 
 564     /* Update our idea of the flags. */
 565     p->config = htonl((ntohl(p->config) & ~c_mask) | (config & c_mask));
 566     p->state = htonl((ntohl(p->state) & ~s_mask) | (state & s_mask));
 567     call_port_changed_callbacks(pw, port_no, &old, p);
 568 
 569     /* Change the flags in the datapath. */
 570     opm = make_openflow(sizeof *opm, OFPT_PORT_MOD, &b);
 571     opm->port_no = p->port_no;
 572     memcpy(opm->hw_addr, p->hw_addr, OFP_ETH_ALEN);
 573     opm->config = p->config;
 574     opm->mask = htonl(c_mask);
 575     opm->advertise = htonl(0);
 576     rconn_send(pw->local_rconn, b, NULL);
 577 
 578     /* Notify the controller that the flags changed. */
 579     ops = make_openflow(sizeof *ops, OFPT_PORT_STATUS, &b);
 580     ops->reason = OFPPR_MODIFY;
 581     ops->desc = *p;
 582     rconn_send(pw->remote_rconn, b, NULL);
 583 }

 592     port_watcher_local_packet_cb,                        /* local_packet_cb */
 593     port_watcher_remote_packet_cb,                       /* remote_packet_cb */
 594     port_watcher_periodic_cb,                            /* periodic_cb */
 595     port_watcher_wait_cb,                                /* wait_cb */
 596     NULL,                                                /* closing_cb */
 597 };

 599 void
 600 port_watcher_start(struct secchan *secchan,
 601                    struct rconn *local_rconn, struct rconn *remote_rconn,
 602                    struct port_watcher **pwp)
 603 {
 604     struct port_watcher *pw;
 605     int retval;
 606 
 607     pw = *pwp = xcalloc(1, sizeof *pw);
 608     pw->local_rconn = local_rconn;
 609     pw->remote_rconn = remote_rconn;
 610     pw->last_feature_request = TIME_MIN;
 611     port_array_init(&pw->ports);
 612     pw->local_port_name[0] = '\0';
 613     retval = netdev_monitor_create(&pw->mon);
 614     if (retval) {
 615         ofp_fatal(retval, "failed to start network device monitoring");
 616     }
 617     shash_init(&pw->port_by_name);
 618     port_watcher_register_callback(pw, log_port_status, NULL);
 619     add_hook(secchan, &port_watcher_hook_class, pw);
 620 }
