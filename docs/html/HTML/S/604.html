3375                     add_child(port_tree, ofp_port_stats_collisions, tvb, &offset, 8);
3376                 }
3377                 break;
3378             }
3379 
3380             case OFPST_QUEUE: {
3381                 /* process each port stats struct in the packet */
3382                 while( offset < len ) {
3383                     proto_item *queue_item = proto_tree_add_item(type_tree, ofp_queue_stats, tvb, offset, -1, FALSE);
3384                     proto_tree *queue_tree = proto_item_add_subtree(queue_item, ett_ofp_queue_stats);

3388                                         dissect_queue_id(queue_tree, ofp_queue_stats_queue_id, tvb, &offset);
3389                     add_child(queue_tree, ofp_queue_stats_tx_bytes, tvb, &offset, 8);
3390                     add_child(queue_tree, ofp_queue_stats_tx_packets, tvb, &offset, 8);
3391                     add_child(queue_tree, ofp_queue_stats_tx_errors, tvb, &offset, 8);
3392                 }
3393                 break;
3394             }
3395 
3396             case OFPST_VENDOR: {
3397                 proto_item* vendor_item = proto_tree_add_item(type_tree, ofp_vendor_stats, tvb, offset, -1, FALSE);
3398                 proto_tree* vendor_tree = proto_item_add_subtree(vendor_item, ett_ofp_vendor_stats);
3399 
3400                 add_child(vendor_tree, ofp_vendor_stats_vendor, tvb, &offset, 4);
3401                 add_child(vendor_tree, ofp_vendor_stats_body, tvb, &offset, len - offset);
3402                 
3403                 break;
3404             }
3405 
3406             default:
3407                 /* add as bytes if type isn't one we know how to dissect */
3408                 add_child(type_tree, ofp_stats_reply_body, tvb, &offset, len - offset);

3412         }

3414         default:
