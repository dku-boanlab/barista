  57         char* name;
  58         char* parent;
  59         struct _ddict_typedefn_t* next;
  60 } ddict_typedefn_t;

  62 typedef struct _ddict_cmd_t {
  63         char* name;
  64         char* vendor;
  65         unsigned code;
  66         struct _ddict_cmd_t* next;
  67 } ddict_cmd_t;

  73         struct _ddict_xmlpi_t* next;
