  46 static uintptr_t UNUSED
  47 get_max_stack(void)
  48 {
  49     static const char file_name[] = "/proc/self/maps";
  50     char line[1024];
  51     int line_number;
  52     FILE *f;
  53 
  54     f = fopen(file_name, "r");
  55     if (f == NULL) {
  56         VLOG_WARN("opening %s failed: %s", file_name, strerror(errno));
  57         return -1;
  58     }
  59 
  60     for (line_number = 1; fgets(line, sizeof line, f); line_number++) {
  61         if (strstr(line, "[stack]")) {
  62             uintptr_t end;
  63             if (sscanf(line, "%*"SCNxPTR"-%"SCNxPTR, &end) != 1) {
  64                 VLOG_WARN("%s:%d: parse error", file_name, line_number);
  65                 continue;
  66             }
  67             fclose(f);
  68             return end;
  69         }
  70     }
  71     fclose(f);
  72 
  73     VLOG_WARN("%s: no stack found", file_name);
  74     return -1;
  75 }

  77 static uintptr_t
  78 stack_high(void)
  79 {
  80     static uintptr_t high;
  81     if (!high) {
  82         high = get_max_stack();
  83     }
  84     return high;
  85 }

  87 static uintptr_t
  88 stack_low(void)
  89 {
  90 #ifdef __i386__
  91     uintptr_t low;
  92     asm("movl %%esp,%0" : "=g" (low));
  93     return low;
  94 #else
  95     /* This causes a warning in GCC that cannot be disabled, so use it only on
  96      * non-x86. */
  97     int dummy;
  98     return (uintptr_t) &dummy;
  99 #endif
 100 }

 102 static bool
 103 in_stack(void *p)
 104 {
 105     uintptr_t address = (uintptr_t) p;
 106     return address >= stack_low() && address < stack_high();
 107 }

 109 void
 110 backtrace_capture(struct backtrace *bt)
 111 {
 112 #ifdef OLD_BACKTRACE_CAPTURE
 113     void **frame;
 114     size_t n;
 115 
 116     n = 0;
 117     for (frame = __builtin_frame_address(1);
 118          frame != NULL && in_stack(frame) && frame[0] != NULL
 119              && n < BACKTRACE_MAX_FRAMES;
 120          frame = frame[0])
 121     {
 122         bt->frames[n++] = (uintptr_t) frame[1];
 123     }
 124     bt->n_frames = n;
 125 #else
 126     bt->n_frames = backtrace(bt->frames, BACKTRACE_MAX_FRAMES);
 127 #endif
 128 }
