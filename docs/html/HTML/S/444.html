  25 #ifndef __PLUGINS_H__
  26 #define __PLUGINS_H__
  27 
  28 #include <glib.h>
  29 #include <gmodule.h>
  30 
  31 #include "packet.h"
  32 
  33 typedef struct _plugin {
  34     GModule     *handle;          /* handle returned by dlopen */
  35     gchar       *name;            /* plugin name */
  36     gchar       *version;         /* plugin version */
  37     void (*register_protoinfo)(void); /* routine to call to register protocol information */
  38     void (*reg_handoff)(void);    /* routine to call to register dissector handoff */
  39     void (*register_tap_listener)(void);   /* routine to call to register tap listener */
  40     void (*register_wtap_module)(void);  /* routine to call to register a wiretap module */
  41     void (*register_codec_module)(void);  /* routine to call to register a codec */
  42     struct _plugin *next;         /* forward link */
  43 } plugin;
  44 
  45 WS_VAR_IMPORT plugin *plugin_list;
  46 
  47 extern void init_plugins(void);
  48 extern void register_all_plugin_registrations(void);
  49 extern void register_all_plugin_handoffs(void);
  50 extern void register_all_plugin_tap_listeners(void);
  51 extern void register_all_wiretap_modules(void);
  52 extern void register_all_codecs(void);
  53 
  54 /* get the personal plugin dir */
  55 /* Return value is g_malloced so the caller should g_free() it. */
  56 extern char *get_plugins_pers_dir(void);
  57 
  58 #endif /* __PLUGINS_H__ */
