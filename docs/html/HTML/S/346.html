  38 /*
  39  * XXX - this is also the traditional default snapshot size in
  40  * tcpdump - but, if IPv6 is enabled, it defaults to 96, to get an
  41  * IPv6 header + TCP + 22 extra bytes.
  42  *
  43  * Some libpcap versions for particular capture devices might happen
  44  * to impose a minimum, but it's not always 68.
  45  */
  46 #define MIN_PACKET_SIZE 68      /* minimum amount of packet data we can read */
  47 
  48 /*
  49  * The list of interfaces returned by "get_interface_list()" is
  50  * a list of these structures.
  51  */
  52 typedef struct {
  53         char    *name;          /* e.g. "eth0" */
  54         char    *description;   /* from OS, e.g. "Local Area Connection" or NULL */
  55         GSList  *ip_addr;       /* containing address values of if_addr_t */
  56         gboolean loopback;      /* TRUE if loopback, FALSE otherwise */
  57 } if_info_t;
  58 
  59 /*
  60  * An address in the "ip_addr" list.
  61  */
  62 typedef struct {
  63         address_type type;      /* AT_IPv4 or AT_IPv6 */
  64         union {
  65                 guint32 ip4_addr;   /*  4 byte IP V4 address, or */
  66                 guint8 ip6_addr[16];/* 16 byte IP V6 address */
  67         } ip_addr;
  68 } if_addr_t;
  69 
  70 GList *get_interface_list(int *err, char **err_str);
  71 #ifdef HAVE_PCAP_REMOTE
  72 GList *get_remote_interface_list(const char *hostname, const char *port,
  73                                  int auth_type, const char *username,
  74                                  const char *passwd, int *err, char **err_str);
  75 #endif
  76 
  77 /* Error values from "get_interface_list()/capture_interface_list()". */
  78 #define CANT_GET_INTERFACE_LIST 1       /* error getting list */
  79 #define NO_INTERFACES_FOUND     2       /* list is empty */
  80 #define CANT_RUN_DUMPCAP        3       /* problem running dumpcap */
  81 
  82 void free_interface_list(GList *if_list);
  83 
  84 /*
  85  * The list of data link types returned by "get_pcap_linktype_list()" is
  86  * a list of these structures.
  87  */
  88 typedef struct {
  89         int     dlt;                /* e.g. DLT_EN10MB (which is 1) */
  90         char    *name;          /* e.g. "EN10MB" or "DLT 1" */
  91         char    *description;   /* descriptive name from wiretap e.g. "Ethernet", NULL if unknown */
  92 } data_link_info_t;
  93 
  94 GList *get_pcap_linktype_list(const char *devname, char **err_str);
  95 void free_pcap_linktype_list(GList *linktype_list);
  96 
  97 /* get/set the link type of an interface */
  98 /* (only used in capture_loop.c / capture-pcap-util.c) */
  99 int get_pcap_linktype(pcap_t *pch, const char *devname);
 100 const char *set_pcap_linktype(pcap_t *pch, char *devname, int dlt);
 101 
 102 
 103 #ifdef HAVE_PCAP_DATALINK_VAL_TO_NAME
 104 const char *linktype_val_to_name(int dlt);
 105 #endif
 106 #ifdef HAVE_PCAP_DATALINK_NAME_TO_VAL
 107 int linktype_name_to_val(const char *linktype);
 108 #endif

 116 /*
 117  * Append to a GString an indication of the version of libpcap/WinPcap
 118  * with which we were compiled, if we were, or an indication that we
 119  * weren't compiled with libpcap/WinPcap, if we weren't.
 120  */
 121 extern void get_compiled_pcap_version(GString *str);
 122 
 123 /*
 124  * Append to a GString an indication of the version of libpcap/WinPcap
 125  * with which we're running, or an indication that we're not running
 126  * with libpcap/WinPcap, if we were compiled with libpcap/WinPcap,
 127  * or nothing, if we weren't compiled with libpcap/WinPcap.
 128  */
 129 extern void get_runtime_pcap_version(GString *str);
