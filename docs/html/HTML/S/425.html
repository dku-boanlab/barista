  50 /* IEEE 802.11 packet type values                                               */
  51 #define AIRPDCAP_TYPE_MANAGEMENT                0
  52 #define AIRPDCAP_TYPE_CONTROL                   1
  53 #define AIRPDCAP_TYPE_DATA                      2

  64 /* Macro definitions                                                            */

  66 /**
  67  * Macros to get various bits of a 802.11 control frame
  68  */
  69 #define AIRPDCAP_TYPE(FrameControl_0)           (UINT8)((FrameControl_0 >> 2) & 0x3)
  70 #define AIRPDCAP_SUBTYPE(FrameControl_0)        (UINT8)((FrameControl_0 >> 4) & 0xF)
  71 #define AIRPDCAP_DS_BITS(FrameControl_1)        (UINT8)(FrameControl_1 & 0x3)
  72 #define AIRPDCAP_TO_DS(FrameControl_1)          (UINT8)(FrameControl_1 & 0x1)
  73 #define AIRPDCAP_FROM_DS(FrameControl_1)        (UINT8)((FrameControl_1 >> 1) & 0x1)
  74 #define AIRPDCAP_WEP(FrameControl_1)            (UINT8)((FrameControl_1 >> 6) & 0x1)
  75 
  76 /**
  77  * Get the Key ID from the Initialization Vector (last byte)
  78  */
  79 #define AIRPDCAP_EXTIV(KeyID)   ((KeyID >> 5) & 0x1)
  80 
  81 /* Macros to get various bits of an EAPOL frame                         */
  82 #define AIRPDCAP_EAP_KEY_DESCR_VER(KeyInfo_1)   ((UCHAR)(KeyInfo_1 & 0x3))
  83 #define AIRPDCAP_EAP_KEY(KeyInfo_1)             ((KeyInfo_1 >> 3) & 0x1)
  84 #define AIRPDCAP_EAP_INST(KeyInfo_1)            ((KeyInfo_1 >> 6) & 0x1)
  85 #define AIRPDCAP_EAP_ACK(KeyInfo_1)             ((KeyInfo_1 >> 7) & 0x1)
  86 #define AIRPDCAP_EAP_MIC(KeyInfo_0)             (KeyInfo_0 & 0x1)
  87 #define AIRPDCAP_EAP_SEC(KeyInfo_0)             ((KeyInfo_0 >> 1) & 0x1)
  88 
  89 /* Note: copied from net80211/ieee80211_airpdcap_tkip.c                 */
  90 #define S_SWAP(a,b) { UINT8 t = S[a]; S[a] = S[b]; S[b] = t; }

  97 /*
  98  * XXX - According to the thread at
  99  * http://www.wireshark.org/lists/wireshark-dev/200612/msg00384.html we

 108 /* Definition of IEEE 802.11 frame (without the address 4)                      */
 109 typedef struct _AIRPDCAP_MAC_FRAME {
 110         UCHAR   fc[2];
 111         UCHAR   dur[2];
 112         UCHAR   addr1[AIRPDCAP_MAC_LEN];
 113         UCHAR   addr2[AIRPDCAP_MAC_LEN];
 114         UCHAR   addr3[AIRPDCAP_MAC_LEN];
 115         UCHAR   seq[2];
 116 } AIRPDCAP_MAC_FRAME, *PAIRPDCAP_MAC_FRAME;
 117 
 118 /* Definition of IEEE 802.11 frame (with the address 4)                 */
 119 typedef struct _AIRPDCAP_MAC_FRAME_ADDR4 {
 120         UCHAR   fc[2];
 121         UCHAR   dur[2];
 122         UCHAR   addr1[AIRPDCAP_MAC_LEN];
 123         UCHAR   addr2[AIRPDCAP_MAC_LEN];
 124         UCHAR   addr3[AIRPDCAP_MAC_LEN];
 125         UCHAR   seq[2];
 126         UCHAR   addr4[AIRPDCAP_MAC_LEN];
 127 } AIRPDCAP_MAC_FRAME_ADDR4, *PAIRPDCAP_MAC_FRAME_ADDR4;
 128 
 129 /* Definition of IEEE 802.11 frame (without the address 4, with QOS)            */
 130 typedef struct _AIRPDCAP_MAC_FRAME_QOS {
 131         UCHAR   fc[2];
 132         UCHAR   dur[2];
 133         UCHAR   addr1[AIRPDCAP_MAC_LEN];
 134         UCHAR   addr2[AIRPDCAP_MAC_LEN];
 135         UCHAR   addr3[AIRPDCAP_MAC_LEN];
 136         UCHAR   seq[2];
 137         UCHAR   qos[2];
 138 } AIRPDCAP_MAC_FRAME_QOS, *PAIRPDCAP_MAC_FRAME_QOS;
 139 
 140 /* Definition of IEEE 802.11 frame (with the address 4 and QOS)         */
 141 typedef struct _AIRPDCAP_MAC_FRAME_ADDR4_QOS {
 142         UCHAR   fc[2];
 143         UCHAR   dur[2];
 144         UCHAR   addr1[AIRPDCAP_MAC_LEN];
 145         UCHAR   addr2[AIRPDCAP_MAC_LEN];
 146         UCHAR   addr3[AIRPDCAP_MAC_LEN];
 147         UCHAR   seq[2];
 148         UCHAR   addr4[AIRPDCAP_MAC_LEN];
 149         UCHAR   qos[2];
 150 } AIRPDCAP_MAC_FRAME_ADDR4_QOS, *PAIRPDCAP_MAC_FRAME_ADDR4_QOS;
