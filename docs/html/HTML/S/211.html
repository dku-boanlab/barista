  90 static int
  91 of_hw_ioctl(of_hw_driver_t *hw_drv, uint32_t op, void **io_param,
  92              int *io_len UNUSED)
  93 {
  94     int val;
  95     of_hw_driver_int_t *hw_int;
  96     int rv = 0;
  97 
  98     hw_int = (of_hw_driver_int_t *)hw_drv;
  99 
 100     switch (op) {
 101     case OF_HW_IOCTL_TABLE_DEBUG_SET:
 102         val = *(int *)(*io_param);
 103         DBG_WARN("Changing table debug value to %d\n", val);
 104         /* FIXME:  Use HW driver for debug level? */
 105         hw_int->table_debug_level = val;
 106         of_hw_debug = val;
 107         break;
 108     case OF_HW_IOCTL_PORT_DEBUG_SET:
 109         val = *(int *)(*io_param);
 110         DBG_WARN("Changing port debug value to %d\n", val);
 111         hw_int->port_debug_level = val;
 112         break;
 113     case OF_HW_IOCTL_BYTE_PKT_CNTR_SET:
 114         val = *(int *)(*io_param);
 115         if ((val != OF_HW_CNTR_PACKETS) &&
 116                 (val != OF_HW_CNTR_BYTES)) {
 117             DBG_ERROR("Bad byte/pkt counters select value %d\n", val);
 118         } else {
 119             hw_int->stat_sel = val;
 120         }
 121         break;
 122     default:
 123         rv = -1; /* Change to UNSUPPORTED */
 124         break;
 125     }
 126 
 127     return 0;
 128 }

 146 static int
 147 of_hw_table_iterate(struct sw_table *table,
 148                      const struct sw_flow_key *key, uint16_t out_port,
 149                      struct sw_table_position *position,
 150                      int (*callback)(struct sw_flow *flow, void *private),
 151                      void *private)
 152 {
 153 
 154     of_hw_driver_int_t *hw_int;
 155     struct sw_flow *flow;
 156     unsigned long start;
 157 
 158     hw_int = (of_hw_driver_int_t *)table;
 159     if (hw_int != NULL) {
 160         start = ~position->private[0];
 161         LIST_FOR_EACH (flow, struct sw_flow, iter_node, &hw_int->iter_flows) {
 162             if (flow->serial <= start
 163                     && flow_matches_2wild(key, &flow->key)
 164                     && flow_has_out_port(flow, out_port)) {
 165                 int error;
 166 
 167                 /* Update stats as that's what's generally used */
 168                 TRY_NR(of_hw_sw_flow_stat_update(flow, NULL, false),
 169                        "sw flow stat update");
 170                 error = callback(flow, private);
 171                 if (error) {
 172                     position->private[0] = ~(flow->serial - 1);
 173                     return error;
 174                 }
 175             }
 176         }
 177     } else {
 178         DBG_ERROR("Table iterate called on NULL driver or SW flow table\n");
 179         return -1;
 180     }
 181 
 182     return 0;
 183 }

 217 of_hw_driver_t *
 218 new_of_hw_driver(struct datapath *dp)
 219 {
 220     struct sw_table *sw_tab;
 221     of_hw_driver_t *hw_drv;
 222     of_hw_driver_int_t *hw_int;
 223 
 224     if (!global_init_done) {
 225         if (hw_drv_global_init(1) < 0) {
 226             REPORT_ERROR("HW driver global init failed\n");
 227             return NULL;
 228         }
 229     }
 230     hw_int = ALLOC(sizeof(*hw_int));
 231     if (hw_int == NULL) {
 232         REPORT_ERROR("Could not allocate HW driver\n");
 233         return NULL;
 234     }
 235     memset(hw_int, 0, sizeof(*hw_int));
 236 
 237     if (hw_drv_int_init(hw_int, dp) < 0) {
 238         FREE(hw_int);
 239         return NULL;
 240     }
 241 
 242 
 243     /* These all point to the same place */
 244     hw_drv = &hw_int->hw_driver;
 245     sw_tab = &hw_drv->sw_table;
 246 
 247     sw_tab->n_lookup = 0;
 248     sw_tab->n_matched = 0;
 249 
 250     /* Fill out the function pointers */
 251     sw_tab->lookup = of_hw_flow_lookup;
 252     sw_tab->insert = of_hw_flow_install;
 253     sw_tab->modify = of_hw_flow_modify;
 254     sw_tab->delete = of_hw_flow_delete;
 255     sw_tab->timeout = of_hw_flow_timeout;
 256 
 257     sw_tab->destroy = of_hw_table_destroy;
 258     sw_tab->iterate = of_hw_table_iterate;
 259     sw_tab->stats = of_hw_sw_table_stats;
 260 
 261     hw_drv->table_stats_get =of_hw_table_stats_get;
 262     hw_drv->port_stats_get = of_hw_port_stats_get;
 263     hw_drv->flow_stats_get = NULL;
 264     hw_drv->aggregate_stats_get = NULL;
 265 
 266     hw_drv->port_add = of_hw_port_add;
 267     hw_drv->port_remove = of_hw_port_remove;
 268     hw_drv->port_link_get = of_hw_port_link_get;
 269     hw_drv->port_enable_set = of_hw_port_enable_set;
 270     hw_drv->port_enable_get = of_hw_port_enable_get;
 271     hw_drv->port_queue_config = of_hw_port_queue_config;
 272     hw_drv->port_queue_remove = of_hw_port_queue_remove;
 273     hw_drv->port_change_register = of_hw_port_change_register;
 274 
 275     hw_drv->packet_send = of_hw_packet_send;
 276     hw_drv->packet_receive_register = of_hw_packet_receive_register;
 277 
 278     hw_drv->ioctl = of_hw_ioctl;
 279 
 280     ++of_hw_drv_instances;
 281 
 282     return hw_drv;
 283 }

 285 /*
 286  * Deallocate a hardware datapath object
 287  * If clear_hw is set, the HW structures related to the
 288  * datapath are also cleared.
 289  *
 290  * In general, clear_hw should be set for now.
 291  */
 292 void
 293 delete_of_hw_driver(of_hw_driver_t *hw_drv)
 294 {
 295     int idx;
 296     of_hw_driver_int_t *hw_int;
 297 
 298     hw_int = (of_hw_driver_int_t *)hw_drv;
 299 
 300     /* Clear the port table of ownership for this dp */
 301     FOREACH_DP_PORT(idx, hw_drv) {
 302         of_hw_ports[idx].owner = NULL;
 303     }
 304 
 305     of_hw_flow_remove_all(hw_int);
 306 
 307     FREE(hw_drv);
 308     --of_hw_drv_instances;
 309 }
