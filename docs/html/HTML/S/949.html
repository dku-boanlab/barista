 133 /** \brief Timer */
 134 #include <time.h>
 135 #define waitsec(sec, nsec) \
 136 { \
 137     struct timespec req; \
 138     req.tv_sec = sec; \
 139     req.tv_nsec = nsec; \
 140     nanosleep(&req, NULL); \
 141 }
