 546                         }
 547                     // external?
 548                     } else {
 549                         app_t *a = app_ctx->app_list[i];
 550 
 551                         char app_cmd[__CONF_WORD_LEN] = {0};
 552                         sprintf(app_cmd, "pkill -SIGTERM %s", a->name);

 554                             cli_print(cli, "Failed to deactivate %s", a->name);
 555                             return -1;
 556                         }
 557 
 558                         a->activated = FALSE;
 559

 561                         zmq_ctx_destroy(a->push_ctx);

 563                         zmq_close(a->req_sock);
 564                         zmq_ctx_destroy(a->req_ctx);
 565 
 566                         cli_print(cli, "%s is deactivated", app_ctx->app_list[i]->name);
 567 
 568                         return 0;
 569                     }
 570                 } else {
 571                     cli_print(cli, "%s is not activated yet", name);
 572                     return -1;
 573                 }
 574             } else {
 575                 cli_print(cli, "%s is not enabled", name);
 576                 return -1;
 577             }
 578             break;
 579         }
 580     }
