  21 /**
  22  * \brief Function to convert the byte array of a MAC address to an integer
  23  * \param mac The pointer of a byte array
  24  * \return The integer converted from the byte array
  25  */
  26 uint64_t mac2int(const uint8_t *mac)
  27 {
  28     const uint8_t *p = mac;
  29     uint64_t res = 0;
  30 
  31     int i;
  32     for (i=5; i>=0; i--) {
  33         res |= (uint64_t) *p++ << (8 * i);
  34     }
  35 
  36     return res;
  37 }

  39 /**
  40  * \brief Function to convert the integer value of a MAC address to a byte array
  41  * \param mac The integer value of a MAC address
  42  * \param macaddr The pointer of a byte array to store the converted result
  43  * \return The byte array converted from the integer

  45 uint8_t *int2mac(const uint64_t mac, uint8_t *macaddr)
  46 {
  47     uint8_t *p = macaddr;
  48 
  49     int i;
  50     for (i=5; i>=0; i--) {
  51         *p++ = mac >> (8 * i);
  52     }
  53 
  54     return macaddr;

  63 uint8_t *str2mac(const char *macaddr, uint8_t *mac)
  64 {
  65     int tmp[ETH_ALEN] = {0};
  66 
  67     sscanf(macaddr, "%x:%x:%x:%x:%x:%x",
  68            &tmp[0], &tmp[1], &tmp[2], &tmp[3], &tmp[4], &tmp[5]);
  69 
  70     int i;
  71     for (i=0; i<ETH_ALEN; i++)
  72         mac[i] = (uint8_t)tmp[i];

  74     return mac;
