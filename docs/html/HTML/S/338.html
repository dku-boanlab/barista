  39 /** Dialog types. */
  40 typedef enum { 
  41     ESD_TYPE_INFO,          /**< tells the user something they should know, but not requiring
  42                                     any action; the only button should be "OK" */
  43     ESD_TYPE_WARN,          /**< tells the user about a problem; the only button should be "OK" */
  44     ESD_TYPE_CONFIRMATION,  /**< asks the user for confirmation; there should be more than
  45                                     one button */
  46     ESD_TYPE_ERROR,         /**< tells the user about a serious problem; the only button should be "OK" */
  47     ESD_TYPE_STOP           /**< tells the user a stop action is in progress, there should be no button */
  48 } ESD_TYPE_E;
  49 
  50 /** display no buttons at all */
  51 #define ESD_BTN_NONE   0x00
  52 /** display an "Ok" button */
  53 #define ESD_BTN_OK     0x01 
  54 /** display a "Cancel" button */
  55 #define ESD_BTN_CANCEL 0x02 
  56 /** display a "Yes" button */
  57 #define ESD_BTN_YES    0x04 
  58 /** display a "No" button */
  59 #define ESD_BTN_NO     0x08 
  60 /** display a "Clear" button */
  61 #define ESD_BTN_CLEAR  0x10
  62 /** display a "Save" button */
  63 #define ESD_BTN_SAVE   0x20
  64 /** display a "Continue without Saving" button */
  65 #define ESD_BTN_DONT_SAVE 0x40 
  66 
  67 /** Standard button combination "Ok" + "Cancel". */
  68 #define ESD_BTNS_OK_CANCEL      (ESD_BTN_OK|ESD_BTN_CANCEL)
  69 /** Standard button combination "Yes" + "No". */
  70 #define ESD_BTNS_YES_NO         (ESD_BTN_YES|ESD_BTN_NO)
  71 /** Standard button combination "Yes" + "No" + "Cancel". */
  72 #define ESD_BTNS_YES_NO_CANCEL  (ESD_BTN_YES|ESD_BTN_NO|ESD_BTN_CANCEL)
  73 /** Standard button combination "No" + "Cancel" + "Save". */
  74 #define ESD_BTNS_SAVE_DONTSAVE_CANCEL (ESD_BTN_DONT_SAVE|ESD_BTN_CANCEL|ESD_BTN_SAVE)

  76 #if __GNUC__ >= 2
  77 /** Create and show a simple dialog.
  78  *
  79  * @param type type of dialog
  80  * @param btn_mask the buttons to display
  81  * @param msg_format printf like message format
  82  * @param ... printf like parameters
  83  * @return the newly created dialog
  84  */
  85 extern gpointer simple_dialog(ESD_TYPE_E type, gint btn_mask,
  86     const gchar *msg_format, ...)
  87     __attribute__((format (printf, 3, 4)));
  88 /** Create and show a simple dialog using a va_list.
  89  *
  90  * @param type type of dialog
  91  * @param btn_mask the buttons to display
  92  * @param msg_format printf like message format
  93  * @param ap parameters
  94  * @return the newly created dialog
  95  */
  96 extern gpointer vsimple_dialog(ESD_TYPE_E type, gint btn_mask,
  97    const gchar *msg_format, va_list ap);
  98 #else
  99 /** Create and show a simple dialog.
 100  *
 101  * @param type type of dialog
 102  * @param btn_mask the buttons to display
 103  * @param msg_format printf like message format
 104  * @param ... printf like parameters
 105  * @return the newly created dialog
 106  */
 107 extern gpointer simple_dialog(ESD_TYPE_E type, gint btn_mask,
 108     const gchar *msg_format, ...);
 109 /** Create and show a simple dialog using a va_list.
 110  *

 112  * @param btn_mask the buttons to display

 114  * @param ap parameters
 115  * @return the newly created dialog

 121 /** Callback function type for simple_dialog_set_cb() */
 122 typedef void (* simple_dialog_cb_t) (gpointer dialog, gint btn, gpointer data);
 123 
 124 /** Set the callback function for the dialog, called when a button was pressed.
 125  *
 126  * @param dialog the dialog from simple_dialog()
 127  * @param callback_fct the callback function to set
 128  * @param data data to be passed to the callback function
 129  */
 130 extern void simple_dialog_set_cb(gpointer dialog, simple_dialog_cb_t callback_fct, gpointer data);
 131 
 132 /** Close the dialog, useful for "no button" dialogs.
 133  *
 134  * @param dialog the dialog to close from simple_dialog()
 135  */
 136 extern void simple_dialog_close(gpointer dialog);
 137 
 138 /** Add a check button to the dialog (e.g. "Don't show this message again")
 139  * 
 140  * @param dialog the dialog from simple_dialog()
 141  * @param text the text to display 
 142  */
 143 extern void simple_dialog_check_set(gpointer dialog, gchar *text);
 144 
 145 /** Get the check buttons state.
 146  * 
 147  * @param dialog the dialog from simple_dialog()
 148  * @return current button state (TRUE is checked)
 149  */
 150 extern gboolean simple_dialog_check_get(gpointer dialog);
 151 
 152 /** Surround the primary dialog message text by 
 153  *  simple_dialog_primary_start() and simple_dialog_primary_end().
 154  *  To highlight the first sentence (will take effect on GTK2 only).
 155  */
 156 extern char *simple_dialog_primary_start(void);
 157 /** Surround the primary dialog message text by 
 158  *  simple_dialog_primary_start() and simple_dialog_primary_end().
 159  *  To highlight the first sentence (will take effect on GTK2 only).
 160  */
 161 extern char *simple_dialog_primary_end(void);
 162 
 163 /** Escape the message text, if it probably contains Pango escape sequences.
 164  *  For example html like tags starting with a <.
 165  *
 166  * @param msg the string to escape
 167  * @return the escaped message text, must be freed with g_free() later
 168  */
 169 extern char *simple_dialog_format_message(const char *msg);
 170 
 171 /**
 172  * Display all queued messages.
 173  * If a routine is called to display a dialog before there are any windows
 174  * open, information to use to display the dialog is queued up.  This
 175  * routine should be called once there are windows open, so that the queued
 176  * up dialogs are displayed on top of those windows.
 177  */
 178 extern void display_queued_messages(void);
