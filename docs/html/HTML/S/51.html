 318 innitialize_generator_packet(struct pkt_details *state, struct traf_gen_det *det) {  
 319   state->data = (void *)malloc(det->pkt_size); 
 320   state->data_len = det->pkt_size;
 321 
 322   bzero((void *)state->data, state->data_len);
 323   if(state->data_len < sizeof(struct ether_vlan_header) + sizeof(struct iphdr) + sizeof(struct tcphdr)) {
 324     printf("packet size is too small\n");
 325     return 0;
 326   }
 327   //ethernet header with default values
 328   state->eth_vlan = (struct ether_vlan_header *) state->data;
 329   state->eth = (struct ether_header *) state->data;
 330   read_mac_addr(state->eth_vlan->ether_dhost, det->mac_dst);
 331   read_mac_addr(state->eth_vlan->ether_shost, det->mac_src);
 332   if(det->vlan != 0 && det->vlan != 0xffff) {
 333     state->eth_vlan->tpid = htons(0x8100);
 334     state->eth_vlan->vid = htons(det->vlan) >>4;
 335     state->eth_vlan->ether_type = htons(ETHERTYPE_IP);
 336     state->ip = (struct iphdr *)(state->data + sizeof(struct ether_vlan_header));
 337     state->ip->tot_len=htons(state->data_len - sizeof(struct ether_vlan_header)); 
 338     state->udp = (struct udphdr *)
 339       (state->data + sizeof(struct ether_vlan_header) + sizeof(struct iphdr));
 340     state->udp->len = htons(state->data_len - sizeof(struct ether_vlan_header) - sizeof(struct iphdr));
 341     state->pktgen = (struct pktgen_hdr *)
 342       (state->data + sizeof(struct ether_vlan_header) + sizeof(struct iphdr) + sizeof(struct udphdr));
 343   } else {
 344     state->eth->ether_type = htons(ETHERTYPE_IP);
 345     state->ip = (struct iphdr *)(state->data + sizeof(struct ether_header));
 346     state->ip->tot_len=htons(state->data_len - sizeof(struct ether_header)); 
 347     state->udp = (struct udphdr *)
 348       (state->data + sizeof(struct ether_header) + sizeof(struct iphdr));
 349     state->udp->len = htons(state->data_len - sizeof(struct ether_header) - sizeof(struct iphdr));
 350     state->pktgen = (struct pktgen_hdr *)
 351       (state->data + sizeof(struct ether_header) + sizeof(struct iphdr) + sizeof(struct udphdr));
 352   }
 353   //ip header with default values
 354   state->ip->protocol=1;
 355   state->ip->ihl=5;
 356   state->ip->version=4;
 357   //state->ip->check = htons(0x9a97);
 358   //total packet size without ethernet header
 359   state->ip->ttl = 100;
 360   state->ip->protocol = IPPROTO_UDP; //udp protocol
 361   state->ip->saddr = inet_addr(det->src_ip); 
 362   state->ip->daddr = inet_addr(det->dst_ip_min); //test.nw_dst;
 363   
 364   state->ip->check=ip_sum_calc(20, (uint16_t *) state->ip);
 365 
 366   state->udp->source = htons(det->udp_src_port);
 367   state->udp->dest = htons(det->udp_dst_port);
 368   
 369   state->pktgen->magic = 0xbe9be955;
 370 
 371   return 1;
 372 }

 376   char *p = str, *tmp;

 379     tmp = index(p, ':');

 382       tmp++;

 393 uint16_t ip_sum_calc(uint16_t len_ip_header, uint16_t buff[]) {
 394   uint16_t word16;
 395   uint32_t sum=0;
 396   uint16_t i;
 397   
 398   // make 16 bit words out of every two adjacent 8 bit words in the packet
 399   // and add them up
 400   for (i=0;i<len_ip_header;i=i+2){
 401     word16 =((buff[i]<<8)&0xFF00)+(buff[i+1]&0xFF);
 402     sum = sum + (uint32_t) word16;      
 403   }
 404   
 405   // take only 16 bits out of the 32 bit sum and add up the carries
 406   while (sum>>16)
 407     sum = (sum & 0xFFFF)+(sum >> 16);
 408   
 409   // one's complement the result
 410   sum = ~sum;
 411   
 412   return ((uint16_t) sum);
 413 }
