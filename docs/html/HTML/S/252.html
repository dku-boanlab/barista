 100 static void
 101 send_bpdu(struct ofpbuf *pkt, int port_no, void *b_)
 102 {
 103     struct bridge *b = b_;
 104     struct lan *lan;
 105 
 106     assert(port_no < b->n_ports);
 107     lan = b->ports[port_no];
 108     if (lan) {
 109         const void *data = pkt->l3;
 110         size_t size = (char *) ofpbuf_tail(pkt) - (char *) data;
 111         int i;
 112 
 113         for (i = 0; i < lan->n_conns; i++) {
 114             struct lan_conn *conn = &lan->conns[i];
 115             if (conn->bridge != b || conn->port_no != port_no) {
 116                 struct bridge *dst = conn->bridge;
 117                 struct bpdu *bpdu = &dst->rxq[dst->rxq_head++ % RXQ_SIZE];
 118                 assert(dst->rxq_head - dst->rxq_tail <= RXQ_SIZE);
 119                 bpdu->data = xmemdup(data, size);
 120                 bpdu->size = size;
 121                 bpdu->port_no = conn->port_no;
 122             }
 123         }
 124     }
 125     ofpbuf_delete(pkt);
 126 }

 156 static void
 157 reconnect_port(struct bridge *b, int port_no, struct lan *new_lan)
 158 {
 159     struct lan *old_lan;
 160     int j;
 161 
 162     assert(port_no < b->n_ports);
 163     old_lan = b->ports[port_no];
 164     if (old_lan == new_lan) {
 165         return;
 166     }
 167 
 168     /* Disconnect from old_lan. */
 169     if (old_lan) {
 170         for (j = 0; j < old_lan->n_conns; j++) {
 171             struct lan_conn *c = &old_lan->conns[j];
 172             if (c->bridge == b && c->port_no == port_no) {
 173                 memmove(c, c + 1, sizeof *c * (old_lan->n_conns - j - 1));
 174                 old_lan->n_conns--;
 175                 break;
 176             }
 177         }
 178     }
 179 
 180     /* Connect to new_lan. */
 181     b->ports[port_no] = new_lan;
 182     if (new_lan) {
 183         int conn_no = new_lan->n_conns++;
 184         assert(conn_no < ARRAY_SIZE(new_lan->conns));
 185         new_lan->conns[conn_no].bridge = b;
 186         new_lan->conns[conn_no].port_no = port_no;
 187     }
 188 }

 190 static void
 191 new_port(struct bridge *b, struct lan *lan, int path_cost)
 192 {
 193     int port_no = b->n_ports++;
 194     struct stp_port *p = stp_get_port(b->stp, port_no);
 195     assert(port_no < ARRAY_SIZE(b->ports));
 196     b->ports[port_no] = NULL;
 197     stp_port_set_path_cost(p, path_cost);
 198     stp_port_enable(p);
 199     reconnect_port(b, port_no, lan);
 200 }

 204 {
 205     int i;
 206 
 207     for (i = 0; i < tc->n_bridges; i++) {
 208         struct bridge *b = tc->bridges[i];
 209         struct stp *stp = b->stp;
 210         int j;
 211 
 212         printf("%s:", stp_get_name(stp));
 213         if (stp_is_root_bridge(stp)) {
 214             printf(" root");
 215         }
 216         printf("\n");
 217         for (j = 0; j < b->n_ports; j++) {
 218             struct stp_port *p = stp_get_port(stp, j);
 219             enum stp_state state = stp_port_get_state(p);
 220 
 221             printf("\tport %d", j);
 222             if (b->ports[j]) {
 223                 printf(" (lan %s)", b->ports[j]->name);
 224             } else {
 225                 printf(" (disconnected)");
 226             }
 227             printf(": %s", stp_state_name(state));
 228             if (p == stp_get_root_port(stp)) {
 229                 printf(" (root port, root_path_cost=%u)", stp_get_root_path_cost(stp));
 230             }
 231             printf("\n");
 232         }
 233     }

 238 static void
 239 dump_bridge_tree(struct test_case *tc, struct bridge *b, int level)
 240 {
 241     int i;
 242 
 243     if (b->reached) {
 244         return;
 245     }
 246     b->reached = true;
 247     for (i = 0; i < level; i++) {
 248         printf("\t");
 249     }
 250     printf("%s\n", stp_get_name(b->stp));
 251     for (i = 0; i < b->n_ports; i++) {
 252         struct lan *lan = b->ports[i];
 253         struct stp_port *p = stp_get_port(b->stp, i);
 254         if (stp_port_get_state(p) == STP_FORWARDING && lan) {
 255             dump_lan_tree(tc, lan, level + 1);
 256         }
 257     }

 279 static void
 280 tree(struct test_case *tc)
 281 {
 282     int i;
 283 
 284     for (i = 0; i < tc->n_bridges; i++) {
 285         struct bridge *b = tc->bridges[i];
 286         b->reached = false;
 287     }
 288     for (i = 0; i < tc->n_lans; i++) {
 289         struct lan *lan = tc->lans[i];
 290         lan->reached = false;
 291     }
 292     for (i = 0; i < tc->n_bridges; i++) {
 293         struct bridge *b = tc->bridges[i];
 294         struct stp *stp = b->stp;
 295         if (stp_is_root_bridge(stp)) {
 296             dump_bridge_tree(tc, b, 0);
 297         }
 298     }
 299 }

 301 static void
 302 simulate(struct test_case *tc, int granularity)
 303 {
 304     int time;
 305 
 306     for (time = 0; time < 1000 * 180; time += granularity) {
 307         int round_trips;
 308         int i;
 309 
 310         for (i = 0; i < tc->n_bridges; i++) {
 311             stp_tick(tc->bridges[i]->stp, granularity);
 312         }
 313         for (round_trips = 0; round_trips < granularity; round_trips++) {
 314             bool any = false;
 315             for (i = 0; i < tc->n_bridges; i++) {
 316                 struct bridge *b = tc->bridges[i];
 317                 for (; b->rxq_tail != b->rxq_head; b->rxq_tail++) {
 318                     struct bpdu *bpdu = &b->rxq[b->rxq_tail % RXQ_SIZE];
 319                     stp_received_bpdu(stp_get_port(b->stp, bpdu->port_no),
 320                                       bpdu->data, bpdu->size);
 321                     any = true;
 322                 }
 323             }
 324             if (!any) {
 325                 break;
 326             }
 327         }
 328     }
 329 }

 368 static bool
 369 get_token(void)
 370 {
 371     char *start;
 372 
 373     while (isspace((unsigned char) *pos)) {
 374         pos++;
 375     }
 376     if (*pos == '\0') {
 377         token = NULL;
 378         return false;
 379     }
 380 
 381     start = pos;
 382     if (isalpha((unsigned char) *pos)) {
 383         while (isalpha((unsigned char) *++pos)) {
 384             continue;
 385         }
 386     } else if (isdigit((unsigned char) *pos)) {
 387         if (*pos == '0' && (pos[1] == 'x' || pos[1] == 'X')) {
 388             pos += 2;
 389             while (isxdigit((unsigned char) *pos)) {
 390                 pos++;
 391             }
 392         } else {
 393             while (isdigit((unsigned char) *++pos)) {
 394                 continue;
 395             }
 396         }
 397     } else {
 398         pos++;
 399     }
 400 
 401     free(token);
 402     token = xmemdup0(start, pos - start);
 403     return true;
 404 }

 406 static bool
 407 get_int(int *intp)
 408 {
 409     char *save_pos = pos;
 410     if (token && isdigit((unsigned char) *token)) {
 411         *intp = strtol(token, NULL, 0);
 412         get_token();
 413         return true;
 414     } else {
 415         pos = save_pos;
 416         return false;
 417     }

 449 int
 450 main(int argc, char *argv[])
 451 {
 452     struct test_case *tc;
 453     FILE *input_file;
 454     int i;
 455 
 456     if (argc != 2) {
 457         ofp_fatal(0, "usage: test-stp INPUT.STP\n");
 458     }
 459     file_name = argv[1];
 460 
 461     input_file = fopen(file_name, "r");
 462     if (!input_file) {
 463         ofp_fatal(errno, "error opening \"%s\"", file_name);
 464     }
 465 
 466     tc = new_test_case();
 467     for (i = 0; i < 26; i++) {
 468         char name[2];
 469         name[0] = 'a' + i;
 470         name[1] = '\0';
 471         new_lan(tc, name);
 472     }
 473 
 474     for (line_number = 1; fgets(line, sizeof line, input_file);
 475          line_number++)
 476     {
 477         char *newline, *hash;
 478 
 479         newline = strchr(line, '\n');
 480         if (newline) {
 481             *newline = '\0';
 482         }
 483         hash = strchr(line, '#');
 484         if (hash) {
 485             *hash = '\0';
 486         }
 487 
 488         pos = line;
 489         if (!get_token()) {
 490             continue;
 491         }
 492         if (match("bridge")) {
 493             struct bridge *bridge;
 494             int bridge_no, port_no;
 495 
 496             bridge_no = must_get_int();
 497             if (bridge_no < tc->n_bridges) {
 498                 bridge = tc->bridges[bridge_no];
 499             } else if (bridge_no == tc->n_bridges) {
 500                 bridge = new_bridge(tc, must_get_int());
 501             } else {
 502                 err("bridges must be numbered consecutively from 0");
 503             }
 504             if (match("^")) {
 505                 stp_set_bridge_priority(bridge->stp, must_get_int());
 506             }
 507 
 508             if (match("=")) {
 509                 for (port_no = 0; port_no < STP_MAX_PORTS; port_no++) {
 510                     struct stp_port *p = stp_get_port(bridge->stp, port_no);
 511                     if (!token || match("X")) {
 512                         stp_port_disable(p);
 513                     } else if (match("_")) {
 514                         /* Nothing to do. */
 515                     } else {
 516                         struct lan *lan;
 517                         int path_cost;
 518 
 519                         if (!strcmp(token, "0")) {
 520                             lan = NULL;
 521                         } else if (strlen(token) == 1 && islower(*token)) {
 522                             lan = tc->lans[*token - 'a']; 
 523                         } else {
 524                             err("%s is not a valid LAN name "
 525                                 "(0 or a lowercase letter)", token);
 526                         }
 527                         get_token();
 528 
 529                         path_cost = match(":") ? must_get_int() : 10;
 530                         if (port_no < bridge->n_ports) {
 531                             stp_port_set_path_cost(p, path_cost);
 532                             stp_port_enable(p);
 533                             reconnect_port(bridge, port_no, lan);
 534                         } else if (port_no == bridge->n_ports) {
 535                             new_port(bridge, lan, path_cost);
 536                         } else {
 537                             err("ports must be numbered consecutively");
 538                         }
 539                         if (match("^")) {
 540                             stp_port_set_priority(p, must_get_int());
 541                         }
 542                     }
 543                 }
 544             }
 545         } else if (match("run")) {
 546             simulate(tc, must_get_int());
 547         } else if (match("dump")) {
 548             dump(tc);
 549         } else if (match("tree")) {
 550             tree(tc);
 551         } else if (match("check")) {
 552             struct bridge *b;
 553             struct stp *stp;
 554             int bridge_no, port_no;
 555 
 556             bridge_no = must_get_int();
 557             if (bridge_no >= tc->n_bridges) {
 558                 err("no bridge numbered %d", bridge_no);
 559             }
 560             b = tc->bridges[bridge_no];
 561             stp = b->stp;
 562 
 563             must_match("=");
 564 
 565             if (match("rootid")) {
 566                 uint64_t rootid;
 567                 must_match(":");
 568                 rootid = must_get_int();
 569                 if (match("^")) {
 570                     rootid |= (uint64_t) must_get_int() << 48;
 571                 } else {
 572                     rootid |= UINT64_C(0x8000) << 48;
 573                 }
 574                 if (stp_get_designated_root(stp) != rootid) {
 575                     warn("%s: root %"PRIx64", not %"PRIx64,
 576                          stp_get_name(stp), stp_get_designated_root(stp),
 577                          rootid);
 578                 }
 579             }
 580 
 581             if (match("root")) {
 582                 if (stp_get_root_path_cost(stp)) {
 583                     warn("%s: root path cost of root is %u but should be 0",
 584                          stp_get_name(stp), stp_get_root_path_cost(stp));
 585                 }
 586                 if (!stp_is_root_bridge(stp)) {
 587                     warn("%s: root is %"PRIx64", not %"PRIx64,
 588                          stp_get_name(stp),
 589                          stp_get_designated_root(stp), stp_get_bridge_id(stp));
 590                 }
 591                 for (port_no = 0; port_no < b->n_ports; port_no++) {
 592                     struct stp_port *p = stp_get_port(stp, port_no);
 593                     enum stp_state state = stp_port_get_state(p);
 594                     if (!(state & (STP_DISABLED | STP_FORWARDING))) {
 595                         warn("%s: root port %d in state %s",
 596                              stp_get_name(b->stp), port_no,
 597                              stp_state_name(state));
 598                     }
 599                 }
 600             } else {
 601                 for (port_no = 0; port_no < STP_MAX_PORTS; port_no++) {
 602                     struct stp_port *p = stp_get_port(stp, port_no);
 603                     enum stp_state state;
 604                     if (token == NULL || match("D")) {
 605                         state = STP_DISABLED;
 606                     } else if (match("B")) {
 607                         state = STP_BLOCKING;
 608                     } else if (match("Li")) {
 609                         state = STP_LISTENING;
 610                     } else if (match("Le")) {
 611                         state = STP_LEARNING;
 612                     } else if (match("F")) {
 613                         state = STP_FORWARDING;
 614                     } else if (match("_")) {
 615                         continue;
 616                     } else {
 617                         err("unknown port state %s", token);
 618                     }
 619                     if (stp_port_get_state(p) != state) {
 620                         warn("%s port %d: state is %s but should be %s",
 621                              stp_get_name(stp), port_no,
 622                              stp_state_name(stp_port_get_state(p)),
 623                              stp_state_name(state));
 624                     }
 625                     if (state == STP_FORWARDING) {
 626                         struct stp_port *root_port = stp_get_root_port(stp);
 627                         if (match(":")) {
 628                             int root_path_cost = must_get_int();
 629                             if (p != root_port) {
 630                                 warn("%s: port %d is not the root port",
 631                                      stp_get_name(stp), port_no);
 632                                 if (!root_port) {
 633                                     warn("%s: (there is no root port)",
 634                                          stp_get_name(stp));
 635                                 } else {
 636                                     warn("%s: (port %d is the root port)",
 637                                          stp_get_name(stp),
 638                                          stp_port_no(root_port));
 639                                 }
 640                             } else if (root_path_cost
 641                                        != stp_get_root_path_cost(stp)) {
 642                                 warn("%s: root path cost is %u, should be %d",
 643                                      stp_get_name(stp),
 644                                      stp_get_root_path_cost(stp),
 645                                      root_path_cost);
 646                             }
 647                         } else if (p == root_port) {
 648                             warn("%s: port %d is the root port but "
 649                                  "not expected to be",
 650                                  stp_get_name(stp), port_no);
 651                         }
 652                     }
 653                 }
 654             }
 655             if (n_warnings) {
 656                 exit(EXIT_FAILURE);
 657             }
 658         }
 659         if (get_token()) {
 660             err("trailing garbage on line");
 661         }
 662     }
 663 
 664     return 0;
 665 }
