  53 /* Sets 'fd' to non-blocking mode.  Returns 0 if successful, otherwise a
  54  * positive errno value. */
  55 int
  56 set_nonblocking(int fd)
  57 {
  58     int flags = fcntl(fd, F_GETFL, 0);
  59     if (flags != -1) {
  60         if (fcntl(fd, F_SETFL, flags | O_NONBLOCK) != -1) {
  61             return 0;
  62         } else {
  63             VLOG_ERR("fcntl(F_SETFL) failed: %s", strerror(errno));
  64             return errno;
  65         }
  66     } else {
  67         VLOG_ERR("fcntl(F_GETFL) failed: %s", strerror(errno));
  68         return errno;
  69     }
  70 }

  81 /* Returns the maximum valid FD value, plus 1. */
  82 int
  83 get_max_fds(void)
  84 {
  85     static int max_fds = -1;
  86     if (max_fds < 0) {
  87         struct rlimit r;
  88         if (!getrlimit(RLIMIT_NOFILE, &r)
  89             && r.rlim_cur != RLIM_INFINITY
  90             && r.rlim_cur != RLIM_SAVED_MAX
  91             && r.rlim_cur != RLIM_SAVED_CUR) {
  92             max_fds = r.rlim_cur;
  93         } else {
  94             VLOG_WARN("failed to obtain fd limit, defaulting to 1024");
  95             max_fds = 1024;
  96         }
  97     }
  98     return max_fds;
  99 }

 101 /* Translates 'host_name', which may be a DNS name or an IP address, into a
 102  * numeric IP address in '*addr'.  Returns 0 if successful, otherwise a
 103  * positive errno value. */
 104 int
 105 lookup_ip(const char *host_name, struct in_addr *addr) 
 106 {
 107     if (!inet_aton(host_name, addr)) {
 108         struct hostent *he = gethostbyname(host_name);
 109         if (he == NULL) {
 110             struct vlog_rate_limit rl = VLOG_RATE_LIMIT_INIT(1, 5);
 111             VLOG_ERR_RL(&rl, "gethostbyname(%s): %s", host_name,
 112                         (h_errno == HOST_NOT_FOUND ? "host not found"
 113                          : h_errno == TRY_AGAIN ? "try again"
 114                          : h_errno == NO_RECOVERY ? "non-recoverable error"
 115                          : h_errno == NO_ADDRESS ? "no address"
 116                          : "unknown error"));
 117             return ENOENT;
 118         }
 119         addr->s_addr = *(uint32_t *) he->h_addr;
 120     }
 121     return 0;
 122 }

 124 /* Returns the error condition associated with socket 'fd' and resets the
 125  * socket's error status. */
 126 int
 127 get_socket_error(int fd) 
 128 {
 129     int error;
 130     socklen_t len = sizeof(error);
 131     if (getsockopt(fd, SOL_SOCKET, SO_ERROR, &error, &len) < 0) {
 132         struct vlog_rate_limit rl = VLOG_RATE_LIMIT_INIT(5, 10);
 133         error = errno;
 134         VLOG_ERR_RL(&rl, "getsockopt(SO_ERROR): %s", strerror(error));
 135     }
 136     return error;
 137 }

 139 int
 140 check_connection_completion(int fd) 
 141 {
 142     struct pollfd pfd;
 143     int retval;
 144 
 145     pfd.fd = fd;
 146     pfd.events = POLLOUT;
 147     do {
 148         retval = poll(&pfd, 1, 0);
 149     } while (retval < 0 && errno == EINTR);
 150     if (retval == 1) {
 151         return get_socket_error(fd);
 152     } else if (retval < 0) {
 153         static struct vlog_rate_limit rl = VLOG_RATE_LIMIT_INIT(5, 10);
 154         VLOG_ERR_RL(&rl, "poll: %s", strerror(errno));
 155         return errno;
 156     } else {
 157         return EAGAIN;
 158     }
 159 }

 161 /* Drain all the data currently in the receive queue of a datagram socket (and
 162  * possibly additional data).  There is no way to know how many packets are in
 163  * the receive queue, but we do know that the total number of bytes queued does
 164  * not exceed the receive buffer size, so we pull packets until none are left
 165  * or we've read that many bytes. */
 166 int
 167 drain_rcvbuf(int fd)
 168 {
 169     socklen_t rcvbuf_len;
 170     size_t rcvbuf;
 171 
 172     rcvbuf_len = sizeof rcvbuf;
 173     if (getsockopt(fd, SOL_SOCKET, SO_RCVBUF, &rcvbuf, &rcvbuf_len) < 0) {
 174         static struct vlog_rate_limit rl = VLOG_RATE_LIMIT_INIT(5, 10);
 175         VLOG_ERR_RL(&rl, "getsockopt(SO_RCVBUF) failed: %s", strerror(errno));
 176         return errno;
 177     }
 178     while (rcvbuf > 0) {
 179         /* In Linux, specifying MSG_TRUNC in the flags argument causes the
 180          * datagram length to be returned, even if that is longer than the
 181          * buffer provided.  Thus, we can use a 1-byte buffer to discard the
 182          * incoming datagram and still be able to account how many bytes were
 183          * removed from the receive buffer.
 184          *
 185          * On other Unix-like OSes, MSG_TRUNC has no effect in the flags
 186          * argument. */
 187 #ifdef __linux__
 188 #define BUFFER_SIZE 1
 189 #else
 190 #define BUFFER_SIZE 2048
 191 #endif
 192         char buffer[BUFFER_SIZE];
 193         ssize_t n_bytes = recv(fd, buffer, sizeof buffer,
 194                                MSG_TRUNC | MSG_DONTWAIT);
 195         if (n_bytes <= 0 || n_bytes >= rcvbuf) {
 196             break;
 197         }
 198         rcvbuf -= n_bytes;
 199     }
 200     return 0;
 201 }

 203 /* Reads and discards up to 'n' datagrams from 'fd', stopping as soon as no
 204  * more data can be immediately read.  ('fd' should therefore be in
 205  * non-blocking mode.)*/
 206 void
 207 drain_fd(int fd, size_t n_packets)
 208 {
 209     for (; n_packets > 0; n_packets--) {
 210         /* 'buffer' only needs to be 1 byte long in most circumstances.  This
 211          * size is defensive against the possibility that we someday want to
 212          * use a Linux tap device without TUN_NO_PI, in which case a buffer
 213          * smaller than sizeof(struct tun_pi) will give EINVAL on read. */
 214         char buffer[128];
 215         if (read(fd, buffer, sizeof buffer) <= 0) {
 216             break;
 217         }
 218     }
 219 }

 221 /* Stores in '*un' a sockaddr_un that refers to file 'name'.  Stores in
 222  * '*un_len' the size of the sockaddr_un. */
 223 static void
 224 make_sockaddr_un(const char *name, struct sockaddr_un* un, socklen_t *un_len)
 225 {
 226     un->sun_family = AF_UNIX;
 227     strncpy(un->sun_path, name, sizeof un->sun_path);
 228     un->sun_path[sizeof un->sun_path - 1] = '\0';
 229     *un_len = (offsetof(struct sockaddr_un, sun_path)
 230                 + strlen (un->sun_path) + 1);
 231 }

 233 /* Creates a Unix domain socket in the given 'style' (either SOCK_DGRAM or
 234  * SOCK_STREAM) that is bound to '*bind_path' (if 'bind_path' is non-null) and
 235  * connected to '*connect_path' (if 'connect_path' is non-null).  If 'nonblock'
 236  * is true, the socket is made non-blocking.  If 'passcred' is true, the socket
 237  * is configured to receive SCM_CREDENTIALS control messages.
 238  *
 239  * Returns the socket's fd if successful, otherwise a negative errno value. */
 240 int
 241 make_unix_socket(int style, bool nonblock, bool passcred UNUSED,
 242                  const char *bind_path, const char *connect_path)
 243 {
 244     int error;
 245     int fd;
 246 
 247     fd = socket(PF_UNIX, style, 0);
 248     if (fd < 0) {
 249         return -errno;
 250     }
 251 
 252     /* Set nonblocking mode right away, if we want it.  This prevents blocking
 253      * in connect(), if connect_path != NULL.  (In turn, that's a corner case:
 254      * it will only happen if style is SOCK_STREAM or SOCK_SEQPACKET, and only
 255      * if a backlog of un-accepted connections has built up in the kernel.)  */
 256     if (nonblock) {
 257         int flags = fcntl(fd, F_GETFL, 0);
 258         if (flags == -1) {
 259             goto error;
 260         }
 261         if (fcntl(fd, F_SETFL, flags | O_NONBLOCK) == -1) {
 262             goto error;
 263         }
 264     }
 265 
 266     if (bind_path) {
 267         struct sockaddr_un un;
 268         socklen_t un_len;
 269         make_sockaddr_un(bind_path, &un, &un_len);
 270         if (unlink(un.sun_path) && errno != ENOENT) {
 271             VLOG_WARN("unlinking \"%s\": %s\n", un.sun_path, strerror(errno));
 272         }
 273         fatal_signal_add_file_to_unlink(bind_path);
 274         if (bind(fd, (struct sockaddr*) &un, un_len)
 275             || fchmod(fd, S_IRWXU)) {
 276             goto error;
 277         }
 278     }
 279 
 280     if (connect_path) {
 281         struct sockaddr_un un;
 282         socklen_t un_len;
 283         make_sockaddr_un(connect_path, &un, &un_len);
 284         if (connect(fd, (struct sockaddr*) &un, un_len)
 285             && errno != EINPROGRESS) {
 286             goto error;
 287         }
 288     }
 289 
 290 #ifdef SCM_CREDENTIALS
 291     if (passcred) {
 292         int enable = 1;
 293         if (setsockopt(fd, SOL_SOCKET, SO_PASSCRED, &enable, sizeof(enable))) {
 294             goto error;
 295         }
 296     }
 297 #endif
 298 
 299     return fd;
 300 
 301 error:
 302     if (bind_path) {
 303         fatal_signal_remove_file_to_unlink(bind_path);
 304     }
 305     error = errno;
 306     close(fd);
 307     return -error;
 308 }
