  40 /* Initializes 'hmap' as an empty hash table. */
  41 void
  42 hmap_init(struct hmap *hmap)
  43 {
  44     hmap->buckets = &hmap->one;
  45     hmap->one = NULL;
  46     hmap->mask = 0;
  47     hmap->n = 0;
  48 }
  49 
  50 /* Frees memory reserved by 'hmap'.  It is the client's responsibility to free
  51  * the nodes themselves, if necessary. */
  52 void
  53 hmap_destroy(struct hmap *hmap)
  54 {
  55     if (hmap && hmap->buckets != &hmap->one) {
  56         free(hmap->buckets);
  57     }
  58 }

  60 /* Exchanges hash maps 'a' and 'b'. */
  61 void
  62 hmap_swap(struct hmap *a, struct hmap *b)
  63 {
  64     struct hmap tmp = *a;
  65     *a = *b;
  66     *b = tmp;
  67     if (a->buckets == &b->one) {
  68         a->buckets = &a->one;
  69     }
  70     if (b->buckets == &a->one) {
  71         b->buckets = &b->one;
  72     }
  73 }

  75 static void
  76 resize(struct hmap *hmap, size_t new_mask)
  77 {
  78     struct hmap tmp;
  79     size_t i;
  80 
  81     assert(!(new_mask & (new_mask + 1)));
  82     assert(new_mask != SIZE_MAX);
  83 
  84     hmap_init(&tmp);
  85     if (new_mask) {
  86         tmp.buckets = xmalloc(sizeof *tmp.buckets * (new_mask + 1));
  87         tmp.mask = new_mask;
  88         for (i = 0; i <= tmp.mask; i++) {
  89             tmp.buckets[i] = NULL;
  90         }
  91     }
  92     for (i = 0; i <= hmap->mask; i++) {
  93         struct hmap_node *node, *next;
  94         for (node = hmap->buckets[i]; node; node = next) {
  95             next = node->next;
  96             hmap_insert_fast(&tmp, node, node->hash);
  97         }
  98     }
  99     hmap_swap(hmap, &tmp);
 100     hmap_destroy(&tmp);
 101 }

 103 static size_t
 104 calc_mask(size_t capacity)
 105 {
 106     size_t mask = capacity / 2;
 107     mask |= mask >> 1;
 108     mask |= mask >> 2;
 109     mask |= mask >> 4;
 110     mask |= mask >> 8;
 111     mask |= mask >> 16;
 112 #if SIZE_MAX > UINT32_MAX
 113     mask |= mask >> 32;
 114 #endif
 115 
 116     /* If we need to dynamically allocate buckets we might as well allocate at
 117      * least 4 of them. */
 118     mask |= (mask & 1) << 1;
 119 
 120     return mask;
 121 }

 123 /* Expands 'hmap', if necessary, to optimize the performance of searches. */
 124 void
 125 hmap_expand(struct hmap *hmap)
 126 {
 127     size_t new_mask = calc_mask(hmap->n);
 128     if (new_mask > hmap->mask) {
 129         resize(hmap, new_mask);
 130     }
 131 }
 132 
 133 /* Shrinks 'hmap', if necessary, to optimize the performance of iteration. */
 134 void
 135 hmap_shrink(struct hmap *hmap)
 136 {
 137     size_t new_mask = calc_mask(hmap->n);
 138     if (new_mask < hmap->mask) {
 139         resize(hmap, new_mask);
 140     }
 141 }
 142 
 143 /* Expands 'hmap', if necessary, to optimize the performance of searches when
 144  * it has up to 'n' elements.  (But iteration will be slow in a hash map whose
 145  * allocated capacity is much higher than its current number of nodes.)  */
 146 void
 147 hmap_reserve(struct hmap *hmap, size_t n)
 148 {
 149     size_t new_mask = calc_mask(n);
 150     if (new_mask > hmap->mask) {
 151         resize(hmap, new_mask);
 152     }
 153 }
