  68 #define AV_PUSH_MSG(i, t, s, d) { \
  69     msg_t msg = {0}; \
  70     msg.id = i; \
  71     msg.type = t; \
  72     memmove(msg.data, d, s); \
  73     zmq_send(av_push_sock, &msg, sizeof(msg_t), 0); \
  74 }

  76 #define AV_WRITE_MSG(i, t, s, d) { \
  77     msg_t msg = {0}; \
  78     msg.id = i; \
  79     msg.type = t; \
  80     memmove(msg.data, d, s); \
  81     zmq_send(av_req_sock, &msg, sizeof(msg_t), 0); \
  82     zmq_recv(av_req_sock, &msg, sizeof(msg_t), 0); \
  83     memmove(d, msg.data, s); \

 180 static void *reply_app_events(void *null)
 181 {
 182     while (av_worker_on) {
 183         msg_t msg = {0};
 184 
 185         zmq_recv(av_rep_sock, &msg, sizeof(msg), 0);
 186 
 187         app_event_out_t out = {0};
 188 
 189         out.id = msg.id;
 190         out.type = msg.type;
 191         out.data = (char *)msg.data;
 192 
 193         const app_event_t *in = (const app_event_t *)&out;
 194 
 195         int ret = 0;
 196 
 197         switch (msg.type) {
 198 
 199         // upstream events
 200 
 201         case AV_DP_RECEIVE_PACKET:
 202             out.length = sizeof(pktin_t);
 203             ret = app.handler(in, &out);
 204             break;
 205         case AV_DP_FLOW_EXPIRED:
 206             out.length = sizeof(flow_t);
 207             ret = app.handler(in, &out);
 208             break;

 220             break;
 221         case AV_DP_PORT_DELETED:
 222             out.length = sizeof(port_t);
 223             ret = app.handler(in, &out);
 224             break;
 225 
 226         // downstream events

 230             break;

 401 int destroy_av_workers(ctx_t *ctx)
 402 {
 403     av_worker_on = FALSE;
 404 
 405     zmq_close(av_push_sock);
 406     zmq_ctx_destroy(av_pull_ctx);
 407 
 408     zmq_close(av_pull_sock);
 409     zmq_ctx_destroy(av_pull_ctx);
 410 
 411     zmq_close(av_req_sock);
 412     zmq_ctx_destroy(av_req_ctx);
 413 
 414     zmq_close(av_rep_sock);
 415     zmq_ctx_destroy(av_rep_ctx);
 416 
 417     return 0;
 418 }

 420 /**
 421  * \brief Function to initialize the app event handler
 422  * \param ctx Context (NULL)
 423  */
 424 int app_event_init(ctx_t *ctx)
 425 {
 426     av_worker_on = TRUE;
 427 
 428     // Push (downstream)
 429 
 430     av_push_ctx = zmq_ctx_new();
 431     av_push_sock = zmq_socket(av_push_ctx, ZMQ_PUSH);
 432     if (zmq_connect(av_push_sock, "ipc://../tmp/ext_app_event_handler")) {
 433         PERROR("zmq_connect");
 434         return -1;
 435     }
 436 
 437     // Pull (upstream, intstream)
 438 
 439     sprintf(app_pull_path, "ipc://../tmp/%s_push_pull", TARGET_APP);
 440 
 441     av_pull_ctx = zmq_ctx_new();
 442     av_pull_sock = zmq_socket(av_pull_ctx, ZMQ_PULL);
 443     if (zmq_connect(av_pull_sock, app_pull_path)) {
 444         PERROR("zmq_connect");
 445         return -1;
 446     }
 447 
 448     int i;
 449     for (i=0; i<__NUM_THREADS; i++) {
 450         pthread_t thread;
 451         if (pthread_create(&thread, NULL, &deliver_app_events, NULL) < 0) {
 452             PERROR("pthread_create");
 453             return -1;
 454         }
 455     }
 456 
 457     // Request (intsteam)
 458 
 459     av_req_ctx = zmq_ctx_new();
 460     av_req_sock = zmq_socket(av_req_ctx, ZMQ_REQ);
 461     if (zmq_connect(av_req_sock, "ipc://../tmp/app_req_handler")) {
 462         PERROR("zmq_connect");
 463         return -1;
 464     }
 465 
 466     // Reply (upstream, intstream)
 467 
 468     sprintf(app_rep_path, "ipc://../tmp/%s_req_rep", TARGET_APP);
 469 
 470     av_rep_ctx = zmq_ctx_new();
 471     av_rep_sock = zmq_socket(av_req_ctx, ZMQ_REP);
 472     if (zmq_connect(av_rep_sock, app_rep_path)) {
 473         PERROR("zmq_connect");
 474         return -1;
 475     }
 476 
 477     for (i=0; i<__NUM_THREADS; i++) {
 478         pthread_t thread;
 479         if (pthread_create(&thread, NULL, &reply_app_events, NULL) < 0) {
 480             PERROR("pthread_create");
 481             return -1;
 482         }
 483     }
 484 
 485     return 0;
 486 }
