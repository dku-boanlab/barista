  55 double run_test(int n_fakeswitches, struct fakeswitch * fakeswitches, int mstestlen, int delay)
  56 {
  57     struct timeval now, then, diff;
  58     struct  pollfd  * pollfds;
  59     int i;
  60     double sum = 0;
  61     double passed;
  62     int count;
  63 
  64     int total_wait = mstestlen + delay;
  65     time_t tNow;
  66     struct tm *tmNow;
  67     pollfds = malloc(n_fakeswitches * sizeof(struct pollfd));
  68     assert(pollfds);
  69     gettimeofday(&then,NULL);
  70     while(1)
  71     {
  72         gettimeofday(&now, NULL);
  73         timersub(&now, &then, &diff);
  74         if( (1000* diff.tv_sec  + (float)diff.tv_usec/1000)> total_wait)
  75             break;
  76 
  77         #ifdef USE_EPOLL
  78         for(i = 0; i < MAX_EVENTS; i++) {
  79             events[i].events = EPOLLIN | EPOLLOUT;
  80         }
  81         
  82         int nfds = epoll_wait(epollfd, events, MAX_EVENTS, -1);
  83 
  84 
  85         for(i = 0; i < nfds; i++) {
  86             fakeswitch_handle_io(events[i].data.ptr, &(events[i].events));
  87         }
  88         #else
  89         for(i = 0; i < n_fakeswitches; i++) 
  90             fakeswitch_set_pollfd(&fakeswitches[i], &pollfds[i]);
  91         
  92         poll(pollfds, n_fakeswitches, 1000);
  93 
  94         for(i = 0; i < n_fakeswitches; i++) 
  95             fakeswitch_handle_io(&fakeswitches[i], &pollfds[i]);
  96         #endif
  97     }
  98     tNow = now.tv_sec;
  99     tmNow = localtime(&tNow);
 100     printf("%02d:%02d:%02d.%03d %-3d switches: flows/sec:  ", tmNow->tm_hour, tmNow->tm_min, tmNow->tm_sec, (int)(now.tv_usec/1000), n_fakeswitches);
 101     usleep(100000); // sleep for 100 ms, to let packets queue
 102     for( i = 0 ; i < n_fakeswitches; i++)
 103     {
 104         count = fakeswitch_get_count(&fakeswitches[i]);
 105         printf("%d  ", count);
 106         sum += count;
 107     }
 108     passed = 1000 * diff.tv_sec + (double)diff.tv_usec/1000;   
 109     passed -= delay;        // don't count the time we intentionally delayed
 110     sum /= passed;  // is now per ms
 111     printf(" total = %lf per ms \n", sum);
 112     free(pollfds);
 113     return sum;
 114 }

 118 int timeout_connect(int fd, const char * hostname, int port, int mstimeout) {
 119         int ret = 0;
 120         int flags;
 121         fd_set fds;
 122         struct timeval tv;
 123         struct addrinfo *res=NULL;
 124         struct addrinfo hints;
 125         char sport[BUFLEN];
 126         int err;
 127 
 128         hints.ai_flags          = 0;
 129         hints.ai_family         = AF_INET;
 130         hints.ai_socktype       = SOCK_STREAM;
 131         hints.ai_protocol       = IPPROTO_TCP;
 132         hints.ai_addrlen        = 0;
 133         hints.ai_addr           = NULL;
 134         hints.ai_canonname      = NULL;
 135         hints.ai_next           = NULL;
 136 
 137         snprintf(sport,BUFLEN,"%d",port);
 138 
 139         err = getaddrinfo(hostname,sport,&hints,&res);
 140         if(err|| (res==NULL))
 141         {
 142                 if(res)
 143                         freeaddrinfo(res);
 144                 return -1;
 145         }
 146         
 147 
 148 
 149         // set non blocking
 150         if((flags = fcntl(fd, F_GETFL)) < 0) {
 151                 fprintf(stderr, "timeout_connect: unable to get socket flags\n");
 152                 freeaddrinfo(res);
 153                 return -1;
 154         }
 155         if(fcntl(fd, F_SETFL, flags | O_NONBLOCK) < 0) {
 156                 fprintf(stderr, "timeout_connect: unable to put the socket in non-blocking mode\n");
 157                 freeaddrinfo(res);
 158                 return -1;
 159         }
 160         
 161     #ifdef USE_EPOLL
 162     struct epoll_event ev;
 163     int epollfd = epoll_create(1);
 164     ev.events = EPOLLIN | EPOLLOUT | EPOLLERR;
 165         ev.data.fd = fd;
 166         if(epoll_ctl(epollfd, EPOLL_CTL_ADD, fd, &ev) == -1) {
 167             printf("Cannot use epoll to create connection\n");
 168             return -1;
 169         }
 170     #else
 171     FD_ZERO(&fds);
 172     FD_SET(fd, &fds);
 173     #endif
 174 
 175         if(mstimeout >= 0) 
 176         {
 177             tv.tv_sec = mstimeout / 1000;
 178             tv.tv_usec = (mstimeout % 1000) * 1000;
 179 
 180             errno = 0;
 181 
 182             if(connect(fd, res->ai_addr, res->ai_addrlen) < 0) 
 183             {
 184                     if((errno != EWOULDBLOCK) && (errno != EINPROGRESS))
 185                     {
 186                             fprintf(stderr, "timeout_connect: error connecting: %d\n", errno);
 187                             freeaddrinfo(res);
 188                             return -1;
 189                     }
 190         }
 191         #ifdef USE_EPOLL
 192             int nfds = epoll_wait(epollfd, &ev, 1, mstimeout);
 193         #else
 194         ret = select(fd + 1, NULL, &fds, NULL, &tv);
 195         #endif
 196         }
 197     freeaddrinfo(res);
 198 
 199     #ifdef USE_EPOLL
 200     close(epollfd);
 201         
 202     if(ev.events & EPOLLERR) {
 203             return -1;
 204     } else {
 205             return 0;
 206         }
 207     #else
 208     if(ret != 1) 
 209     {
 210             if(ret == 0)
 211                 return -1;
 212             else
 213                 return ret;
 214     }
 215     return 0;
 216     #endif
 217 }

 220 int make_tcp_connection_from_port(const char * hostname, unsigned short port, unsigned short sport,
 221         int mstimeout, int nodelay)
 222 {
 223     struct sockaddr_in local;
 224     int s;
 225     int err;
 226     int zero = 0;
 227 
 228     s = socket(AF_INET,SOCK_STREAM,0);
 229     if(s<0){
 230         perror("make_tcp_connection: socket");
 231         exit(1);  // bad socket
 232     }
 233     if(nodelay && (setsockopt(s, IPPROTO_TCP, TCP_NODELAY, &zero, sizeof(zero)) < 0))
 234     {
 235         perror("setsockopt");
 236         fprintf(stderr,"make_tcp_connection::Unable to disable Nagle's algorithm\n");
 237         exit(1);
 238     }
 239     local.sin_family=PF_INET;
 240     local.sin_addr.s_addr=INADDR_ANY;
 241     local.sin_port=htons(sport);
 242 
 243     err=bind(s,(struct sockaddr *)&local, sizeof(local));
 244     if(err)
 245     {
 246         perror("make_tcp_connection_from_port::bind");
 247         return -4;
 248     }
 249 
 250     err = timeout_connect(s,hostname,port, mstimeout);
 251 
 252     if(err)
 253     {
 254         perror("make_tcp_connection: connect");
 255         close(s);
 256         return err; // bad connect
 257     }
 258     return s;
 259 }

 268 int count_bits(int n)
 269 {
 270     int count =0;
 271     int i;
 272     for(i=0; i< 32;i++)
 273         if( n & (1<<i))
 274             count ++;
 275     return count;
 276 }

 353                 break;

 379                 exit(1);

 392                 mode == MODE_THROUGHPUT? "'throughput'": "'latency'",
 393                 controller_hostname,
 394                 controller_port,
 395                 should_test_range ? " from 1 to": "",
 396                 n_fakeswitches,
 397                 dpid_offset,
 398                 tests_per_loop,
 399                 mstestlen,
 400                 total_mac_addresses,
 401                 learn_dst_macs ? "learning" : "NOT learning",
 402                 delay,
 403                 warmup,cooldown,
 404                 connect_delay,connect_group_size,
 405                 debug == 1 ? "on" : "off");
 406     /* done parsing args */
 407     fakeswitches = malloc(n_fakeswitches * sizeof(struct fakeswitch));
 408     assert(fakeswitches);
 409 
 410     double *results;
 411     double  min = DBL_MAX;
 412     double  max = 0.0;
 413     double  v;
 414     results = malloc(tests_per_loop * sizeof(double));
 415 
 416     #ifdef USE_EPOLL
 417     struct epoll_event ev;
 418     epollfd = epoll_create(4096);
 419     if(epollfd == -1) {
 420         fprintf(stderr, "Cannot create epollfd.\n");
 421             exit(1);
 422     }
 423     #endif

 425     for( i = 0; i < n_fakeswitches; i++)
 426     {
 427         int sock;
 428         double sum = 0;
 429         if (connect_delay != 0 && i != 0 && (i % connect_group_size == 0)) {
 430             if(debug)
 431                 fprintf(stderr,"Delaying connection by %dms...", connect_delay*1000);
 432             usleep(connect_delay*1000);

 467             continue;

 469             if ( j > 0 )
