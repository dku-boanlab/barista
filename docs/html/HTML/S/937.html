  25 #define mix(a,b,c) \
  26 { \
  27     a -= c;  a ^= rot(c, 4);  c += b; \
  28     b -= a;  b ^= rot(a, 6);  a += c; \

  48 /** 
  49  * \brief Function to generate a hash value
  50  * \param data The pointer of data
  51  * \param length The length of the data
  52  * \return The hash value of the data
  53  */
  54 uint32_t hash_func(const uint32_t *data, size_t length)
  55 {
  56     uint32_t a,b,c;
  57 
  58     length = length - (length % 4);
  59 
  60     a = b = c = 0xdeadbeef + (((uint32_t)length) << 2);
  61 
  62     while (length > 3) {
  63         a += data[0];
  64         b += data[1];
  65         c += data[2];
  66         mix(a,b,c);
  67         length -= 3;
  68         data += 3;
  69     }
  70 
  71     switch(length) {
  72     case 3:
  73         c += data[2];
  74     case 2:
  75         b += data[1];
  76     case 1:
  77         a += data[0];
  78         final(a,b,c);
  79         break;
  80     }
  81 
  82     return c;
  83 }
