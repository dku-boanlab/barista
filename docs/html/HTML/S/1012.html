 355 /** \brief Function to initialize a flow using pktin */
 356 #define FLOW_INIT(x, y) { \
 357     x.dpid = y->dpid; \
 358     x.port = y->port; \
 359     x.xid = y->xid; \
 360     x.buffer_id = y->buffer_id; \
 361     x.idle_timeout = DEFAULT_IDLE_TIMEOUT; \
 362     x.hard_timeout = DEFAULT_HARD_TIMEOUT; \
 363     x.priority = DEFAULT_PRIORITY; \
 364     x.version = y->version; \
 365     x.vlan_id = y->vlan_id; \
 366     x.vlan_pcp = y->vlan_pcp; \
 367     x.proto = y->proto; \
 368     memmove(x.src_mac, y->src_mac, ETH_ALEN); \
 369     memmove(x.dst_mac, y->dst_mac, ETH_ALEN); \
 370     x.src_ip = y->src_ip; \
 371     x.dst_ip = y->dst_ip; \
 372     x.src_port = y->src_port; \
 373     x.dst_port = y->dst_port; \
 374 }
 375 
 376 /** \brief Function to compare two flows */
 377 #define FLOW_COMPARE(x, y) ( \
 378     (x->dpid == y->dpid) && \
 379     (x->port == y->port) && \
 380     (x->wildcards == y->wildcards) && \
 381     (x->vlan_id == y->vlan_id) && \
 382     (x->vlan_pcp == y->vlan_pcp) && \
 383     (x->proto == y->proto) && \
 384     (memcmp(x->src_mac, y->src_mac, ETH_ALEN) == 0) && \
 385     (memcmp(x->dst_mac, y->dst_mac, ETH_ALEN) == 0) && \
 386     (x->src_ip == y->src_ip) && \
 387     (x->dst_ip == y->dst_ip) && \
 388     (x->src_port == y->src_port) && \
 389     (x->dst_port == y->dst_port) \
 390 )

 405     uint16_t hard_timeout; /**< Hard timeout */
 406 
 407     uint16_t flags; /**< Flow flags */
 408     uint16_t priority; /**< Flow priority */
 409 
 410     uint32_t remote; /**< Remote events */
 411     uint8_t version; /**< OpenFlow version */
 412 
 413     // match fields
 414     uint32_t wildcards; /**< Wildcard bitmasks */
 415 
 416     uint16_t vlan_id; /**< VLAN id */
 417     uint8_t vlan_pcp; /**< VLAN priority */
 418 
 419     uint16_t proto; /**< Protocol */
 420     uint8_t ip_tos; /**< IP type of service */
 421 
 422     uint8_t src_mac[ETH_ALEN]; /**< Source MAC address */
 423     uint8_t dst_mac[ETH_ALEN]; /**< Destination MAC address */
 424 
 425     uint32_t src_ip; /**< Source IP address */
 426     uint32_t dst_ip; /**< Destination IP address */

 429         uint16_t src_port; /**< Source port */

 432     };

 434         uint16_t dst_port; /**< Destination port */

 438     // stats

 441 
 442     uint64_t pkt_count; /**< Packet count */
 443     uint64_t byte_count; /**< Byte count */
 444     uint32_t flow_count; /**< Flow count for aggregate stat */
 445 
 446     // actions

 448     action_t action[__MAX_NUM_ACTIONS]; /**< Actions */
