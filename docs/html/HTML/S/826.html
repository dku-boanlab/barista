  88 /* Creates a new netlink socket for the given netlink 'protocol'
  89  * (NETLINK_ROUTE, NETLINK_GENERIC, ...).  Returns 0 and sets '*sockp' to the
  90  * new socket if successful, otherwise returns a positive errno value.
  91  *
  92  * If 'multicast_group' is nonzero, the new socket subscribes to the specified
  93  * netlink multicast group.  (A netlink socket may listen to an arbitrary
  94  * number of multicast groups, but so far we only need one at a time.)
  95  *
  96  * Nonzero 'so_sndbuf' or 'so_rcvbuf' override the kernel default send or
  97  * receive buffer size, respectively.
  98  */
  99 int
 100 nl_sock_create(int protocol, int multicast_group,
 101                size_t so_sndbuf, size_t so_rcvbuf, struct nl_sock **sockp)
 102 {
 103     struct nl_sock *sock;
 104     struct sockaddr_nl local, remote;
 105     int retval = 0;
 106 
 107     if (next_seq == 0) {
 108         /* Pick initial sequence number. */
 109         next_seq = getpid() ^ time_now();
 110     }
 111 
 112     *sockp = NULL;
 113     sock = malloc(sizeof *sock);
 114     if (sock == NULL) {
 115         return ENOMEM;
 116     }
 117 
 118     sock->fd = socket(AF_NETLINK, SOCK_RAW, protocol);
 119     if (sock->fd < 0) {
 120         VLOG_ERR("fcntl: %s", strerror(errno));
 121         goto error;
 122     }
 123 
 124     retval = alloc_pid(&sock->pid);
 125     if (retval) {
 126         goto error;
 127     }
 128 
 129     if (so_sndbuf != 0
 130         && setsockopt(sock->fd, SOL_SOCKET, SO_SNDBUF,
 131                       &so_sndbuf, sizeof so_sndbuf) < 0) {
 132         VLOG_ERR("setsockopt(SO_SNDBUF,%zu): %s", so_sndbuf, strerror(errno));
 133         goto error_free_pid;
 134     }
 135 
 136     if (so_rcvbuf != 0
 137         && setsockopt(sock->fd, SOL_SOCKET, SO_RCVBUF,
 138                       &so_rcvbuf, sizeof so_rcvbuf) < 0) {
 139         VLOG_ERR("setsockopt(SO_RCVBUF,%zu): %s", so_rcvbuf, strerror(errno));
 140         goto error_free_pid;
 141     }
 142 
 143     /* Bind local address as our selected pid. */
 144     memset(&local, 0, sizeof local);
 145     local.nl_family = AF_NETLINK;
 146     local.nl_pid = sock->pid;
 147     if (multicast_group > 0 && multicast_group <= 32) {
 148         /* This method of joining multicast groups is supported by old kernels,
 149          * but it only allows 32 multicast groups per protocol. */
 150         local.nl_groups |= 1ul << (multicast_group - 1);
 151     }
 152     if (bind(sock->fd, (struct sockaddr *) &local, sizeof local) < 0) {
 153         VLOG_ERR("bind(%"PRIu32"): %s", sock->pid, strerror(errno));
 154         goto error_free_pid;
 155     }
 156 
 157     /* Bind remote address as the kernel (pid 0). */
 158     memset(&remote, 0, sizeof remote);
 159     remote.nl_family = AF_NETLINK;
 160     remote.nl_pid = 0;
 161     if (connect(sock->fd, (struct sockaddr *) &remote, sizeof remote) < 0) {
 162         VLOG_ERR("connect(0): %s", strerror(errno));
 163         goto error_free_pid;
 164     }
 165 
 166     /* Older kernel headers failed to define this macro.  We want our programs
 167      * to support the newer kernel features even if compiled with older
 168      * headers, so define it ourselves in such a case. */
 169 #ifndef NETLINK_ADD_MEMBERSHIP
 170 #define NETLINK_ADD_MEMBERSHIP 1
 171 #endif
 172 
 173     /* This method of joining multicast groups is only supported by newish
 174      * kernels, but it allows for an arbitrary number of multicast groups. */
 175     if (multicast_group > 32
 176         && setsockopt(sock->fd, SOL_NETLINK, NETLINK_ADD_MEMBERSHIP,
 177                       &multicast_group, sizeof multicast_group) < 0) {
 178         VLOG_ERR("setsockopt(NETLINK_ADD_MEMBERSHIP,%d): %s",
 179                  multicast_group, strerror(errno));
 180         goto error_free_pid;
 181     }
 182 
 183     *sockp = sock;
 184     return 0;
 185 
 186 error_free_pid:
 187     free_pid(sock->pid);
 188 error:
 189     if (retval == 0) {
 190         retval = errno;
 191         if (retval == 0) {
 192             retval = EINVAL;
 193         }
 194     }
 195     if (sock->fd >= 0) {
 196         close(sock->fd);
 197     }
 198     free(sock);
 199     return retval;
 200 }

 213 /* Tries to send 'msg', which must contain a Netlink message, to the kernel on
 214  * 'sock'.  nlmsg_len in 'msg' will be finalized to match msg->size before the
 215  * message is sent.
 216  *
 217  * Returns 0 if successful, otherwise a positive errno value.  If
 218  * 'wait' is true, then the send will wait until buffer space is ready;
 219  * otherwise, returns EAGAIN if the 'sock' send buffer is full. */
 220 int
 221 nl_sock_send(struct nl_sock *sock, const struct ofpbuf *msg, bool wait) 
 222 {
 223     int error;
 224 
 225     nl_msg_nlmsghdr(msg)->nlmsg_len = msg->size;
 226     do {
 227         int retval;
 228         retval = send(sock->fd, msg->data, msg->size, wait ? 0 : MSG_DONTWAIT);
 229         error = retval < 0 ? errno : 0;
 230     } while (error == EINTR);
 231     log_nlmsg(__func__, error, msg->data, msg->size);
 232     return error;
 233 }

 235 /* Tries to send the 'n_iov' chunks of data in 'iov' to the kernel on 'sock' as
 236  * a single Netlink message.  (The message must be fully formed and not require
 237  * finalization of its nlmsg_len field.)
 238  *
 239  * Returns 0 if successful, otherwise a positive errno value.  If 'wait' is
 240  * true, then the send will wait until buffer space is ready; otherwise,
 241  * returns EAGAIN if the 'sock' send buffer is full. */
 242 int
 243 nl_sock_sendv(struct nl_sock *sock, const struct iovec iov[], size_t n_iov,
 244               bool wait) 
 245 {
 246     struct msghdr msg;
 247     int error;
 248 
 249     memset(&msg, 0, sizeof msg);
 250     msg.msg_iov = (struct iovec *) iov;
 251     msg.msg_iovlen = n_iov;
 252     do {
 253         int retval;
 254         retval = sendmsg(sock->fd, &msg, wait ? 0 : MSG_DONTWAIT);
 255         error = retval < 0 ? errno : 0;
 256     } while (error == EINTR);
 257     if (error != EAGAIN) {
 258         log_nlmsg(__func__, error, iov[0].iov_base, iov[0].iov_len);
 259     }
 260     return error;
 261 }

 263 /* Tries to receive a netlink message from the kernel on 'sock'.  If
 264  * successful, stores the received message into '*bufp' and returns 0.  The
 265  * caller is responsible for destroying the message with ofpbuf_delete().  On
 266  * failure, returns a positive errno value and stores a null pointer into
 267  * '*bufp'.
 268  *
 269  * If 'wait' is true, nl_sock_recv waits for a message to be ready; otherwise,
 270  * returns EAGAIN if the 'sock' receive buffer is empty. */
 271 int
 272 nl_sock_recv(struct nl_sock *sock, struct ofpbuf **bufp, bool wait) 
 273 {
 274     uint8_t tmp;
 275     ssize_t bufsize = 2048;
 276     ssize_t nbytes, nbytes2;
 277     struct ofpbuf *buf;
 278     struct nlmsghdr *nlmsghdr;
 279     struct iovec iov;
 280     struct msghdr msg = {
 281         .msg_name = NULL,
 282         .msg_namelen = 0,
 283         .msg_iov = &iov,
 284         .msg_iovlen = 1,
 285         .msg_control = NULL,
 286         .msg_controllen = 0,
 287         .msg_flags = 0
 288     };
 289 
 290     buf = ofpbuf_new(bufsize);
 291     *bufp = NULL;
 292 
 293 try_again:
 294     /* Attempt to read the message.  We don't know the size of the data
 295      * yet, so we take a guess at 2048.  If we're wrong, we keep trying
 296      * and doubling the buffer size each time. 
 297      */
 298     nlmsghdr = ofpbuf_put_uninit(buf, bufsize);
 299     iov.iov_base = nlmsghdr;
 300     iov.iov_len = bufsize;
 301     do {
 302         nbytes = recvmsg(sock->fd, &msg, (wait ? 0 : MSG_DONTWAIT) | MSG_PEEK); 
 303     } while (nbytes < 0 && errno == EINTR);
 304     if (nbytes < 0) {
 305         ofpbuf_delete(buf);
 306         return errno;
 307     }
 308     if (msg.msg_flags & MSG_TRUNC) {
 309         bufsize *= 2;
 310         ofpbuf_reinit(buf, bufsize);
 311         goto try_again;
 312     }
 313     buf->size = nbytes;
 314 
 315     /* We successfully read the message, so recv again to clear the queue */
 316     iov.iov_base = &tmp;
 317     iov.iov_len = 1;
 318     do {
 319         nbytes2 = recvmsg(sock->fd, &msg, MSG_DONTWAIT);
 320     } while (nbytes2 < 0 && errno == EINTR);
 321     if (nbytes2 < 0) {
 322         if (errno == ENOBUFS) {
 323             /* The kernel is notifying us that a message it tried to send to us
 324              * was dropped.  We have to pass this along to the caller in case
 325              * it wants to retry a request.  So kill the buffer, which we can
 326              * re-read next time. */
 327             ofpbuf_delete(buf);
 328             return ENOBUFS;
 329         } else {
 330             VLOG_ERR_RL(&rl, "failed to remove nlmsg from socket: %s\n",
 331                         strerror(errno));
 332         }
 333     }
 334     if (nbytes < sizeof *nlmsghdr
 335         || nlmsghdr->nlmsg_len < sizeof *nlmsghdr
 336         || nlmsghdr->nlmsg_len > nbytes) {
 337         VLOG_ERR_RL(&rl, "received invalid nlmsg (%zd bytes < %d)",
 338                     bufsize, NLMSG_HDRLEN);
 339         ofpbuf_delete(buf);
 340         return EPROTO;
 341     }
 342     *bufp = buf;
 343     log_nlmsg(__func__, 0, buf->data, buf->size);
 344     return 0;
 345 }

 347 /* Sends 'request' to the kernel via 'sock' and waits for a response.  If
 348  * successful, stores the reply into '*replyp' and returns 0.  The caller is
 349  * responsible for destroying the reply with ofpbuf_delete().  On failure,
 350  * returns a positive errno value and stores a null pointer into '*replyp'.
 351  *
 352  * Bare Netlink is an unreliable transport protocol.  This function layers
 353  * reliable delivery and reply semantics on top of bare Netlink.
 354  * 
 355  * In Netlink, sending a request to the kernel is reliable enough, because the
 356  * kernel will tell us if the message cannot be queued (and we will in that
 357  * case put it on the transmit queue and wait until it can be delivered).
 358  * 
 359  * Receiving the reply is the real problem: if the socket buffer is full when
 360  * the kernel tries to send the reply, the reply will be dropped.  However, the
 361  * kernel sets a flag that a reply has been dropped.  The next call to recv
 362  * then returns ENOBUFS.  We can then re-send the request.
 363  *
 364  * Caveats:
 365  *
 366  *      1. Netlink depends on sequence numbers to match up requests and
 367  *         replies.  The sender of a request supplies a sequence number, and
 368  *         the reply echos back that sequence number.
 369  *
 370  *         This is fine, but (1) some kernel netlink implementations are
 371  *         broken, in that they fail to echo sequence numbers and (2) this
 372  *         function will drop packets with non-matching sequence numbers, so
 373  *         that only a single request can be usefully transacted at a time.
 374  *
 375  *      2. Resending the request causes it to be re-executed, so the request
 376  *         needs to be idempotent.
 377  */
 378 int
 379 nl_sock_transact(struct nl_sock *sock,
 380                  const struct ofpbuf *request, struct ofpbuf **replyp) 
 381 {
 382     uint32_t seq = nl_msg_nlmsghdr(request)->nlmsg_seq;
 383     struct nlmsghdr *nlmsghdr;
 384     struct ofpbuf *reply;
 385     int retval;
 386 
 387     *replyp = NULL;
 388 
 389     /* Ensure that we get a reply even if this message doesn't ordinarily call
 390      * for one. */
 391     nl_msg_nlmsghdr(request)->nlmsg_flags |= NLM_F_ACK;
 392     
 393 send:
 394     retval = nl_sock_send(sock, request, true);
 395     if (retval) {
 396         return retval;
 397     }
 398 
 399 recv:
 400     retval = nl_sock_recv(sock, &reply, true);
 401     if (retval) {
 402         if (retval == ENOBUFS) {
 403             VLOG_DBG_RL(&rl, "receive buffer overflow, resending request");
 404             goto send;
 405         } else {
 406             return retval;
 407         }
 408     }
 409     nlmsghdr = nl_msg_nlmsghdr(reply);
 410     if (seq != nlmsghdr->nlmsg_seq) {
 411         VLOG_DBG_RL(&rl, "ignoring seq %"PRIu32" != expected %"PRIu32,
 412                     nl_msg_nlmsghdr(reply)->nlmsg_seq, seq);
 413         ofpbuf_delete(reply);
 414         goto recv;
 415     }
 416     if (nl_msg_nlmsgerr(reply, &retval)) {
 417         ofpbuf_delete(reply);
 418         if (retval) {
 419             VLOG_DBG_RL(&rl, "received NAK error=%d (%s)",
 420                         retval, strerror(retval));
 421         }
 422         return retval != EAGAIN ? retval : EPROTO;
 423     }
 424 
 425     *replyp = reply;
 426     return 0;
 427 }

 458 /* If 'buffer' is a NLMSG_ERROR message, stores 0 in '*errorp' if it is an ACK
 459  * message, otherwise a positive errno value, and returns true.  If 'buffer' is
 460  * not an NLMSG_ERROR message, returns false.
 461  *
 462  * 'msg' must be at least as large as a nlmsghdr. */
 463 bool
 464 nl_msg_nlmsgerr(const struct ofpbuf *msg, int *errorp) 
 465 {
 466     if (nl_msg_nlmsghdr(msg)->nlmsg_type == NLMSG_ERROR) {
 467         struct nlmsgerr *err = ofpbuf_at(msg, NLMSG_HDRLEN, sizeof *err);
 468         int code = EPROTO;
 469         if (!err) {
 470             VLOG_ERR_RL(&rl, "received invalid nlmsgerr (%zd bytes < %zd)",
 471                         msg->size, NLMSG_HDRLEN + sizeof *err);
 472         } else if (err->error <= 0 && err->error > INT_MIN) {
 473             code = -err->error;
 474         }
 475         if (errorp) {
 476             *errorp = code;
 477         }
 478         return true;
 479     } else {
 480         return false;
 481     }
 482 }

 492 /* Puts a nlmsghdr at the beginning of 'msg', which must be initially empty.
 493  * Uses the given 'type' and 'flags'.  'sock' is used to obtain a PID and
 494  * sequence number for proper routing of replies.  'expected_payload' should be
 495  * an estimate of the number of payload bytes to be supplied; if the size of
 496  * the payload is unknown a value of 0 is acceptable.
 497  *
 498  * 'type' is ordinarily an enumerated value specific to the Netlink protocol
 499  * (e.g. RTM_NEWLINK, for NETLINK_ROUTE protocol).  For Generic Netlink, 'type'
 500  * is the family number obtained via nl_lookup_genl_family().
 501  *
 502  * 'flags' is a bit-mask that indicates what kind of request is being made.  It
 503  * is often NLM_F_REQUEST indicating that a request is being made, commonly
 504  * or'd with NLM_F_ACK to request an acknowledgement.
 505  *
 506  * nl_msg_put_genlmsghdr is more convenient for composing a Generic Netlink
 507  * message. */
 508 void
 509 nl_msg_put_nlmsghdr(struct ofpbuf *msg, struct nl_sock *sock,
 510                     size_t expected_payload, uint32_t type, uint32_t flags) 
 511 {
 512     struct nlmsghdr *nlmsghdr;
 513 
 514     assert(msg->size == 0);
 515 
 516     nl_msg_reserve(msg, NLMSG_HDRLEN + expected_payload);
 517     nlmsghdr = nl_msg_put_uninit(msg, NLMSG_HDRLEN);
 518     nlmsghdr->nlmsg_len = 0;
 519     nlmsghdr->nlmsg_type = type;
 520     nlmsghdr->nlmsg_flags = flags;
 521     nlmsghdr->nlmsg_seq = ++next_seq;
 522     nlmsghdr->nlmsg_pid = sock->pid;
 523 }

 525 /* Puts a nlmsghdr and genlmsghdr at the beginning of 'msg', which must be
 526  * initially empty.  'sock' is used to obtain a PID and sequence number for
 527  * proper routing of replies.  'expected_payload' should be an estimate of the
 528  * number of payload bytes to be supplied; if the size of the payload is
 529  * unknown a value of 0 is acceptable.
 530  *
 531  * 'family' is the family number obtained via nl_lookup_genl_family().
 532  *
 533  * 'flags' is a bit-mask that indicates what kind of request is being made.  It
 534  * is often NLM_F_REQUEST indicating that a request is being made, commonly
 535  * or'd with NLM_F_ACK to request an acknowledgement.
 536  *
 537  * 'cmd' is an enumerated value specific to the Generic Netlink family
 538  * (e.g. CTRL_CMD_NEWFAMILY for the GENL_ID_CTRL family).
 539  *
 540  * 'version' is a version number specific to the family and command (often 1).
 541  *
 542  * nl_msg_put_nlmsghdr should be used to compose Netlink messages that are not
 543  * Generic Netlink messages. */
 544 void
 545 nl_msg_put_genlmsghdr(struct ofpbuf *msg, struct nl_sock *sock,

 547                       uint8_t cmd, uint8_t version)

 555     genlmsghdr->cmd = cmd;
 556     genlmsghdr->version = version;
 557     genlmsghdr->reserved = 0;

 569 /* Appends 'size' bytes of data, plus Netlink padding if needed, to the tail
 570  * end of 'msg', reallocating and copying its data if necessary.  Returns a

 752 /* Default minimum and maximum payload sizes for each type of attribute. */
 753 static const size_t attr_len_range[][2] = {
 754     [0 ... N_NL_ATTR_TYPES - 1] = { 0, SIZE_MAX },
 755     [NL_A_U8] = { 1, 1 },
 756     [NL_A_U16] = { 2, 2 },
 757     [NL_A_U32] = { 4, 4 },
 758     [NL_A_U64] = { 8, 8 },
 759     [NL_A_STRING] = { 1, SIZE_MAX },
 760     [NL_A_FLAG] = { 0, SIZE_MAX },
 761     [NL_A_NESTED] = { NLMSG_HDRLEN, SIZE_MAX },
 762 };

 764 /* Parses the 'msg' starting at the given 'nla_offset' as a sequence of Netlink
 765  * attributes.  'policy[i]', for 0 <= i < n_attrs, specifies how the attribute
 766  * with nla_type == i is parsed; a pointer to attribute i is stored in
 767  * attrs[i].  Returns true if successful, false on failure.
 768  *
 769  * If the Netlink attributes in 'msg' follow a Netlink header and a Generic
 770  * Netlink header, then 'nla_offset' should be NLMSG_HDRLEN + GENL_HDRLEN. */
 771 bool
 772 nl_policy_parse(const struct ofpbuf *msg, size_t nla_offset,
 773                 const struct nl_policy policy[],
 774                 struct nlattr *attrs[], size_t n_attrs)
 775 {
 776     void *p, *tail;
 777     size_t n_required;
 778     size_t i;
 779 
 780     n_required = 0;
 781     for (i = 0; i < n_attrs; i++) {
 782         attrs[i] = NULL;
 783 
 784         assert(policy[i].type < N_NL_ATTR_TYPES);
 785         if (policy[i].type != NL_A_NO_ATTR
 786             && policy[i].type != NL_A_FLAG
 787             && !policy[i].optional) {
 788             n_required++;
 789         }
 790     }
 791 
 792     p = ofpbuf_at(msg, nla_offset, 0);
 793     if (p == NULL) {
 794         VLOG_DBG_RL(&rl, "missing headers in nl_policy_parse");
 795         return false;
 796     }
 797     tail = ofpbuf_tail(msg);
 798 
 799     while (p < tail) {
 800         size_t offset = (char*)p - (char*)msg->data;
 801         struct nlattr *nla = p;
 802         size_t len, aligned_len;
 803         uint16_t type;
 804 
 805         /* Make sure its claimed length is plausible. */
 806         if (nla->nla_len < NLA_HDRLEN) {
 807             VLOG_DBG_RL(&rl, "%zu: attr shorter than NLA_HDRLEN (%"PRIu16")",
 808                         offset, nla->nla_len);
 809             return false;
 810         }
 811         len = nla->nla_len - NLA_HDRLEN;
 812         aligned_len = NLA_ALIGN(len);
 813         if (aligned_len > (char*)tail - (char*)p) {
 814             VLOG_DBG_RL(&rl, "%zu: attr %"PRIu16" aligned data len (%zu) "
 815                         "> bytes left (%tu)",
 816                         offset, nla->nla_type, aligned_len,
 817                         (char*)tail - (char*)p);
 818             return false;
 819         }
 820 
 821         type = nla->nla_type;
 822         if (type < n_attrs && policy[type].type != NL_A_NO_ATTR) {
 823             const struct nl_policy *p = &policy[type];
 824             size_t min_len, max_len;
 825 
 826             /* Validate length and content. */
 827             min_len = p->min_len ? p->min_len : attr_len_range[p->type][0];
 828             max_len = p->max_len ? p->max_len : attr_len_range[p->type][1];
 829             if (len < min_len || len > max_len) {
 830                 VLOG_DBG_RL(&rl, "%zu: attr %"PRIu16" length %zu not in "
 831                             "allowed range %zu...%zu",
 832                             offset, type, len, min_len, max_len);
 833                 return false;
 834             }
 835             if (p->type == NL_A_STRING) {
 836                 if (((char *) nla)[nla->nla_len - 1]) {
 837                     VLOG_DBG_RL(&rl, "%zu: attr %"PRIu16" lacks null at end",
 838                                 offset, type);
 839                     return false;
 840                 }
 841                 if (memchr(nla + 1, '\0', len - 1) != NULL) {
 842                     VLOG_DBG_RL(&rl, "%zu: attr %"PRIu16" has bad length",
 843                                 offset, type);
 844                     return false;
 845                 }
 846             }
 847             if (!p->optional && attrs[type] == NULL) {
 848                 assert(n_required > 0);
 849                 --n_required;
 850             }
 851             attrs[type] = nla;
 852         } else {
 853             /* Skip attribute type that we don't care about. */
 854         }
 855         p = (char*)p + NLA_ALIGN(nla->nla_len);
 856     }
 857     if (n_required) {
 858         VLOG_DBG_RL(&rl, "%zu required attrs missing", n_required);
 859         return false;
 860     }
 861     return true;
 862 }

 870 static int do_lookup_genl_family(const char *name) 
 871 {
 872     struct nl_sock *sock;
 873     struct ofpbuf request, *reply;
 874     struct nlattr *attrs[ARRAY_SIZE(family_policy)];
 875     int retval;
 876 
 877     retval = nl_sock_create(NETLINK_GENERIC, 0, 0, 0, &sock);
 878     if (retval) {
 879         return -retval;
 880     }
 881 
 882     ofpbuf_init(&request, 0);
 883     nl_msg_put_genlmsghdr(&request, sock, 0, GENL_ID_CTRL, NLM_F_REQUEST,
 884                           CTRL_CMD_GETFAMILY, 1);
 885     nl_msg_put_string(&request, CTRL_ATTR_FAMILY_NAME, name);
 886     retval = nl_sock_transact(sock, &request, &reply);
 887     ofpbuf_uninit(&request);
 888     if (retval) {
 889         nl_sock_destroy(sock);
 890         return -retval;
 891     }
 892 
 893     if (!nl_policy_parse(reply, NLMSG_HDRLEN + GENL_HDRLEN,
 894                          family_policy, attrs, ARRAY_SIZE(family_policy))) {
 895         nl_sock_destroy(sock);
 896         ofpbuf_delete(reply);
 897         return -EPROTO;
 898     }
 899 
 900     retval = nl_attr_get_u16(attrs[CTRL_ATTR_FAMILY_ID]);
 901     if (retval == 0) {
 902         retval = -EPROTO;
 903     }
 904     nl_sock_destroy(sock);
 905     ofpbuf_delete(reply);
 906     return retval;
 907 }

 909 /* If '*number' is 0, translates the given Generic Netlink family 'name' to a
 910  * number and stores it in '*number'.  If successful, returns 0 and the caller
 911  * may use '*number' as the family number.  On failure, returns a positive
 912  * errno value and '*number' caches the errno value. */
 913 int
 914 nl_lookup_genl_family(const char *name, int *number) 
 915 {
 916     if (*number == 0) {
 917         *number = do_lookup_genl_family(name);
 918         assert(*number != 0);
 919     }
 920     return *number > 0 ? 0 : -*number;
 921 }

 923 /* Netlink PID.
 924  *
 925  * Every Netlink socket must be bound to a unique 32-bit PID.  By convention,
 926  * programs that have a single Netlink socket use their Unix process ID as PID,
 927  * and programs with multiple Netlink sockets add a unique per-socket
 928  * identifier in the bits above the Unix process ID.
 929  *
 930  * The kernel has Netlink PID 0.
 931  */
 932 
 933 /* Parameters for how many bits in the PID should come from the Unix process ID
 934  * and how many unique per-socket. */
 935 #define SOCKET_BITS 10
 936 #define MAX_SOCKETS (1u << SOCKET_BITS)
 937 
 938 #define PROCESS_BITS (32 - SOCKET_BITS)
 939 #define MAX_PROCESSES (1u << PROCESS_BITS)
 940 #define PROCESS_MASK ((uint32_t) (MAX_PROCESSES - 1))
 941 
 942 /* Bit vector of unused socket identifiers. */
 943 static uint32_t avail_sockets[ROUND_UP(MAX_SOCKETS, 32)];
 944 
 945 /* Allocates and returns a new Netlink PID. */
 946 static int
 947 alloc_pid(uint32_t *pid)
 948 {
 949     int i;
 950 
 951     for (i = 0; i < MAX_SOCKETS; i++) {
 952         if ((avail_sockets[i / 32] & (1u << (i % 32))) == 0) {
 953             avail_sockets[i / 32] |= 1u << (i % 32);
 954             *pid = (getpid() & PROCESS_MASK) | (i << PROCESS_BITS);
 955             return 0;
 956         }
 957     }
 958     VLOG_ERR("netlink pid space exhausted");
 959     return ENOBUFS;
 960 }
 961 
 962 /* Makes the specified 'pid' available for reuse. */
 963 static void
 964 free_pid(uint32_t pid)
 965 {
 966     int sock = pid >> PROCESS_BITS;
 967     assert(avail_sockets[sock / 32] & (1u << (sock % 32)));
 968     avail_sockets[sock / 32] &= ~(1u << (sock % 32));
 969 }

 971 static void
 972 nlmsghdr_to_string(const struct nlmsghdr *h, struct ds *ds)
 973 {
 974     struct nlmsg_flag {
 975         unsigned int bits;
 976         const char *name;
 977     };
 978     static const struct nlmsg_flag flags[] = { 
 979         { NLM_F_REQUEST, "REQUEST" },
 980         { NLM_F_MULTI, "MULTI" },
 981         { NLM_F_ACK, "ACK" },
 982         { NLM_F_ECHO, "ECHO" },
 983         { NLM_F_DUMP, "DUMP" },
 984         { NLM_F_ROOT, "ROOT" },
 985         { NLM_F_MATCH, "MATCH" },
 986         { NLM_F_ATOMIC, "ATOMIC" },
 987     };
 988     const struct nlmsg_flag *flag;
 989     uint16_t flags_left;
 990 
 991     ds_put_format(ds, "nl(len:%"PRIu32", type=%"PRIu16,
 992                   h->nlmsg_len, h->nlmsg_type);
 993     if (h->nlmsg_type == NLMSG_NOOP) {
 994         ds_put_cstr(ds, "(no-op)");
 995     } else if (h->nlmsg_type == NLMSG_ERROR) {
 996         ds_put_cstr(ds, "(error)");
 997     } else if (h->nlmsg_type == NLMSG_DONE) {
 998         ds_put_cstr(ds, "(done)");
 999     } else if (h->nlmsg_type == NLMSG_OVERRUN) {
1000         ds_put_cstr(ds, "(overrun)");
1001     } else if (h->nlmsg_type < NLMSG_MIN_TYPE) {
1002         ds_put_cstr(ds, "(reserved)");
1003     } else {
1004         ds_put_cstr(ds, "(family-defined)");
1005     }
1006     ds_put_format(ds, ", flags=%"PRIx16, h->nlmsg_flags);
1007     flags_left = h->nlmsg_flags;
1008     for (flag = flags; flag < &flags[ARRAY_SIZE(flags)]; flag++) {
1009         if ((flags_left & flag->bits) == flag->bits) {
1010             ds_put_format(ds, "[%s]", flag->name);
1011             flags_left &= ~flag->bits;
1012         }
1013     }
1014     if (flags_left) {
1015         ds_put_format(ds, "[OTHER:%"PRIx16"]", flags_left);
1016     }
1017     ds_put_format(ds, ", seq=%"PRIx32", pid=%"PRIu32"(%d:%d))",
1018                   h->nlmsg_seq, h->nlmsg_pid,
1019                   (int) (h->nlmsg_pid & PROCESS_MASK),
1020                   (int) (h->nlmsg_pid >> PROCESS_BITS));
1021 }

1023 static char *
1024 nlmsg_to_string(const struct ofpbuf *buffer)
1025 {
1026     struct ds ds = DS_EMPTY_INITIALIZER;
1027     const struct nlmsghdr *h = ofpbuf_at(buffer, 0, NLMSG_HDRLEN);
1028     if (h) {
1029         nlmsghdr_to_string(h, &ds);
1030         if (h->nlmsg_type == NLMSG_ERROR) {
1031             const struct nlmsgerr *e;
1032             e = ofpbuf_at(buffer, NLMSG_HDRLEN,
1033                           NLMSG_ALIGN(sizeof(struct nlmsgerr)));
1034             if (e) {
1035                 ds_put_format(&ds, " error(%d", e->error);
1036                 if (e->error < 0) {
1037                     ds_put_format(&ds, "(%s)", strerror(-e->error));
1038                 }
1039                 ds_put_cstr(&ds, ", in-reply-to(");
1040                 nlmsghdr_to_string(&e->msg, &ds);
1041                 ds_put_cstr(&ds, "))");
1042             } else {
1043                 ds_put_cstr(&ds, " error(truncated)");
1044             }
1045         } else if (h->nlmsg_type == NLMSG_DONE) {
1046             int *error = ofpbuf_at(buffer, NLMSG_HDRLEN, sizeof *error);
1047             if (error) {
1048                 ds_put_format(&ds, " done(%d", *error);
1049                 if (*error < 0) {
1050                     ds_put_format(&ds, "(%s)", strerror(-*error));
1051                 }
1052                 ds_put_cstr(&ds, ")");
1053             } else {
1054                 ds_put_cstr(&ds, " done(truncated)");
1055             }
1056         }
1057     } else {
1058         ds_put_cstr(&ds, "nl(truncated)");
1059     }
1060     return ds.string;
1061 }
