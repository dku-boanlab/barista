  29 /**
  30  * \brief Function to verify an OpenFlow message
  31  * \param id Component ID
  32  * \param msg OpenFlow message
  33  */
  34 static int ofp_msg_verification(uint32_t id, const raw_msg_t *msg)
  35 {
  36     struct ofp_header *ofph = (struct ofp_header *)msg->data;
  37 
  38     if (ofph->version != OFP_VERSION)
  39         return 0;
  40 
  41     if ((ofph->type < OFPT_HELLO) || (ofph->type > OFPT_QUEUE_GET_CONFIG_REPLY)) {
  42         LOG_ERROR(OFP_VERI_ID, " %u -> wrong type", id);
  43         return -1;
  44     }
  45     if (ntohs(ofph->length) < sizeof(struct ofp_header)) {
  46         LOG_ERROR(OFP_VERI_ID, " %u -> wrong length", id);
  47         return -1;
  48     }
  49     // skip xid verification
  50 
  51     switch (ofph->type) {
  52     case OFPT_HELLO:
  53         if (ntohs(ofph->length) != sizeof(struct ofp_header)) {
  54             LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_HELLO length error", id);
  55             return -1;
  56         }
  57         break;
  58     case OFPT_ERROR:
  59         {
  60             struct ofp_error_msg *data = (struct ofp_error_msg *)msg->data;
  61             if ((ntohs(data->type) < OFPET_HELLO_FAILED) || (ntohs(data->type) > (OFPET_QUEUE_OP_FAILED << 1))) {
  62                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_ERROR | type error", id);
  63                 return -1;
  64             }
  65             if (ntohs(data->type) == OFPET_HELLO_FAILED) {
  66                 if ((ntohs(data->code) < OFPHFC_INCOMPATIBLE) || (ntohs(data->code) > (OFPHFC_EPERM << 1))) {
  67                     LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_ERROR | OFPET_HELLO_FAILED error", id);
  68                     return -1;
  69                 }
  70             }
  71             else if (ntohs(data->type) == OFPET_BAD_REQUEST) {
  72                 int dataSize = ntohs(ofph->length) - sizeof(struct ofp_error_msg);
  73                 if (dataSize < 64) {
  74                     LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_ERROR | OFPET_BAD_REQUEST data length error", id);
  75                     return -1;
  76                 }
  77                 if ((ntohs(data->code) < OFPBRC_BAD_VERSION) || (ntohs(data->code) > (OFPBRC_BUFFER_UNKNOWN << 1))) {
  78                     LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_ERROR | OFPET_BAD_REQUEST code error", id);
  79                     return -1;
  80                 }
  81             }
  82             else if (ntohs(data->type) == OFPET_BAD_ACTION) {
  83                 int dataSize = ntohs(ofph->length) - sizeof(struct ofp_error_msg);
  84 
  85                 if (dataSize < 64) {
  86                     LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_ERROR | OFPET_BAD_ACTION data length error", id);
  87                     return -1;
  88                 }
  89                 if ((ntohs(data->code) < OFPBAC_BAD_TYPE) || (ntohs(data->code) > (OFPBAC_BAD_QUEUE << 1))) {
  90                     LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_ERROR | OFPBRC_BAD_ACTION code error", id);
  91                     return -1;
  92                 }
  93             }
  94             else if (ntohs(data->type) == OFPET_FLOW_MOD_FAILED) {
  95                 int dataSize = ntohs(ofph->length) - sizeof(struct ofp_error_msg);
  96 
  97                 if (dataSize < 64) {
  98                     LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_ERROR | OFPET_FLOW_MOD_FAILED data length error", id);
  99                     return -1;
 100                 }
 101                 if ((ntohs(data->code) < OFPFMFC_ALL_TABLES_FULL) || (ntohs(data->code) > (OFPFMFC_UNSUPPORTED << 1))) {
 102                     LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_ERROR | OFPET_FLOW_MOD_FAILED code error", id);
 103                     return -1;
 104                 }
 105             }
 106             else if (ntohs(data->type) == OFPET_PORT_MOD_FAILED) {
 107                 int dataSize = ntohs(ofph->length) - sizeof(struct ofp_error_msg);
 108 
 109                 if (dataSize < 64) {
 110                     LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_ERROR | OFPET_PORT_MOD_FAILED data length error", id);
 111                     return -1;
 112                 }
 113                 if ((ntohs(data->code) < OFPPMFC_BAD_PORT) || (ntohs(data->code) > (OFPPMFC_BAD_HW_ADDR << 1))) {
 114                     LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_ERROR | OFPET_PORT_MOD_FAILED code error", id);
 115                     return -1;
 116                 }
 117             }
 118             else if (ntohs(data->type) == OFPET_QUEUE_OP_FAILED) {
 119                 int dataSize = ntohs(ofph->length) - sizeof(struct ofp_error_msg);
 120 
 121                 if (dataSize < 64) {
 122                     LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_ERROR | OFPET_QUEUE_OP_FAILED data length error", id);
 123                     return -1;
 124                 }
 125                 if ((ntohs(data->code) < OFPQOFC_BAD_PORT) || (ntohs(data->code) > (OFPQOFC_EPERM << 1))) {
 126                     LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_ERROR | OFPET_QUEUE_OP_FAILED code error", id);
 127                     return -1;
 128                 }
 129             }
 130         }
 131         break;
 132     case OFPT_ECHO_REQUEST:
 133         // do nothing
 134         break;
 135     case OFPT_ECHO_REPLY:
 136         // do nothing
 137         break;
 138     case OFPT_VENDOR:
 139         // do nothing
 140         break;
 141     case OFPT_FEATURES_REQUEST:
 142         if (ntohs(ofph->length) != sizeof(struct ofp_header)) {
 143             LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_FEATURES_REQUEST length error", id);
 144             return -1;
 145         }
 146         break;
 147     case OFPT_FEATURES_REPLY:
 148         {
 149             struct ofp_switch_features *data = (struct ofp_switch_features *)msg->data;
 150 
 151             if ((ntohl(data->capabilities) < OFPC_FLOW_STATS) || (ntohl(data->capabilities) > (OFPC_ARP_MATCH_IP << 1))) {
 152                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_FEATURES_REPLY | capabilities error", id);
 153                 return -1;
 154             }
 155             if ((ntohl(data->actions) < OFPAT_OUTPUT) || (ntohl(data->actions) > OFPAT_VENDOR)) {
 156                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_FEATURES_REPLY | action error", id);
 157                 return -1;
 158             }
 159         }
 160         break;
 161     case OFPT_GET_CONFIG_REQUEST:
 162         if (ntohs(ofph->length) != sizeof(struct ofp_header)) {
 163             LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_GET_CONFIG_REQUEST length error", id);
 164             return -1;
 165         }
 166         break;
 167     case OFPT_GET_CONFIG_REPLY:
 168         {
 169             struct ofp_switch_config *data = (struct ofp_switch_config *)msg->data;
 170             if ((ntohs(data->flags)) < OFPC_FRAG_NORMAL || (ntohs(data->flags)) > (OFPC_FRAG_MASK << 1)) {
 171                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_GET_CONFIG_REPLY flag error", id);
 172                 return -1;
 173             }
 174         }
 175         break;
 176     case OFPT_SET_CONFIG:
 177         {
 178             struct ofp_switch_config *data = (struct ofp_switch_config *)msg->data;
 179             if ((ntohs(data->flags)) < OFPC_FRAG_NORMAL || (ntohs(data->flags)) > (OFPC_FRAG_MASK << 1)) {
 180                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_SET_CONFIG flag error", id);
 181                 return -1;
 182             }
 183         }
 184         break;
 185     case OFPT_PACKET_IN:
 186         {
 187             struct ofp_packet_in *data = (struct ofp_packet_in *)msg->data;
 188             if ((data->reason < OFPR_NO_MATCH) || (data->reason > (OFPR_ACTION << 1))) {
 189                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_PACKET_IN reason error", id);
 190                 return -1;
 191             }
 192         }
 193         break;
 194     case OFPT_FLOW_REMOVED:
 195         {
 196             struct ofp_flow_removed *data = (struct ofp_flow_removed *)msg->data;
 197             if ((data->reason < OFPRR_IDLE_TIMEOUT) || (data->reason > (OFPRR_DELETE << 1))) {
 198                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_FLOW_REMOVED reason error", id);
 199                 return -1;
 200             }
 201         }
 202         break;
 203     case OFPT_PORT_STATUS:
 204         {
 205             struct ofp_port_status *data = (struct ofp_port_status *)msg->data;
 206             if ((data->reason < OFPPR_ADD) || (data->reason > (OFPPR_MODIFY << 1))) {
 207                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_PORT_STATUS reason error", id);
 208                 return -1;
 209             }
 210         }
 211         break;
 212     case OFPT_PACKET_OUT:
 213         {
 214             struct ofp_packet_out *data = (struct ofp_packet_out *)msg->data;
 215             if (ntohl(data->buffer_id) == (uint32_t)-1) {
 216                 int size = ntohs(data->header.length) - (sizeof(struct ofp_packet_out) + ntohs(data->actions_len));
 217                 if (size < 0) {
 218                     LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_PACKET_OUT length error", id);
 219                     return -1;
 220                 }
 221             }
 222         }
 223         break;
 224     case OFPT_FLOW_MOD:
 225         {
 226             struct ofp_flow_mod *data = (struct ofp_flow_mod *)msg->data;
 227             if (data->cookie == 0xffffffffffffffff) {
 228                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_FLOW_MOD | cookie reserved data error", id);
 229                 return -1;
 230             }
 231             if ((ntohs(data->command) < OFPFC_ADD) || (ntohs(data->command) > (OFPFC_DELETE_STRICT << 1))) {
 232                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_FLOW_MOD | command error", id);
 233                 return -1;
 234             }
 235             if ((ntohs(data->flags) < OFPFF_SEND_FLOW_REM) || (ntohs(data->flags) > (OFPFF_EMERG << 1))) {
 236                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_FLOW_MOD | flags error", id);
 237                 return -1;
 238             }
 239         }
 240         break;
 241     case OFPT_PORT_MOD:
 242         {
 243             struct ofp_port_mod *data = (struct ofp_port_mod *)msg->data;
 244             if ((ntohl(data->config) < OFPPC_PORT_DOWN) || (ntohl(data->config) > (OFPPC_NO_PACKET_IN << 1))) {
 245                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_PORT_MOD | port config error", id);
 246                 return -1;
 247             }
 248         }
 249         break;
 250     case OFPT_STATS_REQUEST:
 251         {
 252             struct ofp_stats_request *data = (struct ofp_stats_request *)msg->data;
 253             if (((ntohs(data->type) < OFPST_DESC) || (ntohs(data->type) > (OFPST_QUEUE << 1))) && (ntohs(data->type) != OFPST_VENDOR)) {
 254                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_STATS_REQUEST type error", id);
 255                 return -1;
 256             }
 257         }
 258         break;
 259     case OFPT_STATS_REPLY:
 260         {
 261             struct ofp_stats_reply *data = (struct ofp_stats_reply *)msg->data;
 262             if (((ntohs(data->type) < OFPST_DESC) || (ntohs(data->type) > (OFPST_QUEUE << 1))) && (ntohs(data->type) != OFPST_VENDOR)) {
 263                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_STATS_REPLY type error", id);
 264                 return -1;
 265             }
 266         }
 267         break;
 268     case OFPT_BARRIER_REQUEST:
 269         if (ntohs(ofph->length) != sizeof(struct ofp_header)) {
 270             LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_BARRIER_REQUEST length error", id);
 271             return -1;
 272         }
 273         break;
 274     case OFPT_BARRIER_REPLY:
 275         if (ntohs(ofph->length) != sizeof(struct ofp_header)) {
 276             LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_BARRIER_REQUEST length error", id);
 277             return -1;
 278         }
 279         break;
 280     case OFPT_QUEUE_GET_CONFIG_REQUEST:
 281         {
 282             struct ofp_queue_get_config_request *data = (struct ofp_queue_get_config_request *)msg->data;
 283             if (ntohs(data->port) > OFPP_MAX) {
 284                 LOG_ERROR(OFP_VERI_ID, " %u -> OFPT_QUEUE_GET_CONFIG_REQUEST port error", id);
 285                 return -1;
 286             }
 287         }
 288         break;
 289     case OFPT_QUEUE_GET_CONFIG_REPLY:
 290         // do nothing
 291         break;
 292 
 293     default:
 294         break;
 295     }
 296 
 297     return 0;
 298 }

 340 /**
 341  * \brief The handler function
 342  * \param ev Read-only event
 343  * \param ev_out Read-write event (if this component has the write permission)
 344  */
 345 int ofp_veri_handler(const event_t *ev, event_out_t *ev_out)
 346 {
 347     switch (ev->type) {
 348     case EV_OFP_MSG_IN:
 349         PRINT_EV("EV_OFP_MSG_IN\n");
 350         {
 351             const raw_msg_t *msg = ev->msg;
 352             if (ofp_msg_verification(ev->id, msg)) {
 353                 return -1;
 354             }
 355         }
 356         break;
 357     case EV_OFP_MSG_OUT:
 358         PRINT_EV("EV_OFP_MSG_OUT\n");
 359         {
 360             const raw_msg_t *msg = ev->msg;
 361             if (ofp_msg_verification(ev->id, msg)) {
 362                 return -1;
 363             }
 364         }
 365         break;
 366     default:
 367         break;
 368     }
 369 
 370     return 0;
 371 }
