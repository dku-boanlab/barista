  41 #define BITMAP_ULONG_BITS (sizeof(unsigned long) * CHAR_BIT)
  42 
  43 static inline unsigned long *
  44 bitmap_unit__(const unsigned long *bitmap, size_t offset)
  45 {
  46     return (unsigned long *) &bitmap[offset / BITMAP_ULONG_BITS];
  47 }
  48 
  49 static inline unsigned long
  50 bitmap_bit__(size_t offset)
  51 {
  52     return 1UL << (offset % BITMAP_ULONG_BITS);
  53 }
  54 
  55 static inline unsigned long *
  56 bitmap_allocate(size_t n_bits)
  57 {
  58     return xcalloc(1, ROUND_UP(n_bits, BITMAP_ULONG_BITS));
  59 }
  60 
  61 static inline void
  62 bitmap_free(unsigned long *bitmap)
  63 {
  64     free(bitmap);
  65 }
  66 
  67 static inline bool
  68 bitmap_is_set(const unsigned long *bitmap, size_t offset)
  69 {
  70     return (*bitmap_unit__(bitmap, offset) & bitmap_bit__(offset)) != 0;
  71 }
  72 
  73 static inline void
  74 bitmap_set1(unsigned long *bitmap, size_t offset)
  75 {
  76     *bitmap_unit__(bitmap, offset) |= bitmap_bit__(offset);
  77 }
  78 
  79 static inline void
  80 bitmap_set0(unsigned long *bitmap, size_t offset)
  81 {
  82     *bitmap_unit__(bitmap, offset) &= ~bitmap_bit__(offset);
  83 }
  84 
  85 static inline void
  86 bitmap_set(unsigned long *bitmap, size_t offset, bool value)
  87 {
  88     if (value) {
  89         bitmap_set1(bitmap, offset);
  90     } else {
  91         bitmap_set0(bitmap, offset);
  92     }
  93 }
