 110 #define assert_magic(obj, mnum) \
 111         g_assert((obj)); \
 112         if ((obj)->magic != (mnum)) { \
 113                 g_print("\nMagic num is 0x%08x, but should be 0x%08x", \
 114                                 (obj)->magic, (mnum)); \
 115                 g_assert((obj)->magic == (mnum)); \
 116         }

 121 #define STTYPE_ACCESSOR(ret,type,attr,magicnum) \
 122         ret \
 123         CONCAT(CONCAT(CONCAT(sttype_,type),_),attr) (stnode_t *node) \
 124 {\
 125         CONCAT(type,_t) *value; \
 126         value = stnode_data(node);\
 127         assert_magic(value, magicnum); \
 128         return value->attr; \
 129 }
