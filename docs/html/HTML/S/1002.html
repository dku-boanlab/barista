  37 /** \brief The pointer of the first entry in a network socket queue */

  63 /**
  64  * \brief Function to push a socket into a network socket queue
  65  * \param v Network socket
  66  */
  67 static void push_back(int v)
  68 {
  69     if (size == 0) {
  70         head = tail = FIRST;
  71     } else {
  72         tail = (LAST - tail > 0) ? tail + 1 : FIRST;
  73     }
  74 
  75     *tail = v;
  76 
  77     size++;
  78 }
  79 
  80 /**
  81  * \brief Function to pop a socket from a network socket queue
  82  * \return Network socket
  83  */
  84 static int pop_front(void)
  85 {
  86     int ret = -1;
  87 
  88     if (size > 0) {
  89         ret = *head;
  90         head = (LAST - head > 0) ? head + 1 : FIRST;
  91         size--;
  92     }
  93 
  94     return ret;
  95 }

 146 /**
 147  * \brief Function to handle incoming messages from network sockets
 148  * \param sock Network socket
 149  * \param rx_buf Input buffer
 150  * \param bytes The size of the input buffer
 151  */
 152 static int msg_proc(int sock, uint8_t *rx_buf, int bytes)
 153 {
 154     context_t *c = &context[sock];
 155 
 156     uint8_t *temp = c->temp;
 157     int need = c->need;
 158     int done = c->done;
 159 
 160     int buf_ptr = 0;
 161     while (bytes > 0) {
 162         if (0 < done && done < 4) {
 163             if (done + bytes < 4) {
 164                 memmove(temp + done, rx_buf + buf_ptr, bytes);
 165 
 166                 done += bytes;
 167                 bytes = 0;
 168 
 169                 break;
 170             } else {
 171                 memmove(temp + done, rx_buf + buf_ptr, (4 - done));
 172 
 173                 bytes -= (4 - done);
 174                 buf_ptr += (4 - done);
 175 
 176                 need = ntohs(((struct ofp_header *)temp)->length) - 4;
 177                 done = 4;
 178             }
 179         }
 180 
 181         if (need == 0) {
 182             struct ofp_header *ofph = (struct ofp_header *)(rx_buf + buf_ptr);
 183 
 184             if (bytes < 4) {
 185                 memmove(temp, rx_buf + buf_ptr, bytes);
 186 
 187                 need = 0;
 188                 done = bytes;
 189                 bytes = 0;
 190             } else {
 191                 uint16_t len = ntohs(ofph->length);
 192 
 193                 if (bytes < len) {
 194                     memmove(temp, rx_buf + buf_ptr, bytes);
 195 
 196                     need = len - bytes;
 197                     done = bytes;
 198                     bytes = 0;
 199                 } else {
 200                     raw_msg_t msg = {0};
 201 
 202                     msg.fd = sock;
 203                     msg.length = ntohs(ofph->length);
 204 
 205                     if (msg.length == 0)
 206                         return -1;
 207 
 208                     msg.data = rx_buf + buf_ptr;
 209 
 210                     ev_ofp_msg_in(CONN_ID, &msg);
 211 
 212                     bytes -= len;
 213                     buf_ptr += len;
 214                     need = 0;
 215                     done = 0;
 216                 }
 217             }
 218         } else {
 219             struct ofp_header *ofph = (struct ofp_header *)temp;
 220 
 221             if (need > bytes) {
 222                 memmove(temp + done, rx_buf + buf_ptr, bytes);
 223 
 224                 need -= bytes;
 225                 done += bytes;
 226                 bytes = 0;
 227             } else {
 228                 raw_msg_t msg = {0};
 229 
 230                 msg.fd = sock;
 231                 msg.length = ntohs(ofph->length);
 232 
 233                 if (msg.length == 0)
 234                     return -1;
 235 
 236                 memmove(temp + done, rx_buf + buf_ptr, need);
 237                 msg.data = temp;
 238 
 239                 ev_ofp_msg_in(CONN_ID, &msg);
 240 
 241                 bytes -= need;
 242                 buf_ptr += need;
 243                 need = 0;
 244                 done = 0;
 245             }
 246         }
 247     }
 248 
 249     c->need = need;
 250     c->done = done;
 251 
 252     return bytes;
 253 }

 344 /**
 345  * \brief Function to receive raw messages from network sockets
 346  * \return NULL
 347  */
 348 static void *do_tasks(void *null)
 349 {
 350     int wsock, bytes, done = 0;
 351     uint8_t rx_buf[BUFFER_SIZE] = {0};
 352 
 353     pthread_mutex_lock(&queue_mutex);
 354 
 355     while (conn_on) {
 356         pthread_cond_wait(&queue_cond, &queue_mutex);
 357 
 358 FETCH_ONE_FD:
 359         if (conn_on == FALSE) { // clean up
 360             pthread_mutex_unlock(&queue_mutex);
 361             break;
 362         }
 363 
 364         wsock = pop_front();
 365         if (wsock < 0) continue;
 366         pthread_mutex_unlock(&queue_mutex);
 367 
 368         done = 0;
 369         bytes = 1;
 370         while (bytes > 0) {
 371             bytes = read(wsock, rx_buf, BUFFER_SIZE);
 372             if (bytes < 0) {
 373                 if (errno != EAGAIN) {
 374                     done = 1;
 375                 }
 376                 break;
 377             } else if (bytes == 0) {
 378                 done = 1;
 379                 break;
 380             }
 381 
 382             bytes = recv_cb(wsock, rx_buf, bytes);
 383             if (bytes == -1) {
 384                 done = 1;
 385                 break;
 386             }
 387         }
 388 
 389         if (done) {
 390             // closed connection
 391             switch_t sw = {0};
 392             sw.fd = wsock;
 393             ev_sw_expired_conn(CONN_ID, &sw);
 394 
 395             clean_context(wsock);
 396             close(wsock);
 397         } else {
 398             if (relink_epoll(epoll, wsock, EPOLLIN | EPOLLET | EPOLLONESHOT) < 0) {
 399                 // closed connection
 400                 switch_t sw = {0};
 401                 sw.fd = wsock;
 402                 ev_sw_expired_conn(CONN_ID, &sw);
 403 
 404                 clean_context(wsock);
 405                 close(wsock);
 406             }
 407         }
 408 
 409         pthread_mutex_lock(&queue_mutex);
 410 
 411         if (get_qsize() > 0) {
 412             goto FETCH_ONE_FD;
 413         }
 414     }
 415 
 416     return NULL;
 417 }

 444 /**
 445  * \brief Function to set the non-blocking mode to a socket
 446  * \param fd Socket
 447  */
 448 static int nonblocking_mode(const int fd)
 449 {
 450     int flags;
 451 
 452     if ((flags = fcntl(fd, F_GETFL, 0)) < 0) {
 453         if (errno != EAGAIN) {
 454             PERROR("fcntl");
 455             return -1;
 456         }
 457     }
 458 
 459     flags |= O_NONBLOCK;
 460     if (fcntl(fd, F_SETFL, flags) < 0) {
 461         if (errno != EAGAIN) {
 462             PERROR("fcntl");
 463             return -1;
 464         }
 465     }
 466 
 467     return 0;
 468 }

 470 /**
 471  * \brief Function to initialize a socket
 472  * \param port Port number
 473  */
 474 static int init_socket(uint16_t port)
 475 {
 476     struct sockaddr_in server;
 477     int option = 1;
 478 
 479     if ((sc = socket(AF_INET, SOCK_STREAM, 0)) < 0) {
 480         PERROR("socket");
 481         return -1;
 482     }
 483 
 484     if (setsockopt(sc, SOL_SOCKET, SO_REUSEADDR, &option, sizeof(option)) < 0) {
 485         PERROR("setsockopt");
 486     }
 487 
 488     if (nonblocking_mode(sc) < 0) {
 489         close(sc);
 490         return -1;
 491     }
 492 
 493     server.sin_family = AF_INET;
 494     server.sin_addr.s_addr = INADDR_ANY;
 495     server.sin_port = htons(port);
 496 
 497     if (bind(sc, (struct sockaddr *)&server, sizeof(server)) < 0) {
 498         PERROR("bind");
 499         close(sc);
 500         return -1;
 501     }
 502 
 503     if (listen(sc, SOMAXCONN) < 0) {
 504         PERROR("listen");
 505         close(sc);
 506         return -1;
 507     }
 508 
 509     if (link_epoll(epoll, sc, EPOLLIN | EPOLLET) < 0) {
 510         close(sc);
 511         return -1;
 512     }
 513 
 514     return sc;
 515 }

 517 /**
 518  * \brief Function to receive connections from network sockets
 519  * \return NULL
 520  */
 521 static void *socket_listen(void *arg)
 522 {
 523     int csock;
 524     int nums = 0;
 525     struct sockaddr_in client;
 526     socklen_t len = sizeof(struct sockaddr);
 527 
 528     while (conn_on) {
 529         nums = epoll_wait(epoll, events, MAXEVENTS, 100);
 530 
 531         if (conn_on == FALSE) break;
 532 
 533         int i;
 534         for (i=0; i<nums; i++) {
 535             if (events[i].data.fd == sc) {
 536                 while (1) {
 537                     if ((csock = accept(sc, (struct sockaddr *)&client, &len)) < 0) {
 538                         if (errno != EAGAIN && errno != EWOULDBLOCK) {
 539                             PERROR("accept");
 540                         }
 541                         break;
 542                     }
 543 
 544                     // clean up the previous segments
 545                     clean_context(csock);
 546 
 547                     // new connection
 548                     switch_t sw = {0};
 549                     sw.fd = csock;
 550                     ev_sw_new_conn(CONN_ID, &sw);
 551 
 552                     if (nonblocking_mode(csock) < 0) {
 553                         // closed connection
 554                         switch_t sw = {0};
 555                         sw.fd = csock;
 556                         ev_sw_expired_conn(CONN_ID, &sw);
 557 
 558                         close(csock);
 559                         break;
 560                     }
 561 
 562                     if (link_epoll(epoll, csock, EPOLLIN | EPOLLET | EPOLLONESHOT) < 0) {
 563                         // closed connection
 564                         switch_t sw = {0};
 565                         sw.fd = csock;
 566                         ev_sw_expired_conn(CONN_ID, &sw);
 567 
 568                         close(csock);
 569                         break;
 570                     }
 571                 }
 572             } else {
 573                 pthread_mutex_lock(&queue_mutex);
 574                 push_back(events[i].data.fd);
 575                 pthread_mutex_unlock(&queue_mutex);
 576                 pthread_cond_signal(&queue_cond);
 577             }
 578         }
 579 
 580         if (nums < 0 && errno != EINTR)
 581             break;
 582     }
 583 
 584     if (nums < 0)
 585         PERROR("epoll_wait");
 586 
 587     close(epoll);
 588     close(sc);
 589 
 590     LOG_INFO(CONN_ID, "Closed a socket");
 591 
 592     return NULL;
 593 }

 625 /**
 626  * \brief The cleanup function
 627  * \param activated The activation flag of this component
 628  */
 629 int conn_cleanup(int *activated)
 630 {
 631     LOG_INFO(CONN_ID, "Clean up - Packet I/O engine");
 632 
 633     deactivate();
 634 
 635     conn_on = FALSE;
 636 
 637     int i;
 638     for (i=0; i<__NUM_THREADS; i++) {
 639         pthread_mutex_lock(&queue_mutex);
 640         push_back(0);
 641         pthread_mutex_unlock(&queue_mutex);
 642         pthread_cond_signal(&queue_cond);
 643     }
 644 
 645     pthread_cond_destroy(&queue_cond);
 646     pthread_mutex_destroy(&queue_mutex);
 647 
 648     FREE(context);
 649 
 650     signal(SIGPIPE, SIG_DFL);
 651 
 652     return 0;
 653 }

 665 /**
 666  * \brief The handler function
 667  * \param ev Read-only event
 668  * \param ev_out Read-write event (if this component has the write permission)
 669  */
 670 int conn_handler(const event_t *ev, event_out_t *ev_out)
 671 {
 672     switch (ev->type) {
 673     case EV_OFP_MSG_OUT:
 674         PRINT_EV("EV_OFP_MSG_OUT\n");
 675         {
 676             const raw_msg_t *msg = ev->msg;
 677             int fd = msg->fd;
 678 
 679             int remain = msg->length;
 680             int bytes = 0;
 681             int done = 0;
 682 
 683             while (remain > 0) {
 684                 bytes = write(fd, msg->data + done, remain);
 685                 if (bytes < 0 && errno != EAGAIN) {
 686                     break;
 687                 } else if (bytes < 0) {
 688                     continue;
 689                 }
 690 
 691                 remain -= bytes;
 692                 done += bytes;
 693             }
 694         }
 695         break;
 696     default:
 697         break;
 698     }
 699 
 700     return 0;
 701 }
