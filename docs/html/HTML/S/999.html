 128 /**
 129  * \brief Function to summarize the resource history for the last n seconds
 130  * \param cli The CLI pointer
 131  * \param seconds Time period to query
 132  */
 133 static int resource_stat_summary(cli_t *cli, char *seconds)
 134 {
 135     int sec = atoi(seconds);
 136     if (sec > RESOURCE_MGMT_HISTORY) sec = RESOURCE_MGMT_HISTORY;
 137     else if (rs_history_ptr < RESOURCE_MGMT_HISTORY && rs_history_ptr < sec) sec = rs_history_ptr;
 138 
 139     int ptr = rs_history_ptr - sec;
 140     if (ptr < 0) ptr = RESOURCE_MGMT_HISTORY + ptr;
 141 
 142     resource_t rs = {0.0};
 143 
 144     int i;
 145     for (i=0; i<sec; i++) {
 146         rs.cpu += rs_history[ptr % RESOURCE_MGMT_HISTORY].cpu;
 147         rs.mem += rs_history[ptr % RESOURCE_MGMT_HISTORY].mem;
 148 
 149         ptr++;
 150     }
 151 
 152     rs.cpu /= sec;
 153     rs.mem /= sec;
 154 
 155     cli_print(cli, "<Resource Statistics for %d seconds>", sec);
 156     cli_print(cli, "  CPU: %.2f %%, MEM: %.2f %%", rs.cpu, rs.mem);
 157 
 158     return 0;
 159 }
