  34 /**********************************************************************
  35  * hook for the test module to get access to a raw file descriptor bound
  36  *      to the data channel's device
  37  **/
  38 
  39 int oflops_get_channel_raw_fd(struct oflops_context * ctx, oflops_channel_name ch)
  40 {
  41         struct ifreq ifr;
  42         struct sockaddr_ll saddrll;
  43         struct channel_info * ch_info;
  44         if(ch >= ctx->n_channels)
  45                 return -1;      // no such channel
  46         ch_info = &ctx->channels[ch];
  47         if(ch_info->raw_sock != -1)     // already allocated?
  48                 return ch_info->raw_sock;
  49         // else, setup raw socket
  50         ch_info->raw_sock = socket(AF_PACKET,SOCK_RAW, htons(ETH_P_ALL));
  51         if( ch_info->raw_sock == -1)
  52                 perror_and_exit("raw socket(AF_PACKET,SOCK_RAW,htons(ETH_P_ALL))",1);
  53         // bind to a specific port
  54         strncpy(ifr.ifr_name,ch_info->dev,IFNAMSIZ);
  55         if( ioctl( ch_info->raw_sock, SIOCGIFINDEX, &ifr)  == -1 )
  56                 perror_and_exit("ioctl()",1);
  57         memset(&saddrll, 0, sizeof(saddrll));
  58         saddrll.sll_family = AF_PACKET;
  59         saddrll.sll_protocol = ETH_P_ALL;
  60         saddrll.sll_ifindex = ifr.ifr_ifindex;
  61         if ( bind(ch_info->raw_sock, (struct sockaddr *) &saddrll, sizeof(struct sockaddr_ll)) == -1 )
  62                 perror_and_exit("bind()",1);
  63         return ch_info->raw_sock;
  64 }

  86                 perror_and_exit("ioctl() bind to dev",1);

 178 int oflops_snmp_get(struct oflops_context * ctx, oid query[], size_t len)
 179 {
 180         struct snmp_channel* ch = ctx->snmp_channel_info;
 181         struct snmp_session* sess;
 182 
 183         //Open session for async request
 184         if(!(sess = snmp_open(&(ch->session))))
 185         {
 186                 snmp_perror("snmp_open");
 187                 return 1;
 188         }
 189 
 190         //Build and send packet
 191         if (ch->req != NULL)
 192                 snmp_free_pdu(ch->req);
 193         ch->req = snmp_pdu_create(SNMP_MSG_GET);
 194         snmp_add_null_var(ch->req, query, len);
 195         if (!snmp_send(sess, ch->req))
 196                 snmp_perror("snmp_send");
 197 
 198         return 0;
 199 }
