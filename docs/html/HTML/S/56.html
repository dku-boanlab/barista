  45 int parse_args(oflops_context * ctx, int argc, char * argv[])
  46 {
  47   int c;
  48   int snmp_arg = 0;
  49   int options_index;
  50   char* snmp_client;
  51   char* snmp_community;
  52   char * short_options = make_short_from_long(oflops_options);
  53 
  54   while(1)
  55     {
  56       c = getopt_long(argc, argv, short_options, oflops_options, &options_index);
  57       if( c == -1 )
  58         break;  // done args parsing
  59       switch(c)
  60         {
  61         case 'c':
  62           assert(OFLOPS_CONTROL == 0);
  63           assert(ctx->n_channels > 0);
  64           if(!optarg)
  65             usage(argv[optind], "requires argument");
  66           channel_info_init(&ctx->channels[OFLOPS_CONTROL],optarg);
  67           fprintf(stderr,"Setting control channel to %s\n", optarg);
  68           break;
  69         case 's':
  70           snmp_arg++;
  71           assert(snmp_arg < 2);
  72           if(!optarg)
  73             usage(argv[optind], "requires argument");
  74           snmp_client = strtok(optarg, SNMP_DELIMITER);
  75           snmp_community = strtok(NULL, SNMP_DELIMITER);
  76           snmp_channel_init(ctx->snmp_channel_info, snmp_client, snmp_community);
  77           fprintf(stderr,"Adding SNMP channel on %s with community string %s.\n", 
  78                   snmp_client, snmp_community);
  79           break;
  80         case 'd':
  81           if(ctx->n_channels >= ctx->max_channels)      // resize array if needed
  82             {
  83               ctx->max_channels *= 2;
  84               ctx->channels = realloc_and_check(ctx->channels, ctx->max_channels * sizeof(channel_info));
  85             }
  86           if(!optarg)
  87             usage(argv[optind], "requires argument");
  88           channel_info_init(&ctx->channels[ctx->n_channels++], optarg);
  89           if(ctx->channels[ctx->n_channels-1].of_port == -1)
  90             ctx->channels[ctx->n_channels-1].of_port = ctx->n_channels-1;
  91           fprintf(stderr,"Adding a data channel on %s\n", optarg);
  92           break;
  93         case 'p':
  94           ctx->listen_port = atoi(optarg);
  95           fprintf(stderr,"Setting Control listen port to %d\n", ctx->listen_port);
  96           break;
  97         case 'o':
  98           free(ctx->log);
  99           ctx->log = malloc(strlen(optarg) + 1);
 100           strcpy(ctx->log, optarg);
 101           fprintf(stderr,"Setting log file to %s\n", optarg);
 102           break;
 103         case 'g':
 104           ctx->trafficGen = strtol(optarg, NULL, 10);
 105           if((ctx->trafficGen < 1) || (ctx->trafficGen > 2)) {
 106             fprintf(stderr,"traffic generator %d is invalid\n", ctx->trafficGen);
 107             ctx->trafficGen = 2;
 108           } else {
 109             fprintf(stderr,"Using traffic generator %d\n", ctx->trafficGen);
 110           }
 111           break;
 112         case 'i':
 113           printf("configuration file: %s\n", optarg);
 114           load_config_file(ctx, optarg);
 115           break;
 116         case 't':
 117           ctx->dump_controller = 1;
 118           break;
 119         default:
 120           usage("unknown option", argv[optind]);
 121         }
 122     }
 123 
 124   // skip ahead to any other args
 125   argc-=optind;
 126   argv+=optind;
 127   if(argc > 0)
 128     parse_test_module(ctx, argc, argv);
 129   return 0;
 130 }

 133  * static char * make_short_from_long(struct option long_options[]);
 134  **/
 135 static char * make_short_from_long(struct option long_options[])
 136 {
 137   static char buf[BUFLEN];
 138   int buf_index=0;
 139   int opt_index=0;
 140 
 141   bzero(buf,BUFLEN);
 142   while(oflops_options[opt_index].name != NULL)
 143     {
 144       buf[buf_index++] = oflops_options[opt_index].val;
 145       if(oflops_options[opt_index].has_arg)
 146         buf[buf_index++] = ':';
 147       opt_index++;
 148     }
 149   return buf;
 150 }

 152 /***************************************************************
 153  * void usage(char * s1, char *s2);
 154  *      print usage information and exit
 155  **/
 156 void usage(char * s1, char *s2)
 157 {
 158   struct option * o;
 159   int i = 0;
 160   if(s1)
 161     fprintf(stderr, "%s",s1);
 162   if(s2)
 163     fprintf(stderr, " %s",s2);
 164   if (s1|| s2)
 165     fprintf(stderr, "\n");
 166   fprintf( stderr, "Usage:\noflops [options]\n");
 167   o = &oflops_options[i];
 168   do {
 169     fprintf(stderr, "\t-%c|--%s\t%s\n",
 170             o->val,
 171             o->name,
 172             option_args[o->has_arg]
 173             );
 174 
 175     i++;
 176     o = &oflops_options[i];
 177   } while(o->name);
 178 
 179   fprintf(stderr, "\n\nExample invocation:\n"
 180           "oflops -o oflops.log -c eth0 -s 10.2.3.4:public -d eth1 -d eth2 -p 6633 liboflops_debug.so test_args\n");
 181 
 182   exit(1);
 183 }

 186  *      parse a test module from argc/argv and try loading it

 188 static void parse_test_module(oflops_context * ctx, int argc, char * argv[])
 189 {
 190   char buf[BUFLEN];
 191   int count=0;
 192   int i;
 193 
 194   if(argc==0)
 195     usage("need to specify a test_module to load\n",NULL);
 196   // turn all of the args into a single string
 197   for(i=1;((count < BUFLEN) && (i<argc)); i++)
 198     count += snprintf((buf+count),BUFLEN-count-1, " %s", argv[i]);
 199   if(load_test_module(ctx,argv[0],buf))
 200     fprintf(stderr, "Failed to load test_module %s\n", argv[0]);
 201 }
