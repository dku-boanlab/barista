  44 static int
  45 new_queue(struct sw_port * port, struct sw_queue * queue,
  46           uint32_t queue_id, uint16_t class_id,
  47           struct ofp_queue_prop_min_rate * mr)
  48 {
  49     memset(queue, '\0', sizeof *queue);
  50     queue->port = port;
  51     queue->queue_id = queue_id;
  52     /* class_id is the internal mapping to class. It is the offset
  53      * in the array of queues for each port. Note that class_id is
  54      * local to port, so we don't have any conflict.
  55      * tc uses 16-bit class_id, so we cannot use the queue_id
  56      * field */
  57     queue->class_id = class_id;
  58     queue->property = ntohs(mr->prop_header.property);
  59     queue->min_rate = ntohs(mr->rate);
  60 
  61     list_push_back(&port->queue_list, &queue->node);
  62 
  63     return 0;
  64 }

  66 static int
  67 port_add_queue(struct sw_port *p, uint32_t queue_id,
  68                struct ofp_queue_prop_min_rate * mr)
  69 {
  70     int queue_no;
  71     for (queue_no = 1; queue_no < p->num_queues; queue_no++) {
  72         struct sw_queue *q = &p->queues[queue_no];
  73         if (!q->port) {
  74             return new_queue(p,q,queue_id,queue_no,mr);
  75         }
  76     }
  77     return EXFULL;
  78 }

  80 static int
  81 port_delete_queue(struct sw_port *p UNUSED, struct sw_queue *q)
  82 {
  83     list_remove(&q->node);
  84     memset(q,'\0', sizeof *q);
  85     return 0;
  86 }

  88 static void
  89 recv_of_exp_queue_delete(struct datapath *dp,
  90                          const struct sender *sender,
  91                          const void *oh)
  92 {
  93     struct sw_port *p;
  94     struct sw_queue *q;
  95     struct openflow_queue_command_header * ofq_delete;
  96     struct ofp_packet_queue *opq;
  97 
  98     uint16_t port_no;
  99     uint32_t queue_id;
 100 
 101     ofq_delete = (struct openflow_queue_command_header *)oh;
 102     opq = (struct ofp_packet_queue *)ofq_delete->body;
 103     port_no = ntohs(ofq_delete->port);
 104     queue_id = ntohl(opq->queue_id);
 105 
 106     p = dp_lookup_port(dp,port_no);
 107     if (p->netdev) {
 108         q = dp_lookup_queue(p,queue_id);
 109         if (q) {
 110             netdev_delete_class(p->netdev,q->class_id);
 111             port_delete_queue(p,q);
 112         }
 113         else {
 114             dp_send_error_msg(dp, sender, OFPET_QUEUE_OP_FAILED,
 115                               OFPQOFC_BAD_PORT, oh,
 116                               ntohs(ofq_delete->header.header.length));
 117         }
 118     }
 119     else {
 120         dp_send_error_msg(dp, sender, OFPET_QUEUE_OP_FAILED,
 121                           OFPQOFC_BAD_PORT, oh,
 122                           ntohs(ofq_delete->header.header.length));
 123     }
 124 }

 126 /** Modifies/adds a queue. It first searches if a queue with
 127  * id exists for this port. If yes it modifies it, otherwise adds
 128  * a new configuration.
 129  *
 130  * @param dp the related datapath
 131  * @param sender request source
 132  * @param oh the openflow message for queue mod.
 133  */
 134 static void
 135 recv_of_exp_queue_modify(struct datapath *dp,
 136                          const struct sender *sender UNUSED,
 137                          const void *oh)
 138 {
 139     struct sw_port *p;
 140     struct sw_queue *q;
 141     struct openflow_queue_command_header * ofq_modify;
 142     struct ofp_packet_queue *opq;
 143     struct ofp_queue_prop_min_rate *mr;
 144 
 145     int error = 0;
 146     uint16_t port_no;
 147     uint32_t queue_id;
 148 
 149 
 150     ofq_modify = (struct openflow_queue_command_header *)oh;
 151     opq = (struct ofp_packet_queue *)ofq_modify->body;
 152     mr = (struct ofp_queue_prop_min_rate*)opq->properties;
 153 
 154     /* Currently, we only accept queues with a single, min-rate property */
 155     if ((ntohs(opq->len) != 24) ||
 156         ntohs(mr->prop_header.property) != OFPQT_MIN_RATE) {
 157         VLOG_ERR("Unknown queue configuration");
 158         dp_send_error_msg(dp, sender, OFPET_QUEUE_OP_FAILED,
 159                           OFQ_ERR_DISCIPLINE, oh,
 160                           ntohs(ofq_modify->header.header.length));
 161         return;
 162     }
 163 
 164 
 165 
 166     port_no = ntohs(ofq_modify->port);
 167     queue_id = ntohl(opq->queue_id);
 168 
 169     p = dp_lookup_port(dp, port_no);
 170     if (PORT_IN_USE(p)) {
 171         q = dp_lookup_queue(p, queue_id);
 172         if (q) {
 173             /* queue exists - modify it */
 174             error = netdev_change_class(p->netdev,q->class_id, ntohs(mr->rate));
 175             if (error) {
 176                 VLOG_ERR("Failed to update queue %d", queue_id);
 177                 dp_send_error_msg(dp, sender, OFPET_QUEUE_OP_FAILED,
 178                                   OFPQOFC_EPERM, oh,
 179                                   ntohs(ofq_modify->header.header.length));
 180             }
 181             else {
 182                 q->property = ntohs(mr->prop_header.property);
 183                 q->min_rate = ntohs(mr->rate);
 184             }
 185         }
 186         else {
 187             /* create new queue */
 188             error = port_add_queue(p,queue_id, mr);
 189             if (error == EXFULL) {
 190                 dp_send_error_msg(dp, sender, OFPET_QUEUE_OP_FAILED,
 191                                   OFPQOFC_EPERM, oh,
 192                                   ntohs(ofq_modify->header.header.length));
 193                 return;
 194             }
 195             q = dp_lookup_queue(p, queue_id);
 196             error = netdev_setup_class(p->netdev,q->class_id, ntohs(mr->rate));
 197             if (error) {
 198                 VLOG_ERR("Failed to configure queue %d", queue_id);
 199                 dp_send_error_msg(dp, sender, OFPET_QUEUE_OP_FAILED,
 200                                   OFPQOFC_BAD_QUEUE, oh,
 201                                   ntohs(ofq_modify->header.header.length));
 202             }
 203         }
 204     }
 205     else {
 206         dp_send_error_msg(dp, sender, OFPET_QUEUE_OP_FAILED, OFPQOFC_BAD_PORT,
 207                           oh, ntohs(ofq_modify->header.header.length));
 208         VLOG_ERR("Failed to create/modify queue - port %d doesn't exist",
 209                  port_no);
 210     }
 211     if (!error) {
 212         if (IS_HW_PORT(p)) {
 213 #if defined(OF_HW_PLAT) && !defined(USE_NETDEV)
 214             error = dp->hw_drv->port_queue_config(dp->hw_drv, port_no,
 215                                                   queue_id, ntohs(mr->rate));
 216             if (error < 0) {
 217                 VLOG_ERR("Failed to update HW port %d queue %d",
 218                          port_no, queue_id);
 219             }
 220 #endif
 221         }
 222     }
 223 }
 224 /**
 225  * Parses a set dp_desc message and uses it to set
 226  *  the dp_desc string in dp
 227  */
 228 static void
 229 recv_of_set_dp_desc(struct datapath *dp,
 230                          const struct sender *sender UNUSED,
 231                          const struct ofp_extension_header * exth)
 232 {
 233     struct openflow_ext_set_dp_desc * set_dp_desc = (struct openflow_ext_set_dp_desc * )
 234         exth;
 235     strncpy(dp->dp_desc, set_dp_desc->dp_desc, DESC_STR_LEN);
 236     dp->dp_desc[DESC_STR_LEN-1] = 0;        // force null for safety
 237 }

 239 /**
 240  * Receives an experimental message and pass it
 241  * to the appropriate handler
 242  */
 243 int of_ext_recv_msg(struct datapath *dp, const struct sender *sender,
 244         const void *oh)
 245 {
 246     const struct ofp_extension_header  *ofexth = oh;
 247 
 248     switch (ntohl(ofexth->subtype)) {
 249     case OFP_EXT_QUEUE_MODIFY: {
 250         recv_of_exp_queue_modify(dp,sender,oh);
 251         return 0;
 252     }
 253     case OFP_EXT_QUEUE_DELETE: {
 254         recv_of_exp_queue_delete(dp,sender,oh);
 255         return 0;
 256     }
 257     case OFP_EXT_SET_DESC:
 258         recv_of_set_dp_desc(dp,sender,ofexth);
 259         return 0;
 260     default:
 261         VLOG_ERR("Received unknown command of type %d",
 262                  ntohl(ofexth->subtype));
 263         return -EINVAL;
 264     }
 265 
 266     return -EINVAL;
 267 }
