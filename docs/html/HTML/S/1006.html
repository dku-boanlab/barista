  66 /**
  67  * \brief The main function
  68  * \param activated The activation flag of this component
  69  * \param argc The number of arguments
  70  * \param argv Arguments
  71  */
  72 int stat_mgmt_main(int *activated, int argc, char **argv)
  73 {
  74     LOG_INFO(STAT_MGMT_ID, "Init - Statistics management");
  75 
  76     switch_list = (uint64_t *)MALLOC(sizeof(uint64_t) * __DEFAULT_TABLE_SIZE);
  77     if (switch_list == NULL) {
  78         PERROR("malloc");
  79         return -1;
  80     }
  81 
  82     memset(switch_list, 0, sizeof(uint64_t) * __DEFAULT_TABLE_SIZE);
  83 
  84     pthread_rwlock_init(&stat_lock, NULL);
  85 
  86     LOG_INFO(STAT_MGMT_ID, "Statistics request period: %d sec", STAT_MGMT_REQUEST_TIME);
  87 
  88     activate();
  89 
  90     while (*activated) {
  91         pthread_rwlock_rdlock(&stat_lock);
  92 
  93         int i;
  94         for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
  95             if (switch_list[i]) {
  96                 // aggregate stats
  97                 aggregate_stats_request(switch_list[i]);
  98 
  99                 // port stats
 100                 port_stats_request(switch_list[i]);
 101             }
 102         }
 103 
 104         pthread_rwlock_unlock(&stat_lock);
 105 
 106         waitsec(STAT_MGMT_REQUEST_TIME, 0);
 107     }
 108 
 109     return 0;
 110 }

 139 /**
 140  * \brief The handler function
 141  * \param ev Read-only event
 142  * \param ev_out Read-write event (if this component has the write permission)
 143  */
 144 int stat_mgmt_handler(const event_t *ev, event_out_t *ev_out)
 145 {
 146     switch (ev->type) {
 147     case EV_SW_CONNECTED:
 148         PRINT_EV("EV_SW_CONNECTED\n");
 149         {
 150             if (ev->sw->remote == TRUE)
 151                 break;
 152 
 153             pthread_rwlock_wrlock(&stat_lock);
 154 
 155             int i;
 156             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 157                 if (switch_list[i] == 0) {
 158                     switch_list[i] = ev->sw->dpid;
 159                     break;
 160                 }
 161             }
 162 
 163             pthread_rwlock_unlock(&stat_lock);
 164         }
 165         break;
 166     case EV_SW_DISCONNECTED:
 167         PRINT_EV("EV_SW_DISCONNECTED\n");
 168         {
 169             if (ev->sw->remote == TRUE)
 170                 break;
 171 
 172             pthread_rwlock_wrlock(&stat_lock);
 173 
 174             int i;
 175             for (i=0; i<__DEFAULT_TABLE_SIZE; i++) {
 176                 if (switch_list[i] == ev->sw->dpid) {
 177                     switch_list[i] = 0;
 178                     break;
 179                 }
 180             }
 181 
 182             pthread_rwlock_unlock(&stat_lock);
 183         }
 184         break;
 185     default:
 186         break;
 187     }
 188 
 189     return 0;
 190 }
