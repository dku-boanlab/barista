 109 start(struct oflops_context * ctx) {  
 110   struct flow *fl = (struct flow*)xmalloc(sizeof(struct flow));
 111   fl_probe = (struct flow*)xmalloc(sizeof(struct flow));
 112   void *b; //somewhere to store message data
 113   int res, len;
 114   struct timeval now;
 115   char msg[1024];
 116 
 117   //init h
 118   TAILQ_INIT(&head); 
 119 
 120   //Initialize pap-based  tcp flow reassembler for the communication 
 121   //channel
 122   msg_init();  
 123   snprintf(msg, 1024,  "Intializing module %s", name());
 124 
 125   //log when I start module
 126   gettimeofday(&now, NULL);
 127   oflops_log(now, GENERIC_MSG, msg);
 128   oflops_log(now,GENERIC_MSG , cli_param);
 129 
 130   //start openflow session with switch
 131   make_ofp_hello(&b);
 132   res = oflops_send_of_mesgs(ctx, b, sizeof(struct ofp_hello));
 133   free(b);  
 134   
 135   //send a message to clean up flow tables. 
 136   printf("cleaning up flow table...\n");
 137   res = make_ofp_flow_del(&b);
 138   res = oflops_send_of_mesg(ctx, b);  
 139   free(b);
 140   
 141   /**
 142    * Send flow records to start routing packets.
 143    */
 144   printf("Sending measurement probe flow...\n");
 145   bzero(fl, sizeof(struct flow));
 146   printf("table value:%d\n", table);
 147   if(table == 0)
 148     fl->mask = 0; //if table is 0 the we generate an exact match */
 149   else 
 150     fl->mask = OFPFW_DL_DST | OFPFW_DL_SRC | (32 << OFPFW_NW_SRC_SHIFT) | 
 151       (8 << OFPFW_NW_DST_SHIFT) | OFPFW_DL_VLAN | OFPFW_TP_DST | OFPFW_NW_PROTO | 
 152       OFPFW_TP_SRC | OFPFW_DL_VLAN_PCP | OFPFW_NW_TOS;
 153   fl->in_port = htons(ctx->channels[OFLOPS_DATA2].of_port);
 154   fl->dl_type = htons(ETHERTYPE_IP);          
 155   fl->dl_src[0] = 00; 
 156   fl->dl_src[1] = 0x1e; 
 157   fl->dl_src[2] = 0x68; 
 158   fl->dl_src[3] = 0x9a; 
 159   fl->dl_src[4] = 0xc5; 
 160   fl->dl_src[5] = 0x75;

 162   fl->dl_dst[1] = 0x15; 
 163   fl->dl_dst[2] = 0x17; 
 164   fl->dl_dst[3] = 0x7b; 
 165   fl->dl_dst[4] = 0x92; 
 166   fl->dl_dst[5] = 0x0a; 
 167   fl->dl_vlan = 0xffff;
 168   fl->nw_proto = IPPROTO_UDP;
 169   fl->nw_src =  inet_addr("10.1.1.1");
 170   fl->nw_dst =  inet_addr("10.1.1.2");

 172   fl->tp_dst = htons(8080);

 174   res = oflops_send_of_mesg(ctx, b);
 175   free(b);
 176 
 177   //store locally the probe to manipulate it later during the modification phase
 178   memcpy(fl_probe, fl, sizeof(struct flow));
 179 
 180   /**
 181    * Shceduling events
 182    */
 183   //send the flow modyfication command in 30 seconds. 
 184   gettimeofday(&now, NULL);
 185   add_time(&now, 20, 0);
 186   oflops_schedule_timer_event(ctx,&now, SND_ACT);
 187 
 188   //get port and cpu status from switch 
 189   gettimeofday(&now, NULL);
 190   add_time(&now, 1, 0);
 191   oflops_schedule_timer_event(ctx,&now, SNMPGET);
 192 
 193   //end process 
 194   gettimeofday(&now, NULL);
 195   add_time(&now, 240, 0);
 196   oflops_schedule_timer_event(ctx,&now, BYESTR);
 197   return 0;

 200 int destroy(struct oflops_context *ctx) {
 201   FILE *out = fopen(logfile, "w");
 202   struct entry *np;
 203   long long t[3], t_sq[3], mean, std;
 204   int count[3], min_id[3], max_id[3];
 205   int ch, delay;
 206   float loss;
 207   char msg[1024];
 208   struct timeval now;
 209   gettimeofday(&now, NULL);
 210 
 211   //init storage
 212   for(ch = 0; ch < ctx->n_channels-1; ch++) {
 213     count[ch] = 0;
 214     min_id[ch] =  INT_MAX;
 215     max_id[ch] =  INT_MIN;
 216     t[ch] = 0;
 217     t_sq[ch] = 0;    
 218   }
 219 
 220   for (np = head.tqh_first; np != NULL; np = np->entries.tqe_next) {
 221     if(fprintf(out, "%lu;%lu.%06lu;%lu.%06lu;%d\n", 
 222                (long unsigned int)np->id,  
 223                (long unsigned int)np->snd.tv_sec, 
 224                (long unsigned int)np->snd.tv_usec,
 225                (long unsigned int)np->rcv.tv_sec, 
 226                (long unsigned int)np->rcv.tv_usec,  np->ch) < 0)  
 227       perror_and_exit("fprintf fail", 1); 
 228     ch = np->ch - 1;
 229     count[ch]++; 
 230     min_id[ch] = (np->id < min_id[ch])?np->id:min_id[ch];
 231     max_id[ch] = (np->id > max_id[ch])?np->id:max_id[ch];
 232     delay = time_diff(&np->snd, &np->rcv);
 233     t[ch] += delay;
 234     t_sq[ch] += delay*delay;
 235     free(np);
 236   }
 237   
 238   for(ch = 0; ch < ctx->n_channels-1; ch++) {
 239     if(count[ch] == 0) continue;
 240     mean = t[ch]/count[ch];
 241     std = (t_sq[ch]/count[ch]) - mean*mean;
 242     std = (std >= 0)?sqrt(std):LONG_MAX;
 243     //if(std >= 0) std = sqrt(std); else std = LLONG_MAX;
 244     loss = (float)count[ch]/(float)(max_id[ch] - min_id[ch]);
 245     snprintf(msg, 1024, "statistics:port:%d:%lld:%lld:%.4f:%d", 
 246              ctx->channels[ch + 1].of_port, mean, std, loss, count[ch]);
 247     oflops_log(now, GENERIC_MSG, msg);
 248   }
 249 
 250   return 0;
 251 }

 257 int handle_timer_event(struct oflops_context * ctx, struct timer_event *te) {

 259   int len, i, inc=1, res;
 260   void *b;

 269     return 0;

 286     for(i=0; i< flows; i++) {
 287       ip_addr.s_addr += inc;
 288       fl_probe->nw_dst =  htonl(ip_addr.s_addr);
 289       len = make_ofp_flow_add(&b, fl_probe, ctx->channels[OFLOPS_DATA2].of_port, 1, 1200);
 290       res = oflops_send_of_mesgs(ctx, b, len);

 292     }
 293     
 294     fl_probe->in_port = htons(ctx->channels[OFLOPS_DATA2].of_port);
 295     fl_probe->dl_src[5] = 0x75;  
 296     fl_probe->dl_dst[0] = 00; 
 297     fl_probe->dl_dst[1] = 0x15; 
 298     fl_probe->dl_dst[2] = 0x17; 
 299     fl_probe->dl_dst[3] = 0x7b; 
 300     fl_probe->dl_dst[4] = 0x92; 
 301     fl_probe->dl_dst[5] = 0x0a;

 303     len = make_ofp_flow_add(&b, fl_probe, ctx->channels[OFLOPS_DATA3].of_port, 1, 1200);
 304     oflops_send_of_mesg(ctx, b);
 305     free(b);
 306 
 307   } else if(strcmp(str, SNMPGET) == 0) {
 308     oflops_snmp_get(ctx, ctx->cpuOID, ctx->cpuOID_len);
 309     for(i=0;i<ctx->n_channels;i++) {
 310       oflops_snmp_get(ctx, ctx->channels[i].inOID, ctx->channels[i].inOID_len);

 312     }

 314     add_time(&now, 1, 0);
 315     oflops_schedule_timer_event(ctx,&now, SNMPGET);
 316   }

 337 /** Handle pcap event.
 338  * @param ctx pointer to opaque context
 339  * @param pe pcap event
 340  * @param ch enumeration of channel that pcap event is triggered
 341  */
 342 int 
 343 handle_pcap_event(struct oflops_context *ctx, struct pcap_event * pe, oflops_channel_name ch) {
 344   struct pktgen_hdr *pktgen;
 345   int dir, len;
 346   struct ofp_header *ofp;
 347   struct pcap_event *ofp_msg;
 348   struct ofp_error_msg *err_p = NULL;
 349   char msg[1024];
 350 
 351   if (ch == OFLOPS_CONTROL) {
 352     dir = append_data_to_flow(pe->data,pe->pcaphdr);
 353     while(contains_next_msg(dir) > 0) {
 354       len = get_next_msg(dir, &ofp_msg);
 355       ofp = (struct ofp_header *)ofp_msg->data;
 356       switch(ofp->type) {
 357       case OFPT_FLOW_MOD:
 358         if((send_flow_mod) && (!stored_flow_mod_time)) {
 359           memcpy(&flow_mod_timestamp, &pe->pcaphdr.ts, sizeof(struct timeval));
 360           stored_flow_mod_time = 1;
 361         }
 362         oflops_log(pe->pcaphdr.ts,OFPT_FLOW_MOD_ADD, "flow modification send");
 363         break;  
 364       case OFPT_ERROR:
 365         err_p = (struct ofp_error_msg *)ofp;
 366         snprintf(msg, 1024, "OFPT_ERROR(type: %d, code: %d)", ntohs(err_p->type), ntohs(err_p->code));
 367         oflops_log(pe->pcaphdr.ts, OFPT_ERROR_MSG, msg);
 368         fprintf(stderr, "%s\n", msg);
 369         break;   
 370       }
 371     }
 372   } else if ((ch == OFLOPS_DATA1) || (ch == OFLOPS_DATA2) || (ch == OFLOPS_DATA3)) {
 373     struct flow fl;
 374     struct timeval now;
 375     pktgen = extract_pktgen_pkt(pe->data, pe->pcaphdr.caplen, &fl);
 376     if((ch == OFLOPS_DATA3) && (!first_pkt)) {
 377       oflops_log(pe->pcaphdr.ts, GENERIC_MSG, "FIRST_PKT_RCV");
 378       snprintf(msg, 1024, "INSERT_DELAY:%d", time_diff(&flow_mod_timestamp, &pe->pcaphdr.ts));
 379       oflops_log(pe->pcaphdr.ts, GENERIC_MSG, msg);
 380       first_pkt = 1;
 381       gettimeofday(&now, NULL);
 382       add_time(&now, 10, 0);
 383       oflops_schedule_timer_event(ctx,&now, BYESTR);
 384     }
 385     if(htonl(pktgen->seq_num) % 100000 == 0)
 386       printf("data packet received %d\n", htonl(pktgen->seq_num));
 387     
 388     struct entry *n1 = malloc(sizeof(struct entry));
 389     n1->snd.tv_sec = htonl(pktgen->tv_sec);
 390     n1->snd.tv_usec = htonl(pktgen->tv_usec);
 391     memcpy(&n1->rcv, &pe->pcaphdr.ts, sizeof(struct timeval));
 392     n1->id = htonl(pktgen->seq_num);
 393     n1->ch = ch;
 394     TAILQ_INSERT_TAIL(&head, n1, entries);
 395   }
 396   return 0;
 397 }

 426 int 
 427 handle_snmp_event(struct oflops_context * ctx, struct snmp_event * se) {
 428   netsnmp_variable_list *vars;
 429   int len = 1024, i;
 430   char msg[1024], log[1024];
 431   struct timeval now;
 432 
 433   for(vars = se->pdu->variables; vars; vars = vars->next_variable)  {
 434     snprint_value(msg, len, vars->name, vars->name_length, vars);
 435     if((vars->name_length == ctx->cpuOID_len) &&
 436        (memcmp(vars->name, ctx->cpuOID,  ctx->cpuOID_len * sizeof(oid)) == 0) ) {
 437       snprintf(log, len, "cpu : %s %%", msg);
 438       oflops_log(now, SNMP_MSG, log);
 439     } else {
 440       for(i=0;i<ctx->n_channels;i++) {
 441         if((vars->name_length == ctx->channels[i].inOID_len) &&
 442            (memcmp(vars->name, ctx->channels[i].inOID,  
 443                    ctx->channels[i].inOID_len * sizeof(oid)) == 0) ) {
 444           snprintf(log, len, "port %d : rx %s pkts",  
 445                    (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 446           oflops_log(now, SNMP_MSG, log);
 447           break;
 448         }
 449         
 450         if((vars->name_length == ctx->channels[i].outOID_len) &&
 451            (memcmp(vars->name, ctx->channels[i].outOID,  
 452                    ctx->channels[i].outOID_len * sizeof(oid))==0) ) {
 453           snprintf(log, len, "port %d : tx %s pkts",  
 454                    (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 455           //    printf("port %d : tx %s pkts\n",  (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 456           oflops_log(now, SNMP_MSG, log);
 457           break;
 458         }
 459       } //for
 460     }// if cpu
 461   }
 462   return 0;
 463 }

 488 /**
 489  * Initialization code with parameters
 490  * @param ctx 
 491  */
 492 int init(struct oflops_context *ctx, char * config_str) {
 493   char *pos = NULL;
 494   char *param = config_str;
 495   char *value = NULL;
 496   struct timeval now;
 497 
 498   //init counters
 499   finished = 0;
 500 
 501   gettimeofday(&now, NULL);
 502 
 503   cli_param = strdup(config_str);
 504 
 505 
 506   while(*config_str == ' ') {
 507     config_str++;
 508   }
 509   param = config_str;
 510   while(1) {
 511     pos = index(param, ' ');
 512 
 513     if((pos == NULL)) {
 514       if (*param != '\0') {
 515         pos = param + strlen(param) + 1;
 516       } else
 517         break;
 518     }
 519     *pos='\0';
 520     pos++;
 521     value = index(param,'=');
 522     *value = '\0';
 523     value++;
 524     //fprintf(stderr, "param = %s, value = %s\n", param, value);
 525     if(value != NULL) {
 526       if(strcmp(param, "pkt_size") == 0) {
 527         //parse int to get pkt size
 528         pkt_size = strtol(value, NULL, 0);
 529         if((pkt_size < MIN_PKT_SIZE) && (pkt_size > MAX_PKT_SIZE))
 530           perror_and_exit("Invalid packet size value", 1);
 531       }  else if(strcmp(param, "probe_rate") == 0) {
 532         //parse int to get measurement probe rate
 533         proberate = strtol(value, NULL, 0);
 534         if((proberate <= 0) || (proberate >= 1010)) 
 535           perror_and_exit("Invalid probe rate param(Value between 1 and 1010)", 1);
 536       } else if(strcmp(param, "table") == 0) {
 537         //parse int to get pkt size
 538         table = strtol(value, NULL, 0);
 539         if((table < 0) && (table > 2))  
 540           perror_and_exit("Invalid table number", 1);
 541       } else if(strcmp(param, "flows") == 0) {
 542         //parse int to get pkt size
 543         flows = strtol(value, NULL, 0);
 544         if(flows <= 0)  
 545           perror_and_exit("Invalid flow number", 1);
 546       } else 
 547         fprintf(stderr, "Invalid parameter:%s\n", param);
 548       param = pos;
 549     }
 550   } 
 551 
 552   //calculate sendind interval
 553   probe_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (proberate * mbits_to_bits);
 554   fprintf(stderr, "Sending probe interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 555           (uint32_t)probe_snd_interval, (uint32_t)pkt_size, (uint32_t)proberate);
 556   return 0;
 557 }
