  57 struct sw_queue {
  58     struct list node; /* element in port.queues */
  59     unsigned long long int tx_packets;
  60     unsigned long long int tx_bytes;
  61     unsigned long long int tx_errors;
  62     uint32_t queue_id;
  63     uint16_t class_id; /* internal mapping from OF queue_id to tc class_id */
  64     struct sw_port *port; /* reference to the parent port */
  65     /* keep it simple for now, only one property (assuming min_rate) */
  66     uint16_t property; /* one from OFPQT_ */
  67     uint16_t min_rate;
  68 };

  70 #define MAX_HW_NAME_LEN 32
  71 enum sw_port_flags {
  72     SWP_USED             = 1 << 0,    /* Is port being used */
  73     SWP_HW_DRV_PORT      = 1 << 1,    /* Port controlled by HW driver */
  74 };
  75 #if defined(OF_HW_PLAT) && !defined(USE_NETDEV)
  76 #define IS_HW_PORT(p) ((p)->flags & SWP_HW_DRV_PORT)
  77 #else
  78 #define IS_HW_PORT(p) 0
  79 #endif
  80 
  81 #define PORT_IN_USE(p) (((p) != NULL) && (p)->flags & SWP_USED)
  82 
  83 struct sw_port {
  84     uint32_t config;            /* Some subset of OFPPC_* flags. */
  85     uint32_t state;             /* Some subset of OFPPS_* flags. */
  86     uint32_t flags;             /* SWP_* flags above */
  87     struct datapath *dp;
  88     struct netdev *netdev;
  89     char hw_name[OFP_MAX_PORT_NAME_LEN];
  90     struct list node; /* Element in datapath.ports. */
  91     unsigned long long int rx_packets, tx_packets;
  92     unsigned long long int rx_bytes, tx_bytes;
  93     unsigned long long int tx_dropped;
  94     uint16_t port_no;
  95     /* port queues */
  96     uint16_t num_queues;
  97     struct sw_queue queues[NETDEV_MAX_QUEUES];
  98     struct list queue_list; /* list of all queues for this port */
  99 };

 110 #define DP_MAX_PORTS 255
 111 BUILD_ASSERT_DECL(DP_MAX_PORTS <= OFPP_MAX);
 112 
 113 struct datapath {
 114     /* Remote connections. */
 115     struct list remotes;        /* All connections (including controller). */
 116 
 117     /* Listeners. */
 118     struct pvconn **listeners;
 119     size_t n_listeners;
 120 
 121     time_t last_timeout;
 122 
 123     /* Unique identifier for this datapath */
 124     uint64_t  id;
 125     char dp_desc[DESC_STR_LEN]; /* human readible comment to ID this DP */
 126 
 127     struct sw_chain *chain;  /* Forwarding rules. */
 128 
 129     /* Configuration set from controller. */
 130     uint16_t flags;
 131     uint16_t miss_send_len;
 132 
 133     /* Switch ports. */
 134     struct sw_port ports[DP_MAX_PORTS];
 135     struct sw_port *local_port;  /* OFPP_LOCAL port, if any. */
 136     struct list port_list; /* All ports, including local_port. */
 137 
 138 #if defined(OF_HW_PLAT)
 139     /* Although the chain maintains the pointer to the HW driver
 140      * for flow operations, the datapath needs the port functions
 141      * in the driver structure
 142      */
 143     of_hw_driver_t *hw_drv;
 144     struct hw_pkt_q_entry *hw_pkt_list_head, *hw_pkt_list_tail;
 145 #endif
 146 };
