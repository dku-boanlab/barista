  24 #ifndef __W_BUFFER_H__
  25 #define __W_BUFFER_H__
  26 
  27 #define SOME_FUNCTIONS_ARE_DEFINES
  28 
  29 typedef struct Buffer {
  30         guchar          *data;
  31         unsigned int    allocated;
  32         unsigned int    start;
  33         unsigned int    first_free;
  34 } Buffer;
  35 
  36 void buffer_init(Buffer* buffer, unsigned int space);
  37 void buffer_free(Buffer* buffer);
  38 void buffer_assure_space(Buffer* buffer, unsigned int space);
  39 void buffer_append(Buffer* buffer, guchar *from, unsigned int bytes);
  40 void buffer_remove_start(Buffer* buffer, unsigned int bytes);
  41 
  42 #ifdef SOME_FUNCTIONS_ARE_DEFINES
  43 # define buffer_clean(buffer) buffer_remove_start((buffer), buffer_length(buffer))
  44 # define buffer_increase_length(buffer,bytes) (buffer)->first_free += (bytes)
  45 # define buffer_length(buffer) ((buffer)->first_free - (buffer)->start)
  46 # define buffer_start_ptr(buffer) ((buffer)->data + (buffer)->start)
  47 # define buffer_end_ptr(buffer) ((buffer)->data + (buffer)->first_free)
  48 # define buffer_append_buffer(buffer,src_buffer) buffer_append((buffer), buffer_start_ptr(src_buffer), buffer_length(src_buffer))
  49 #else
  50  void buffer_clean(Buffer* buffer);
  51  void buffer_increase_length(Buffer* buffer, unsigned int bytes);
  52  unsigned int buffer_length(Buffer* buffer);
  53  guchar* buffer_start_ptr(Buffer* buffer);
  54  guchar* buffer_end_ptr(Buffer* buffer);
  55  void buffer_append_buffer(Buffer* buffer, Buffer* src_buffer);
  56 #endif
  57 
  58 #endif
