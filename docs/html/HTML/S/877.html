  52 #define MAC_HASH_BITS 10
  53 #define MAC_HASH_MASK (MAC_HASH_SIZE - 1)

  58 /* A MAC learning table entry. */
  59 struct mac_entry {
  60     struct list hash_node;      /* Element in a mac_learning 'table' list. */
  61     struct list lru_node;       /* Element in 'lrus' or 'free' list. */
  62     time_t expires;             /* Expiration time. */
  63     uint8_t mac[ETH_ADDR_LEN];  /* Known MAC address. */
  64     uint16_t vlan;              /* VLAN tag. */
  65     int port;                   /* Port on which MAC was most recently seen. */
  66     tag_type tag;               /* Tag for this learning entry. */
  67 };

  69 /* MAC learning table. */
  70 struct mac_learning {
  71     struct list free;           /* Not-in-use entries. */
  72     struct list lrus;           /* In-use entries, least recently used at the
  73                                    front, most recently used at the back. */
  74     struct list table[MAC_HASH_SIZE]; /* Hash table. */
  75     struct mac_entry entries[MAC_MAX]; /* All entries. */
  76     uint32_t secret;            /* Secret for  */
  77 };

  79 static uint32_t
  80 mac_table_hash(const uint8_t mac[ETH_ADDR_LEN], uint16_t vlan)
  81 {
  82     return hash_bytes(mac, ETH_ADDR_LEN, vlan);

  91 /* Returns a tag that represents that 'mac' is on an unknown port in 'vlan'.
  92  * (When we learn where 'mac' is in 'vlan', this allows flows that were
  93  * flooded to be revalidated.) */
  94 static tag_type
  95 make_unknown_mac_tag(const struct mac_learning *ml,
  96                      const uint8_t mac[ETH_ADDR_LEN], uint16_t vlan)
  97 {
  98     uint32_t h = hash_bytes(&ml->secret, sizeof ml->secret,
  99                             mac_table_hash(mac, vlan));
 100     return tag_create_deterministic(h);
 101 }

 103 static struct list *
 104 mac_table_bucket(const struct mac_learning *ml,
 105                  const uint8_t mac[ETH_ADDR_LEN],
 106                  uint16_t vlan)
 107 {
 108     uint32_t hash = mac_table_hash(mac, vlan);
 109     const struct list *list = &ml->table[hash & MAC_HASH_BITS];
 110     return (struct list *) list;
 111 }

 113 static struct mac_entry *
 114 search_bucket(struct list *bucket, const uint8_t mac[ETH_ADDR_LEN],
 115               uint16_t vlan)
 116 {
 117     struct mac_entry *e;
 118     LIST_FOR_EACH (e, struct mac_entry, hash_node, bucket) {
 119         if (eth_addr_equals(e->mac, mac) && e->vlan == vlan) {
 120             return e;
 121         }
 122     }
 123     return NULL;
 124 }

 126 /* If the LRU list is not empty, stores the least-recently-used entry in '*e'
 127  * and returns true.  Otherwise, if the LRU list is empty, stores NULL in '*e'
 128  * and return false. */
 129 static bool
 130 get_lru(struct mac_learning *ml, struct mac_entry **e)
 131 {
 132     if (!list_is_empty(&ml->lrus)) {
 133         *e = mac_entry_from_lru_node(ml->lrus.next);
 134         return true;
 135     } else {
 136         *e = NULL;
 137         return false;
 138     }
 139 }

 141 /* Removes 'e' from the 'ml' hash table.  'e' must not already be on the free
 142  * list. */
 143 static void
 144 free_mac_entry(struct mac_learning *ml, struct mac_entry *e)
 145 {
 146     list_remove(&e->hash_node);
 147     list_remove(&e->lru_node);
 148     list_push_front(&ml->free, &e->lru_node);
 149 }

 151 /* Creates and returns a new MAC learning table. */
 152 struct mac_learning *
 153 mac_learning_create(void)
 154 {
 155     struct mac_learning *ml;
 156     int i;
 157 
 158     ml = xmalloc(sizeof *ml);
 159     list_init(&ml->lrus);
 160     list_init(&ml->free);
 161     for (i = 0; i < MAC_HASH_SIZE; i++) {
 162         list_init(&ml->table[i]);
 163     }
 164     for (i = 0; i < MAC_MAX; i++) {
 165         struct mac_entry *s = &ml->entries[i];
 166         list_push_front(&ml->free, &s->lru_node);
 167     }
 168     ml->secret = random_uint32();
 169     return ml;
 170 }

 179 /* Attempts to make 'ml' learn from the fact that a frame from 'src_mac' was
 180  * just observed arriving from 'src_port' on the given 'vlan'.
 181  *
 182  * Returns nonzero if we actually learned something from this, zero if it just
 183  * confirms what we already knew.  The nonzero return value is the tag of flows
 184  * that now need revalidation.
 185  *
 186  * The 'vlan' parameter is used to maintain separate per-VLAN learning tables.
 187  * Specify 0 if this behavior is undesirable. */
 188 tag_type
 189 mac_learning_learn(struct mac_learning *ml,
 190                    const uint8_t src_mac[ETH_ADDR_LEN], uint16_t vlan,
 191                    uint16_t src_port)
 192 {
 193     struct mac_entry *e;
 194     struct list *bucket;
 195 
 196     if (eth_addr_is_multicast(src_mac)) {
 197         static struct vlog_rate_limit rl = VLOG_RATE_LIMIT_INIT(30, 30);
 198         VLOG_DBG_RL(&rl, "multicast packet source "ETH_ADDR_FMT,
 199                     ETH_ADDR_ARGS(src_mac));
 200         return 0;
 201     }
 202 
 203     bucket = mac_table_bucket(ml, src_mac, vlan);
 204     e = search_bucket(bucket, src_mac, vlan);
 205     if (!e) {
 206         if (!list_is_empty(&ml->free)) {
 207             e = mac_entry_from_lru_node(ml->free.next);
 208         } else {
 209             e = mac_entry_from_lru_node(ml->lrus.next);
 210             list_remove(&e->hash_node);
 211         }
 212         memcpy(e->mac, src_mac, ETH_ADDR_LEN);
 213         list_push_front(bucket, &e->hash_node);
 214         e->port = -1;
 215         e->vlan = vlan;
 216         e->tag = make_unknown_mac_tag(ml, src_mac, vlan);
 217     }
 218 
 219     /* Make the entry most-recently-used. */
 220     list_remove(&e->lru_node);
 221     list_push_back(&ml->lrus, &e->lru_node);
 222     e->expires = time_now() + 60;
 223 
 224     /* Did we learn something? */
 225     if (e->port != src_port) {
 226         tag_type old_tag = e->tag;
 227         e->port = src_port;
 228         e->tag = tag_create_random();
 229         return old_tag;
 230     }
 231     return 0;
 232 }

 234 /* Looks up MAC 'dst' for VLAN 'vlan' in 'ml'.  Returns the port on which a
 235  * frame destined for 'dst' should be sent, OFPP_FLOOD if unknown. */
 236 uint16_t
 237 mac_learning_lookup(const struct mac_learning *ml,
 238                     const uint8_t dst[ETH_ADDR_LEN], uint16_t vlan)
 239 {
 240     tag_type tag = 0;
 241     return mac_learning_lookup_tag(ml, dst, vlan, &tag);
 242 }
 243 
 244 /* Looks up MAC 'dst' for VLAN 'vlan' in 'ml'.  Returns the port on which a
 245  * frame destined for 'dst' should be sent, OFPP_FLOOD if unknown.
 246  *
 247  * Adds to '*tag' (which the caller must have initialized) the tag that should
 248  * be attached to any flow created based on the return value, if any, to allow
 249  * those flows to be revalidated when the MAC learning entry changes. */
 250 uint16_t
 251 mac_learning_lookup_tag(const struct mac_learning *ml,
 252                         const uint8_t dst[ETH_ADDR_LEN], uint16_t vlan,
 253                         tag_type *tag)
 254 {
 255     if (eth_addr_is_multicast(dst)) {
 256         return OFPP_FLOOD;
 257     } else {
 258         struct mac_entry *e = search_bucket(mac_table_bucket(ml, dst, vlan),
 259                                             dst, vlan);
 260         if (e) {
 261             *tag |= e->tag;
 262             return e->port;
 263         } else {
 264             *tag |= make_unknown_mac_tag(ml, dst, vlan);
 265             return OFPP_FLOOD;
 266         }
 267     }
 268 }

 270 /* Expires all the mac-learning entries in 'ml'.  The tags in 'ml' are
 271  * discarded, so the client is responsible for revalidating any flows that
 272  * depend on 'ml', if necessary. */
 273 void
 274 mac_learning_flush(struct mac_learning *ml)
 275 {
 276     struct mac_entry *e;
 277     while (get_lru(ml, &e)){
 278         free_mac_entry(ml, e);
 279     }
 280 }
 281 
 282 void
 283 mac_learning_run(struct mac_learning *ml, struct tag_set *set)
 284 {
 285     struct mac_entry *e;
 286     while (get_lru(ml, &e) && time_now() >= e->expires) {
 287         if (set) {
 288             tag_set_add(set, e->tag);
 289         }
 290         free_mac_entry(ml, e);
 291     }
 292 }

 294 void
 295 mac_learning_wait(struct mac_learning *ml)
 296 {
 297     if (!list_is_empty(&ml->lrus)) {
 298         struct mac_entry *e = mac_entry_from_lru_node(ml->lrus.next);
 299         poll_timer_wait((e->expires - time_now()) * 1000);
 300     }
 301 }
