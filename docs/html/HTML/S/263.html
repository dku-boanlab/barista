  25 static struct sw_flow *table_linear_lookup(struct sw_table *swt,
  26                                          const struct sw_flow_key *key)
  27 {
  28         struct sw_table_linear *tl = (struct sw_table_linear *) swt;
  29         struct sw_flow *flow;
  30         list_for_each_entry_rcu (flow, &tl->flows, node) {
  31                 if (flow_matches_1wild(key, &flow->key))
  32                         return flow;
  33         }
  34         return NULL;
  35 }

  37 static int table_linear_insert(struct sw_table *swt, struct sw_flow *flow)
  38 {
  39         struct sw_table_linear *tl = (struct sw_table_linear *) swt;
  40         struct sw_flow *f;
  41 
  42 
  43         /* Loop through the existing list of entries.  New entries will
  44          * always be placed behind those with equal priority.  Just replace 
  45          * any flows that match exactly.
  46          */
  47         list_for_each_entry (f, &tl->flows, node) {
  48                 if (f->priority == flow->priority
  49                                 && f->key.wildcards == flow->key.wildcards
  50                                 && flow_matches_2wild(&f->key, &flow->key)) {
  51                         flow->serial = f->serial;
  52                         list_replace_rcu(&f->node, &flow->node);
  53                         list_replace_rcu(&f->iter_node, &flow->iter_node);
  54                         flow_deferred_free(f);
  55                         return 1;
  56                 }
  57 
  58                 if (f->priority < flow->priority)
  59                         break;
  60         }
  61 
  62         /* Make sure there's room in the table. */
  63         if (tl->n_flows >= tl->max_flows) {
  64                 return 0;
  65         }
  66         tl->n_flows++;
  67 
  68         /* Insert the entry immediately in front of where we're pointing. */
  69         flow->serial = tl->next_serial++;
  70         list_add_tail_rcu(&flow->node, &f->node);
  71         list_add_rcu(&flow->iter_node, &tl->iter_flows);
  72         return 1;
  73 }

 108 static int do_delete(struct datapath *dp, struct sw_table *swt, 
 109                         struct sw_flow *flow, enum ofp_flow_removed_reason reason)
 110 {
 111         dp_send_flow_end(dp, flow, reason);
 112         list_del_rcu(&flow->node);

 118 static int table_linear_delete(struct datapath *dp, struct sw_table *swt,
 119                                 const struct sw_flow_key *key, uint16_t out_port,
 120                                 uint16_t priority, int strict)
 121 {
 122         struct sw_table_linear *tl = (struct sw_table_linear *) swt;
 123         struct sw_flow *flow;
 124         unsigned int count = 0;
 125 
 126         list_for_each_entry (flow, &tl->flows, node) {
 127                 if (flow_matches_desc(&flow->key, key, strict)
 128                                 && flow_has_out_port(flow, out_port)
 129                                 && (!strict || (flow->priority == priority)))
 130                         count += do_delete(dp, swt, flow, OFPRR_DELETE);
 131         }
 132         tl->n_flows -= count;
 133         return count;
 134 }

 136 static int table_linear_timeout(struct datapath *dp, struct sw_table *swt)
 137 {
 138         struct sw_table_linear *tl = (struct sw_table_linear *) swt;
 139         struct sw_flow *flow;
 140         int count = 0;
 141 
 142         if (mutex_lock_interruptible(&dp_mutex))
 143                 return 0;
 144         list_for_each_entry (flow, &tl->flows, node) {
 145                 int reason = flow_timeout(flow);
 146                 if (reason >= 0) {
 147                         count += do_delete(dp, swt, flow, reason);
 148                 }
 149         }
 150         tl->n_flows -= count;
 151         mutex_unlock(&dp_mutex);
 152         return count;
 153 }

 155 static void table_linear_destroy(struct sw_table *swt)
 156 {
 157         struct sw_table_linear *tl = (struct sw_table_linear *) swt;
 158 
 159         while (!list_empty(&tl->flows)) {
 160                 struct sw_flow *flow = list_entry(tl->flows.next,
 161                                                   struct sw_flow, node);
 162                 list_del(&flow->node);
 163                 flow_free(flow);
 164         }
 165         kfree(tl);
 166 }

 168 static int table_linear_iterate(struct sw_table *swt,
 169                                 const struct sw_flow_key *key, uint16_t out_port,
 170                                 struct sw_table_position *position,
 171                                 int (*callback)(struct sw_flow *, void *),
 172                                 void *private)
 173 {
 174         struct sw_table_linear *tl = (struct sw_table_linear *) swt;
 175         struct sw_flow *flow;
 176         unsigned long start;
 177 
 178         start = position->private[0];
 179         list_for_each_entry (flow, &tl->iter_flows, iter_node) {
 180                 if (flow->serial >= start
 181                                 && flow_matches_2wild(key, &flow->key)
 182                                 && flow_has_out_port(flow, out_port)) {
 183                         int error = callback(flow, private);
 184                         if (error) {
 185                                 position->private[0] = flow->serial;
 186                                 return error;
 187                         }
 188                 }
 189         }
 190         return 0;
 191 }
