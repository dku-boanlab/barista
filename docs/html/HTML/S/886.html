  39 static size_t
  40 hash_name(const char *name)
  41 {
  42     return hash_bytes(name, strlen(name), 0);
  43 }

  45 void
  46 shash_init(struct shash *sh)
  47 {
  48     hmap_init(&sh->map);
  49 }
  50 
  51 void
  52 shash_destroy(struct shash *sh)
  53 {
  54     if (sh) {
  55         shash_clear(sh);
  56         free(sh);
  57     }
  58 }
  59 
  60 void
  61 shash_clear(struct shash *sh)
  62 {
  63     struct shash_node *node, *next;
  64 
  65     HMAP_FOR_EACH_SAFE (node, next, struct shash_node, node, &sh->map) {
  66         hmap_remove(&sh->map, &node->node);
  67         free(node->name);
  68         free(node);
  69     }
  70 }
  71 
  72 /* It is the caller's responsible to avoid duplicate names, if that is
  73  * desirable. */
  74 void
  75 shash_add(struct shash *sh, const char *name, void *data)
  76 {
  77     struct shash_node *node = xmalloc(sizeof *node);
  78     node->name = xstrdup(name);
  79     node->data = data;
  80     hmap_insert(&sh->map, &node->node, hash_name(name));
  81 }
  82 
  83 void
  84 shash_delete(struct shash *sh, struct shash_node *node)
  85 {
  86     hmap_remove(&sh->map, &node->node);
  87     free(node->name);
  88     free(node);
  89 }
  90 
  91 /* If there are duplicates, returns a random element. */
  92 struct shash_node *
  93 shash_find(const struct shash *sh, const char *name)
  94 {
  95     struct shash_node *node;
  96 
  97     HMAP_FOR_EACH_WITH_HASH (node, struct shash_node, node,
  98                              hash_name(name), &sh->map) {
  99         if (!strcmp(node->name, name)) {
 100             return node;
 101         }
 102     }
 103     return NULL;
 104 }
 105 
 106 void *
 107 shash_find_data(const struct shash *sh, const char *name)
 108 {
 109     struct shash_node *node = shash_find(sh, name);
 110     return node ? node->data : NULL;
 111 }
