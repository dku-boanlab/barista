  47 /* Returns a randomly selected tag. */
  48 tag_type
  49 tag_create_random(void)
  50 {
  51     int x, y;
  52     do {
  53         uint16_t r = random_uint16();
  54         x = r & (N_TAG_BITS - 1);
  55         y = r >> (16 - LOG2_N_TAG_BITS);
  56     } while (x == y);
  57     return (1u << x) | (1u << y);
  58 }

  60 /* Returns a tag deterministically generated from 'seed'.
  61  *
  62  * 'seed' should have data in all of its bits; if it has data only in its
  63  * low-order bits then the resulting tags will be poorly distributed.  Use a
  64  * hash function such as hash_bytes() to generate 'seed' if necessary. */
  65 tag_type
  66 tag_create_deterministic(uint32_t seed)
  67 {
  68     int x = seed & (N_TAG_BITS - 1);
  69     int y = (seed >> LOG2_N_TAG_BITS) % 31;
  70     y += y >= x;
  71     return (1u << x) | (1u << y);
  72 }

  81 /* Adds 'tag' to 'set'. */
  82 void
  83 tag_set_add(struct tag_set *set, tag_type tag)
  84 {
  85     if (tag && (!tag_is_valid(tag) || !tag_set_intersects(set, tag))) {
  86         /* XXX We could do better by finding the set member to which we would
  87          * add the fewest number of 1-bits.  This would reduce the amount of
  88          * ambiguity, since e.g. three 1-bits match 3 * 2 / 2 = 3 unique tags
  89          * whereas four 1-bits match 4 * 3 / 2 = 6 unique tags. */
  90         tag_type *t = &set->tags[set->n++ % TAG_SET_SIZE];
  91         *t |= tag;
  92         if (*t == TYPE_MAXIMUM(tag_type)) {
  93             set->tags[0] = *t;
  94         }
  95 
  96         set->total |= tag;
  97     }
  98 }
