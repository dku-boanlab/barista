  34 #ifndef OFPBUF_H
  35 #define OFPBUF_H 1
  36 
  37 #include <stddef.h>
  38 
  39 /* Buffer for holding arbitrary data.  An ofpbuf is automatically reallocated
  40  * as necessary if it grows too large for the available memory. */
  41 struct ofpbuf {
  42     void *base;                 /* First byte of area malloc()'d area. */
  43     size_t allocated;           /* Number of bytes allocated. */
  44 
  45     void *data;                 /* First byte actually in use. */
  46     size_t size;                /* Number of bytes in use. */
  47 
  48     void *l2;                   /* Link-level header. */
  49     void *l3;                   /* Network-level header. */
  50     void *l4;                   /* Transport-level header. */
  51     void *l7;                   /* Application data. */
  52 
  53     struct ofpbuf *next;        /* Next in a list of ofpbufs. */
  54     void *private;              /* Private pointer for use by owner. */
  55 };
  56 
  57 void ofpbuf_use(struct ofpbuf *, void *, size_t);

  61 void ofpbuf_reinit(struct ofpbuf *, size_t);

  63 struct ofpbuf *ofpbuf_new(size_t);
  64 struct ofpbuf *ofpbuf_clone(const struct ofpbuf *);
  65 struct ofpbuf *ofpbuf_clone_data(const void *, size_t);
