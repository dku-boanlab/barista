 117 void heapify(wc_queue * pq, int i){
 118         int l,r;
 119         int largest;
 120         qelm tmp;
 121         
 122         assert(pq);
 123         l = LEFT(i);
 124         r = RIGHT(i);
 125         if((l<= pq->size)&& (timercmp(&pq->array[l].key,&pq->array[i].key,<)))
 126                 largest=l;
 127         else largest=i;
 128         if((r<= pq->size)&& (timercmp(&pq->array[r].key,&pq->array[largest].key,<)))
 129                 largest=r;
 130         if(largest!=i){
 131                 // swap 'em
 132                 tmp=pq->array[i];
 133                 pq->array[i]=pq->array[largest];
 134                 pq->array[largest]=tmp;
 135                 heapify(pq,largest);
 136         }
 137 }

 167 static void wc_schedule_next_event(wc_queue *pq)
 168 {
 169         struct timeval now,diff;
 170         struct itimerval itv;
 171         qelm *data;
 172         assert(pq->timersEnabled);      // should only be called when timers are enabled
 173         assert(pq->size>0);
 174 
 175         diff.tv_sec = diff.tv_usec = 0;
 176 
 177         while(!wc_queue_isempty(pq))            
 178         {
 179                 // grab first element
 180                 data=&pq->array[1];             // DUMBASS: we skip array[0] b/c it makes the math easier
 181                 gettimeofday(&now,NULL);
 182                 if(timercmp(&now,&data->key,>))
 183                 {       // event already should have happened; sched for NOW
 184                         fprintf(stderr,"Scheduling event %d NOW b/c it has passed already\n",data->id);
 185                         wc_run_next_event(pq);
 186                         continue;
 187                 }
 188                 // schedule event for some amount of time in future
 189                 timersub(&data->key,&now,&diff);
 190                 break;
 191         }
 192         if(wc_queue_isempty(pq))
 193                 return;         // nothing to schedule
 194 
 195         itv.it_value=diff;
 196         itv.it_interval.tv_sec = itv.it_interval.tv_usec=0;
 197         signal(SIGALRM,wc_event_handler);       // set this here...
 198         assert(setitimer(ITIMER_REAL,&itv,NULL)==0);
 199 }

 201  * int wc_queue_extract(wc_queue *pq , struct timeval * key, wc_event * wc){
 202  *      remove from the top of the heap and reheap 
 203  *              ASSUMES timer will not go off, which is true if called from signal handler
 204  */

 206 int wc_queue_extract(wc_queue *pq , int * id, struct timeval * key,void (**fun)(void *), void **arg) {
 207         wc_event * data;
 208         assert(pq);
 209         assert(pq->timersEnabled==0);
 210         if(pq->size<1){         //empty
 211                 return -1;
 212         }
 213         data=pq->array[1].data;
 214         *key=pq->array[1].key;
 215         *id=pq->array[1].id;
 216         pq->array[1]=pq->array[pq->size];       // move first elm to end
 217         pq->size--;
 218         heapify(pq,1);
 219 
 220         *fun=data->fun;
 221         *arg=data->arg;
 222         free(data);
 223         return 0;
 224 
 225 }
 226 /***********************************************************************
 227  * int wc_event_remove(wc_queue * pq, int id);
 228  *      remove the event specified by id
 229  *
 230  *              YES this is a linear search; but hopefully it won't matter (GULP)
 231  *      alg: move last elm to deleted elm slot, then call heapify on deleted slot
 232  *              
 233  */
 234 int wc_event_remove(wc_queue * pq, int id,void (**fun)(void *),void **arg)
 235 {
 236         int i;
 237         struct wc_event * data;
 238         assert(pq->timersEnabled==0);
 239         for(i=1;i<=pq->size;i++)
 240                 if(pq->array[i].id==id)
 241                         break;
 242         if(i>pq->size)
 243                 return -1;      // tried to delete something that didn't exist
 244 #ifdef HEAP_DEBUG
 245         verify_heap(pq);
 246 #endif
 247         data = pq->array[i].data;
 248         *fun=data->fun;
 249         *arg=data->arg;
 250         free(data);
 251         pq->array[i]=pq->array[pq->size];
 252         pq->size--;
 253         do
 254         {       // need to fix up the entire heap after arbitary delete
 255                 heapify(pq,i);          // this is a log(log(n)) op, 
 256                 i=PARENT(i);            // but that is the cost of arbt dels in a heap
 257         } while(i>0);
 258 #ifdef HEAP_DEBUG
 259         verify_heap(pq);
 260 #endif
 261         // don't worry about deleting the currently scheduled event;
 262         //  that will be rescheduled elsewhere
 263         return 0;
 264 }

 305 /**************************************************************************
 306  * int wc_get_next_event_delta(struct wc_queue *, struct timeval *delta);
 307  *      return the time delta (not absolute time) until the next event
 308  *              - used for select()
 309  *              - return:       
 310  *                      0 if *delta is valid
 311  *                      -1 if no event exists
 312  *                      1 if first event has already passed
 313  */
 314 int wc_get_next_event_delta(struct wc_queue * pq, struct timeval *delta)
 315 {
 316         struct timeval now;
 317         qelm *data;
 318         assert(pq->timersEnabled==0);   // should only be called when timers are disabled
 319         if(wc_queue_isempty(pq))
 320                 return -1;
 321         data=&pq->array[1];             // DUMBASS: we skip array[0] b/c it makes the math easier
 322         gettimeofday(&now,NULL);
 323         if(timercmp(&now,&data->key,>)) {
 324           return 1;             // event already should have happened; sched for NOW
 325         }
 326         // schedule event for some amount of time in future
 327         timersub(&data->key,&now,delta);
 328         return 0;                       // there exists an event in the future
 329 }

 338 void wc_event_handler(int ignore)
 339 {
 340         wc_queue *pq=WC_Queue;          // HACK; how do we pass state to a signal handler?; need to use globalvar
 341 
 342         assert(pq->timersEnabled);      // it would suck if this went off when timers were disabled
 343         pq->timersEnabled=0;            // mark timers as disabled (they can't go off anyway while we are here)
 344         wc_run_next_event(pq);
 345         pq->timersEnabled=1;            // re-enabled them
 346         if(!wc_queue_isempty(pq))
 347                 wc_schedule_next_event(pq);     // schedule next event, if exists
 348                                                 // this will also reset the SIGALRM handler and call setitimer()
 349 }

 373 void wc_disable_timers(struct wc_queue *pq)
 374 {
 375         struct itimerval stop,old;

 377         signal(SIGALRM,SIG_IGN);                        // this should stop a race condition of

 379                                                         // been proceeded; won't drop event, b/c it will
 380                                                         // be resched when we re-enable
 381         assert(setitimer(ITIMER_REAL,&stop,&old)==0);   // stop the current timer, so it doesn't go 
 382         fprintf(stderr," timers disabled\n");
 383         pq->timersEnabled=0;

 481                 times[j]=times[i];
 482                 t[j]=t[i];
 483                 times[i]=k;
 484                 t[i]=tmp;
 485         }

 487         // insert
 488         for(i=0;i<10;i++){
 489                 printf("Adding event %d with id==%d\n",t[i],
