  88 nf2_lookup_flowtable(struct sw_table *flowtab, const struct sw_flow_key *key)
  89 {
  90         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;

 102 static int
 103 nf2_install_flow(struct sw_table *flowtab, struct sw_flow *flow)
 104 {
 105         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 106 
 107         /* Delete flows that match exactly. */
 108         nf2_uninstall_flow(NULL, flowtab, &flow->key, OFPP_NONE,
 109                            flow->priority, true, KEEP_FLOW);
 110 
 111         if (nf2_are_actions_supported(flow)) {
 112                 if (nf2_build_and_write_flow(flow)) {
 113                         /* Not successful */
 114                         return 0;
 115                 }
 116         } else {
 117                 /* Unsupported actions or no device. */
 118                 return 0;
 119         }
 120 
 121         nf2flowtab->num_flows++;
 122         list_push_front(&nf2flowtab->flows, &flow->node);
 123         list_push_front(&nf2flowtab->iter_flows, &flow->iter_node);
 124 
 125         return 1;
 126 }

 128 static int
 129 nf2_modify_flow(struct sw_table *flowtab, const struct sw_flow_key *key,
 130                 uint16_t priority, int strict,
 131                 const struct ofp_action_header *actions, size_t actions_len)
 132 {
 133         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 134         struct sw_flow *flow;
 135         unsigned int count = 0;
 136 
 137         LIST_FOR_EACH(flow, struct sw_flow, node, &nf2flowtab->flows) {
 138                 if (flow_matches_desc(&flow->key, key, strict)
 139                     && (!strict || flow->priority == priority)) {
 140                         flow_replace_acts(flow, actions, actions_len);
 141                         if (nf2_are_actions_supported(flow)) {
 142                                 count += nf2_modify_acts(flow);
 143                         }
 144                 } else {
 145                         return 0;
 146                 }
 147         }
 148 
 149         return count;
 150 }

 152 static int
 153 nf2_has_conflict(struct sw_table *flowtab, const struct sw_flow_key *key,
 154                  uint16_t priority, int strict)
 155 {
 156         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 157         struct sw_flow *flow;
 158 
 159         LIST_FOR_EACH(flow, struct sw_flow, node, &nf2flowtab->flows) {
 160 
 161                 if (flow_matches_2desc(&flow->key, key, strict)
 162                     && (flow->priority == priority)) {
 163                         return true;
 164                 }
 165         }
 166 
 167         return false;
 168 }

 170 static int
 171 do_uninstall(struct sw_flow *flow, struct list *deleted)
 172 {
 173         if (flow != NULL && flow->private != NULL) {
 174                 list_remove(&flow->node);
 175                 list_remove(&flow->iter_node);
 176                 list_push_back(deleted, &flow->node);
 177                 return 1;
 178         }
 179 
 180         return 0;
 181 }

 192 static int
 193 nf2_uninstall_flow(struct datapath *dpinst, struct sw_table *flowtab,
 194                    const struct sw_flow_key *key, uint16_t out_port,
 195                    uint16_t priority, int strict, int keep_flow)
 196 {
 197         struct nf2device *dev;
 198         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 199         struct sw_flow *flow, *n;
 200         struct nf2_flow *nf2flow;
 201         unsigned int count = 0;
 202         struct list deleted;
 203         list_init(&deleted);
 204 
 205         dev = nf2_get_net_device();
 206         if (dev == NULL)
 207                 return 0;
 208 
 209         LIST_FOR_EACH_SAFE (flow, n, struct sw_flow, node, &nf2flowtab->flows) {
 210                 if (flow_matches_desc(&flow->key, key, strict)
 211                     && (!strict || flow->priority == priority)
 212                     && flow_has_out_port(flow, out_port)) {
 213                         nf2flow = flow->private;
 214 
 215                         if (nf2flow != NULL) {
 216                                 flow->packet_count
 217                                         += nf2_get_packet_count(dev,
 218                                                                 nf2flow);
 219                                 flow->byte_count += nf2_get_byte_count(dev,
 220                                                                        nf2flow);
 221                         }
 222                         count += do_uninstall(flow, &deleted);
 223                         if (keep_flow == KEEP_FLOW) {
 224                                 /* Delete private in sw_flow here */
 225                                 nf2_delete_private(flow->private);
 226                         }
 227                 }
 228         }
 229         nf2flowtab->num_flows -= count;
 230 
 231         nf2_free_net_device(dev);
 232 
 233         if (keep_flow == DELETE_FLOW) {
 234                 /* Notify DP of deleted flows and delete the flow */
 235                 LIST_FOR_EACH_SAFE (flow, n, struct sw_flow, node, &deleted) {
 236                         dp_send_flow_end(dpinst, flow, flow->reason);
 237                         list_remove(&flow->node);
 238                         nf2_delete_private(flow->private);
 239                         flow_free(flow);
 240                         }
 241         }
 242 
 243         return count;
 244 }

 247 nf2_flow_timeout(struct sw_table *flowtab, struct list *deleted)
 248 {
 249         struct nf2device *dev;
 250         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 251         struct sw_flow *flow, *n;
 252         struct nf2_flow *nf2flow;
 253         int num_uninst_flows = 0;
 254         uint64_t num_forw_packets = 0;
 255         uint64_t now = time_msec();
 256 
 257         dev = nf2_get_net_device();
 258         if (dev == NULL) {
 259                 DBG_ERROR("Could not open NetFPGA device\n");
 260                 return;
 261         }
 262 
 263         /* LOCK; */
 264         /* FIXME */
 265         LIST_FOR_EACH_SAFE (flow, n, struct sw_flow, node, &nf2flowtab->flows) {
 266                 nf2flow = flow->private;
 267                 if (nf2flow != NULL) {
 268                         num_forw_packets = flow->packet_count
 269                                 + nf2_get_packet_count(dev, nf2flow);
 270                         flow->byte_count += nf2_get_byte_count(dev, nf2flow);
 271                 }
 272                 if (num_forw_packets > flow->packet_count) {
 273                         flow->packet_count = num_forw_packets;
 274                         flow->used = now;
 275                 }
 276 
 277                 if (flow_timeout(flow)) {
 278                         num_uninst_flows += do_uninstall(flow, deleted);
 279                         nf2_delete_private(flow->private);
 280                 }
 281         }
 282 
 283         /* UNLOCK; */
 284 
 285         nf2_clear_watchdog(dev);
 286         nf2_free_net_device(dev);
 287 
 288         nf2flowtab->num_flows -= num_uninst_flows;
 289 }

 291 static void
 292 nf2_destroy_flowtable(struct sw_table *flowtab)
 293 {
 294         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 295         struct nf2_flow *nf2flow = NULL;
 296 
 297         if (nf2flowtab == NULL)
 298                 return;
 299 
 300         while (!list_is_empty(&nf2flowtab->flows)) {
 301                 struct sw_flow *flow
 302                         = CONTAINER_OF(list_front(&nf2flowtab->flows),
 303                                                   struct sw_flow, node);
 304                 list_remove(&flow->node);
 305                 if (flow->private) {
 306                         nf2flow = (struct nf2_flow *)flow->private;
 307 
 308                         if (nf2flow->type == NF2_TABLE_EXACT) {
 309                                 nf2_add_free_exact(nf2flow);
 310                         } else if (nf2flow->type == NF2_TABLE_WILDCARD) {
 311                                 nf2_add_free_wildcard(nf2flow);
 312                         }
 313                         flow->private = NULL;
 314                 }
 315                 flow_free(flow);
 316         }
 317         free(nf2flowtab);
 318 
 319         nf2_destroy_exact_freelist();
 320         nf2_destroy_wildcard_freelist();
 321 }

 323 static int
 324 nf2_iterate_flowtable(struct sw_table *flowtab, const struct sw_flow_key *key,
 325                       uint16_t out_port, struct sw_table_position *position,
 326                       int (*callback) (struct sw_flow *, void *),
 327                       void *private)
 328 {
 329         unsigned long start;
 330         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 331         struct sw_flow *flow;
 332         int error = 0;
 333 
 334         start = ~position->private[0];
 335         LIST_FOR_EACH(flow, struct sw_flow, iter_node, &nf2flowtab->iter_flows) {
 336                 if (flow->serial <= start
 337                     && flow_matches_2wild(key, &flow->key)
 338                     && flow_has_out_port(flow, out_port)) {
 339                         error = callback(flow, private);
 340                         if (error != 0) {
 341                                 position->private[0] = ~flow->serial;
 342                                 return error;
 343                         }
 344                 }
 345         }
 346 
 347         return error;
 348 }

 350 static void
 351 nf2_get_flowstats(struct sw_table *flowtab, struct sw_table_stats *stats)
 352 {
 353         struct nf2_flowtable *nf2flowtab = (struct nf2_flowtable *)flowtab;
 354         struct nf2device *dev;
 355         unsigned long int num_matched = 0;
 356         unsigned long int num_missed = 0;
 357 
 358         dev = nf2_get_net_device();
 359         if (dev == NULL) {
 360                 DBG_VERBOSE("Could not open NetFPGA device\n");
 361         } else {
 362                 num_matched = nf2_get_matched_count(dev);
 363                 num_missed = nf2_get_missed_count(dev);
 364                 nf2_free_net_device(dev);
 365         }
 366 
 367         stats->name = "nf2";
 368         stats->wildcards = OPENFLOW_WILDCARD_TABLE_SIZE
 369                          - RESERVED_FOR_CPU2NETFPGA;
 370         stats->n_flows = nf2flowtab->num_flows;
 371         stats->max_flows = nf2flowtab->max_flows;
 372         stats->n_lookup = num_matched + num_missed;
 373         stats->n_matched = num_matched;
 374 }

 376 static int
 377 nf2_get_portstats(of_hw_driver_t *hw_drv, int of_port,
 378                         struct ofp_port_stats *stats)
 379 {
 380         int nf2_port;
 381         struct nf2_port_info *nf2portinfo;
 382         struct nf2device *dev;
 383 
 384         if ((of_port > NF2_PORT_NUM) || (of_port <= 0)) {
 385                 return 1;
 386         }
 387         nf2_port = of_port - 1;
 388 
 389         nf2portinfo = calloc(1, sizeof(struct nf2_port_info));
 390         if (nf2portinfo == NULL) {
 391                 return 1;
 392         }
 393 
 394         dev = nf2_get_net_device();
 395         if (dev == NULL) {
 396                 free(nf2portinfo);
 397                 return 1;
 398         }
 399 
 400         if (nf2_get_port_info(dev, nf2_port, nf2portinfo)) {
 401                 nf2_free_net_device(dev);
 402                 free(nf2portinfo);
 403                 return 1;
 404         }
 405 
 406         stats->rx_packets = (uint64_t)(nf2portinfo->rx_q_num_pkts_stored);
 407         stats->rx_dropped = (uint64_t)(nf2portinfo->rx_q_num_pkts_dropped_full
 408                           + nf2portinfo->rx_q_num_pkts_dropped_bad);
 409         stats->rx_bytes = (uint64_t)(nf2portinfo->rx_q_num_bytes_pushed);
 410         stats->tx_packets = (uint64_t)(nf2portinfo->tx_q_num_pkts_sent);
 411         stats->tx_bytes = (uint64_t)(nf2portinfo->tx_q_num_bytes_pushed);
 412 
 413         /* Not supported */
 414         stats->tx_dropped = -1;
 415         stats->rx_errors = -1;
 416         stats->tx_errors = -1;
 417         stats->rx_frame_err = -1;
 418         stats->rx_over_err = -1;
 419         stats->rx_crc_err = -1;
 420         stats->collisions = -1;
 421 
 422         nf2_free_net_device(dev);
 423         free(nf2portinfo);
 424         return 0;
 425 }

 431 of_hw_driver_t *
 432 new_of_hw_driver(struct datapath *dp)
 433 {
 434         struct sw_table *sw_tab;
 435         of_hw_driver_t *hw_drv;
 436         struct nf2device *dev;
 437         struct nf2_flowtable *nf2flowtab;
 438 
 439         dev = nf2_get_net_device();
 440         if (dev == NULL) {
 441                 return NULL;
 442         }
 443         nf2_reset_card(dev);
 444         nf2_free_net_device(dev);
 445 
 446         nf2flowtab = calloc(1, sizeof(*nf2flowtab));
 447         if (nf2flowtab == NULL) {
 448                 return NULL;
 449         }
 450 
 451         /* These all point to the same place */
 452         hw_drv = &nf2flowtab->hw_driver;
 453         sw_tab = &hw_drv->sw_table;
 454 
 455         sw_tab->n_lookup = 0;
 456         sw_tab->n_matched = 0;
 457 
 458         /* Fill out the function pointers */
 459         sw_tab->lookup = nf2_lookup_flowtable;
 460         sw_tab->insert = nf2_install_flow;
 461         sw_tab->modify = nf2_modify_flow;
 462         sw_tab->has_conflict = nf2_has_conflict;
 463 
 464         sw_tab->delete = nf2_uninstall_flow_wrap;
 465         sw_tab->timeout = nf2_flow_timeout;
 466 
 467         sw_tab->destroy = nf2_destroy_flowtable;
 468         sw_tab->iterate = nf2_iterate_flowtable;
 469         sw_tab->stats = nf2_get_flowstats;
 470 
 471         nf2flowtab->max_flows = OPENFLOW_NF2_EXACT_TABLE_SIZE
 472             + OPENFLOW_WILDCARD_TABLE_SIZE - RESERVED_FOR_CPU2NETFPGA;
 473         nf2flowtab->num_flows = 0;
 474         list_init(&nf2flowtab->flows);
 475         list_init(&nf2flowtab->iter_flows);
 476         nf2flowtab->next_serial = 0;
 477 
 478         if (nf2_init_wildcard_freelist()) {
 479                 DBG_ERROR("Could not create wildcard freelist\n");
 480                 free(nf2flowtab);
 481                 return NULL;
 482         }
 483         if (nf2_write_static_wildcard()) {
 484                 DBG_ERROR("Could not create wildcard freelist\n");
 485                 free(nf2flowtab);
 486                 return NULL;
 487         }
 488         if (nf2_init_exact_freelist()) {
 489                 DBG_ERROR("Could not create exact freelist\n");
 490                 free(nf2flowtab);
 491                 return NULL;
 492         }
 493 
 494         hw_drv->table_stats_get = NULL;
 495         hw_drv->port_stats_get = nf2_get_portstats;
 496         hw_drv->flow_stats_get = NULL;
 497         hw_drv->aggregate_stats_get = NULL;
 498 
 499         hw_drv->port_add = NULL;
 500         hw_drv->port_remove = NULL;
 501         hw_drv->port_link_get = NULL;
 502         hw_drv->port_enable_set = NULL;
 503         hw_drv->port_enable_get = NULL;
 504         hw_drv->port_queue_config = NULL;
 505         hw_drv->port_queue_remove = NULL;
 506         hw_drv->port_change_register = NULL;
 507 
 508         hw_drv->packet_send = NULL;
 509         hw_drv->packet_receive_register = NULL;
 510 
 511         hw_drv->ioctl = NULL;
 512 
 513         return hw_drv;
 514 }
