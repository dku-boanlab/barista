  24 #ifndef  __COLOR_FILTERS_H__
  25 #define  __COLOR_FILTERS_H__
  26 
  27 #define TEMP_COLOR_PREFIX       "___tmp_color_filter___"
  28 /** @file
  29  *  Color filters.
  30  */
  31 
  32 /* Data for a color filter. */
  33 typedef struct _color_filter {
  34         gchar     *filter_name;   /* name of the filter */
  35         gchar     *filter_text;   /* text of the filter expression */
  36         color_t    bg_color;      /* background color for packets that match */
  37         color_t    fg_color;      /* foreground color for packets that match */
  38         gboolean   disabled;      /* set if the filter is disabled */
  39         gboolean   selected;      /* set if the filter is selected in the color dialog box */
  40 
  41         /* only used inside of color_filters.c */
  42         dfilter_t *c_colorfilter; /* compiled filter expression */
  43 
  44         /* only used outside of color_filters.c (beside init) */
  45         void      *edit_dialog;   /* if filter is being edited, dialog
  46                                    * box for it */
  47 } color_filter_t;
  48 
  49 
  50 /** Init the color filters (incl. initial read from file). */
  51 void color_filters_init(void);
  52 
  53 /** Reload the color filters */
  54 void color_filters_reload(void);
  55 
  56 /** Cleanup remaining color filter zombies */
  57 void color_filters_cleanup(void);
  58 
  59 /** Color filters currently used?
  60  *
  61  * @return TRUE, if filters are used
  62  */
  63 gboolean color_filters_used(void);
  64 
  65 /** Are there any temporary coloring filters used?
  66  *
  67  * @return TRUE, if temporary coloring filters are used
  68  */
  69 gboolean tmp_color_filters_used(void);
  70 
  71 /** En-/disable color filters
  72  *
  73  * @param enable TRUE to enable (default)
  74  */
  75 void
  76 color_filters_enable(gboolean enable);
  77 
  78 /** Set the filter string of a temporary color filter
  79  *
  80  * @param filt_nr a number 1-10 pointing to a temporary color
  81  * @param filter the new filter-string
  82  * @param disabled whether the filter-rule should be disabled
  83  */
  84 void
  85 color_filters_set_tmp(guint8 filt_nr, gchar *filter, gboolean disabled);
  86 
  87 /** Reset the temporary color filters
  88  *
  89  */
  90 void
  91 color_filters_reset_tmp(void);
  92 
  93 /* Prime the epan_dissect_t with all the compiler
  94  * color filters of the current filter list. 
  95  *
  96  * @param the epan dissector details
  97  */
  98 void color_filters_prime_edt(epan_dissect_t *edt);
  99 
 100 /** Colorize a specific packet.
 101  *
 102  * @param row the row in the packet list
 103  * @param edt the dissected packet
 104  * @return the matching color filter or NULL
 105  */
 106 color_filter_t *
 107 color_filters_colorize_packet(gint row, epan_dissect_t *edt);
 108

 111 /** Clone the currently active filter list.
 112  *
 113  * @param user_data will be returned by each call to to color_filter_add_cb()
 114  */
 115 void color_filters_clone(gpointer user_data);
 116 
 117 /** Load filters (import) from some other filter file.
 118  *
 119  * @param path the path to the import file
 120  * @param user_data will be returned by each call to to color_filter_add_cb()
 121  * @return TRUE, if read succeeded
 122  */
 123 gboolean color_filters_import(gchar *path, gpointer user_data);
 124 
 125 /** Read filters from the global filter file (not the users file).
 126  *
 127  * @param user_data will be returned by each call to to color_filter_add_cb()
 128  * @return TRUE, if read succeeded
 129  */
 130 gboolean color_filters_read_globals(gpointer user_data);
 131 
 132 /** A color filter was added (while importing).
 133  * (color_filters.c calls this for every filter coming in)
 134  *
 135  * @param colorf the new color filter
 136  * @param user_data from caller
 137  */
 138 void color_filter_add_cb (color_filter_t *colorf, gpointer user_data);
 139 
 140 
 141 
 142 /** Apply a changed filter list.
 143  *
 144  * @param tmp_cfl the temporary color filter list to apply
 145  * @param edit_cfl the edited permanent color filter list to apply
 146  */
 147 void color_filters_apply(GSList *tmp_cfl, GSList *edit_cfl);
 148 
 149 /** Save filters in users filter file.
 150  *
 151  * @param cfl the filter list to write
 152  * @return TRUE if write succeeded
 153  */
 154 gboolean color_filters_write(GSList *cfl);
 155 
 156 /** Save filters (export) to some other filter file.
 157  *
 158  * @param path the path to the filter file
 159  * @param cfl the filter list to write
 160  * @param only_selected TRUE if only the selected filters should be saved
 161  * @return TRUE, if write succeeded
 162  */
 163 gboolean color_filters_export(gchar *path, GSList *cfl, gboolean only_selected);
 164

 167 /** Create a new color filter (g_malloc'ed).
 168  *
 169  * @param name the name of the filter
 170  * @param filter_string the filter string
 171  * @param bg_color background color
 172  * @param fg_color foreground color
 173  * @return the new color filter
 174  */
 175 color_filter_t *color_filter_new(
 176     const gchar *name, const gchar *filter_string,
 177     color_t *bg_color, color_t *fg_color, gboolean disabled);
 178 
 179 /** Delete a single color filter (g_free'ed).
 180  *
 181  * @param colorf the color filter to be removed
 182  */
 183 void color_filter_delete(color_filter_t *colorf);

 188 /** Delete a filter list including all entries.

 191  */
 192 void color_filter_list_delete(GSList **cfl);
