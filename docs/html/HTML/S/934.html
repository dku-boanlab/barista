 525     }
 526 
 527     int i, conn = -1, cluster = -1;
 528     for (i=0; i<compnt_ctx->num_compnts; i++) {
 529         if (strncmp(compnt_ctx->compnt_list[i]->name, "conn", 4) == 0) {

 531                 conn = i;

 534             if (compnt_ctx->compnt_list[i]->status == COMPNT_ENABLED)

 689                 cli_print(cli, "%s is deactivated", compnt_ctx->compnt_list[conn]->name);
 690             }
 691         }
 692     }

 705             // already activated?
 706             if (compnt_ctx->compnt_list[i]->activated) {
 707                 // internal?
 708                 if (compnt_ctx->compnt_list[i]->site == COMPNT_INTERNAL) {
 709                     if (compnt_ctx->compnt_list[i]->cleanup(&compnt_ctx->compnt_list[i]->activated) < 0) {
 710                         cli_print(cli, "Failed to deactivate %s", compnt_ctx->compnt_list[i]->name);
 711                         return -1;
 712                     } else {
 713                         compnt_ctx->compnt_list[i]->activated = FALSE;
 714                         cli_print(cli, "%s is deactivated", compnt_ctx->compnt_list[i]->name);
 715                     }
 716                 }
 717             }
 718         }
 719     }
 720 
 721     // deactivate cluster
 722     if (cluster != -1 && compnt_ctx->compnt_list[cluster]->activated) {
 723         // internal?
 724         if (compnt_ctx->compnt_list[cluster]->site == COMPNT_INTERNAL) {
 725             if (compnt_ctx->compnt_list[cluster]->cleanup(&compnt_ctx->compnt_list[cluster]->activated) < 0) {
 726                 cli_print(cli, "Failed to deactivate %s", compnt_ctx->compnt_list[cluster]->name);

 730                 cli_print(cli, "%s is deactivated", compnt_ctx->compnt_list[cluster]->name);
 731             }
 732         }
 733     }
 734 
 735     compnt_ctx->compnt_on = FALSE;
 736 
 737     return 0;

1199

1248         json_t *data = json_array_get(json, i);
