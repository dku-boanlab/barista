 134 start(struct oflops_context * ctx) {  
 135   struct flow *fl = (struct flow*)xmalloc(sizeof(struct flow));
 136   fl_probe = (struct flow*)xmalloc(sizeof(struct flow));
 137   void *b; //somewhere to store message data
 138   int res, len;
 139   struct timeval now;
 140   char msg[1024];
 141 
 142   //init h
 143   TAILQ_INIT(&head); 
 144 
 145   //Initialize pap-based  tcp flow reassembler for the communication 
 146   //channel
 147   msg_init();  
 148   snprintf(msg, 1024,  "Intializing module %s", name());
 149 
 150   get_mac_address(ctx->channels[OFLOPS_DATA1].dev, local_mac);
 151   printf("%s: %02x:%02x:%02x:%02x:%02x:%02x\n", ctx->channels[OFLOPS_DATA2].dev,
 152          (unsigned char)local_mac[0], (unsigned char)local_mac[1], (unsigned char)local_mac[2], 
 153          (unsigned char)local_mac[3], (unsigned char)local_mac[4], (unsigned char)local_mac[5]);
 154   get_mac_address(ctx->channels[OFLOPS_DATA2].dev, data_mac);
 155 
 156   //log when I start module
 157   gettimeofday(&now, NULL);
 158   oflops_log(now, GENERIC_MSG, msg);
 159   oflops_log(now,GENERIC_MSG , cli_param);
 160 
 161   //start openflow session with switch
 162   make_ofp_hello(&b);
 163   res = oflops_send_of_mesgs(ctx, b, sizeof(struct ofp_hello));
 164   free(b);  
 165   
 166   //send a message to clean up flow tables. 
 167   printf("cleaning up flow table...\n");
 168   res = make_ofp_flow_del(&b);
 169   res = oflops_send_of_mesg(ctx, b);  
 170   free(b);
 171   
 172   /**
 173    * Send flow records to start routing packets.
 174    */
 175   printf("Sending measurement probe flow...\n");
 176   bzero(fl, sizeof(struct flow));
 177   if(table == 0)
 178     fl->mask = 0; //if table is 0 the we generate an exact match */
 179   else 
 180     fl->mask = OFPFW_DL_DST | OFPFW_DL_SRC | (32 << OFPFW_NW_SRC_SHIFT) | 
 181       (8 << OFPFW_NW_DST_SHIFT) | OFPFW_DL_VLAN | OFPFW_TP_DST | OFPFW_NW_PROTO | 
 182       OFPFW_TP_SRC | OFPFW_DL_VLAN_PCP | OFPFW_NW_TOS;

 185   fl->in_port = htons(ctx->channels[OFLOPS_DATA1].of_port);
 186   fl->dl_type = htons(ETHERTYPE_IP);         
 187   memcpy(fl->dl_src, local_mac, 6);
 188   memcpy(fl->dl_dst, "\x00\x15\x17\x7b\x92\x0a", 6);
 189 
 190   fl->dl_vlan = 0xffff;
 191   fl->nw_proto = IPPROTO_UDP;
 192   fl->nw_src =  inet_addr("10.1.1.1");
 193   fl->nw_dst =  inet_addr("10.1.1.2");
 194   fl->tp_src = htons(8080);
 195   fl->tp_dst = htons(8080);
 196   len = make_ofp_flow_add(&b, fl, ctx->channels[OFLOPS_DATA3].of_port, 1, 120);
 197   res = oflops_send_of_mesg(ctx, b);
 198   free(b);
 199 
 200   //store locally the probe to manipulate it later during the modification phase
 201   memcpy(fl_probe, fl, sizeof(struct flow));
 202 
 203   ip_received = xmalloc(flows*sizeof(int));
 204   memset(ip_received, 0, flows*sizeof(int));
 205   ip_received_count++;
 206 
 207   /**
 208    * Shceduling events
 209    */
 210   //send the flow modyfication command in 30 seconds. 
 211   gettimeofday(&now, NULL);
 212   add_time(&now, 20, 0);
 213   oflops_schedule_timer_event(ctx,&now, SND_ACT);
 214 
 215   //get port and cpu status from switch 
 216   gettimeofday(&now, NULL);
 217   add_time(&now, 1, 0);
 218   oflops_schedule_timer_event(ctx,&now, SNMPGET);
 219 
 220   //end process 
 221   gettimeofday(&now, NULL);
 222   add_time(&now, 30, 0);
 223   oflops_schedule_timer_event(ctx,&now, BYESTR);
 224   return 0;

 227 int destroy(struct oflops_context *ctx) {
 228   FILE *out;
 229   struct entry *np;
 230   int  min_id[] = {INT_MAX, INT_MAX, INT_MAX};
 231   int ix[] = {0,0,0};
 232 
 233   int max_id[] = {INT_MIN, INT_MIN, INT_MIN}, ch, i;
 234   char msg[1024];
 235   struct timeval now;
 236   double **data;
 237   uint32_t mean, std, median;
 238   float loss;
 239 
 240   //get what time we start printin output
 241   gettimeofday(&now, NULL);
 242 
 243   //open log file if required
 244   if(print) {
 245     out = fopen(logfile, "w");
 246     if(out == NULL)
 247       perror_and_exit("fopen_logfile", 1);
 248   }
 249 
 250   //init tmp data storage
 251   data = xmalloc(3*sizeof(double *));
 252   for(ch = 0; ch < 3; ch++) {
 253     data[ch] = xmalloc(count[ch]*sizeof(double));
 254   }
 255 
 256   // for every measurement save the delay in the appropriate entry on the 
 257   // measurement matrix
 258   for (np = head.tqh_first; np != NULL; np = np->entries.tqe_next) {
 259     ch = np->ch - 1;
 260     min_id[ch] = (np->id < min_id[ch])?np->id:min_id[ch];
 261     max_id[ch] = (np->id > max_id[ch])?np->id:max_id[ch];
 262     data[ch][ix[ch]++] = time_diff(&np->snd, &np->rcv);
 263     //print also packet details on otuput if required
 264     if(print)
 265       if(fprintf(out, "%lu;%lu.%06lu;%lu.%06lu;%d\n", 
 266                  (long unsigned int)np->id,  
 267                  (long unsigned int)np->snd.tv_sec, 
 268                  (long unsigned int)np->snd.tv_usec,
 269                  (long unsigned int)np->rcv.tv_sec, 
 270                  (long unsigned int)np->rcv.tv_usec,  np->ch) < 0)  
 271         perror_and_exit("fprintf fail", 1); 
 272 
 273     free(np);
 274   }
 275   
 276   for(ch = 0; ch < 3; ch++) {
 277     if(ix[ch] > 0) {
 278       gsl_sort (data[ch], 1, ix[ch]);
 279       mean = (uint32_t)gsl_stats_mean(data[ch], 1, ix[ch]);
 280       std = (uint32_t)sqrt(gsl_stats_variance(data[ch], 1, ix[ch]));
 281       median = (uint32_t)gsl_stats_median_from_sorted_data (data[ch], 1, ix[ch]);
 282       loss = (float)ix[ch]/(float)(max_id[ch] - min_id[ch]);
 283       
 284       snprintf(msg, 1024, "statistics:port:%d:%u:%u:%u:%.4f:%d", 
 285                ctx->channels[ch + 1].of_port, mean, median, std, loss, count[ch]);
 286       printf("statistics:port:%d:%u:%u:%u:%.4f:%d\n", 
 287              ctx->channels[ch + 1].of_port, mean, median, std, loss, count[ch]);
 288       oflops_log(now, GENERIC_MSG, msg);
 289     }
 290   }
 291   
 292   free(data[0]);
 293   data[0] = xmalloc(echo_data_count * sizeof(double));
 294   for(i = 1; i <= echo_data_count; i++) {
 295     snprintf(msg, 1024, "OFP_ECHO:%d:%ld.%06ld:%ld.%06ld", i, echo_data[i].snd.tv_sec, echo_data[i].snd.tv_usec, 
 296              echo_data[i].rcv.tv_sec, echo_data[i].rcv.tv_usec);
 297     oflops_log(now, GENERIC_MSG, msg);
 298     data[0][i-1] = time_diff(& echo_data[i].snd, &echo_data[i].rcv);
 299   }
 300   if(echo_data_count > 0) {
 301     gsl_sort (data[0], 1, echo_data_count);
 302     mean = (uint32_t)gsl_stats_mean(data[0], 1, echo_data_count);
 303     std = (uint32_t)sqrt(gsl_stats_variance(data[0], 1, echo_data_count));
 304     median = (uint32_t)gsl_stats_median_from_sorted_data (data[0], 1, echo_data_count);
 305     printf("statistics:echo:%u:%u:%u:%d\n", mean, median, std, echo_data_count);
 306     snprintf(msg, 1024, "statistics:echo:%u:%u:%u:%d", mean, median, std, echo_data_count);
 307     oflops_log(now, GENERIC_MSG, msg);
 308   }
 309 
 310   return 0;
 311 }

 317 int handle_timer_event(struct oflops_context * ctx, struct timer_event *te) {  
 318   char *str = te->arg; 
 319   int len, i, inc=1;
 320   void *b;
 321   struct timeval now;
 322   struct in_addr ip_addr;
 323 
 324   //terminate process 
 325   if (strcmp(str, BYESTR) == 0) {
 326     printf("terminating test....\n");
 327     oflops_end_test(ctx,1);
 328     finished = 0;    
 329     return 0;    
 330   } else if (strcmp(str, SND_ACT) == 0) {
 331     //first create new rules
 332     send_flow_mod = 1;
 333     //if(table == 1)
 334     //  inc = inc << 8;
 335     if(table == 0)
 336       fl_probe->mask = 0; //if table is 0 the we generate an exact match */
 337     else 
 338       fl_probe->mask = OFPFW_DL_DST | OFPFW_DL_SRC | (32 << OFPFW_NW_SRC_SHIFT) | 
 339         (32 << OFPFW_NW_DST_SHIFT) | OFPFW_DL_VLAN | OFPFW_TP_DST | OFPFW_NW_PROTO | 
 340         OFPFW_TP_SRC | OFPFW_DL_VLAN_PCP | OFPFW_NW_TOS;
 341 
 342     memcpy(fl_probe->dl_src, data_mac, 6);
 343     memcpy(fl_probe->dl_dst, "\x00\x15\x17\x7b\x92\x0a", 6);
 344     fl_probe->in_port = htons(ctx->channels[OFLOPS_DATA2].of_port);
 345     ip_addr.s_addr = inet_addr(network);
 346     ip_addr.s_addr =  ntohl(ip_addr.s_addr);
 347     for(i=0; i< flows; i++) {
 348     fl_probe->nw_dst =  htonl(ip_addr.s_addr);
 349       len = make_ofp_flow_add(&b, fl_probe, OFPP_IN_PORT, 1, 1200);
 350       oflops_send_of_mesgs(ctx, b, len);
 351       free(b);
 352       ip_addr.s_addr += inc;
 353     }
 354     memcpy(fl_probe->dl_src, local_mac, 6);
 355     fl_probe->in_port = htons(ctx->channels[OFLOPS_DATA1].of_port);
 356     fl_probe->nw_dst =  inet_addr("10.1.1.2");
 357     len = make_ofp_flow_add(&b, fl_probe, OFPP_IN_PORT, 1, 1200);
 358     oflops_send_of_mesg(ctx, b);
 359     free(b);
 360     printf("sending flow modifications ....\n"); 
 361 
 362   } else if(strcmp(str, SNMPGET) == 0) {
 363     for(i = 0; i < ctx->cpuOID_count; i++) {
 364       oflops_snmp_get(ctx, ctx->cpuOID[i], ctx->cpuOID_len[i]);
 365     }
 366     for(i=0;i<ctx->n_channels;i++) {
 367       oflops_snmp_get(ctx, ctx->channels[i].inOID, ctx->channels[i].inOID_len);
 368       oflops_snmp_get(ctx, ctx->channels[i].outOID, ctx->channels[i].outOID_len);
 369     }      
 370     gettimeofday(&now, NULL);
 371     add_time(&now, 1, 0);
 372     oflops_schedule_timer_event(ctx,&now, SNMPGET);
 373   } else if(strcmp(str, SEND_ECHO_REQ) == 0) {
 374     struct ofp_header *p;
 375     make_ofp_echo_req(&p);
 376     p->xid = htonl(++echo_data_count);
 377     oflops_send_of_mesg(ctx, p);
 378     free(p);
 379     gettimeofday(&now, NULL);
 380     add_time(&now, 4, 0);
 381     oflops_schedule_timer_event(ctx,&now, SEND_ECHO_REQ);
 382     
 383   }
 384   return 0;
 385 }

 404 /** Handle pcap event.
 405  * @param ctx pointer to opaque context

 407  * @param ch enumeration of channel that pcap event is triggered

 409 int 
 410 handle_pcap_event(struct oflops_context *ctx, struct pcap_event * pe, oflops_channel_name ch) {
 411   struct pktgen_hdr *pktgen;
 412   int dir, len;
 413   struct ofp_header *ofp;
 414   struct pcap_event *ofp_msg;
 415   struct ofp_error_msg *err_p = NULL;
 416   char msg[1024];
 417 
 418   if (ch == OFLOPS_CONTROL) {
 419     dir = append_data_to_flow(pe->data,pe->pcaphdr);
 420     while(contains_next_msg(dir) > 0) {
 421       len = get_next_msg(dir, &ofp_msg);
 422       ofp = (struct ofp_header *)ofp_msg->data;
 423       switch(ofp->type) {
 424       case OFPT_FLOW_MOD:
 425         if((send_flow_mod) && (!stored_flow_mod_time)) {
 426           memcpy(&flow_mod_timestamp, &pe->pcaphdr.ts, sizeof(struct timeval));
 427           stored_flow_mod_time = 1;
 428         }
 429         oflops_log(pe->pcaphdr.ts,OFPT_FLOW_MOD_ADD, "flow modification send");
 430         break;  
 431       case OFPT_ERROR:
 432         err_p = (struct ofp_error_msg *)ofp;
 433         snprintf(msg, 1024, "OFPT_ERROR(type: %d, code: %d)", ntohs(err_p->type), ntohs(err_p->code));
 434         oflops_log(pe->pcaphdr.ts, OFPT_ERROR_MSG, msg);
 435         fprintf(stderr, "%s\n", msg);
 436         break;   
 437       case  OFPT_ECHO_REQUEST:
 438         err_p = (struct ofp_error_msg *)ofp;
 439         printf("%ld.%06ld: OFPT_ECHO_REQUEST %d\n", pe->pcaphdr.ts.tv_sec, pe->pcaphdr.ts.tv_usec, ntohl(ofp->xid));
 440         if((ntohl(ofp->xid) < 100) && (ntohl(ofp->xid) > 0))
 441           memcpy(&echo_data[ntohl(ofp->xid)].snd, &pe->pcaphdr.ts, sizeof(struct timeval));
 442         break;   
 443       case  OFPT_ECHO_REPLY:
 444         err_p = (struct ofp_error_msg *)ofp;
 445         printf("%ld.%06ld: OFPT_ECHO_REPLY %d\n", pe->pcaphdr.ts.tv_sec, pe->pcaphdr.ts.tv_usec, ntohl(ofp->xid));
 446         if( (ntohl(ofp->xid) < 100) && (ntohl(ofp->xid) > 0))
 447           memcpy(&echo_data[ntohl(ofp->xid)].rcv, &pe->pcaphdr.ts, sizeof(struct timeval));
 448         break;  
 449       }
 450     }
 451   } else if ((ch == OFLOPS_DATA1) || (ch == OFLOPS_DATA2) || (ch == OFLOPS_DATA3)) {
 452     struct flow fl;
 453     struct timeval now;
 454     pktgen = extract_pktgen_pkt(pe->data, pe->pcaphdr.caplen, &fl);
 455     if((ch == OFLOPS_DATA1) && (!first_pkt)) {
 456       oflops_log(pe->pcaphdr.ts, GENERIC_MSG, "FIRST_PKT_RCV");
 457       oflops_log(pe->pcaphdr.ts, GENERIC_MSG, msg);
 458       printf("INSERT_DELAY:%d\n", time_diff(&flow_mod_timestamp, &pe->pcaphdr.ts));
 459       snprintf(msg, 1024, "INSERT_DELAY:%d", time_diff(&flow_mod_timestamp, &pe->pcaphdr.ts));
 460       oflops_log(pe->pcaphdr.ts, GENERIC_MSG, msg);
 461       first_pkt = 1;
 462       gettimeofday(&now, NULL);
 463       add_time(&now, 1, 0);
 464       //oflops_schedule_timer_event(ctx,&now, BYESTR);
 465     } else if (ch == OFLOPS_DATA2) {
 466       int id = ntohl(fl.nw_dst) - ntohl(inet_addr(network));
 467       //printf("id %d %x %x\n", id, ntohl(fl.nw_dst),  ntohl(inet_addr(network)));
 468       if ((id >= 0) && (id < flows) && (!ip_received[id])) {
 469         ip_received_count++;
 470         ip_received[id] = 1;
 471         if (ip_received_count == flows) {
 472           gettimeofday(&now, NULL);
 473           add_time(&now, 1, 0);
 474           oflops_schedule_timer_event(ctx,&now, BYESTR);
 475           printf("Received all packets to channel 2\n");
 476           oflops_log(pe->pcaphdr.ts, GENERIC_MSG, "LAST_PKT_RCV");
 477           oflops_log(pe->pcaphdr.ts, GENERIC_MSG, msg);
 478 
 479           printf("COMPLETE_INSERT_DELAY:%u\n", time_diff(&flow_mod_timestamp, &pe->pcaphdr.ts));
 480           snprintf(msg, 1024, "INSERT_DELAY:%u", time_diff(&flow_mod_timestamp, &pe->pcaphdr.ts));
 481           oflops_log(pe->pcaphdr.ts, GENERIC_MSG, msg);
 482         }
 483       }
 484     }
 485     if(htonl(pktgen->seq_num) % 100000 == 0)
 486       printf("data packet received %d\n", htonl(pktgen->seq_num));
 487     
 488     struct entry *n1 = malloc(sizeof(struct entry));
 489     n1->snd.tv_sec = htonl(pktgen->tv_sec);
 490     n1->snd.tv_usec = htonl(pktgen->tv_usec);
 491     memcpy(&n1->rcv, &pe->pcaphdr.ts, sizeof(struct timeval));
 492     n1->id = htonl(pktgen->seq_num);
 493     n1->ch = ch;
 494     count[ch - 1]++;
 495     TAILQ_INSERT_TAIL(&head, n1, entries);
 496   }
 497   return 0;
 498 }

 527 int 
 528 handle_snmp_event(struct oflops_context * ctx, struct snmp_event * se) {
 529   netsnmp_variable_list *vars;
 530   int len = 1024, i;
 531   char msg[1024], log[1024];
 532   struct timeval now;
 533 
 534   for(vars = se->pdu->variables; vars; vars = vars->next_variable)  {
 535     snprint_value(msg, len, vars->name, vars->name_length, vars);
 536 
 537     for (i = 0; i < ctx->cpuOID_count; i++) {
 538       if((vars->name_length == ctx->cpuOID_len[i]) &&
 539          (memcmp(vars->name, ctx->cpuOID[i],  ctx->cpuOID_len[i] * sizeof(oid)) == 0) ) {
 540         snprintf(log, len, "cpu:%ld:%d:%s",
 541                  se->pdu->reqid, 
 542                  (int)vars->name[ vars->name_length - 1], msg);
 543         oflops_log(now, SNMP_MSG, log);
 544       }
 545     }
 546 
 547     for(i=0;i<ctx->n_channels;i++) {
 548       if((vars->name_length == ctx->channels[i].inOID_len) &&
 549          (memcmp(vars->name, ctx->channels[i].inOID,  
 550                  ctx->channels[i].inOID_len * sizeof(oid)) == 0) ) {
 551         snprintf(log, len, "port:rx:%ld:%d:%s",  
 552                  se->pdu->reqid, 
 553                  (int)(int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 554         oflops_log(now, SNMP_MSG, log);
 555         break;
 556       }
 557         
 558       if((vars->name_length == ctx->channels[i].outOID_len) &&
 559          (memcmp(vars->name, ctx->channels[i].outOID,  
 560                  ctx->channels[i].outOID_len * sizeof(oid))==0) ) {
 561         snprintf(log, len, "port:tx:%ld:%d:%s",  
 562                  se->pdu->reqid, 
 563                  (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 564         //      printf("port %d : tx %s pkts\n",  (int)ctx->channels[i].outOID[ctx->channels[i].outOID_len-1], msg);
 565         oflops_log(now, SNMP_MSG, log);
 566         break;
 567       }
 568     } //for
 569   }// if cpu
 570   return 0;
 571 }

 622 /**
 623  * Initialization code with parameters
 624  * @param ctx 
 625  */
 626 int init(struct oflops_context *ctx, char * config_str) {
 627   char *pos = NULL;
 628   char *param = config_str;
 629   char *value = NULL;
 630   struct timeval now;
 631 
 632   //init counters
 633   finished = 0;
 634 
 635   gettimeofday(&now, NULL);
 636 
 637   cli_param = strdup(config_str);
 638 
 639 
 640   while(*config_str == ' ') {
 641     config_str++;
 642   }
 643   param = config_str;
 644   while(1) {
 645     pos = index(param, ' ');
 646 
 647     if((pos == NULL)) {
 648       if (*param != '\0') {
 649         pos = param + strlen(param) + 1;
 650       } else
 651         break;
 652     }
 653     *pos='\0';
 654     pos++;
 655     value = index(param,'=');
 656     *value = '\0';
 657     value++;
 658     //fprintf(stderr, "param = %s, value = %s\n", param, value);
 659     if(value != NULL) {
 660       if(strcmp(param, "pkt_size") == 0) {
 661         //parse int to get pkt size
 662         pkt_size = strtol(value, NULL, 0);
 663         if((pkt_size < MIN_PKT_SIZE) && (pkt_size > MAX_PKT_SIZE))
 664           perror_and_exit("Invalid packet size value", 1);
 665       }  else if(strcmp(param, "probe_rate") == 0) {
 666         //parse int to get measurement probe rate
 667         proberate = strtol(value, NULL, 0);
 668         if((proberate <= 0) || (proberate >= 1010)) 
 669           perror_and_exit("Invalid probe rate param(Value between 1 and 1010)", 1);
 670       }  else if(strcmp(param, "data_rate") == 0) {
 671         //parse int to get measurement probe rate
 672         datarate = strtol(value, NULL, 0);
 673         if((datarate <= 0) || (datarate >= 1010)) 
 674           perror_and_exit("Invalid data rate param(Value between 1 and 1010)", 1);
 675       } else if(strcmp(param, "table") == 0) {
 676         //parse int to get pkt size
 677         table = strtol(value, NULL, 0);
 678         if((table < 0) && (table > 2))  
 679           perror_and_exit("Invalid table number", 1);
 680       } else if(strcmp(param, "flows") == 0) {
 681         //parse int to get pkt size
 682         flows = strtol(value, NULL, 0);
 683         if(flows <= 0)  
 684           perror_and_exit("Invalid flow number", 1);
 685       } else if(strcmp(param, "print") == 0) {
 686         //parse int to get pkt size
 687         print = strtol(value, NULL, 0);
 688       } else 
 689         fprintf(stderr, "Invalid parameter:%s\n", param);
 690       param = pos;
 691     }
 692   } 
 693 
 694   //calculate sendind interval
 695   probe_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (proberate * mbits_to_bits);
 696   fprintf(stderr, "Sending probe interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 697           (uint32_t)probe_snd_interval, (uint32_t)pkt_size, (uint32_t)proberate);
 698   data_snd_interval = (pkt_size * byte_to_bits * sec_to_usec) / (datarate * mbits_to_bits);
 699   fprintf(stderr, "Sending probe interval : %u usec (pkt_size: %u bytes, rate: %u Mbits/sec )\n", 
 700           (uint32_t)data_snd_interval, (uint32_t)pkt_size, (uint32_t)datarate);
 701   return 0;
 702 }
