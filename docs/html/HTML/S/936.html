  48 /**
  49  * \brief Function to daemonize the Barista NOS
  50  * \return None
  51  */
  52 static void daemonize(void)
  53 {
  54     pid_t pid = fork();
  55     if (pid == -1) {
  56         PERROR("fork");
  57         exit(-1);
  58     } else if (pid != 0) {
  59         exit(0);
  60     }
  61 
  62     if (setsid() == -1) {
  63         PERROR("setsid");
  64     }
  65 
  66     signal(SIGHUP, SIG_IGN);
  67     pid = fork();
  68     if (pid == -1) {
  69         PERROR("fork");
  70         exit(-1);
  71     } else if (pid != 0) {
  72         exit(0);
  73     }
  74 
  75     umask(0);
  76 
  77     close(STDIN_FILENO);
  78     close(STDOUT_FILENO);
  79     close(STDERR_FILENO);
  80 
  81     if (open("/dev/null", O_RDONLY) == -1) {
  82         PERROR("open");
  83         exit(0);
  84     }
  85     if (open("/dev/null", O_WRONLY) == -1) {
  86         PERROR("open");
  87         exit(0);
  88     }
  89     if (open("/dev/null", O_RDWR) == -1) {
  90         PERROR("open");
  91         exit(0);
  92     }
  93 }

 149 /**
 150  * \brief Function to start the Barista NOS
 151  * \param argc The number of arguments
 152  * \param argv Arguments
 153  */
 154 int main(int argc, char **argv)
 155 {
 156     int opt;
 157     int autostart = FALSE;
 158     int daemon = FALSE;
 159 
 160     print_logo();
 161 
 162     if (argc >= 2) {
 163         while ((opt = getopt(argc, argv, "c:a:r:")) != -1) {
 164             switch (opt) {
 165             case 'c':
 166                 memset(conf_file, 0, __CONF_WORD_LEN);
 167                 strcpy(conf_file, optarg);
 168                 break;
 169             case 'a':
 170                 memset(app_conf_file, 0, __CONF_WORD_LEN);
 171                 strcpy(app_conf_file, optarg);
 172                 break;
 173             case 'r':
 174                 if (strcmp(optarg, "auto") == 0) {
 175                     autostart = TRUE;
 176                 } else if (strcmp(optarg, "base") == 0) {
 177                     memset(conf_file, 0, __CONF_WORD_LEN);
 178                     strcpy(conf_file, COMPNT_BASE_CONFIG_FILE);
 179 
 180                     memset(app_conf_file, 0, __CONF_WORD_LEN);
 181                     strcpy(app_conf_file, APP_BASE_CONFIG_FILE);
 182                 } else if (strcmp(optarg, "daemon") == 0) {
 183                     daemon = TRUE;
 184                 } else {
 185                     print_usage(argv[0]);
 186                     return -1;
 187                 }
 188                 break;
 189             default:
 190                 print_usage(argv[0]);
 191                 return -1;
 192             }
 193         }
 194     }
 195 
 196     // debug messages
 197     DEBUG("Structures in type.h\n");
 198     DEBUG("raw_msg_t: %lu\n", sizeof(raw_msg_t));
 199     DEBUG("msg_t: %lu\n", sizeof(msg_t));
 200     DEBUG("switch_t: %lu\n", sizeof(switch_t));
 201     DEBUG("port_t: %lu\n", sizeof(port_t));
 202     DEBUG("host_t: %lu\n", sizeof(host_t));
 203     DEBUG("pktin_t: %lu\n", sizeof(pktin_t));
 204     DEBUG("action_t: %lu\n", sizeof(action_t));
 205     DEBUG("pktout_t: %lu\n", sizeof(pktout_t));
 206     DEBUG("flow_t: %lu\n", sizeof(flow_t));
 207     DEBUG("traffic_t: %lu\n", sizeof(traffic_t));
 208     DEBUG("resource_t: %lu\n", sizeof(resource_t));
 209     DEBUG("odp_t: %lu\n", sizeof(odp_t));
 210     DEBUG("meta_event_t: %lu\n", sizeof(meta_event_t));
 211 
 212     // kill external packet I/O engine
 213     int pid;
 214     char *argb[] = {"pkill", "-9", "barista_sb", NULL};
 215     posix_spawn(&pid, "/usr/bin/pkill", NULL, NULL, argb, NULL);
 216 
 217     // kill resource monitor
 218     char *argr[] = {"pkill", "-9", "barista_rsm", NULL};
 219     posix_spawn(&pid, "/usr/bin/pkill", NULL, NULL, argr, NULL);
 220 
 221     if (daemon == TRUE)
 222         daemonize();
 223 
 224     // initialize context
 225     ctx_init(&ctx);
 226 
 227     // set autostart
 228     if (autostart == TRUE)
 229         ctx.autostart = TRUE;
 230 
 231     // store the conf file in the context
 232     strcpy(ctx.conf_file, conf_file);
 233 
 234     // store the app conf file in the context
 235     strcpy(ctx.app_conf_file, app_conf_file);
 236 
 237     // initialize event handler
 238     event_init(&ctx);
 239 
 240     // initialize app event handler
 241     app_event_init(&ctx);
 242 
 243     // execute cli
 244     start_cli(&ctx);
 245 
 246     return 0;
 247 }
