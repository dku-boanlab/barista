  33                     match++;
  34                     break;
  35                 }
  36             }
  37         }
  38

  40             cnt++;
  41 
  42             int j;

  44                 if (odp[i].port[j] == 0) // No more port

  48                     match++;
  49                     break;
  50                 }
  51             }
  52         }
