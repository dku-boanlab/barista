  60 /**
  61  * \brief Function to encode binary data
  62  * \param bin Binary data
  63  * \param len The length of the data
  64  * \return Encoded string (newly allocated)
  65  */
  66 char *base64_encode(const char *bin, int len)
  67 {
  68     int i = 0, j = 0;
  69     int bs = len - (len % 3);
  70     int remain_size = len - bs;
  71     char buffer[MAX_BUFFER_SIZE];
  72 
  73     for (; i < bs; i += 3) {
  74         buffer[j++] = enc_table[(bin[i] >> 2) & 0x3f]; 
  75         buffer[j++] = enc_table[((bin[i] & 0x03) << 4) + ((bin[i+1] >> 4) & 0x0f)];
  76         buffer[j++] = enc_table[((bin[i+1] & 0x0f) << 2) + ((bin[i+2] >> 6) & 0x03)];
  77         buffer[j++] = enc_table[bin[i+2] & 0x3f];
  78     }
  79 
  80     if (remain_size != 0) {
  81         buffer[j++] = enc_table[bin[i] >> 2 & 0x3f];
  82 
  83         if (len - i == 2) {
  84             buffer[j++] = enc_table[((bin[i] & 0x03) << 4) + ((bin[i + 1] >> 4) & 0x0f)];
  85             buffer[j++] = enc_table[(bin[i + 1] & 0x0f) << 2];
  86         }else {
  87             buffer[j++] = enc_table[(bin[i] & 0x03) << 4];
  88             buffer[j++] = enc_table[PADDING];
  89         }
  90 
  91         buffer[j++] = enc_table[PADDING];
  92     }
  93 
  94     buffer[j++] = '\0';
  95 
  96     int size = sizeof(char) * j;
  97 
  98     char *result = (char *)MALLOC(size);
  99     memset(result, 0, size);
 100     memmove(result, buffer, size);
 101 
 102     return result;
 103 }

 105 /**
 106  * \brief Function to decode an encoded string
 107  * \param encoded Encoded string
 108  * \return Binary data (newly allocated)
 109  */
 110 char *base64_decode(const char *encoded)
 111 {
 112     int i = 0, j = 0;
 113     int len = strlen(encoded);
 114 
 115     char buffer[MAX_BUFFER_SIZE];
 116     
 117     for (; i < len; i+=4) {
 118         int c0 = dec_table[(int)encoded[i]];
 119         int c1 = dec_table[(int)encoded[i+1]];
 120         int c2 = dec_table[(int)encoded[i+2]];
 121         int c3 = dec_table[(int)encoded[i+3]];
 122         
 123         buffer[j++] = (c0 << 2) | (c1 >> 4);
 124         buffer[j++] = (c1 << 4) | (c2 >> 2);
 125         buffer[j++] = (c2 << 6) | c3;
 126     }
 127 
 128     buffer[j++] = '\0';
 129 
 130     int size = sizeof(char) * j;
 131     char *result = (char *)MALLOC(size);
 132     memset(result, 0, size);
 133     memmove(result, buffer, size);
 134 
 135     return result;
 136 }
