  95 /*
  96  * 802.11i defines an extended IV for use with non-WEP ciphers.
  97  * When the EXTIV bit is set in the key id byte an additional
  98  * 4 bytes immediately follow the IV for TKIP.  For CCMP the
  99  * EXTIV bit is likewise set but the 8 bytes represent the
 100  * CCMP header rather than IV+extended-IV.
 101  */
 102 #define AIRPDCAP_RSNA_EXTIV     0x20
 103 #define AIRPDCAP_RSNA_EXTIVLEN  4       /* extended IV length */
 104 #define AIRPDCAP_RSNA_MICLEN    8       /* trailing MIC */

 169 /**
 170  * Given an 802.11 packet, either extract its key data (in the case of
 171  * WPA handshaking) or try to decrypt it.
 172  * @param ctx [IN] Pointer to the current context
 173  * @param data [IN] Pointer to a buffer with an 802.11 frame, including MAC
 174  *   header and payload
 175  * @param data_off [IN] Payload offset (aka the MAC header length)
 176  * @param data_len [IN] Total length of the MAC header and the payload
 177  * @param decrypt_data [OUT] Pointer to a buffer that will contain
 178  *   decrypted data
 179  * @param decrypt_len [OUT] Length of decrypted data
 180  * @param key [OUT] Pointer to a preallocated key structure containing
 181  *   the key used during the decryption process (if done). If this parameter
 182  *   is set to NULL, the key will be not returned.
 183  * @param mngHandshake [IN] If TRUE this function will manage the 4-way
 184  *   handshake for WPA/WPA2
 185  * @param mngDecrypt [IN] If TRUE this function will manage the WEP or
 186  * WPA/WPA2 decryption
 187  * @return
 188  * - AIRPDCAP_RET_SUCCESS: Decryption has been done (decrypt_data and
 189  *   decrypt_length will contain the packet data decrypted and the length of
 190  *   the new packet)
 191  * - AIRPDCAP_RET_SUCCESS_HANDSHAKE: A step of the 4-way handshake for
 192  *   WPA key has been successfully done
 193  * - AIRPDCAP_RET_NO_DATA: The packet is not a data packet
 194  * - AIRPDCAP_RET_WRONG_DATA_SIZE: The size of the packet is below the
 195  *   accepted minimum
 196  * - AIRPDCAP_RET_REQ_DATA: Required data is not available and the
 197  *   processing must be interrupted
 198  * - AIRPDCAP_RET_NO_VALID_HANDSHAKE: The authentication is not for WPA or RSNA
 199  * - AIRPDCAP_RET_NO_DATA_ENCRYPTED: No encrypted data
 200  * - AIRPDCAP_RET_UNSUCCESS: No decryption has been done (decrypt_data
 201  *   and decrypt_length will be not modified).
 202  * Some other errors could be:
 203  *   data not correct
 204  *   data not encrypted
 205  *   key handshake, not encryption
 206  *   decryption not successful
 207  *   key handshake not correct
 208  *   replay check not successful
 209  * @note
 210  * The decrypted buffer should be allocated for a size equal or greater
 211  * than the packet data buffer size. Before decryption process original
 212  * data is copied in the buffer pointed by decrypt_data not to modify the
 213  * original packet.
 214  * @note
 215  * The length of decrypted data will consider the entire 802.11 frame
 216  * (thus the MAC header, the frame body and the recalculated FCS -if
 217  * initially present-)
 218  * @note
 219  * This function is not thread-safe when used in parallel with context
 220  *  management functions on the same context.
 221  */
 222 extern INT AirPDcapPacketProcess(
 223         PAIRPDCAP_CONTEXT ctx,
 224         const guint8 *data,
 225         const guint data_off,
 226         const guint data_len,
 227         UCHAR *decrypt_data,
 228         guint32 *decrypt_len,
 229         PAIRPDCAP_KEY_ITEM key,
 230         gboolean mngHandshake,
 231         gboolean mngDecrypt)
 232         ;

 234 /**
 235  * It sets a new keys collection to use during packet processing.
 236  * Any key should be well-formed, thus: it should have a defined key
 237  * type and the specified length should be conforming WEP or WPA/WPA2
 238  * standards. A general WEP keys could be of any length (in the range
 239  * defined in AIRPDCAP_KEY_ITEM), if a specific WEP key is used, the
 240  * length of the key will be the one specified in 802.11i-2004 (40 bits or
 241  * 104 bits).
 242  * For WPA/WPA2 the password (passphrase and SSID), the PSK and the PMK
 243  * are in alternative, as explain in the AIRPDCAP_KEY_ITEM structure
 244  * description.
 245  * @param ctx [IN] pointer to the current context
 246  * @param keys [IN] an array of keys to set.
 247  * @param keys_nr [IN] the size of the keys array
 248  * @return The number of keys correctly inserted in the current database.
 249  * @note Before inserting new keys, the current database will be cleaned.
 250  * @note
 251  * This function is not thread-safe when used in parallel with context
 252  * management functions and the packet process function on the same
 253  * context.
 254  */
 255 extern INT AirPDcapSetKeys(
 256         PAIRPDCAP_CONTEXT ctx,
 257         AIRPDCAP_KEY_ITEM keys[],
 258         const size_t keys_nr)
 259         ;
 260 
 261 /**
 262  * It gets the keys collection fom the specified context.
 263  * @param ctx [IN] pointer to the current context
 264  * @param key [IN] a preallocated array of keys to be returned
 265  * @param keys_nr [IN] the number of keys to return (the key array must
 266  * be able to contain at least keys_nr keys)
 267  * @return The number of keys returned
 268  * @note
 269  * Any key could be modified, as stated in the AIRPDCAP_KEY_ITEM description.
 270  * @note
 271  * This function is not thread-safe when used in parallel with context
