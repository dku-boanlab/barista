  32 /**
  33  * \brief Function to pop a MAC entry from a MAC pool
  34  * \return Empty MAC entry
  35  */
  36 static mac_entry_t *mac_dequeue(void)
  37 {
  38     mac_entry_t *new = NULL;
  39 
  40     pthread_spin_lock(&mac_q.lock);
  41 
  42     if (mac_q.head == NULL) {
  43         pthread_spin_unlock(&mac_q.lock);
  44 
  45         new = (mac_entry_t *)CALLOC(1, sizeof(mac_entry_t));
  46         if (new == NULL) {
  47             PERROR("calloc");
  48             return NULL;
  49         }
  50 
  51         pthread_spin_lock(&mac_q.lock);
  52         mac_q.size--;
  53         pthread_spin_unlock(&mac_q.lock);
  54 
  55         return new;
  56     } else if (mac_q.head == mac_q.tail) {
  57         new = mac_q.head;
  58         mac_q.head = mac_q.tail = NULL;
  59     } else {
  60         new = mac_q.head;
  61         mac_q.head = mac_q.head->next;
  62         mac_q.head->prev = NULL;
  63     }
  64 
  65     mac_q.size--;
  66 
  67     pthread_spin_unlock(&mac_q.lock);
  68 
  69     memset(new, 0, sizeof(mac_entry_t));
  70 
  71     return new;
  72 }

  74 /**
  75  * \brief Function to push a used MAC entry into a MAC pool
  76  * \param old Used MAC entry
  77  */
  78 static int mac_enqueue(mac_entry_t *old)
  79 {
  80     if (old == NULL)
  81         return -1;
  82 
  83     memset(old, 0, sizeof(mac_entry_t));
  84 
  85     pthread_spin_lock(&mac_q.lock);
  86 
  87     if (mac_q.size < 0) {
  88         mac_q.size++;
  89 
  90         pthread_spin_unlock(&mac_q.lock);
  91 
  92         FREE(old);
  93 
  94         return 0;
  95     }
  96 
  97     if (mac_q.head == NULL && mac_q.tail == NULL) {
  98         mac_q.head = mac_q.tail = old;
  99     } else {
 100         mac_q.tail->next = old;
 101         old->prev = mac_q.tail;
 102         mac_q.tail = old;
 103     }
 104 
 105     mac_q.size++;
 106 
 107     pthread_spin_unlock(&mac_q.lock);
 108 
 109     return 0;
 110 }

 112 /**
 113  * \brief Function to initialize a MAC pool
 114  * \return None
 115  */
 116 static int mac_q_init(void)
 117 {
 118     memset(&mac_q, 0, sizeof(mac_queue_t));
 119     pthread_spin_init(&mac_q.lock, PTHREAD_PROCESS_PRIVATE);
 120 
 121     int i;
 122     for (i=0; i<MAC_PRE_ALLOC; i++) {
 123         mac_entry_t *new = (mac_entry_t *)MALLOC(sizeof(mac_entry_t));
 124         if (new == NULL) {
 125             PERROR("malloc");
 126             return -1;
 127         }
 128 
 129         mac_enqueue(new);
 130     }
 131 
 132     return 0;
 133 }

 135 /**
 136  * \brief Function to destroy a MAC pool
 137  * \return None
 138  */
 139 static int mac_q_destroy(void)
 140 {
 141     pthread_spin_lock(&mac_q.lock);
 142 
 143     mac_entry_t *curr = mac_q.head;
 144     while (curr != NULL) {
 145         mac_entry_t *tmp = curr;
 146         curr = curr->next;
 147         FREE(tmp);
 148     }
 149 
 150     pthread_spin_unlock(&mac_q.lock);
 151     pthread_spin_destroy(&mac_q.lock);
 152 
 153     memset(&mac_q, 0, sizeof(mac_queue_t));
 154 
 155     return 0;
 156 }
