 130 static bool
 131 protocol_stat_remote_packet_cb(struct relay *relay, void *context_)
 132 {
 133         struct protocol_stat_context *context = context_;
 134         struct rconn *mgmt_rconn = relay->halves[HALF_REMOTE].rconn;
 135         struct ofpbuf *qbuf = relay->halves[HALF_REMOTE].rxbuf;
 136         struct ofpbuf *pbuf = NULL;
 137         struct private_vxhdr *qvxhdr = NULL;
 138         struct private_vxhdr *pvxhdr = NULL;
 139         struct private_vxopt *qvxopt = NULL;
 140         struct private_vxopt *pvxopt = NULL;
 141         struct ofpstat *ofps = NULL;
 142         struct ofpstat ofps_rcvd;
 143         struct ofpstat ofps_sent;
 144         int error = 0;
 145 
 146         if (qbuf->size < sizeof(*qvxhdr))
 147                 return false;
 148         qvxhdr = qbuf->data;
 149         if (qvxhdr->ofp_hdr.type != OFPT_VENDOR)
 150                 return false;
 151         if (ntohl(qvxhdr->ofp_vxid) != PRIVATE_VENDOR_ID) {
 152                 return false;
 153         }
 154         qvxopt = (struct private_vxopt *)(qvxhdr + 1);
 155         if (ntohs(qvxopt->pvo_type) != PRIVATEOPT_PROTOCOL_STATS_REQUEST) {
 156                 return true;
 157         }
 158 
 159         pvxhdr = make_openflow_xid(sizeof(*pvxhdr) + sizeof(*pvxopt)
 160                                    + (sizeof(*ofps) * 2),
 161                                    OFPT_VENDOR, qvxhdr->ofp_hdr.xid, &pbuf);
 162         pvxopt = (struct private_vxopt *)(pvxhdr + 1);
 163         pvxhdr->ofp_vxid = qvxhdr->ofp_vxid;
 164         pvxopt->pvo_type = htons(PRIVATEOPT_PROTOCOL_STATS_REPLY);
 165         pvxopt->pvo_len = htons(sizeof(*ofps) * 2);
 166 
 167         rconn_update_protocol_stat(context->remote_rconn,
 168                                    &ofps_rcvd, &ofps_sent);
 169         ofps = (struct ofpstat *)((uint8_t *)(pvxhdr + 1) + sizeof(*pvxopt));
 170         COPY_OFP_STAT(ofps, &ofps_rcvd);
 171         ofps = ofps + 1;
 172         COPY_OFP_STAT(ofps, &ofps_sent);
 173 
 174         error = rconn_send(mgmt_rconn, pbuf, NULL);
 175         if (error && error != EAGAIN) {
 176                 VLOG_WARN("send failed (%s)", strerror(error));
 177         }
 178 
 179         return true;
 180 }

 182 void
 183 protocol_stat_start(struct secchan *secchan, const struct settings *settings,
 184                     struct rconn *local_rconn, struct rconn *remote_rconn)
 185 {
 186         struct protocol_stat_context *context = NULL;
 187         static struct hook_class protocol_stat_hook_class = {
 188                 NULL,           /* local_packet_cb */
 189                 protocol_stat_remote_packet_cb, /* remote_packet_cb */
 190                 NULL,           /* periodic_cb */
 191                 NULL,           /* wait_cb */
 192                 NULL,           /* closing_cb */
 193         };
 194 
 195         context = xmalloc(sizeof(*context));
 196         context->settings = settings;
 197         context->secchan = secchan;
 198         context->local_rconn = local_rconn;
 199         context->remote_rconn = remote_rconn;
 200         memset(&context->ofps_rcvd, 0, sizeof(context->ofps_rcvd));
 201         memset(&context->ofps_sent, 0, sizeof(context->ofps_sent));
 202 
 203         add_hook(secchan, &protocol_stat_hook_class, context);
 204 }
