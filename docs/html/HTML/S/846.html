 115 static void
 116 inc_error_notification(struct ofpstat *ifps, struct ofp_header *hdr)
 117 {
 118         struct ofp_error_msg *errmsg = (struct ofp_error_msg *)hdr;
 119         uint16_t errtype = ntohs(errmsg->type);
 120         uint16_t errcode = ntohs(errmsg->code);
 121 
 122         switch (errtype) {
 123         case OFPET_HELLO_FAILED:
 124                 INC_IFP_STAT(ifps, ofps_error_type.hello_fail);
 125                 switch (errcode) {
 126                 case OFPHFC_INCOMPATIBLE:
 127                         INC_IFP_STAT(ifps, ofps_error_code.hf_incompat);
 128                         break;
 129                 case OFPHFC_EPERM:
 130                         INC_IFP_STAT(ifps, ofps_error_code.hf_eperm);
 131                         break;
 132                 default:
 133                         INC_IFP_STAT(ifps, ofps_error_code.unknown);
 134                         break;
 135                 }
 136                 break;
 137         case OFPET_BAD_REQUEST:
 138                 INC_IFP_STAT(ifps, ofps_error_type.bad_request);
 139                 switch (errcode) {
 140                 case OFPBRC_BAD_VERSION:
 141                         INC_IFP_STAT(ifps, ofps_error_code.br_bad_version);
 142                         break;
 143                 case OFPBRC_BAD_TYPE:
 144                         INC_IFP_STAT(ifps, ofps_error_code.br_bad_type);
 145                         break;
 146                 case OFPBRC_BAD_STAT:
 147                         INC_IFP_STAT(ifps, ofps_error_code.br_bad_stat);
 148                         break;
 149                 case OFPBRC_BAD_VENDOR:
 150                         INC_IFP_STAT(ifps, ofps_error_code.br_bad_vendor);
 151                         break;
 152                 case OFPBRC_EPERM:
 153                         INC_IFP_STAT(ifps, ofps_error_code.br_eperm);
 154                         break;
 155                 default:
 156                         INC_IFP_STAT(ifps, ofps_error_code.unknown);
 157                         break;
 158                 }
 159                 break;
 160         case OFPET_BAD_ACTION:
 161                 INC_IFP_STAT(ifps, ofps_error_type.bad_action);
 162                 switch (errcode) {
 163                 case OFPBAC_BAD_TYPE:
 164                         INC_IFP_STAT(ifps, ofps_error_code.ba_bad_type);
 165                         break;
 166                 case OFPBAC_BAD_LEN:
 167                         INC_IFP_STAT(ifps, ofps_error_code.ba_bad_len);
 168                         break;
 169                 case OFPBAC_BAD_VENDOR:
 170                         INC_IFP_STAT(ifps, ofps_error_code.ba_bad_vendor);
 171                         break;
 172                 case OFPBAC_BAD_VENDOR_TYPE:
 173                         INC_IFP_STAT(ifps, ofps_error_code.ba_bad_vendor_type);
 174                         break;
 175                 case OFPBAC_BAD_OUT_PORT:
 176                         INC_IFP_STAT(ifps, ofps_error_code.ba_bad_out_port);
 177                         break;
 178                 case OFPBAC_BAD_ARGUMENT:
 179                         INC_IFP_STAT(ifps, ofps_error_code.ba_bad_argument);
 180                         break;
 181                 case OFPBAC_EPERM:
 182                         INC_IFP_STAT(ifps, ofps_error_code.ba_eperm);
 183                         break;
 184                 default:
 185                         INC_IFP_STAT(ifps, ofps_error_code.unknown);
 186                         break;
 187                 }
 188                 break;
 189         case OFPET_FLOW_MOD_FAILED:
 190                 INC_IFP_STAT(ifps, ofps_error_type.flow_mod_fail);
 191                 switch (errcode) {
 192                 case OFPFMFC_ALL_TABLES_FULL:
 193                         INC_IFP_STAT(ifps, ofps_error_code.fmf_all_tables_full);
 194                         break;
 195                 case OFPFMFC_OVERLAP:
 196                         INC_IFP_STAT(ifps, ofps_error_code.fmf_overlap);
 197                         break;
 198                 case OFPFMFC_EPERM:
 199                         INC_IFP_STAT(ifps, ofps_error_code.fmf_eperm);
 200                         break;
 201                 case OFPFMFC_BAD_EMERG_TIMEOUT:
 202                         INC_IFP_STAT(ifps, ofps_error_code.fmf_emerg);
 203                         break;
 204                 default:
 205                         INC_IFP_STAT(ifps, ofps_error_code.unknown);
 206                         break;
 207                 }
 208                 break;
 209         default:
 210                 INC_IFP_STAT(ifps, ofps_error_type.unknown);
 211                 break;
 212         }
 213 }

 215 static void
 216 inc_flow_manipulation(struct ofpstat *ifps, struct ofp_header *hdr)
 217 {
 218         struct ofp_flow_mod *flowmodmsg = (struct ofp_flow_mod *)hdr;
 219         uint16_t flowmodops = ntohs(flowmodmsg->command);
 220 
 221         switch (flowmodops) {
 222         case OFPFC_ADD:
 223                 INC_IFP_STAT(ifps, ofps_flow_mod_ops.add);
 224                 break;
 225         case OFPFC_MODIFY:
 226                 INC_IFP_STAT(ifps, ofps_flow_mod_ops.modify);
 227                 break;
 228         case OFPFC_MODIFY_STRICT:
 229                 INC_IFP_STAT(ifps, ofps_flow_mod_ops.modify_strict);
 230                 break;
 231         case OFPFC_DELETE:
 232                 INC_IFP_STAT(ifps, ofps_flow_mod_ops.delete);
 233                 break;
 234         case OFPFC_DELETE_STRICT:
 235                 INC_IFP_STAT(ifps, ofps_flow_mod_ops.delete_strict);
 236                 break;
 237         default:
 238                 INC_IFP_STAT(ifps, ofps_flow_mod_ops.unknown);
 239                 break;
 240         }
 241 }

 243 void
 244 ofpstat_inc_protocol_stat(struct ofpstat *ifps, struct ofp_header *hdr)
 245 {
 246         ++ifps->ofps_total;
 247         inc_protocol_message(ifps, hdr);
 248 
 249         switch (hdr->type) {
 250         case OFPT_ERROR:
 251                 inc_error_notification(ifps, hdr);
 252                 break;
 253         case OFPT_FLOW_MOD:
 254                 inc_flow_manipulation(ifps, hdr);
 255                 break;
 256         default:
 257                 break;
 258         }
 259 }
