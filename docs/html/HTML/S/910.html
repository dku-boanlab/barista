  40 /* Saturating addition: overflow yields UINT_MAX. */
  41 static inline unsigned int
  42 sat_add(unsigned int x, unsigned int y)
  43 {
  44     return x + y >= x ? x + y : UINT_MAX;
  45 }
  46 
  47 /* Saturating subtraction: underflow yields 0. */
  48 static inline unsigned int
  49 sat_sub(unsigned int x, unsigned int y)
  50 {
  51     return x >= y ? x - y : 0;
  52 }
  53 
  54 /* Saturating multiplication: overflow yields UINT_MAX. */
  55 static inline unsigned int
  56 sat_mul(unsigned int x, unsigned int y)
  57 {
  58     return (!y ? 0
  59             : x <= UINT_MAX / y ? x * y
  60             : UINT_MAX);
  61 }
