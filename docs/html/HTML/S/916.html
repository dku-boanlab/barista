  73 #define LIST_FOR_EACH(ITER, STRUCT, MEMBER, LIST)                   \
  74     for (ITER = CONTAINER_OF((LIST)->next, STRUCT, MEMBER);         \
  75          &(ITER)->MEMBER != (LIST);                                 \
  76          ITER = CONTAINER_OF((ITER)->MEMBER.next, STRUCT, MEMBER))
  77 #define LIST_FOR_EACH_REVERSE(ITER, STRUCT, MEMBER, LIST)           \
  78     for (ITER = CONTAINER_OF((LIST)->prev, STRUCT, MEMBER);         \
  79          &(ITER)->MEMBER != (LIST);                                 \
  80          ITER = CONTAINER_OF((ITER)->MEMBER.prev, STRUCT, MEMBER))
  81 #define LIST_FOR_EACH_SAFE(ITER, NEXT, STRUCT, MEMBER, LIST)        \
  82     for (ITER = CONTAINER_OF((LIST)->next, STRUCT, MEMBER);         \
  83          (NEXT = CONTAINER_OF((ITER)->MEMBER.next, STRUCT, MEMBER), \
  84           &(ITER)->MEMBER != (LIST));                               \
  85          ITER = NEXT)
