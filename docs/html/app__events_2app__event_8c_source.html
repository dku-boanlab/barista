242 int app_event_init(ctx_t *ctx)
243 {
244  av_ctx = ctx;
245 
246  if (av_ctx->av_on == FALSE) {
247  pthread_t thread;
248 
249 #ifdef __ENABLE_META_EVENTS
250  if (pthread_create(&thread, NULL, &meta_app_events, NULL) < 0) {
251  PERROR("pthread_create");
252  return -1;
253  }
254 #endif /* __ENABLE_META_EVENTS */
255 
256  av_ctx->av_on = TRUE;
257 
258  // pull
259 
260  av_pull_in_ctx = zmq_ctx_new();
261  av_pull_in_sock = zmq_socket(av_pull_in_ctx, ZMQ_PULL);
262 
263  if (zmq_bind(av_pull_in_sock, EXT_APP_PULL_ADDR)) {
264  PERROR("zmq_bind");
265  return -1;
266  }
267 
268  av_pull_out_ctx = zmq_ctx_new();
269  av_pull_out_sock = zmq_socket(av_pull_out_ctx, ZMQ_PUSH);
270 
271  if (zmq_bind(av_pull_out_sock, "inproc://av_pull_workers")) {
272  PERROR("zmq_bind");
273  return -1;
274  }
275 
276  if (pthread_create(&thread, NULL, &receive_app_events, NULL) < 0) {
277  PERROR("pthread_create");
278  return -1;
279  }
280 
281  int i;
282  for (i=0; i<__NUM_PULL_THREADS; i++) {
283  if (pthread_create(&thread, NULL, &deliver_app_events, NULL) < 0) {
284  PERROR("pthread_create");
285  return -1;
286  }
287  }
288 
289  // reply
290 
291  av_rep_ctx = zmq_ctx_new();
292  av_rep_app = zmq_socket(av_rep_ctx, ZMQ_ROUTER);
293 
294  if (zmq_bind(av_rep_app, EXT_APP_REPLY_ADDR)) {
295  PERROR("zmq_bind");
296  return -1;
297  }
298 
299  av_rep_work = zmq_socket(av_rep_ctx, ZMQ_DEALER);
300 
301  if (zmq_bind(av_rep_work, "inproc://av_reply_workers")) {
302  PERROR("zmq_bind");
303  return -1;
304  }
305 
306  for (i=0; i<__NUM_REP_THREADS; i++) {
307  if (pthread_create(&thread, NULL, &reply_app_events, NULL) < 0) {
308  PERROR("pthread_create");
309  return -1;
310  }
311  }
312 
313  if (pthread_create(&thread, NULL, &reply_proxy, ctx) < 0) {
314  PERROR("pthread_create");
315  return -1;
316  }
317  }
318 
319  return 0;
320 }
