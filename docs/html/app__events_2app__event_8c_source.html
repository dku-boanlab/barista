210  //char **args = split_line(meta[j].cmd);
211  }
212  break;
213  case META_LTE: // <=
214  if (num_event <= meta[j].threshold) {
215  //char **args = split_line(meta[j].cmd);

217  break;

245 int app_event_init(ctx_t *ctx)
246 {
247  av_ctx = ctx;
248 
249  if (av_on == FALSE) {
250  pthread_t thread;
251  if (pthread_create(&thread, NULL, &meta_app_events, NULL) < 0) {
252  PERROR("pthread_create");
253  return -1;
254  }
255 
256  av_on = TRUE;
257 
258  // pull (outside)
259 
260  av_pull_ctx = zmq_ctx_new();
261  av_pull_sock = zmq_socket(av_pull_ctx, ZMQ_PULL);
262 
263  if (zmq_bind(av_pull_sock, EXT_APP_PULL_ADDR)) {
264  PERROR("zmq_bind");
265  return -1;
266  }
267 
268  int i;
269  for (i=0; i<__NUM_THREADS; i++) {
270  if (pthread_create(&thread, NULL, &receive_app_events, NULL) < 0) {
271  PERROR("pthread_create");
272  return -1;
273  }
274  }
275 
276  // reply
277 
278  av_rep_ctx = zmq_ctx_new();
279  av_rep_app = zmq_socket(av_rep_ctx, ZMQ_ROUTER);
280 
281  if (zmq_bind(av_rep_app, EXT_APP_REPLY_ADDR)) {
282  PERROR("zmq_bind");
283  return -1;
284  }
285 
286  av_rep_work = zmq_socket(av_rep_ctx, ZMQ_DEALER);
287 
288  if (zmq_bind(av_rep_work, "inproc://av_reply_workers")) {
289  PERROR("zmq_bind");
290  return -1;
291  }
292 
293  for (i=0; i<__NUM_THREADS; i++) {
294  if (pthread_create(&thread, NULL, &reply_app_events, NULL) < 0) {
295  PERROR("pthread_create");
296  return -1;
297  }
298  }
299 
300  if (pthread_create(&thread, NULL, &reply_proxy, ctx) < 0) {
301  PERROR("pthread_create");
302  return -1;
303  }
304  }
305 
306  return 0;
307 }
