114  PERROR("json_loads");

117  PERROR("json_is_array");

119  return -1;

121 
122  int i;

124  json_t *data = json_array_get(json, i);
125

127  json_t *j_name = json_object_get(data, "name");
128  if (json_is_string(j_name)) {
129  strcpy(name, json_string_value(j_name));
130  }
131 
132  char role[__CONF_WORD_LEN] = {0};
133  json_t *j_role = json_object_get(data, "role");
134  if (json_is_string(j_role)) {
135  strcpy(role, json_string_value(j_role));
136  }
137 
138  if (strlen(name) == 0) {
139  ALOG_ERROR(RBAC_ID, "No application name");
140  return -1;

142  app_to_role[app_to_role_cnt].id = hash_func((uint32_t *)&name, __HASHING_NAME_LENGTH);

144  }
145 
146  if (strlen(role) == 0) {

148  } else {

150  app_to_role[app_to_role_cnt].role = APP_BASE;
151  else if (strcmp(role, "network") == 0)
152  app_to_role[app_to_role_cnt].role = APP_NETWORK;

162  }
163  }

264 static int check_app_role(uint32_t id, uint16_t type)
265 {
266  int i;

268  if (app_to_role[i].id == id) { // found the matched app

270  return -1;
