253 static void *do_tasks(void *null)
254 {
255  int wsock, bytes, done = 0;
256  uint8_t rx_buf[BUFFER_SIZE];
257 
258  pthread_mutex_lock(&queue_mutex);
259 
260  while (listening) {
261  struct timeval now;
262  gettimeofday(&now, NULL);
263 
264  struct timespec time_wait;
265  time_wait.tv_sec = now.tv_sec+1;
266  time_wait.tv_nsec = now.tv_usec;
267 
268  pthread_cond_timedwait(&queue_cond, &queue_mutex, &time_wait);
269

271  if (listening == FALSE) { // clean up
272  pthread_mutex_unlock(&queue_mutex);
273  break;
274  }

278  pthread_mutex_unlock(&queue_mutex);

281  bytes = 1;

291  break;
292  }

295  if (bytes == -1) {
296  done = 1;
297  break;
298  }
299  }
300 
301  if (done) {
302  // closed connection
303  closed_connection(wsock);
304  clean_context(wsock);
305  close(wsock);
306  } else {
307  if (relink_epoll(epoll, wsock, EPOLLIN | EPOLLET | EPOLLONESHOT) < 0) {

310  clean_context(wsock);

312  }
313  }

315  pthread_mutex_lock(&queue_mutex);
316 
317  if (get_qsize() > 0) {

319  }
320  }
321

323 }
