254  if (num_event < meta[j].threshold) {
255  //char **args = split_line(meta[j].cmd);
256  }
257  break;
258  case META_LTE: // <=
259  if (num_event <= meta[j].threshold) {
260  //char **args = split_line(meta[j].cmd);
261  }
262  break;
263  case META_EQ: // ==

265  //char **args = split_line(meta[j].cmd);

290 int event_init(ctx_t *ctx)
291 {
292  ev_ctx = ctx;
293 
294  if (ev_on == FALSE) {
295  pthread_t thread;
296  if (pthread_create(&thread, NULL, &meta_events, NULL) < 0) {
297  PERROR("pthread_create");
298  return -1;
299  }
300 
301  ev_on = TRUE;
302 
303  // pull (outside)
304 
305  ev_pull_ctx = zmq_ctx_new();
306  ev_pull_sock = zmq_socket(ev_pull_ctx, ZMQ_PULL);
307 
308  const int timeout = 1000;
309  zmq_setsockopt(ev_pull_sock, ZMQ_RCVTIMEO, &timeout, sizeof(int));
310 
311  if (zmq_bind(ev_pull_sock, EXT_COMP_PULL_ADDR)) {
312  PERROR("zmq_bind");
313  return -1;
314  }
315 
316  int i;
317  for (i=0; i<__NUM_THREADS; i++) {
318  if (pthread_create(&thread, NULL, &receive_events, NULL) < 0) {
319  PERROR("pthread_create");
320  return -1;
321  }
322  }
323 
324  // reply
325 
326  ev_rep_ctx = zmq_ctx_new();
327  ev_rep_comp = zmq_socket(ev_rep_ctx, ZMQ_ROUTER);
328  if (zmq_bind(ev_rep_comp, EXT_COMP_REPLY_ADDR)) {
329  PERROR("zmq_bind");
330  return -1;
331  }
332 
333  ev_rep_work = zmq_socket(ev_rep_ctx, ZMQ_DEALER);
334  if (zmq_bind(ev_rep_work, "inproc://ev_reply_workers")) {
335  PERROR("zmq_bind");
336  return -1;
337  }
338 
339  for (i=0; i<__NUM_THREADS; i++) {
340  if (pthread_create(&thread, NULL, &reply_events, NULL) < 0) {
341  PERROR("pthread_create");
342  return -1;
343  }
344  }
345 
346  if (pthread_create(&thread, NULL, &reply_proxy, ctx) < 0) {
347  PERROR("pthread_create");
348  return -1;
349  }
350  }
351 
352  return 0;
353 }
