285 int event_init(ctx_t *ctx)
286 {
287  ev_ctx = ctx;
288 
289  if (ev_ctx->ev_on == FALSE) {
290  pthread_t thread;
291 
292 #ifdef __EANBLE_META_EVENTS
293  if (pthread_create(&thread, NULL, &meta_events, NULL) < 0) {
294  PERROR("pthread_create");
295  return -1;
296  }
297 #endif /* __ENABLE_META_EVENTS */
298 
299  ev_ctx->ev_on = TRUE;
300 
301  // pull
302 
303  ev_pull_in_ctx = zmq_ctx_new();
304  ev_pull_in_sock = zmq_socket(ev_pull_in_ctx, ZMQ_PULL);
305 
306  if (zmq_bind(ev_pull_in_sock, EXT_COMP_PULL_ADDR)) {
307  PERROR("zmq_bind");
308  return -1;
309  }
310 
311  ev_pull_out_ctx = zmq_ctx_new();
312  ev_pull_out_sock = zmq_socket(ev_pull_out_ctx, ZMQ_PUSH);
313 
314  if (zmq_bind(ev_pull_out_sock, "inproc://ev_pull_workers")) {
315  PERROR("zmq_bind");
316  return -1;
317  }
318 
319  if (pthread_create(&thread, NULL, &receive_events, NULL) < 0) {
320  PERROR("pthread_create");
321  return -1;
322  }
323 
324  int i;
325  for (i=0; i<__NUM_PULL_THREADS; i++) {
326  if (pthread_create(&thread, NULL, &deliver_events, NULL) < 0) {
327  PERROR("pthread_create");
328  return -1;
329  }
330  }
331 
332  // reply
333 
334  ev_rep_ctx = zmq_ctx_new();
335  ev_rep_comp = zmq_socket(ev_rep_ctx, ZMQ_ROUTER);
336 
337  if (zmq_bind(ev_rep_comp, EXT_COMP_REPLY_ADDR)) {
338  PERROR("zmq_bind");
339  return -1;
340  }
341 
342  ev_rep_work = zmq_socket(ev_rep_ctx, ZMQ_DEALER);
343 
344  if (zmq_bind(ev_rep_work, "inproc://ev_reply_workers")) {
345  PERROR("zmq_bind");
346  return -1;
347  }
348 
349  for (i=0; i<__NUM_REP_THREADS; i++) {
350  if (pthread_create(&thread, NULL, &reply_events, NULL) < 0) {
351  PERROR("pthread_create");
352  return -1;
353  }
354  }
355 
356  if (pthread_create(&thread, NULL, &reply_proxy, ctx) < 0) {
357  PERROR("pthread_create");
358  return -1;
359  }
360  }
361 
362  return 0;
363 }
