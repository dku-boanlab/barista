318 static void *reply_events(void *null)
319 {
320  void *recv = zmq_socket(ev_rep_ctx, ZMQ_REP);
321  zmq_connect(recv, "inproc://ev_reply_workers");
322 
323  while (ev_on) {
324  msg_t msg = {0};
325 
326  zmq_recv(recv, &msg, sizeof(msg_t), 0);
327 
328  if (!ev_on) break;
329  else if (msg.id == 0) continue;
330 
331  switch (msg.type) {
332 
333  // request-response events
334 
335  case EV_SW_GET_DPID:
336  sw_rw_raise(msg.id, EV_SW_GET_DPID, sizeof(switch_t), (switch_t *)msg.data);
337  break;
338  case EV_SW_GET_FD:

340  break;

343  break;

347  }

349  zmq_send(recv, &msg, sizeof(msg_t), 0);

546 int event_init(ctx_t *ctx)
547 {
548  ev_ctx = ctx;
549 
550  if (ev_on == FALSE) {
551  pthread_t thread;
552  if (pthread_create(&thread, NULL, &meta_events, NULL) < 0) {
553  PERROR("pthread_create");
554  return -1;
555  }
556 
557  ev_on = TRUE;
558 
559  // pull (outside)
560 
561  ev_pull_ctx = zmq_ctx_new();
562  ev_pull_sock = zmq_socket(ev_pull_ctx, ZMQ_PULL);
563 
564  const int timeout = 1000;
565  zmq_setsockopt(ev_pull_sock, ZMQ_RCVTIMEO, &timeout, sizeof(int));
566 
567  if (zmq_bind(ev_pull_sock, EXT_COMP_PULL_ADDR)) {
568  PERROR("zmq_bind");
569  return -1;
570  }
571 
572  int i;
573  for (i=0; i<__NUM_THREADS; i++) {
574  if (pthread_create(&thread, NULL, &receive_events, NULL) < 0) {
575  PERROR("pthread_create");
576  return -1;
577  }
578  }
579 
580  // reply
581 
582  ev_rep_ctx = zmq_ctx_new();
583  ev_rep_comp = zmq_socket(ev_rep_ctx, ZMQ_ROUTER);
584  if (zmq_bind(ev_rep_comp, EXT_COMP_REPLY_ADDR)) {
585  PERROR("zmq_bind");
586  return -1;
587  }
588
