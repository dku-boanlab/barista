122 static int cli_load(struct cli_def *cli, UNUSED(const char *command), char *argv[], int argc)
123 {
124  // load components
125  if (component_load(cli, cli_ctx)) {
126  cli_print(cli, "Failed to load component configurations");
127  return CLI_ERROR;
128  }
129 
130  // load applications
131  if (application_load(cli, cli_ctx)) {
132  cli_print(cli, "Failed to load application configurations");
133  return CLI_ERROR;
134  }
135 
136  // autostart
137  if (component_activate(cli, "log")) {
138  cli_print(cli, "Failed to activate the log component");
139  return CLI_ERROR;
140  }
141 
142  // automatically load operator-defined policies
143  FILE *fp = fopen(DEFAULT_ODP_FILE, "r");
144  if (fp != NULL) {
145  char buf[__CONF_STR_LEN] = {0};
146 
147  cli_print(cli, "Operator-defined policy file: %s", DEFAULT_ODP_FILE);
148 
149  while (fgets(buf, __CONF_STR_LEN-1, fp) != NULL) {
150  if (buf[0] == '#') continue;
151 
152  int cnt = 0;
153  char *args[__CONF_ARGC + 1] = {0};
154 
155  str2args(buf, &cnt, args, __CONF_ARGC);
156 
157  if (cnt != 3) continue;
158 
159  if (strcmp(args[0], "component") == 0) {
160  component_add_policy(cli, args[1], args[2]);
161  } else if (strcmp(args[0], "application") == 0) {
162  application_add_policy(cli, args[1], args[2]);
163  }
164  }
165 
166  fclose(fp);
167  } else {
168  cli_print(cli, "No operator-defined policy");
169  }
170 
171  return CLI_OK;
172 }
