140 static int traffic_stat_summary(cli_t *cli, char *seconds)
141 {
142  int sec = atoi(seconds);
143  if (sec > TRAFFIC_MGMT_HISTORY) sec = TRAFFIC_MGMT_HISTORY;
144  else if (sec > tr_history_ptr) sec = tr_history_ptr;
145 
146  int ptr = tr_history_ptr - sec;
147  if (ptr < 0) ptr = TRAFFIC_MGMT_HISTORY + ptr;
148 
149  traffic_t tr = {0};
150 
151  int i;
152  for (i=0; i<sec; i++) {
153  tr.in_pkt_cnt += tr_history[ptr % TRAFFIC_MGMT_HISTORY].in_pkt_cnt;
154  tr.in_byte_cnt += tr_history[ptr % TRAFFIC_MGMT_HISTORY].in_byte_cnt;
155 
156  tr.out_pkt_cnt += tr_history[ptr % TRAFFIC_MGMT_HISTORY].out_pkt_cnt;
157  tr.out_byte_cnt += tr_history[ptr % TRAFFIC_MGMT_HISTORY].out_byte_cnt;
158 
159  ptr++;
160  }
161 
162  double in_pkt_cnt = tr.in_pkt_cnt * 1.0 / sec;
163  double in_byte_cnt = tr.in_byte_cnt * 1.0 / sec;
164 
165  double out_pkt_cnt = tr.out_pkt_cnt * 1.0 / sec;
166  double out_byte_cnt = tr.out_byte_cnt * 1.0 / sec;
167 
168  cli_print(cli, "<Traffic Statistics for %d seconds>", sec);
169  cli_print(cli, " - Cumulative values");
170  cli_print(cli, " Inbound packet count : %lu packets", tr.in_pkt_cnt);
171  cli_print(cli, " Inbound byte count : %lu bytes", tr.in_byte_cnt);
172  cli_print(cli, " Outbound packet count: %lu packets", tr.out_pkt_cnt);
173  cli_print(cli, " Outbound byte count : %lu bytes", tr.out_byte_cnt);
174  cli_print(cli, " - Average values");
175  cli_print(cli, " Inbound packet count : %.2f packets", in_pkt_cnt);
176  cli_print(cli, " Inbound byte count : %.2f bytes", in_byte_cnt);
177  cli_print(cli, " Outbound packet count: %.2f packets", out_pkt_cnt);
178  cli_print(cli, " Outbound byte count : %.2f bytes", out_byte_cnt);
179 
180  return 0;
181 }
