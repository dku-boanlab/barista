640 {

642

646  ev_push_sock = zmq_socket(ev_push_ctx, ZMQ_PUSH);
647 
648  const int timeout = 1000;
649  zmq_setsockopt(ev_push_sock, ZMQ_SNDTIMEO, &timeout, sizeof(int));
650 
651  if (zmq_connect(ev_push_sock, EXT_COMP_PULL_ADDR)) {
652  PERROR("zmq_connect");
653  return -1;
654  }
655 
656  // Pull (upstream, intstream)
657 
658  ev_pull_ctx = zmq_ctx_new();
659  ev_pull_sock = zmq_socket(ev_pull_ctx, ZMQ_PULL);
660 
661  zmq_setsockopt(ev_pull_sock, ZMQ_RCVTIMEO, &timeout, sizeof(int));
662 
663  if (zmq_connect(ev_pull_sock, TARGET_COMP_PULL_ADDR)) {
664  PERROR("zmq_connect");
665  return -1;
666  }
667 
668  int i;
