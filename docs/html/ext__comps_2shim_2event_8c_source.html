66  memmove(msg.data, _data, _size); \

69  } \

72 #define EV_WRITE_MSG(_id, _type, _size, _data) { \
73  msg_t msg = {0}; \
74  msg.id = _id; \
75  msg.type = _type; \
76  memmove(msg.data, _data, _size); \
77  if (ev_req_sock) { \
78  if (zmq_send(ev_req_sock, &msg, sizeof(msg_t), 0) != -1) { \
79  zmq_recv(ev_req_sock, &msg, sizeof(msg_t), 0); \
80  memmove(_data, msg.data, _size); \
81  } \
82  } \
83 }

648 
649  const int timeout = 1000;
650  zmq_setsockopt(ev_push_sock, ZMQ_SNDTIMEO, &timeout, sizeof(int));
651 
652  if (zmq_connect(ev_push_sock, EXT_COMP_PULL_ADDR)) {
653  PERROR("zmq_connect");
654  return -1;
655  }
656

661
