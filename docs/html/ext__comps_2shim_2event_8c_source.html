49 static int handshake(uint32_t id, char *name)
50 {
51  void *sock = zmq_socket(ev_req_ctx, ZMQ_REQ);
52  if (zmq_connect(sock, EXT_COMP_REPLY_ADDR)) {
53  PERROR("zmq_connect");
54  return -1;
55  } else {
56  char out_str[__MAX_EXT_MSG_SIZE];
57  sprintf(out_str, "#{\"id\": %u, \"name\": \"%s\"}", compnt.component_id, compnt.name);
58 
59  zmq_msg_t out_msg;
60  zmq_msg_init_size(&out_msg, strlen(out_str));
61  memcpy(zmq_msg_data(&out_msg), out_str, strlen(out_str));
62 
63  if (zmq_send(sock, out_str, strlen(out_str)+1, 0) < 0) {
64  PERROR("zmq_send");
65  zmq_msg_close(&out_msg);
66  zmq_close(sock);
67  return -1;
68  }
69 
70  zmq_msg_close(&out_msg);
71 
72  zmq_msg_t in_msg;
73  zmq_msg_init(&in_msg);
74  int zmq_ret = zmq_msg_recv(&in_msg, sock, 0);
75  if (zmq_ret < 0) {
76  PERROR("zmq_recv");
77  zmq_msg_close(&in_msg);
78  zmq_close(sock);
79  return -1;
80  } else {
81  char *in_str = zmq_msg_data(&in_msg);
82  in_str[zmq_ret] = '\0';
83 
84  if (strcmp(in_str, "#{\"return\": 0}") != 0) {
85  zmq_msg_close(&in_msg);
86  zmq_close(sock);
87  return -1;
88  }
89  }
90 
91  zmq_msg_close(&in_msg);
92 
93  zmq_close(sock);
94  }
95 
96  return 0;
97 }
