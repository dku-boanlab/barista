242  }

250

253  break;
254  }
255 
256  switch_t *curr = switch_table[sw->fd];
257  switch_t *tmp = curr;
258

260  curr->prev->next = curr->next;
261  curr->next->prev = curr->prev;
262  } else if (curr->prev == NULL && curr->next != NULL) {
263  switches.head = curr->next;
264  curr->next->prev = NULL;
265  } else if (curr->prev != NULL && curr->next == NULL) {
266  switches.tail = curr->prev;
267  curr->prev->next = NULL;

271  }
272

274 
275  pthread_rwlock_unlock(&switches.lock);
276

367
