45 
46  int i;

48  if (av_ctx->app_list[i]->app_id == id) {
49  if (strcmp(av_ctx->app_list[i]->name, name) == 0) {
50  av_ctx->app_list[i]->activated = TRUE;
51  json_decref(json);
52  return 0;
53  } else {
54  json_decref(json);
55  return -1;
56  }
